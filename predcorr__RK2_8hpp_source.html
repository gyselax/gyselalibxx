83 void operator()(DFieldXY allfdistribu, double const dt, int const nbiter)
84 {
85 // Index range
86 IdxRangeXY const meshXY = get_idx_range(allfdistribu);
87
88 // Output of the Poisson solver
89 DFieldMemXY electrostatic_potential_alloc(meshXY);
90 DFieldXY electrostatic_potential = get_field(electrostatic_potential_alloc);
91
92 VectorFieldMemXY_XY electric_field_alloc(meshXY);
93 VectorFieldXY_XY electric_field = get_field(electric_field_alloc);
94
95 // Definition of the RK2
96 RK2<DFieldMemXY, VectorFieldMemXY_XY> predictor_corrector(meshXY);
97
98 // Computation of the advection field: Poisson equation ---
99 std::function<void(VectorFieldXY_XY, DConstFieldXY)> define_electric_field
100 = [&](VectorFieldXY_XY electric_field, DConstFieldXY allfdistribu_const) {
101 IdxRangeXY idx_range_xy(get_idx_range<GridX, GridY>(allfdistribu_const));
102
103 // --- compute electrostatic potential and electric field:
104 DFieldMemXY electrostatic_potential_alloc(idx_range_xy);
105 DFieldXY electrostatic_potential = get_field(electrostatic_potential_alloc);
106
107 /*
108 The applied Poisson solver needs a modifiable field for allfdistribu.
109 The time stepper uses a constant field type. So we create a temporary
110 allfdistribu_alloc chunk containing the values of the constant
111 allfdistribu to solve the type conflict in the Poisson solver.
112 */
113 DFieldMemXY allfdistribu_alloc(idx_range_xy);
114 DFieldXY allfdistribu = get_field(allfdistribu_alloc);
115 ddc::parallel_deepcopy(
116 Kokkos::DefaultExecutionSpace(),
117 allfdistribu,
118 allfdistribu_const);
119
120
121 m_poisson_solver(electrostatic_potential, electric_field, allfdistribu);
122 };
123
124 // Advection operator ---
125 std::function<void(DFieldXY, VectorConstFieldXY_XY, double)> advect_allfdistribu
126 = [&](DFieldXY allfdistribu, VectorConstFieldXY_XY electric_field, double dt) {
127 DConstFieldXY electric_field_x(ddcHelper::get<X>(electric_field));
128 DConstFieldXY electric_field_y(ddcHelper::get<Y>(electric_field));
129
130 // --- compute advection field:
131 IdxRangeXY idx_range = get_idx_range(electric_field);
132 DFieldMemXY advection_field_x_alloc(idx_range);
133 DFieldMemXY advection_field_y_alloc(idx_range);
134 DFieldXY advection_field_x = get_field(advection_field_x_alloc);
135 DFieldXY advection_field_y = get_field(advection_field_y_alloc);
136 ddc::parallel_for_each(
137 Kokkos::DefaultExecutionSpace(),
138 meshXY,
139 KOKKOS_LAMBDA(IdxXY const i_xy) {
140 advection_field_x(i_xy) = -electric_field_y(i_xy);
141 advection_field_y(i_xy) = electric_field_x(i_xy);
142 });
143
144
145 // --- Strang splitting for the advection
146 m_advection_x(allfdistribu, advection_field_x, dt / 2);
147 m_advection_y(allfdistribu, advection_field_y, dt);
148 m_advection_x(allfdistribu, advection_field_x, dt / 2);
149 };
150
151
152
153 // Iteration on the number of steps.
154 for (int iter(1); iter < nbiter + 1; ++iter) {
155 predictor_corrector
156 .update(Kokkos::DefaultExecutionSpace(),
157 allfdistribu,
158 dt,
159 define_electric_field,
160 advect_allfdistribu);
161
162 // Save the data ---
163 m_poisson_solver(electrostatic_potential, electric_field, allfdistribu);
164 auto allfdistribu_host = ddc::create_mirror_and_copy(allfdistribu);
165 auto electrostatic_potential_host
166 = ddc::create_mirror_and_copy(electrostatic_potential);
167 auto electric_field_x_host
168 = ddc::create_mirror_and_copy(ddcHelper::get<X>(electric_field));
169 auto electric_field_y_host
170 = ddc::create_mirror_and_copy(ddcHelper::get<Y>(electric_field));
171 ddc::PdiEvent("iteration")
172 .with("iter", iter)
173 .with("time_saved", iter * dt)
174 .with("fdistribu", allfdistribu_host)
175 .with("electrostatic_potential", electrostatic_potential_host)
176 .with("electric_field_x", electric_field_x_host)
177 .with("electric_field_y", electric_field_y_host);
178 }
179 };
