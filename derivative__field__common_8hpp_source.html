198  template <class QueryDDim, class... ODDims>
199  KOKKOS_FUNCTION constexpr auto get_slicer_for(
200  ddc::DiscreteElement<ODDims...> const& slice_idx,
201  int array_idx) const
202  {
203  if constexpr (!ddc::in_tags_v<QueryDDim, ddc::detail::TypeSeq<ODDims...>>) {
204  return std::experimental::full_extent;
205  } else {
206  if constexpr (ddc::in_tags_v<QueryDDim, physical_deriv_dims>) {
207  // Physical dimension along which derivatives are known
208  // If information is available about the physical domain
209  if (array_idx & (1 << ddc::type_seq_rank_v<ddc::Deriv<QueryDDim>, deriv_tags>)) {
210  // If the derivative is being requested
211  return m_cross_derivative_domain.get_index(ddc::select<QueryDDim>(slice_idx));
212  }
213  }
214  if constexpr (ddc::in_tags_v<QueryDDim, physical_dims>) {
215  // Physical dimension along which derivatives are not known
216  return std::size_t(
217  (ddc::select<QueryDDim>(slice_idx)
218  - ddc::select<QueryDDim>(m_physical_domain).front()));
219  }
220  if constexpr (ddc::in_tags_v<QueryDDim, deriv_tags>) {
221  // Derivative dimension
222  if (array_idx & (1 << ddc::type_seq_rank_v<QueryDDim, deriv_tags>)) {
223  // If array contains derivatives
224  return std::size_t(
225  (ddc::select<QueryDDim>(slice_idx)
226  - ddc::DiscreteElement<QueryDDim>(1)));
227  } else {
228  // If array doesn't contain derivatives
229  assert(ddc::select<QueryDDim>(slice_idx) == ddc::DiscreteElement<QueryDDim>(0));
230  return std::size_t(0);
231  }
232  }
233  }
234  }

246  template <class QueryDDim, class... ODDims>
247  KOKKOS_FUNCTION constexpr auto get_slicer_for(
248  ddc::DiscreteDomain<ODDims...> const& slice_domain,
249  int array_idx) const
250  {
251  if constexpr (!ddc::in_tags_v<QueryDDim, ddc::detail::TypeSeq<ODDims...>>) {
252  return std::experimental::full_extent;
253  } else {
254  if constexpr (ddc::in_tags_v<QueryDDim, physical_deriv_dims>) {
255  // Physical dimension along which derivatives are known
256  // If information is available about the physical domain
257  ddc::DiscreteDomain<QueryDDim> requested_dom(slice_domain);
258  if (array_idx & (1 << ddc::type_seq_rank_v<ddc::Deriv<QueryDDim>, deriv_tags>)) {
259  // If the derivative is being requested
260  assert(m_cross_derivative_domain.contains(requested_dom));
261  return std::pair<std::size_t, std::size_t>(
262  m_cross_derivative_domain.get_index(requested_dom.front()),
263  m_cross_derivative_domain.get_index(requested_dom.back()) + 1);
264  }
265  }
266  if constexpr (ddc::in_tags_v<QueryDDim, physical_dims>) {
267  // Physical dimension along which derivatives are not known
268  return std::pair<std::size_t, std::size_t>(
269  ddc::select<QueryDDim>(slice_domain).front()
270  - ddc::select<QueryDDim>(m_physical_domain).front(),
271  ddc::select<QueryDDim>(slice_domain).back() + 1
272  - ddc::select<QueryDDim>(m_physical_domain).front());
273  }
274  if constexpr (ddc::in_tags_v<QueryDDim, deriv_tags>) {
275  // Derivative dimension
276  if (array_idx & (1 << ddc::type_seq_rank_v<QueryDDim, deriv_tags>)) {
277  // If array contains derivatives
278  return std::pair<std::size_t, std::size_t>(
279  ddc::select<QueryDDim>(slice_domain).front()
280  - ddc::DiscreteElement<QueryDDim>(1),
281  ddc::select<QueryDDim>(slice_domain).back() + 1
282  - ddc::DiscreteElement<QueryDDim>(1));
283  } else {
284  // If array doesn't contain derivatives
285  assert(ddc::select<QueryDDim>(slice_domain).front()
286  == ddc::DiscreteElement<QueryDDim>(0));
287  assert(ddc::select<QueryDDim>(slice_domain).back()
288  == ddc::select<QueryDDim>(slice_domain).front());
289  return std::pair<std::size_t, std::size_t>(0, 1);
290  }
291  }
292  }

306  template <class... ODims>
307  KOKKOS_FUNCTION auto get_internal_chunk(ddc::DiscreteDomain<ODims...> dom) const
308  {
309  // Get the types related to the provided information
310  using provided_tags = ddc::detail::TypeSeq<ODims...>;
311  using provided_deriv_tags = detail::deriv_sub_set_t<provided_tags>;
312 
313  // Get the types related to the implicit information
314  using remaining_deriv_tags = ddc::type_seq_remove_t<deriv_tags, provided_deriv_tags>;
315  using remaining_deriv_domain_type =
316  typename ddc::detail::convert_type_seq_to_discrete_domain<remaining_deriv_tags>;
317 
318  // Find the domain of the derivatives (either provided or a domain containing only the 0-th derivative)
319  remaining_deriv_domain_type no_deriv_domain = detail::get_domain_from_element(
320  detail::no_derivative_element<remaining_deriv_tags>());
321  discrete_deriv_domain_type deriv_domain(dom, no_deriv_domain);
322 
323  // Find the physical domain of the chunk
324  physical_domain_type local_physical_domain(detail::select_default(dom, m_physical_domain));
325 
326  // Find the discrete domain of the chunk
327  discrete_domain_type full_domain(local_physical_domain, deriv_domain);
328 
329  // Find the index of the internal chunk
330  int const array_idx = get_array_index(deriv_domain.front());
331 
332  // Get the relevant internal chunk
333  internal_mdspan_type internal_view = internal_chunks[array_idx];
334  // Slice the relevant section of the internal chunk
335  auto subview = std::experimental::
336  submdspan(internal_view, get_slicer_for<DDims>(full_domain, array_idx)...);
337  // Create a ddc::ChunkSpan with the expected domain
338  ddc::ChunkSpan<
339  element_type,
340  discrete_domain_type,
341  typename decltype(subview)::layout_type,
342  typename chunk_type::memory_space>
343  local_chunk(subview, full_domain);
344 
345  // If necessary, slice off the derivative dimensions deduced implicitly
346  if constexpr (ddcHelper::type_seq_length_v<remaining_deriv_tags> == 0) {
347  return local_chunk;
348  } else {
349  return local_chunk[no_deriv_domain.front()];
350  }
351  }

364  template <class... ODims>
365  KOKKOS_FUNCTION auto get_internal_chunk(ddc::DiscreteElement<ODims...> elem) const
366  {
367  // Get the types related to the provided information
368  using provided_tags = ddc::detail::TypeSeq<ODims...>;
369  using provided_deriv_tags = detail::deriv_sub_set_t<provided_tags>;
370  using provided_physical_tags = ddc::type_seq_remove_t<provided_tags, provided_deriv_tags>;
371  using provided_deriv_domain_type
372  = ddc::detail::convert_type_seq_to_discrete_domain<provided_deriv_tags>;
373  using provided_deriv_element_type =
374  typename provided_deriv_domain_type::discrete_element_type;
375 
376  // Get the types related to the implicit information
377  using remaining_deriv_tags = ddc::type_seq_remove_t<deriv_tags, provided_deriv_tags>;
378  using remaining_deriv_domain_type =
379  typename ddc::detail::convert_type_seq_to_discrete_domain<remaining_deriv_tags>;
380  using remaining_deriv_element_type =
381  typename remaining_deriv_domain_type::discrete_element_type;
382 
383  // Get the types related to the final chunk type
384  using sliced_tags = ddc::type_seq_merge_t<provided_physical_tags, deriv_tags>;
385  using sliced_domain_type =
386  typename ddc::detail::convert_type_seq_to_discrete_domain<sliced_tags>;
387  using sliced_element_type = typename sliced_domain_type::discrete_element_type;
388  using final_tags = ddc::type_seq_remove_t<ddc::detail::TypeSeq<DDims...>, sliced_tags>;
389  using final_domain_type =
390  typename ddc::detail::convert_type_seq_to_discrete_domain<final_tags>;
391 
392  // Get the index of the relevant derivatives
393  provided_deriv_element_type requested_derivs(elem);
394  remaining_deriv_element_type no_deriv_elements(
395  detail::no_derivative_element<remaining_deriv_tags>());
396  discrete_deriv_element_type deriv_index(requested_derivs, no_deriv_elements);
397 
398  // Get the element which will slice the mdspan
399  sliced_element_type slice_idx(elem, no_deriv_elements);
400 
401  // Find the index of the internal chunk
402  int const array_idx = get_array_index(deriv_index);
403 
404  // Get the final domain
405  final_domain_type final_domain(m_physical_domain);
406 
407  // Get the relevant internal chunk
408  internal_mdspan_type internal_view = internal_chunks[array_idx];
409  // Slice the relevant section of the internal chunk
410  auto subview = std::experimental::
411  submdspan(internal_view, get_slicer_for<DDims>(slice_idx, array_idx)...);
412  // Create a ddc::ChunkSpan with the expected domain
413  ddc::ChunkSpan<
414  element_type,
415  final_domain_type,
416  typename decltype(subview)::layout_type,
417  typename chunk_type::memory_space>
418  local_chunk(subview, final_domain);
419 
420  return local_chunk;
421  }

520  template <class... ODims>
521  auto get_mdspan(ddc::DiscreteDomain<ODims...> provided_deriv_domain)
522  {
523  static_assert(((ddc::in_tags_v<ODims, deriv_tags>)&&...));
524  using provided_deriv_tags = ddc::detail::TypeSeq<ODims...>;
525  using remaining_deriv_tags = ddc::type_seq_remove_t<deriv_tags, provided_deriv_tags>;
526  using remaining_deriv_domain_type =
527  typename ddc::detail::convert_type_seq_to_discrete_domain<remaining_deriv_tags>;
528  using remaining_deriv_element_type =
529  typename remaining_deriv_domain_type::discrete_element_type;
530 
531  discrete_deriv_element_type default_derivatives
532  = detail::no_derivative_element<deriv_tags>();
533 
534  remaining_deriv_element_type deriv_elements(default_derivatives);
535  discrete_deriv_element_type deriv_index(
536  detail::select_default(provided_deriv_domain.front(), default_derivatives));
537 
538  int const array_idx = get_array_index(deriv_index);
539 
540  internal_mdspan_type internal_view = internal_chunks[array_idx];
541  auto subview_all_dims = std::experimental::submdspan(
542  internal_view,
543  get_slicer_for<DDims>(provided_deriv_domain, array_idx)...);
544  auto subview = std::experimental::
545  submdspan(subview_all_dims, get_slicer_for<DDims>(deriv_elements, array_idx)...);
546  return subview;
547  }
