213 template <class QueryDDim, class... ODDims>
214 KOKKOS_FUNCTION constexpr auto get_slicer_for(Idx<ODDims...> const& slice_idx, int array_idx)
215 const
216 {
217 if constexpr (!ddc::in_tags_v<QueryDDim, ddc::detail::TypeSeq<ODDims...>>) {
218 return Kokkos::full_extent;
219 } else {
220 if constexpr (ddc::in_tags_v<QueryDDim, physical_deriv_grids>) {
221 // Physical dimension along which derivatives are known
222 // If information is available about the physical index range
223 if (array_idx & (1 << ddc::type_seq_rank_v<ddc::Deriv<QueryDDim>, deriv_tags>)) {
224 // If the derivative is being requested
225 return m_cross_derivative_idx_range.get_index(
226 ddc::select<QueryDDim>(slice_idx));
227 }
228 }
229 if constexpr (ddc::in_tags_v<QueryDDim, physical_grids>) {
230 // Physical dimension along which derivatives are not known
231 return std::size_t(
232 (ddc::select<QueryDDim>(slice_idx)
233 - ddc::select<QueryDDim>(m_physical_idx_range).front()));
234 }
235 if constexpr (ddc::in_tags_v<QueryDDim, deriv_tags>) {
236 // Derivative dimension
237 if (array_idx & (1 << ddc::type_seq_rank_v<QueryDDim, deriv_tags>)) {
238 // If array contains derivatives
239 return std::size_t((ddc::select<QueryDDim>(slice_idx) - Idx<QueryDDim>(1)));
240 } else {
241 // If array doesn't contain derivatives
242 assert(ddc::select<QueryDDim>(slice_idx) == Idx<QueryDDim>(0));
243 return std::size_t(0);
244 }
245 }
246 }
247 }

259 template <class QueryDDim, class... ODDims>
260 KOKKOS_FUNCTION constexpr auto get_slicer_for(
261 IdxRange<ODDims...> const& slice_idx_range,
262 int array_idx) const
263 {
264 if constexpr (!ddc::in_tags_v<QueryDDim, ddc::detail::TypeSeq<ODDims...>>) {
265 return Kokkos::full_extent;
266 } else {
267 if constexpr (ddc::in_tags_v<QueryDDim, physical_deriv_grids>) {
268 // Physical dimension along which derivatives are known
269 // If information is available about the physical index range
270 IdxRange<QueryDDim> idx_range_requested(slice_idx_range);
271 if (array_idx & (1 << ddc::type_seq_rank_v<ddc::Deriv<QueryDDim>, deriv_tags>)) {
272 // If the derivative is being requested
273 assert(m_cross_derivative_idx_range.contains(idx_range_requested));
274 return std::pair<std::size_t, std::size_t>(
275 m_cross_derivative_idx_range.get_index(idx_range_requested.front()),
276 m_cross_derivative_idx_range.get_index(idx_range_requested.back()) + 1);
277 }
278 }
279 if constexpr (ddc::in_tags_v<QueryDDim, physical_grids>) {
280 // Physical dimension along which derivatives are not known
281 return std::pair<std::size_t, std::size_t>(
282 ddc::select<QueryDDim>(slice_idx_range).front()
283 - ddc::select<QueryDDim>(m_physical_idx_range).front(),
284 ddc::select<QueryDDim>(slice_idx_range).back() + 1
285 - ddc::select<QueryDDim>(m_physical_idx_range).front());
286 }
287 if constexpr (ddc::in_tags_v<QueryDDim, deriv_tags>) {
288 // Derivative dimension
289 if (array_idx & (1 << ddc::type_seq_rank_v<QueryDDim, deriv_tags>)) {
290 // If array contains derivatives
291 return std::pair<std::size_t, std::size_t>(
292 ddc::select<QueryDDim>(slice_idx_range).front() - Idx<QueryDDim>(1),
293 ddc::select<QueryDDim>(slice_idx_range).back() + 1 - Idx<QueryDDim>(1));
294 } else {
295 // If array doesn't contain derivatives
296 assert(ddc::select<QueryDDim>(slice_idx_range).front() == Idx<QueryDDim>(0));
297 assert(ddc::select<QueryDDim>(slice_idx_range).back()
298 == ddc::select<QueryDDim>(slice_idx_range).front());
299 return std::pair<std::size_t, std::size_t>(0, 1);
300 }
301 }
302 }

316 template <class... ODims>
317 KOKKOS_FUNCTION auto get_internal_field(IdxRange<ODims...> idx_range) const
318 {
319 // Get the types related to the provided information
320 using provided_tags = ddc::detail::TypeSeq<ODims...>;
321 using provided_deriv_tags = detail::deriv_sub_set_t<provided_tags>;
322
323 // Get the types related to the implicit information
324 using remaining_deriv_tags = ddc::type_seq_remove_t<deriv_tags, provided_deriv_tags>;
325 using remaining_deriv_idx_range_type =
326 typename ddc::detail::convert_type_seq_to_discrete_domain_t<remaining_deriv_tags>;
327
328 // Find the index range of the derivatives (either provided or an index range containing only the 0-th derivative)
329 remaining_deriv_idx_range_type no_deriv_idx_range = detail::get_idx_range_from_element(
330 detail::no_derivative_element<remaining_deriv_tags>());
331 discrete_deriv_idx_range_type deriv_idx_range(idx_range, no_deriv_idx_range);
332
333 // Find the physical index range of the field
334 physical_idx_range_type local_physical_idx_range(
335 detail::select_default(idx_range, m_physical_idx_range));
336
337 // Find the discrete index range of the field
338 index_range_type full_idx_range(local_physical_idx_range, deriv_idx_range);
339
340 // Find the index of the internal field
341 int const array_idx = get_array_index(deriv_idx_range.front());
342
343 // Get the relevant internal field
344 internal_mdspan_type internal_view = internal_fields[array_idx];
345 // Slice the relevant section of the internal field
346 auto subview = Kokkos::
347 submdspan(internal_view, get_slicer_for<DDims>(full_idx_range, array_idx)...);
348 // Create a Field with the expected index range
349 Field<element_type,
350 index_range_type,
351 typename chunk_type::memory_space,
352 typename decltype(subview)::layout_type>
353 local_field(subview, full_idx_range);
354
355 // If necessary, slice off the derivative dimensions deduced implicitly
356 if constexpr (ddc::type_seq_size_v<remaining_deriv_tags> == 0) {
357 return local_field;
358 } else {
359 return local_field[no_deriv_idx_range.front()];
360 }
361 }

374 template <class... ODims>
375 KOKKOS_FUNCTION auto get_internal_field(Idx<ODims...> elem) const
376 {
377 // Get the types related to the provided information
378 using provided_tags = ddc::detail::TypeSeq<ODims...>;
379 using provided_deriv_tags = detail::deriv_sub_set_t<provided_tags>;
380 using provided_physical_tags = ddc::type_seq_remove_t<provided_tags, provided_deriv_tags>;
381 using provided_deriv_idx_range_type
382 = ddc::detail::convert_type_seq_to_discrete_domain_t<provided_deriv_tags>;
383 using provided_deriv_index_type =
384 typename provided_deriv_idx_range_type::discrete_element_type;
385
386 // Get the types related to the implicit information
387 using remaining_deriv_tags = ddc::type_seq_remove_t<deriv_tags, provided_deriv_tags>;
388 using remaining_deriv_idx_range_type =
389 typename ddc::detail::convert_type_seq_to_discrete_domain_t<remaining_deriv_tags>;
390 using remaining_deriv_index_type =
391 typename remaining_deriv_idx_range_type::discrete_element_type;
392
393 // Get the types related to the final field type
394 using sliced_tags = ddc::type_seq_merge_t<provided_physical_tags, deriv_tags>;
395 using sliced_idx_range_type =
396 typename ddc::detail::convert_type_seq_to_discrete_domain_t<sliced_tags>;
397 using sliced_index_type = typename sliced_idx_range_type::discrete_element_type;
398 using final_tags = ddc::type_seq_remove_t<ddc::detail::TypeSeq<DDims...>, sliced_tags>;
399 using final_idx_range_type =
400 typename ddc::detail::convert_type_seq_to_discrete_domain_t<final_tags>;
401
402 // Get the index of the relevant derivatives
403 provided_deriv_index_type requested_derivs(elem);
404 remaining_deriv_index_type no_deriv_elements(
405 detail::no_derivative_element<remaining_deriv_tags>());
406 discrete_deriv_index_type deriv_index(requested_derivs, no_deriv_elements);
407
408 // Get the element which will slice the mdspan
409 sliced_index_type slice_idx(elem, no_deriv_elements);
410
411 // Find the index of the internal field
412 int const array_idx = get_array_index(deriv_index);
413
414 // Get the final index range
415 final_idx_range_type final_idx_range(m_physical_idx_range);
416
417 // Get the relevant internal field
418 internal_mdspan_type internal_view = internal_fields[array_idx];
419 // Slice the relevant section of the internal field
420 auto subview
421 = Kokkos::submdspan(internal_view, get_slicer_for<DDims>(slice_idx, array_idx)...);
422 // Create a Field with the expected index range
423 Field<element_type,
424 final_idx_range_type,
425 typename chunk_type::memory_space,
426 typename decltype(subview)::layout_type>
427 local_field(subview, final_idx_range);
428
429 return local_field;
430 }

528 template <class... ODims>
529 auto get_mdspan(IdxRange<ODims...> provided_deriv_idx_range)
530 {
531 static_assert(((ddc::in_tags_v<ODims, deriv_tags>)&&...));
532 using provided_deriv_tags = ddc::detail::TypeSeq<ODims...>;
533 using remaining_deriv_tags = ddc::type_seq_remove_t<deriv_tags, provided_deriv_tags>;
534 using remaining_deriv_idx_range_type =
535 typename ddc::detail::convert_type_seq_to_discrete_domain_t<remaining_deriv_tags>;
536 using remaining_deriv_index_type =
537 typename remaining_deriv_idx_range_type::discrete_element_type;
538
539 discrete_deriv_index_type default_derivatives = detail::no_derivative_element<deriv_tags>();
540
541 remaining_deriv_index_type deriv_elements(default_derivatives);
542 discrete_deriv_index_type deriv_index(
543 detail::select_default(provided_deriv_idx_range.front(), default_derivatives));
544
545 int const array_idx = get_array_index(deriv_index);
546
547 internal_mdspan_type internal_view = internal_fields[array_idx];
548 auto subview_all_dims = Kokkos::submdspan(
549 internal_view,
550 get_slicer_for<DDims>(provided_deriv_idx_range, array_idx)...);
551 auto subview = Kokkos::
552 submdspan(subview_all_dims, get_slicer_for<DDims>(deriv_elements, array_idx)...);
553 return subview;
554 }
