253  template <class ExecSpace>
254  bool have_converged(ExecSpace const& exec_space, ValConstField y_old, ValConstField y_new) const
255  {
256  IdxRange const idx_range = get_idx_range(y_old);
257 
258  double norm_old = ddc::parallel_transform_reduce(
259  exec_space,
260  idx_range,
261  0.,
262  ddc::reducer::max<double>(),
263  KOKKOS_LAMBDA(Idx const idx) { return norm_inf(y_old(idx)); });
264 
265  double max_diff = ddc::parallel_transform_reduce(
266  exec_space,
267  idx_range,
268  0.,
269  ddc::reducer::max<double>(),
270  KOKKOS_LAMBDA(Idx const idx) { return norm_inf(y_old(idx) - y_new(idx)); });
271 
272  return (max_diff / norm_old) < m_epsilon;
273  }
