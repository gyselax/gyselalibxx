158 void update(
159 ExecSpace const& exec_space,
160 ValField y,
161 double dt,
162 std::function<void(DerivFieldMem, ValConstField)> dy,
163 std::function<void(ValField, DerivConstField, double)> y_update) const
164 {
165 static_assert(
166 Kokkos::SpaceAccessibility<ExecSpace, typename FieldMemType::memory_space>::
167 accessible,
168 "MemorySpace has to be accessible for ExecutionSpace.");
169 static_assert(
170 Kokkos::SpaceAccessibility<ExecSpace, typename DerivFieldMemType::memory_space>::
171 accessible,
172 "MemorySpace has to be accessible for ExecutionSpace.");
173 FieldMemType m_y_init_alloc(m_idx_range);
174 FieldMemType m_y_old_alloc(m_idx_range);
175 DerivFieldMemType m_k1_alloc(m_idx_range);
176 DerivFieldMemType m_k_new_alloc(m_idx_range);
177 DerivFieldMemType m_k_total_alloc(m_idx_range);
178 ValField m_y_init = get_field(m_y_init_alloc);
179 ValField m_y_old = get_field(m_y_old_alloc);
180 DerivFieldMem m_k1 = get_field(m_k1_alloc);
181 DerivFieldMem m_k_new = get_field(m_k_new_alloc);
182 DerivFieldMem m_k_total = get_field(m_k_total_alloc);
183
184
185 copy(m_y_init, y);
186
187 // --------- Calculate k1 ------------
188 // Calculate k1 = f(y_n)
189 dy(m_k1, y);
190
191 // -------- Calculate k_new ----------
192 bool not_converged = true;
193 int counter = 0;
194 do {
195 counter++;
196
197 // Calculate k_new = f(y_new)
198 dy(m_k_new, y);
199
200 // Calculation of step
201 if constexpr (is_field_v<DerivFieldMemType>) {
202 ddc::parallel_for_each(
203 exec_space,
204 get_idx_range(m_k_total),
205 KOKKOS_CLASS_LAMBDA(Idx const i) {
206 // k_total = k1 + k_new
207 fill_k_total(i, m_k_total, m_k1(i) + m_k_new(i));
208 });
209 } else {
210 ddc::parallel_for_each(
211 exec_space,
212 get_idx_range(m_k_total),
213 KOKKOS_CLASS_LAMBDA(Idx const i) {
214 // k_total = k1 + k_new
215 m_k_total(i) = m_k1(i) + m_k_new(i);
216 });
217 }
218
219 // Save the old characteristic feet
220 copy(m_y_old, y);
221
222 // Re-initiliase the characteristic feet
223 copy(y, m_y_init);
224
225 // Calculate y_new := y_n + h/2*(k_1 + k_new)
226 y_update(y, m_k_total, 0.5 * dt);
227
228
229 // Check convergence
230 not_converged = not have_converged(exec_space, m_y_old, y);
231
232
233 } while (not_converged and (counter < m_max_counter));
234 }

253 template <class ExecSpace>
254 bool have_converged(ExecSpace const& exec_space, ValConstField y_old, ValConstField y_new) const
255 {
256 IdxRange const idx_range = get_idx_range(y_old);
257
258 double norm_old = ddc::parallel_transform_reduce(
259 exec_space,
260 idx_range,
261 0.,
262 ddc::reducer::max<double>(),
263 KOKKOS_LAMBDA(Idx const idx) { return norm_inf(y_old(idx)); });
264
265 double max_diff = ddc::parallel_transform_reduce(
266 exec_space,
267 idx_range,
268 0.,
269 ddc::reducer::max<double>(),
270 KOKKOS_LAMBDA(Idx const idx) { return norm_inf(y_old(idx) - y_new(idx)); });
271
272 return (max_diff / norm_old) < m_epsilon;
273 }
