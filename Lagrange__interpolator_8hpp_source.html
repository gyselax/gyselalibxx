96  auto batch_domain = ddc::remove_dims_of<DDimI>(inout_data.domain());
97  ddc::parallel_for_each(

101  Lagrange<Kokkos::DefaultExecutionSpace, DDimI, BcMin, BcMax> evaluator(
102  deg,
103  inout_data_tmp[i],
104  inout_data.template domain<DDimI>(),
105  ghost);
106  for (ddc::DiscreteElement<DDimI> j : inout_data.template domain<DDimI>()) {
107  inout_data(i, j) = evaluator.evaluate(coordinates(i, j));
108  }
109  });
