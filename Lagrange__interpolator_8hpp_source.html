79 derivs_xmax
80 = std::nullopt) const override
81 {
82 static_assert(
83 BcMin != BCond::PERIODIC,
84 "PERIODIC Boundary condition is not supported yet in LagrangeInterpolator.");
85
86 int const deg = m_degree;
87 auto const ghost = m_ghost;
88 auto inout_data_tmp_alloc
89 = ddc::create_mirror_and_copy(Kokkos::DefaultExecutionSpace(), inout_data);
90 auto inout_data_tmp = get_field(inout_data_tmp_alloc);
91 auto batch_idx_range = ddc::remove_dims_of<GridInterp>(get_idx_range(inout_data));
92 auto const interp_range = get_idx_range<GridInterp>(inout_data);
93 ddc::parallel_for_each(
94 Kokkos::DefaultExecutionSpace(),
95 batch_idx_range,
96 KOKKOS_LAMBDA(typename decltype(batch_idx_range)::discrete_element_type const i) {
97 Lagrange<Kokkos::DefaultExecutionSpace, GridInterp, BcMin, BcMax>
98 evaluator(deg, inout_data_tmp[i], interp_range, ghost);
99 for (Idx<GridInterp> j : interp_range) {
100 inout_data(i, j) = evaluator.evaluate(coordinates(i, j));
101 }
102 });
103 return inout_data;
104 }
