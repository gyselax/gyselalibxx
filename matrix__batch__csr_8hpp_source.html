189  void setup_solver() final
190  {
191  std::shared_ptr const gko_exec = m_batch_matrix_csr->get_executor();
192  //Check if the indices array is sorted, and sort it if necessary.
193  //The values array corresponding to the indices is also reordered.
194  for (size_t i = 0; i < batch_size(); i++) {
195  std::unique_ptr<gko::matrix::Csr<double, int>> tmp_matrix
196  = m_batch_matrix_csr->create_view_for_item(i);
197  if (!tmp_matrix->is_sorted_by_column_index()) {
198  tmp_matrix->sort_by_column_index();
199  }
200  }
201  if constexpr (
202  Solver == MatrixBatchCsrSolver::CG || Solver == MatrixBatchCsrSolver::BICGSTAB) {
203  // Create the solver factory
204  std::shared_ptr const residual_criterion
205  = gko::stop::ResidualNorm<double>::build().with_reduction_factor(m_tol).on(
206  gko_exec);
207 
208  std::shared_ptr const iterations_criterion
209  = gko::stop::Iteration::build().with_max_iters(m_max_iter).on(gko_exec);
210 
211  std::shared_ptr const preconditioner
212  = gko::preconditioner::Jacobi<double>::build()
213  .with_max_block_size(m_preconditionner_max_block_size)
214  .on(gko_exec);
215 
216  std::unique_ptr const solver_factory
217  = solver_type::build()
218  .with_preconditioner(preconditioner)
219  .with_criteria(residual_criterion, iterations_criterion)
220  .on(gko_exec);
221 
222  // Create the solvers
223  for (size_t i = 0; i < batch_size(); i++) {
224  m_solver.emplace_back(solver_factory->generate(
225  m_batch_matrix_csr->create_const_view_for_item(i)));
226  }
227  } else {
228  // Create the solver factory
229  std::shared_ptr const preconditioner
230  = gko::batch::preconditioner::Jacobi<double, int>::build()
231  .with_max_block_size(m_preconditionner_max_block_size)
232  .on(gko_exec);

320  double norm(int batch_idx) const
321  {
322  int const tmp_mat_size = size();
323  int const tmp_batch_size = batch_size();
324  double result = 0.;
325  double* vals_proxy = m_batch_matrix_csr->get_values();
326  int* row_ptr_proxy = m_batch_matrix_csr->get_row_ptrs();
327  Kokkos::View<double**, Kokkos::LayoutRight, typename ExecSpace::memory_space> vals_view(
328  vals_proxy,
329  tmp_batch_size,
330  m_batch_matrix_csr->get_num_elements_per_item());
331  Kokkos::View<int*, Kokkos::LayoutRight, typename ExecSpace::memory_space>
332  row_ptr_view(row_ptr_proxy, tmp_mat_size + 1);
333 
334  Kokkos::parallel_reduce(
335  "L-infinity norm",
336  Kokkos::RangePolicy<ExecSpace>(0, tmp_mat_size),
337  KOKKOS_LAMBDA(int const i, double& res) {
338  double row_sum = 0.;
339  for (int k = row_ptr_view[i]; k < row_ptr_view[i + 1]; k++) {
340  row_sum += Kokkos::abs(vals_view(batch_idx, k));
341  }
342  if (row_sum > res) {
343  res = row_sum;
344  }
345  },
346  Kokkos::Max<double>(result));
347 
348  return result;
349  }

363  std::unique_ptr<MatrixBatchCsr<Kokkos::DefaultExecutionSpace, Solver>>& matrix,

367 {

370  int const non_zero_per_system = vals_coo_host.extent(0) / batch_size;
371 
372  Kokkos::View<double**, Kokkos::LayoutRight, Kokkos::DefaultHostExecutionSpace>
373  values_view_host("", batch_size, non_zero_per_system);
374  Kokkos::View<int*, Kokkos::LayoutRight, Kokkos::DefaultHostExecutionSpace>
375  idx_view_host("", non_zero_per_system);
376  Kokkos::View<int*, Kokkos::LayoutRight, Kokkos::DefaultHostExecutionSpace>
377  nnz_per_row_view_host("", mat_size + 1);
378  nnz_per_row_view_host(0) = 0;
379  nnz_per_row_view_host(mat_size) = non_zero_per_system;
380 
381  for (int i = 0; i < mat_size; i++) {
382  int n_non_zero_in_row = 0;
383  for (int k = 0; k < non_zero_per_system; k++) {
384  if (row_coo_host(k) == i) {
385  int const idx = nnz_per_row_view_host(i) + n_non_zero_in_row;
386  idx_view_host(idx) = col_coo_host(k);
387  values_view_host(0, idx) = vals_coo_host(k);
388  n_non_zero_in_row++;
389  }
390  }
391  nnz_per_row_view_host(i + 1) = nnz_per_row_view_host(i) + n_non_zero_in_row;
392  }
393  for (int batch_idx = 1; batch_idx < batch_size; batch_idx++) {
394  for (int idx = 0; idx < non_zero_per_system; idx++) {
395  values_view_host(batch_idx, idx) = vals_coo_host(batch_idx * non_zero_per_system + idx);
396  }
397  }
398  auto [values, col_idx, nnz_per_row] = matrix->get_batch_csr();
399  Kokkos::deep_copy(values, values_view_host);
400  Kokkos::deep_copy(col_idx, idx_view_host);
401  Kokkos::deep_copy(nnz_per_row, nnz_per_row_view_host);
