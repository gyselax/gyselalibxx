215  m_solver.emplace_back(solver_factory->generate(

217  }

223  .on(gko_exec);

226  .with_max_iterations(m_max_iter)

316  {
317  int const tmp_mat_size = get_size();
318  int const tmp_batch_size = get_batch_size();
319  double result = 0.;
320  double* vals_proxy = m_batch_matrix_csr->get_values();
321  int* row_ptr_proxy = m_batch_matrix_csr->get_row_ptrs();
322  Kokkos::View<double**, Kokkos::LayoutRight, typename ExecSpace::memory_space> vals_view(

327  row_ptr_view(row_ptr_proxy, tmp_mat_size + 1);
328 
329  Kokkos::parallel_reduce(
330  "L-infinity norm",
331  Kokkos::RangePolicy<ExecSpace>(0, tmp_mat_size),
332  KOKKOS_LAMBDA(int const i, double& res) {
333  double row_sum = 0.;
334  for (int k = row_ptr_view[i]; k < row_ptr_view[i + 1]; k++) {
335  row_sum += Kokkos::abs(vals_view(batch_idx, k));
336  }
337  if (row_sum > res) {
338  res = row_sum;
339  }
340  },
341  Kokkos::Max<double>(result));
342 
343  return result;

362 {

366 
367  Kokkos::View<double**, Kokkos::LayoutRight, Kokkos::DefaultHostExecutionSpace>
368  values_view_host("", batch_size, non_zero_per_system);
369  Kokkos::View<int*, Kokkos::LayoutRight, Kokkos::DefaultHostExecutionSpace>
370  idx_view_host("", non_zero_per_system);
371  Kokkos::View<int*, Kokkos::LayoutRight, Kokkos::DefaultHostExecutionSpace>
372  nnz_per_row_view_host("", mat_size + 1);
373  nnz_per_row_view_host(0) = 0;
374  nnz_per_row_view_host(mat_size) = non_zero_per_system;
375 
376  for (int i = 0; i < mat_size; i++) {
377  int n_non_zero_in_row = 0;
378  for (int k = 0; k < non_zero_per_system; k++) {
379  if (row_coo_host(k) == i) {
380  int const idx = nnz_per_row_view_host(i) + n_non_zero_in_row;
381  idx_view_host(idx) = col_coo_host(k);
382  values_view_host(0, idx) = vals_coo_host(k);
383  n_non_zero_in_row++;
384  }
385  }
386  nnz_per_row_view_host(i + 1) = nnz_per_row_view_host(i) + n_non_zero_in_row;
387  }
388  for (int batch_idx = 1; batch_idx < batch_size; batch_idx++) {
389  for (int idx = 0; idx < non_zero_per_system; idx++) {
390  values_view_host(batch_idx, idx) = vals_coo_host(batch_idx * non_zero_per_system + idx);
391  }
392  }
393  auto [values, col_idx, nnz_per_row] = matrix->get_batch_csr();
394  Kokkos::deep_copy(values, values_view_host);
395  Kokkos::deep_copy(col_idx, idx_view_host);
396  Kokkos::deep_copy(nnz_per_row, nnz_per_row_view_host);
