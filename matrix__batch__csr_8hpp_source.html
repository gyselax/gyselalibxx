190 void setup_solver() final
191 {
192 std::shared_ptr const gko_exec = m_batch_matrix_csr->get_executor();
193 //Check if the indices array is sorted, and sort it if necessary.
194 //The values array corresponding to the indices is also reordered.
195 for (size_t i = 0; i < batch_size(); i++) {
196 std::unique_ptr<gko::matrix::Csr<double, int>> tmp_matrix
197 = m_batch_matrix_csr->create_view_for_item(i);
198 if (!tmp_matrix->is_sorted_by_column_index()) {
199 tmp_matrix->sort_by_column_index();
200 }
201 }
202 if constexpr (
203 Solver == MatrixBatchCsrSolver::CG || Solver == MatrixBatchCsrSolver::BICGSTAB) {
204 // Create the solver factory
205 std::shared_ptr const residual_criterion
206 = gko::stop::ResidualNorm<double>::build().with_reduction_factor(m_tol).on(
207 gko_exec);
208
209 std::shared_ptr const iterations_criterion
210 = gko::stop::Iteration::build().with_max_iters(m_max_iter).on(gko_exec);
211
212 std::shared_ptr const preconditioner
213 = gko::preconditioner::Jacobi<double>::build()
214 .with_max_block_size(m_preconditionner_max_block_size)
215 .on(gko_exec);
216
217 std::unique_ptr const solver_factory
218 = solver_type::build()
219 .with_preconditioner(preconditioner)
220 .with_criteria(residual_criterion, iterations_criterion)
221 .on(gko_exec);
222
223 // Create the solvers
224 for (size_t i = 0; i < batch_size(); i++) {
225 m_solver.emplace_back(solver_factory->generate(
226 m_batch_matrix_csr->create_const_view_for_item(i)));
227 }
228 } else {
229 // Create the solver factory
230 std::shared_ptr const preconditioner
231 = gko::batch::preconditioner::Jacobi<double, int>::build()
232 .with_max_block_size(m_preconditionner_max_block_size)
233 .on(gko_exec);
234
235 std::shared_ptr solver_factory = solver_type::build()
236 .with_max_iterations(m_max_iter)
237 .with_tolerance(m_tol)
238 .with_preconditioner(preconditioner)
239 .on(gko_exec);
240
241 // Create the solver
242 m_solver = solver_factory->generate(m_batch_matrix_csr);
243 }
244 gko_exec->synchronize();
245 }

267 void solve(BatchedRHS const x, BatchedRHS const b) const
268 {
269 if constexpr (
270 Solver == MatrixBatchCsrSolver::CG || Solver == MatrixBatchCsrSolver::BICGSTAB) {
271 for (size_t i = 0; i < batch_size(); i++) {
272 std::shared_ptr const gko_exec = m_solver[i]->get_executor();
273
274 // Create a logger to obtain the iteration counts and "implicit" residual norms for every system after the solve.
275 std::shared_ptr const logger = gko::log::Convergence<double>::create();
276
277 // Solve & log
278 m_solver[i]->add_logger(logger);
279 m_solver[i]
280 ->apply(to_gko_multivector(gko_exec, b)->create_const_view_for_item(i),
281 to_gko_multivector(gko_exec, x)->create_view_for_item(i));
282 m_solver[i]->remove_logger(logger);
283 // save logger data
284 if (m_with_logger) {
285 std::fstream log_file("csr_log.txt", std::ios::out | std::ios::app);
286 save_logger(
287 log_file,
288 i,
289 m_batch_matrix_csr->create_const_view_for_item(i),
290 Kokkos::subview(x, i, Kokkos::ALL),
291 Kokkos::subview(b, i, Kokkos::ALL),
292 logger,
293 m_tol);
294 log_file.close();
295 }
296 // Check convergency
297 if (!logger->has_converged()) {
298 throw std::runtime_error("Ginkgo did not converge in MatrixBatchCsr");
299 }
300 }
301 } else {
302 std::shared_ptr const gko_exec = m_solver->get_executor();
303
304 // Create a logger to obtain the iteration counts and "implicit" residual norms for every system after the solve.
305 std::shared_ptr const logger = gko::batch::log::BatchConvergence<double>::create();
306
307 // Solve & log
308 m_solver->add_logger(logger);
309 m_solver->apply(to_gko_multivector(gko_exec, b), to_gko_multivector(gko_exec, x));
310 m_solver->remove_logger(logger);
311
312 // Save logger data
313 if (m_with_logger) {
314 std::fstream log_file("csr_log.txt", std::ios::out | std::ios::app);
315 save_logger(log_file, m_batch_matrix_csr, x, b, logger, m_tol);
316 log_file.close();
317 }
318 // Check convergence
319 check_conv(batch_size(), m_tol, gko_exec, logger);
320 }
321 }

328 double norm(int batch_idx) const
329 {
330 int const tmp_mat_size = size();
331 int const tmp_batch_size = batch_size();
332 double result = 0.;
333 double* vals_proxy = m_batch_matrix_csr->get_values();
334 int* row_ptr_proxy = m_batch_matrix_csr->get_row_ptrs();
335 Kokkos::View<double**, Kokkos::LayoutRight, typename ExecSpace::memory_space> vals_view(
336 vals_proxy,
337 tmp_batch_size,
338 m_batch_matrix_csr->get_num_elements_per_item());
339 Kokkos::View<int*, Kokkos::LayoutRight, typename ExecSpace::memory_space>
340 row_ptr_view(row_ptr_proxy, tmp_mat_size + 1);
341
342 Kokkos::parallel_reduce(
343 "L-infinity norm",
344 Kokkos::RangePolicy<ExecSpace>(0, tmp_mat_size),
345 KOKKOS_LAMBDA(int const i, double& res) {
346 double row_sum = 0.;
347 for (int k = row_ptr_view[i]; k < row_ptr_view[i + 1]; k++) {
348 row_sum += Kokkos::abs(vals_view(batch_idx, k));
349 }
350 if (row_sum > res) {
351 res = row_sum;
352 }
353 },
354 Kokkos::Max<double>(result));
355
356 return result;
357 }

369template <MatrixBatchCsrSolver Solver = MatrixBatchCsrSolver::BICGSTAB>
370void convert_coo_to_csr(
371 std::unique_ptr<MatrixBatchCsr<Kokkos::DefaultExecutionSpace, Solver>>& matrix,
372 Kokkos::View<double*, Kokkos::LayoutRight, Kokkos::DefaultHostExecutionSpace> vals_coo_host,
373 Kokkos::View<int*, Kokkos::LayoutRight, Kokkos::DefaultHostExecutionSpace> row_coo_host,
374 Kokkos::View<int*, Kokkos::LayoutRight, Kokkos::DefaultHostExecutionSpace> col_coo_host)
375{
376 int const mat_size = matrix->size();
377 int const batch_size = matrix->batch_size();
378 int const non_zero_per_system = vals_coo_host.extent(0) / batch_size;
379
380 Kokkos::View<double**, Kokkos::LayoutRight, Kokkos::DefaultHostExecutionSpace>
381 values_view_host("", batch_size, non_zero_per_system);
382 Kokkos::View<int*, Kokkos::LayoutRight, Kokkos::DefaultHostExecutionSpace>
383 idx_view_host("", non_zero_per_system);
384 Kokkos::View<int*, Kokkos::LayoutRight, Kokkos::DefaultHostExecutionSpace>
385 nnz_per_row_view_host("", mat_size + 1);
386 nnz_per_row_view_host(0) = 0;
387 nnz_per_row_view_host(mat_size) = non_zero_per_system;
388
389 for (int i = 0; i < mat_size; i++) {
390 int n_non_zero_in_row = 0;
391 for (int k = 0; k < non_zero_per_system; k++) {
392 if (row_coo_host(k) == i) {
393 int const idx = nnz_per_row_view_host(i) + n_non_zero_in_row;
394 idx_view_host(idx) = col_coo_host(k);
395 values_view_host(0, idx) = vals_coo_host(k);
396 n_non_zero_in_row++;
397 }
398 }
399 nnz_per_row_view_host(i + 1) = nnz_per_row_view_host(i) + n_non_zero_in_row;
400 }
401 for (int batch_idx = 1; batch_idx < batch_size; batch_idx++) {
402 for (int idx = 0; idx < non_zero_per_system; idx++) {
403 values_view_host(batch_idx, idx) = vals_coo_host(batch_idx * non_zero_per_system + idx);
404 }
405 }
406 auto [values, col_idx, nnz_per_row] = matrix->get_batch_csr();
407 Kokkos::deep_copy(values, values_view_host);
408 Kokkos::deep_copy(col_idx, idx_view_host);
409 Kokkos::deep_copy(nnz_per_row, nnz_per_row_view_host);
410}
