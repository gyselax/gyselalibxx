178 void is_unified(
179 Field<T, IdxRangeRTheta, std::experimental::layout_right, Kokkos::HostSpace> const&
180 values) const
181 {
182 IdxRangeR const r_idx_range = get_idx_range<GridR>(values);
183 IdxRangeTheta const theta_idx_range = get_idx_range<GridTheta>(values);
184 if (std::fabs(ddc::coordinate(r_idx_range.front())) < 1e-15) {
185 ddc::for_each(theta_idx_range, [&](const IdxTheta ip) {
186 if (norm_inf(
187 values(r_idx_range.front(), ip)
188 - values(r_idx_range.front(), theta_idx_range.front()))
189 > 1e-15) {
190 std::cout << "WARNING ! -> Discontinous at the center point." << std::endl;
191 }
192 assert(values(r_idx_range.front(), ip)
