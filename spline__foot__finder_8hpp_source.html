133 void operator()(
134 host_t<FieldRTheta<CoordRTheta>> feet,
135 host_t<DConstVectorFieldRTheta<X, Y>> advection_field,
136 double dt) const final
137 {
138 host_t<VectorSplineCoeffsMem2D<X_adv, Y_adv>> advection_field_in_adv_domain_coefs(
139 get_spline_idx_range(m_builder_advection_field));
140
141 // Compute the advection field in the advection domain.
142 auto advection_field_in_adv_domain = create_geometry_mirror_view(
143 Kokkos::DefaultHostExecutionSpace(),
144 advection_field,
145 m_pseudo_physical_to_physical);
146
147 // Get the coefficients of the advection field in the advection domain.
148 m_builder_advection_field(
149 ddcHelper::get<X_adv>(advection_field_in_adv_domain_coefs),
150 ddcHelper::get<X_adv>(get_const_field(advection_field_in_adv_domain)));
151 m_builder_advection_field(
152 ddcHelper::get<Y_adv>(advection_field_in_adv_domain_coefs),
153 ddcHelper::get<Y_adv>(get_const_field(advection_field_in_adv_domain)));
154
155
156 // The function describing how the derivative of the evolve function is calculated.
157 std::function<
158 void(host_t<DVectorFieldRTheta<X_adv, Y_adv>>,
159 host_t<ConstFieldRTheta<CoordRTheta>>)>
160 dy = [&](host_t<DVectorFieldRTheta<X_adv, Y_adv>> updated_advection_field,
161 host_t<ConstFieldRTheta<CoordRTheta>> feet) {
162 m_evaluator_advection_field(
163 get_field(ddcHelper::get<X_adv>(updated_advection_field)),
164 get_const_field(feet),
165 get_const_field(
166 ddcHelper::get<X_adv>(advection_field_in_adv_domain_coefs)));
167 m_evaluator_advection_field(
168 get_field(ddcHelper::get<Y_adv>(updated_advection_field)),
169 get_const_field(feet),
170 get_const_field(
171 ddcHelper::get<Y_adv>(advection_field_in_adv_domain_coefs)));
172 };
173
174 IdxRangeRTheta const idx_range_rp = get_idx_range<GridR, GridTheta>(feet);
175
176 CoordXY_adv coord_center(m_logical_to_pseudo_physical(CoordRTheta(0, 0)));
177
178 // The function describing how the value(s) are updated using the derivative.
179 std::function<
180 void(host_t<FieldRTheta<CoordRTheta>>,
181 host_t<DConstVectorFieldRTheta<X_adv, Y_adv>>,
182 double)>
183 update_function = [&](host_t<FieldRTheta<CoordRTheta>> feet,
184 host_t<DConstVectorFieldRTheta<X_adv, Y_adv>> advection_field,
185 double dt) {
186 // Compute the characteristic feet at t^n:
187 ddc::for_each(idx_range_rp, [&](IdxRTheta const irp) {
188 CoordRTheta const coord_rp(feet(irp));
189 CoordXY_adv const coord_xy = m_logical_to_pseudo_physical(coord_rp);
190
191 CoordXY_adv const feet_xy = coord_xy - dt * advection_field(irp);
192
193 if (norm_inf(feet_xy - coord_center) < 1e-15) {
194 feet(irp) = CoordRTheta(0, 0);
195 } else {
196 feet(irp) = m_pseudo_physical_to_logical(feet_xy);
197 ddc::select<Theta>(feet(irp)) = ddcHelper::restrict_to_idx_range(
198 ddc::select<Theta>(feet(irp)),
199 IdxRangeTheta(idx_range_rp));
200 }
201 });
202
203 // Treatment to conserve the C0 property of the advected function:
204 unify_value_at_center_pt(feet);
205 // Test if the values are the same at the center point
206 is_unified(feet);
207 };
208
209
210 // Solve the characteristic equation
211 m_time_stepper.update(Kokkos::DefaultHostExecutionSpace(), feet, dt, dy, update_function);
212
213 is_unified(feet);
214 }

231 template <class T>
232 void is_unified(Field<T, IdxRangeRTheta, Kokkos::HostSpace> const& values) const
233 {
234 IdxRangeR const r_idx_range = get_idx_range<GridR>(values);
235 IdxRangeTheta const theta_idx_range = get_idx_range<GridTheta>(values);
236 if (std::fabs(ddc::coordinate(r_idx_range.front())) < 1e-15) {
237 ddc::for_each(theta_idx_range, [&](const IdxTheta ip) {
238 if (norm_inf(
239 values(r_idx_range.front(), ip)
240 - values(r_idx_range.front(), theta_idx_range.front()))
241 > 1e-15) {
242 std::cout << "WARNING ! -> Discontinous at the center point." << std::endl;
243 }
244 assert(values(r_idx_range.front(), ip)
245 == values(r_idx_range.front(), theta_idx_range.front()));
246 });
247 }
248 }
