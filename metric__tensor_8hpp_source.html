81 KOKKOS_FUNCTION void inverse(Matrix_2x2& matrix, PositionCoordinate const& coord) const
82 {
83 const double J_11 = m_mapping.jacobian_11(coord);

85 const double J_21 = m_mapping.jacobian_21(coord);

88 matrix[0][0] = (J_12 * J_12 + J_22 * J_22) / jacob_2;
89 matrix[0][1] = (-J_11 * J_12 - J_21 * J_22) / jacob_2;
90 matrix[1][0] = (-J_11 * J_12 - J_21 * J_22) / jacob_2;
91 matrix[1][1] = (J_11 * J_11 + J_21 * J_21) / jacob_2;
