149 KOKKOS_FUNCTION DTensor<VectorIndexSet<X, Y>, VectorIndexSet<R_cov, Theta_cov>> jacobian_matrix(
150 Coord<X, Y> const& coord) const
151
152 {
153 const double x = ddc::get<X>(coord);
154 const double y = ddc::get<Y>(coord);
155 DTensor<VectorIndexSet<R, Theta>, VectorIndexSet<X_cov, Y_cov>> jacobian_matrix;
156 ddcHelper::get<R, X_cov>(jacobian_matrix) = 2 * x / Kokkos::pow(x * x + y * y, 0.5);
157 ddcHelper::get<R, Y_cov>(jacobian_matrix) = 2 * y / Kokkos::pow(x * x + y * y, 0.5);
158 ddcHelper::get<Theta, X_cov>(jacobian_matrix) = -y / Kokkos::pow(x * x + y * y, 2.);
159 ddcHelper::get<Theta, Y_cov>(jacobian_matrix) = x / Kokkos::pow(x * x + y * y, 2.);
160 return jacobian_matrix;
161 }

171 template <class IndexTag1, class IndexTag2>
172 KOKKOS_INLINE_FUNCTION double jacobian_component(Coord<X, Y> const& coord) const
173 {
174 static_assert(ddc::in_tags_v<IndexTag1, VectorIndexSet<X, Y>>);
175 static_assert(ddc::in_tags_v<IndexTag2, VectorIndexSet<R_cov, Theta_cov>>);
176
177 const double x = ddc::get<X>(coord);
178 const double y = ddc::get<Y>(coord);
179 if constexpr (std::is_same_v<IndexTag1, X> && std::is_same_v<IndexTag2, R_cov>) {
180 // Component (1,1), i.e dx/dr
181 return 2 * x / Kokkos::pow(x * x + y * y, 0.5);
182 } else if constexpr (std::is_same_v<IndexTag1, X> && std::is_same_v<IndexTag2, Theta_cov>) {
183 // Component (1,2), i.e dx/dtheta
184 return 2 * y / Kokkos::pow(x * x + y * y, 0.5);
185 } else if constexpr (std::is_same_v<IndexTag1, Y> && std::is_same_v<IndexTag2, R_cov>) {
186 // Component (2,1), i.e dy/dr
187 return -y / Kokkos::pow(x * x + y * y, 2.);
188 } else {
189 // Component (2,2), i.e dy/dtheta
190 return x / Kokkos::pow(x * x + y * y, 2.);
191 }
192 }
