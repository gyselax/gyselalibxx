153 matrix[0][0] = 2 * x / Kokkos::pow(x * x + y * y, 0.5);
154 matrix[0][1] = 2 * y / Kokkos::pow(x * x + y * y, 0.5);

156 matrix[1][1] = x / Kokkos::pow(x * x + y * y, 2.);
157 }
