59 KOKKOS_FUNCTION DTensor<Dims, Dims> operator()(PositionCoordinate const& coord) const
60 {
61 const double J_11 = m_mapping.jacobian_11(coord);
62 const double J_12 = m_mapping.jacobian_12(coord);
63 const double J_21 = m_mapping.jacobian_21(coord);
64 const double J_22 = m_mapping.jacobian_22(coord);
65 DTensor<Dims, Dims> metric_tensor;

82 KOKKOS_FUNCTION DTensor<Dims_cov, Dims_cov> inverse(PositionCoordinate const& coord) const
83 {
84 const double J_11 = m_mapping.jacobian_11(coord);
85 const double J_12 = m_mapping.jacobian_12(coord);
86 const double J_21 = m_mapping.jacobian_21(coord);

90 DTensor<Dims_cov, Dims_cov> inverse_metric_tensor;
91 ddcHelper::get<Dim0_cov, Dim0_cov>(inverse_metric_tensor)
92 = (J_12 * J_12 + J_22 * J_22) / jacob_2;
93 ddcHelper::get<Dim0_cov, Dim1_cov>(inverse_metric_tensor)
94 = (-J_11 * J_12 - J_21 * J_22) / jacob_2;

98 = (J_11 * J_11 + J_21 * J_21) / jacob_2;
