154template <class StartDims, class EndDims>
155void check_inverse_tensor(
156 DTensor<StartDims, EndDims> const& tensor,
157 DTensor<vector_index_set_dual_t<EndDims>, vector_index_set_dual_t<StartDims>> const&
158 inv_tensor,
159 double TOL)
160{
161 using StartDim0 = ddc::type_seq_element_t<0, StartDims>;
162 using StartDim1 = ddc::type_seq_element_t<1, StartDims>;
163 using StartDim0_cov = typename StartDim0::Dual;
164 using StartDim1_cov = typename StartDim1::Dual;
165
166 DTensor<StartDims, vector_index_set_dual_t<StartDims>> identity
167 = tensor_mul(index<'i', 'j'>(tensor), index<'j', 'k'>(inv_tensor));
168
169 EXPECT_NEAR((ddcHelper::get<StartDim0, StartDim0_cov>(identity)), 1., TOL);
170 EXPECT_NEAR((ddcHelper::get<StartDim0, StartDim1_cov>(identity)), 0., TOL);
171 EXPECT_NEAR((ddcHelper::get<StartDim1, StartDim0_cov>(identity)), 0., TOL);
172 EXPECT_NEAR((ddcHelper::get<StartDim1, StartDim1_cov>(identity)), 1., TOL);
173}
