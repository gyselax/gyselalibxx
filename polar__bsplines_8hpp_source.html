238 template <class DiscreteMapping>

240 {
241 using DimX = typename DiscreteMapping::cartesian_tag_x;
242 using DimY = typename DiscreteMapping::cartesian_tag_y;
243 using mapping_tensor_product_index_type = ddc::DiscreteElement<
244 typename DiscreteMapping::BSplineR,
245 typename DiscreteMapping::BSplineTheta>;
246 if constexpr (C > -1) {
247 const ddc::Coordinate<DimX, DimY> pole
248 = curvilinear_to_cartesian(ddc::Coordinate<DimR, DimTheta>(0.0, 0.0));
249 const double x0 = ddc::get<DimX>(pole);
250 const double y0 = ddc::get<DimY>(pole);
251 double tau = 0.0;
252 for (std::size_t i(0); i < ddc::discrete_space<BSplinesTheta>().size(); ++i) {
253 const ddc::Coordinate<DimX, DimY> point
254 = curvilinear_to_cartesian.control_point(
255 mapping_tensor_product_index_type(1, i));
256
257 const double c_x = ddc::get<DimX>(point);
258 const double c_y = ddc::get<DimY>(point);
259
260 double tau1 = -2.0 * (c_x - x0);
261 double tau2 = c_x - x0 - sqrt(3.0) * (c_y - y0);
262 double tau3 = c_x - x0 + sqrt(3.0) * (c_y - y0);
263 tau = tau > tau1 ? tau : tau1;
264 tau = tau > tau2 ? tau : tau2;
265 tau = tau > tau3 ? tau : tau3;
266 }
267 // Determine the corners for the barycentric coordinates
268 const ddc::Coordinate<DimX, DimY> corner1(x0 + tau, y0);
269 const ddc::Coordinate<DimX, DimY>
270 corner2(x0 - 0.5 * tau, y0 + 0.5 * tau * sqrt(3.0));
271 const ddc::Coordinate<DimX, DimY>
272 corner3(x0 - 0.5 * tau, y0 - 0.5 * tau * sqrt(3.0));
273
274 const CartesianToBarycentricCoordinates<
275 DimX,
276 DimY,
277 Corner1Tag,
278 Corner2Tag,
279 Corner3Tag>
280 barycentric_coordinate_converter(corner1, corner2, corner3);
281
282 using BernsteinBasis = IntermediateBernsteinBasis<DiscreteMapping>;
283
284 ddc::init_discrete_space<BernsteinBasis>(barycentric_coordinate_converter);

286 // The number of radial bases used to construct the bsplines traversing the singular point.
287 constexpr IdxStepR nr_in_singular(C + 1);
288 assert(nr_in_singular.value() < int(ddc::discrete_space<BSplinesR>().size()));
289
290 // The number of poloidal bases used to construct the bsplines traversing the singular point.
291 const IdxStepTheta np_in_singular(ddc::discrete_space<BSplinesTheta>().nbasis());
292
293 // The number of elements of the poloidal basis which will have an associated coefficient
294 // (This will be larger than np_in_singular as it includes the periodicity)
295 const IdxStepTheta np_tot(ddc::discrete_space<BSplinesTheta>().size());
296
297 // The index range of the 2D bsplines in the innermost circles from which the polar bsplines
298 // traversing the singular point will be constructed.
299 tensor_product_idx_range_type const dom_bsplines_inner(
300 tensor_product_index_type(0, 0),
301 tensor_product_idx_step_type(nr_in_singular, np_tot));
302
303 // Initialise memory
304 m_singular_basis_elements
305 = ddc::Chunk<double, singular_basis_linear_combination_idx_range_type>(
306 singular_basis_linear_combination_idx_range_type(
307 singular_idx_range<DDim>(),
308 dom_bsplines_inner));
309
310 ddc::DiscreteDomain<BernsteinBasis> bernstein_idx_range(
311 ddc::DiscreteElement<BernsteinBasis> {0},
312 ddc::DiscreteVector<BernsteinBasis> {n_singular_basis()});
313
314 ddc::DiscreteDomain<BSplinesTheta> poloidal_spline_idx_range
315 = ddc::discrete_space<BSplinesTheta>().full_domain();
316
317 for (IdxR const ir : ddc::DiscreteDomain<BSplinesR>(IdxR(0), IdxStepR(C + 1))) {
318 for (IdxTheta const ip : poloidal_spline_idx_range.take_first(np_in_singular)) {
319 const ddc::Coordinate<DimX, DimY> point
320 = curvilinear_to_cartesian.control_point(
321 mapping_tensor_product_index_type(ir, ip));
322 ddc::Chunk<double, ddc::DiscreteDomain<BernsteinBasis>> bernstein_vals(
323 bernstein_idx_range);
324 ddc::discrete_space<BernsteinBasis>().eval_basis(bernstein_vals, point);
325 // Fill spline coefficients
326 for (auto k : bernstein_idx_range) {
327 m_singular_basis_elements(discrete_element_type {k.uid()}, ir, ip)
328 = bernstein_vals(k);
329 }
330 }
331 for (discrete_element_type k : singular_idx_range<DDim>()) {
332 for (IdxTheta const ip : poloidal_spline_idx_range.take_first(
333 IdxStepTheta {BSplinesTheta::degree()})) {
334 m_singular_basis_elements(k, ir, ip + np_in_singular)
335 = m_singular_basis_elements(k, ir, ip);
336 }
337 }
338 }
339 } else {
340 // Initialise m_singular_basis_elements to avoid any problems in the copy constructor
341 tensor_product_idx_range_type const empty_dom_bsplines(
342 tensor_product_index_type(0, 0),
343 tensor_product_idx_step_type(0, 0));
344 m_singular_basis_elements
345 = ddc::Chunk<double, singular_basis_linear_combination_idx_range_type>(
346 singular_basis_linear_combination_idx_range_type(
347 singular_idx_range<DDim>(),
348 empty_dom_bsplines));
349 }

584template <class BSplinesR, class BSplinesTheta, int C>
585template <class DDim, class MemorySpace>
586template <class EvalTypeR, class EvalTypeTheta>
587ddc::DiscreteElement<BSplinesR, BSplinesTheta> PolarBSplines<BSplinesR, BSplinesTheta, C>::
588 Impl<DDim, MemorySpace>::eval(
589 DSpan1D singular_values,
590 DSpan2D values,
591 ddc::Coordinate<DimR, DimTheta> coord_eval,
592 EvalTypeR const,
593 EvalTypeTheta const) const
594{
595 assert(singular_values.extent(0) == n_singular_basis());
596 assert(values.extent(0) == BSplinesR::degree() + 1);
597 assert(values.extent(1) == BSplinesTheta::degree() + 1);
598 static_assert(
599 std::is_same_v<EvalTypeR, eval_type> || std::is_same_v<EvalTypeR, eval_deriv_type>);
600 static_assert(
601 std::is_same_v<
602 EvalTypeTheta,
603 eval_type> || std::is_same_v<EvalTypeTheta, eval_deriv_type>);
604
605 ddc::DiscreteElement<BSplinesR> jmin_r;
606 ddc::DiscreteElement<BSplinesTheta> jmin_theta;
607
608 std::size_t constexpr nr = BSplinesR::degree() + 1;
609 std::size_t constexpr ntheta = BSplinesTheta::degree() + 1;
610
611 std::array<double, nr> vals_r_ptr;
612 std::array<double, ntheta> vals_theta_ptr;
613 DSpan1D const vals_r(vals_r_ptr.data(), nr);
614 DSpan1D const vals_theta(vals_theta_ptr.data(), ntheta);
615
616 if constexpr (std::is_same_v<EvalTypeR, eval_type>) {
617 jmin_r = ddc::discrete_space<BSplinesR>().eval_basis(vals_r, ddc::select<DimR>(coord_eval));
618 } else if constexpr (std::is_same_v<EvalTypeR, eval_deriv_type>) {
619 jmin_r = ddc::discrete_space<BSplinesR>().eval_deriv(vals_r, ddc::select<DimR>(coord_eval));
620 }
621 if constexpr (std::is_same_v<EvalTypeTheta, eval_type>) {
622 jmin_theta = ddc::discrete_space<BSplinesTheta>()
623 .eval_basis(vals_theta, ddc::select<DimTheta>(coord_eval));
624 } else if constexpr (std::is_same_v<EvalTypeTheta, eval_deriv_type>) {
625 jmin_theta = ddc::discrete_space<BSplinesTheta>()
626 .eval_deriv(vals_theta, ddc::select<DimTheta>(coord_eval));
627 }
628
629 std::size_t nr_done = 0;
630
631 if (jmin_r.uid() < C + 1) {
632 nr_done = C + 1 - jmin_r.uid();
633 for (discrete_element_type k : singular_idx_range<DDim>()) {
634 singular_values(k.uid()) = 0.0;
635 for (std::size_t i(0); i < nr_done; ++i) {
636 for (std::size_t j(0); j < ntheta; ++j) {
637 singular_values(k.uid())
638 += m_singular_basis_elements(k, jmin_r + i, jmin_theta + j) * vals_r[i]
639 * vals_theta[j];
640 }
641 }
642 }
643 } else {
644 for (std::size_t k(0); k < n_singular_basis(); ++k) {
645 singular_values(k) = 0.0;
646 }
647 }
648
649 for (std::size_t i(0); i < nr - nr_done; ++i) {
650 for (std::size_t j(0); j < ntheta; ++j) {
651 values(i, j) = vals_r[i + nr_done] * vals_theta[j];
652 }
653 }
654 for (std::size_t i(nr - nr_done); i < nr; ++i) {
655 for (std::size_t j(0); j < ntheta; ++j) {
656 values(i, j) = 0.0;
657 }
658 }
659 return ddc::DiscreteElement<BSplinesR, BSplinesTheta>(jmin_r, jmin_theta);
660}

664void PolarBSplines<BSplinesR, BSplinesTheta, C>::Impl<DDim, MemorySpace>::integrals(
665 PolarSplineSpan<DDim> int_vals) const
666{
667 auto r_bspl_space = ddc::discrete_space<BSplinesR>();
668 auto theta_bspl_space = ddc::discrete_space<BSplinesTheta>();
669
670 assert(int_vals.singular_spline_coef.domain().extents() == n_singular_basis());
671 assert(int_vals.spline_coef.domain().front().template uid<BSplinesR>() == C + 1);
672 assert(int_vals.spline_coef.domain().back().template uid<BSplinesR>()
673 == r_bspl_space.nbasis() - 1);
674 assert(int_vals.spline_coef.domain().template extent<BSplinesTheta>()
675 == theta_bspl_space.nbasis()
676 || int_vals.spline_coef.domain().template extent<BSplinesTheta>()
677 == theta_bspl_space.size());
678
679 ddc::Chunk<double, ddc::DiscreteDomain<BSplinesR>> r_integrals_alloc(
680 r_bspl_space.full_domain().take_first(
681 ddc::DiscreteVector<BSplinesR> {r_bspl_space.nbasis()}));
682 ddc::Chunk<double, ddc::DiscreteDomain<BSplinesTheta>> theta_integrals_alloc(
683 theta_bspl_space.full_domain().take_first(
684 ddc::DiscreteVector<BSplinesTheta> {theta_bspl_space.size()}));
685 ddc::ChunkSpan r_integrals = r_integrals_alloc.span_view();
686 ddc::ChunkSpan theta_integrals = theta_integrals_alloc.span_view();
687
688 ddc::integrals(Kokkos::DefaultHostExecutionSpace(), r_integrals);
689 ddc::integrals(Kokkos::DefaultHostExecutionSpace(), theta_integrals);
690
691 ddc::for_each(singular_idx_range<DDim>(), [&](auto k) {
692 int_vals.singular_spline_coef(k) = ddc::transform_reduce(
693 ddc::select<BSplinesR, BSplinesTheta>(m_singular_basis_elements.domain()),
694 0.0,
695 ddc::reducer::sum<double>(),
696 [&](tensor_product_index_type const idx) {
697 IdxR i = ddc::select<BSplinesR>(idx);
698 IdxTheta j = ddc::select<BSplinesTheta>(idx);
699 return m_singular_basis_elements(k, i, j) * r_integrals(i) * theta_integrals(j);
700 });
701 });
702
703 ddc::DiscreteDomain<BSplinesR> r_tensor_product_dom(
704 ddc::select<BSplinesR>(int_vals.spline_coef.domain()));
705
706 tensor_product_idx_range_type
707 tensor_bspline_idx_range(r_tensor_product_dom, theta_integrals.domain());
708
709 ddc::for_each(tensor_bspline_idx_range, [&](auto idx) {
710 int_vals.spline_coef(idx) = r_integrals(ddc::select<BSplinesR>(idx))
711 * theta_integrals(ddc::select<BSplinesTheta>(idx));
712 });
713
714 if (int_vals.spline_coef.domain().template extent<BSplinesTheta>() == theta_bspl_space.size()) {
715 ddc::DiscreteDomain<BSplinesTheta> periodic_points(theta_integrals.domain().take_last(
716 ddc::DiscreteVector<BSplinesTheta> {BSplinesTheta::degree()}));
717 tensor_product_idx_range_type repeat_idx_range(r_tensor_product_dom, periodic_points);
718 ddc::for_each(repeat_idx_range, [&](auto idx) { int_vals.spline_coef(idx) = 0.0; });
719 }
