241  template <class DiscreteMapping>
242  Impl(const DiscreteMapping& curvilinear_to_cartesian)
243  {
244  using DimX = typename DiscreteMapping::cartesian_tag_x;
245  using DimY = typename DiscreteMapping::cartesian_tag_y;
246  using mapping_tensor_product_index_type = ddc::DiscreteElement<
247  typename DiscreteMapping::BSplineR,
248  typename DiscreteMapping::BSplineP>;
249  if constexpr (C > -1) {
250  const ddc::Coordinate<DimX, DimY> pole
251  = curvilinear_to_cartesian(ddc::Coordinate<DimR, DimTheta>(0.0, 0.0));
252  const double x0 = ddc::get<DimX>(pole);
253  const double y0 = ddc::get<DimY>(pole);
254  double tau = 0.0;
255  for (std::size_t i(0); i < ddc::discrete_space<BSplinesTheta>().size(); ++i) {
256  const ddc::Coordinate<DimX, DimY> point
257  = curvilinear_to_cartesian.control_point(
258  mapping_tensor_product_index_type(1, i));
259 
260  const double c_x = ddc::get<DimX>(point);
261  const double c_y = ddc::get<DimY>(point);
262 
263  double tau1 = -2.0 * (c_x - x0);
264  double tau2 = c_x - x0 - sqrt(3.0) * (c_y - y0);
265  double tau3 = c_x - x0 + sqrt(3.0) * (c_y - y0);
266  tau = tau > tau1 ? tau : tau1;
267  tau = tau > tau2 ? tau : tau2;
268  tau = tau > tau3 ? tau : tau3;
269  }
270  // Determine the corners for the barycentric coordinates
271  const ddc::Coordinate<DimX, DimY> corner1(x0 + tau, y0);
272  const ddc::Coordinate<DimX, DimY>
273  corner2(x0 - 0.5 * tau, y0 + 0.5 * tau * sqrt(3.0));
274  const ddc::Coordinate<DimX, DimY>
275  corner3(x0 - 0.5 * tau, y0 - 0.5 * tau * sqrt(3.0));

278  DimX,
279  DimY,
280  Corner1Tag,
281  Corner2Tag,
282  Corner3Tag>
283  barycentric_coordinate_converter(corner1, corner2, corner3);

587 template <class BSplinesR, class BSplinesTheta, int C>
588 template <class DDim, class MemorySpace>
589 template <class EvalTypeR, class EvalTypeP>
590 ddc::DiscreteElement<BSplinesR, BSplinesTheta> PolarBSplines<BSplinesR, BSplinesTheta, C>::
591  Impl<DDim, MemorySpace>::eval(
592  DSpan1D singular_values,
593  DSpan2D values,
594  ddc::Coordinate<DimR, DimTheta> coord_eval,
595  EvalTypeR const,
596  EvalTypeP const) const
597 {
598  assert(singular_values.extent(0) == n_singular_basis());
599  assert(values.extent(0) == BSplinesR::degree() + 1);
600  assert(values.extent(1) == BSplinesTheta::degree() + 1);
601  static_assert(
602  std::is_same_v<EvalTypeR, eval_type> || std::is_same_v<EvalTypeR, eval_deriv_type>);
603  static_assert(
604  std::is_same_v<EvalTypeP, eval_type> || std::is_same_v<EvalTypeP, eval_deriv_type>);
605 
606  ddc::DiscreteElement<BSplinesR> jmin_r;
607  ddc::DiscreteElement<BSplinesTheta> jmin_theta;
608 
609  std::size_t constexpr nr = BSplinesR::degree() + 1;
610  std::size_t constexpr ntheta = BSplinesTheta::degree() + 1;
611 
612  std::array<double, nr> vals_r_ptr;
613  std::array<double, ntheta> vals_theta_ptr;
614  DSpan1D const vals_r(vals_r_ptr.data(), nr);
615  DSpan1D const vals_theta(vals_theta_ptr.data(), ntheta);
616 
617  if constexpr (std::is_same_v<EvalTypeR, eval_type>) {
618  jmin_r = ddc::discrete_space<BSplinesR>().eval_basis(vals_r, ddc::select<DimR>(coord_eval));
619  } else if constexpr (std::is_same_v<EvalTypeR, eval_deriv_type>) {
620  jmin_r = ddc::discrete_space<BSplinesR>().eval_deriv(vals_r, ddc::select<DimR>(coord_eval));
621  }
622  if constexpr (std::is_same_v<EvalTypeP, eval_type>) {
623  jmin_theta = ddc::discrete_space<BSplinesTheta>()
624  .eval_basis(vals_theta, ddc::select<DimTheta>(coord_eval));
625  } else if constexpr (std::is_same_v<EvalTypeP, eval_deriv_type>) {
626  jmin_theta = ddc::discrete_space<BSplinesTheta>()
627  .eval_deriv(vals_theta, ddc::select<DimTheta>(coord_eval));
628  }
629 
630  std::size_t nr_done = 0;
631 
632  if (jmin_r.uid() < C + 1) {
633  nr_done = C + 1 - jmin_r.uid();
634  for (discrete_element_type k : singular_idx_range<DDim>()) {
635  singular_values(k.uid()) = 0.0;
636  for (std::size_t i(0); i < nr_done; ++i) {
637  for (std::size_t j(0); j < ntheta; ++j) {
638  singular_values(k.uid())
639  += m_singular_basis_elements(k, jmin_r + i, jmin_theta + j) * vals_r[i]
640  * vals_theta[j];
641  }
642  }
643  }
644  } else {
645  for (std::size_t k(0); k < n_singular_basis(); ++k) {
646  singular_values(k) = 0.0;
647  }
648  }
649 
650  for (std::size_t i(0); i < nr - nr_done; ++i) {
651  for (std::size_t j(0); j < ntheta; ++j) {
652  values(i, j) = vals_r[i + nr_done] * vals_theta[j];
653  }
654  }
655  for (std::size_t i(nr - nr_done); i < nr; ++i) {
656  for (std::size_t j(0); j < ntheta; ++j) {
657  values(i, j) = 0.0;
658  }
659  }
660  return ddc::DiscreteElement<BSplinesR, BSplinesTheta>(jmin_r, jmin_theta);
661 }

664 template <class DDim, class MemorySpace>
665 void PolarBSplines<BSplinesR, BSplinesTheta, C>::Impl<DDim, MemorySpace>::integrals(
666  PolarSplineSpan<DDim> int_vals) const
667 {
668  auto r_bspl_space = ddc::discrete_space<BSplinesR>();
669  auto theta_bspl_space = ddc::discrete_space<BSplinesTheta>();
670 
671  assert(int_vals.singular_spline_coef.domain().extents() == n_singular_basis());
672  assert(int_vals.spline_coef.domain().front().template uid<BSplinesR>() == C + 1);
673  assert(int_vals.spline_coef.domain().back().template uid<BSplinesR>()
674  == r_bspl_space.nbasis() - 1);
675  assert(int_vals.spline_coef.domain().template extent<BSplinesTheta>()
676  == theta_bspl_space.nbasis()
677  || int_vals.spline_coef.domain().template extent<BSplinesTheta>()
678  == theta_bspl_space.size());
679 
680  ddc::Chunk<double, ddc::DiscreteDomain<BSplinesR>> r_integrals_alloc(
681  r_bspl_space.full_domain().take_first(
682  ddc::DiscreteVector<BSplinesR> {r_bspl_space.nbasis()}));
683  ddc::Chunk<double, ddc::DiscreteDomain<BSplinesTheta>> theta_integrals_alloc(
684  theta_bspl_space.full_domain().take_first(
685  ddc::DiscreteVector<BSplinesTheta> {theta_bspl_space.size()}));
686  ddc::ChunkSpan r_integrals = r_integrals_alloc.span_view();
687  ddc::ChunkSpan theta_integrals = theta_integrals_alloc.span_view();
688 
689  r_bspl_space.integrals(r_integrals);
690  theta_bspl_space.integrals(theta_integrals);
691 
692  ddc::for_each(singular_idx_range<DDim>(), [&](auto k) {
693  int_vals.singular_spline_coef(k) = ddc::transform_reduce(
694  ddc::select<BSplinesR, BSplinesTheta>(m_singular_basis_elements.domain()),
695  0.0,
696  ddc::reducer::sum<double>(),
697  [&](tensor_product_index_type const idx) {
698  IdxR i = ddc::select<BSplinesR>(idx);
699  IdxTheta j = ddc::select<BSplinesTheta>(idx);
700  return m_singular_basis_elements(k, i, j) * r_integrals(i) * theta_integrals(j);
701  });
702  });
703 
704  ddc::DiscreteDomain<BSplinesR> r_tensor_product_dom(
705  ddc::select<BSplinesR>(int_vals.spline_coef.domain()));
706 
707  tensor_product_idx_range_type
708  tensor_bspline_idx_range(r_tensor_product_dom, theta_integrals.domain());
709 
710  ddc::for_each(tensor_bspline_idx_range, [&](auto idx) {
711  int_vals.spline_coef(idx) = r_integrals(ddc::select<BSplinesR>(idx))
712  * theta_integrals(ddc::select<BSplinesTheta>(idx));
713  });
714 
715  if (int_vals.spline_coef.domain().template extent<BSplinesTheta>() == theta_bspl_space.size()) {
716  ddc::DiscreteDomain<BSplinesTheta> periodic_points(theta_integrals.domain().take_last(
717  ddc::DiscreteVector<BSplinesTheta> {BSplinesTheta::degree()}));
718  tensor_product_idx_range_type repeat_idx_range(r_tensor_product_dom, periodic_points);
719  ddc::for_each(repeat_idx_range, [&](auto idx) { int_vals.spline_coef(idx) = 0.0; });
720  }
