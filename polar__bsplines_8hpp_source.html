133 template <class DDim>
134 static KOKKOS_FUNCTION Idx<DDim> get_polar_index(tensor_product_index_type const& idx)
135 {
136 Idx<BSplinesR> idx_r(idx);
137 Idx<BSplinesTheta> idx_theta(idx);
138 int const r_idx = idx_r - Idx<BSplinesR>(C + 1);
139 int const theta_idx = idx_theta - Idx<BSplinesTheta>(0);
140 assert(r_idx >= 0);

142 return Idx<DDim>(n_singular_basis() + local_idx);

246 template <class DiscreteMapping>
247 explicit Impl(const DiscreteMapping& curvilinear_to_cartesian)
248 {
249 static_assert(is_accessible_v<Kokkos::DefaultHostExecutionSpace, DiscreteMapping>);
250 static_assert(std::is_same_v<MemorySpace, Kokkos::HostSpace>);
251 using DimX = typename DiscreteMapping::cartesian_tag_x;
252 using DimY = typename DiscreteMapping::cartesian_tag_y;
253 using mapping_tensor_product_index_type
254 = Idx<typename DiscreteMapping::BSplineR,
255 typename DiscreteMapping::BSplineTheta>;
256 if constexpr (C > -1) {
257 const ddc::Coordinate<DimX, DimY> pole
258 = curvilinear_to_cartesian(ddc::Coordinate<DimR, DimTheta>(0.0, 0.0));
259 const double x0 = ddc::get<DimX>(pole);
260 const double y0 = ddc::get<DimY>(pole);
261 double tau = 0.0;
262 for (std::size_t i(0); i < ddc::discrete_space<BSplinesTheta>().size(); ++i) {
263 const ddc::Coordinate<DimX, DimY> point
264 = curvilinear_to_cartesian.control_point(
265 mapping_tensor_product_index_type(1, i));
266
267 const double c_x = ddc::get<DimX>(point);
268 const double c_y = ddc::get<DimY>(point);
269
270 double tau1 = -2.0 * (c_x - x0);
271 double tau2 = c_x - x0 - sqrt(3.0) * (c_y - y0);
272 double tau3 = c_x - x0 + sqrt(3.0) * (c_y - y0);
273 tau = tau > tau1 ? tau : tau1;
274 tau = tau > tau2 ? tau : tau2;
275 tau = tau > tau3 ? tau : tau3;
276 }
277 // Determine the corners for the barycentric coordinates
278 const ddc::Coordinate<DimX, DimY> corner1(x0 + tau, y0);
279 const ddc::Coordinate<DimX, DimY>
280 corner2(x0 - 0.5 * tau, y0 + 0.5 * tau * sqrt(3.0));
281 const ddc::Coordinate<DimX, DimY>
282 corner3(x0 - 0.5 * tau, y0 - 0.5 * tau * sqrt(3.0));
283
284 const CartesianToBarycentric<DimX, DimY, Corner1Tag, Corner2Tag, Corner3Tag>
285 barycentric_coordinate_converter(corner1, corner2, corner3);
286
287 using BernsteinBasis = IntermediateBernsteinBasis<DiscreteMapping>;
288
289 ddc::init_discrete_space<BernsteinBasis>(barycentric_coordinate_converter);
290
291 // The number of radial bases used to construct the B-splines traversing the singular point.
292 constexpr IdxStepR nr_in_singular(C + 1);
293 assert(nr_in_singular.value() < int(ddc::discrete_space<BSplinesR>().size()));
294
295 // The number of poloidal bases used to construct the B-splines traversing the singular point.
296 const IdxStepTheta np_in_singular(ddc::discrete_space<BSplinesTheta>().nbasis());
297
298 // The number of elements of the poloidal basis which will have an associated coefficient
299 // (This will be larger than np_in_singular as it includes the periodicity)
300 const IdxStepTheta np_tot(ddc::discrete_space<BSplinesTheta>().size());
301
302 // The index range of the 2D B-splines in the innermost circles from which the polar B-splines
303 // traversing the singular point will be constructed.
304 tensor_product_idx_range_type const dom_bsplines_inner(
305 tensor_product_index_type(0, 0),
306 tensor_product_idx_step_type(nr_in_singular, np_tot));
307
308 // Initialise memory
309 m_singular_basis_elements_alloc
310 = host_t<DFieldMem<singular_basis_linear_combination_idx_range_type>>(
311 singular_basis_linear_combination_idx_range_type(
312 singular_idx_range<DDim>(),
313 dom_bsplines_inner));
314 m_singular_basis_elements = get_field(m_singular_basis_elements_alloc);
315
316 IdxRange<BernsteinBasis> bernstein_idx_range(
317 Idx<BernsteinBasis> {0},
318 IdxStep<BernsteinBasis> {n_singular_basis()});
319
320 IdxRange<BSplinesTheta> poloidal_spline_idx_range
321 = ddc::discrete_space<BSplinesTheta>().full_domain();
322
323 for (IdxR const ir : IdxRange<BSplinesR>(IdxR(0), IdxStepR(C + 1))) {
324 for (IdxTheta const ip : poloidal_spline_idx_range.take_first(np_in_singular)) {
325 const ddc::Coordinate<DimX, DimY> point
326 = curvilinear_to_cartesian.control_point(
327 mapping_tensor_product_index_type(ir, ip));
328 host_t<DFieldMem<IdxRange<BernsteinBasis>>> bernstein_vals(
329 bernstein_idx_range);
330 ddc::discrete_space<BernsteinBasis>()
331 .eval_basis(get_field(bernstein_vals), point);
332 // Fill spline coefficients
333 for (Idx<BernsteinBasis> k : bernstein_idx_range) {
334 m_singular_basis_elements(
335 discrete_element_type {
336 (k - bernstein_idx_range.front()).value()},
337 ir,
338 ip)
339 = bernstein_vals(k);
340 }
341 }
342 for (discrete_element_type k : singular_idx_range<DDim>()) {
343 for (IdxTheta const ip : poloidal_spline_idx_range.take_first(
344 IdxStepTheta {BSplinesTheta::degree()})) {
345 m_singular_basis_elements(k, ir, ip + np_in_singular)
346 = m_singular_basis_elements(k, ir, ip);
347 }
348 }
349 }
350 } else {
351 // Initialise m_singular_basis_elements to avoid any problems in the copy constructor
352 tensor_product_idx_range_type const empty_dom_bsplines(
353 tensor_product_index_type(0, 0),
354 tensor_product_idx_step_type(0, 0));
355 m_singular_basis_elements_alloc
356 = host_t<DFieldMem<singular_basis_linear_combination_idx_range_type>>(
357 singular_basis_linear_combination_idx_range_type(
358 singular_idx_range<DDim>(),
359 empty_dom_bsplines));
360 m_singular_basis_elements = get_field(m_singular_basis_elements_alloc);
361 }
362 }

597template <class BSplinesR, class BSplinesTheta, int C>
598template <class DDim, class MemorySpace>
599template <class EvalTypeR, class EvalTypeTheta>
600KOKKOS_FUNCTION Idx<BSplinesR, BSplinesTheta> PolarBSplines<BSplinesR, BSplinesTheta, C>::
601 Impl<DDim, MemorySpace>::eval(
602 DSpan1D singular_values,
603 DSpan2D values,
604 ddc::Coordinate<DimR, DimTheta> coord_eval,
605 EvalTypeR const,
606 EvalTypeTheta const) const
607{
608 assert(singular_values.extent(0) == n_singular_basis());
609 assert(values.extent(0) == BSplinesR::degree() + 1);
610 assert(values.extent(1) == BSplinesTheta::degree() + 1);
611 static_assert(
612 std::is_same_v<EvalTypeR, eval_type> || std::is_same_v<EvalTypeR, eval_deriv_type>);
613 static_assert(
614 std::is_same_v<
615 EvalTypeTheta,
616 eval_type> || std::is_same_v<EvalTypeTheta, eval_deriv_type>);
617
618 Idx<BSplinesR> jmin_r;
619 Idx<BSplinesTheta> jmin_theta;
620
621 std::size_t constexpr nr = BSplinesR::degree() + 1;
622 std::size_t constexpr ntheta = BSplinesTheta::degree() + 1;
623
624 std::array<double, nr> vals_r_ptr;
625 std::array<double, ntheta> vals_theta_ptr;
626 DSpan1D const vals_r(vals_r_ptr.data(), nr);
627 DSpan1D const vals_theta(vals_theta_ptr.data(), ntheta);
628
629 if constexpr (std::is_same_v<EvalTypeR, eval_type>) {
630 jmin_r = ddc::discrete_space<BSplinesR>().eval_basis(vals_r, ddc::select<DimR>(coord_eval));
631 } else if constexpr (std::is_same_v<EvalTypeR, eval_deriv_type>) {
632 jmin_r = ddc::discrete_space<BSplinesR>().eval_deriv(vals_r, ddc::select<DimR>(coord_eval));
633 }
634 if constexpr (std::is_same_v<EvalTypeTheta, eval_type>) {
635 jmin_theta = ddc::discrete_space<BSplinesTheta>()
636 .eval_basis(vals_theta, ddc::select<DimTheta>(coord_eval));
637 } else if constexpr (std::is_same_v<EvalTypeTheta, eval_deriv_type>) {
638 jmin_theta = ddc::discrete_space<BSplinesTheta>()
639 .eval_deriv(vals_theta, ddc::select<DimTheta>(coord_eval));
640 }
641
642 std::size_t nr_done = 0;
643
644 Idx<BSplinesR> first_tensor_product_radial_spline(C + 1);
645
646 if (jmin_r < first_tensor_product_radial_spline) {
647 nr_done = first_tensor_product_radial_spline - jmin_r;
648 for (discrete_element_type k : singular_idx_range<DDim>()) {
649 singular_values(k - discrete_element_type(0)) = 0.0;
650 for (std::size_t i(0); i < nr_done; ++i) {
651 for (std::size_t j(0); j < ntheta; ++j) {
652 singular_values(k - discrete_element_type(0))
653 += m_singular_basis_elements(k, jmin_r + i, jmin_theta + j) * vals_r[i]
654 * vals_theta[j];
655 }
656 }
657 }
658 } else {
659 for (std::size_t k(0); k < n_singular_basis(); ++k) {
660 singular_values(k) = 0.0;
661 }
662 }
663
664 for (std::size_t i(0); i < nr - nr_done; ++i) {
665 for (std::size_t j(0); j < ntheta; ++j) {
666 values(i, j) = vals_r[i + nr_done] * vals_theta[j];
667 }
668 }
669 for (std::size_t i(nr - nr_done); i < nr; ++i) {
670 for (std::size_t j(0); j < ntheta; ++j) {
671 values(i, j) = 0.0;
672 }
673 }
674 return Idx<BSplinesR, BSplinesTheta>(jmin_r, jmin_theta);
675}

677template <class ExecSpace, class DDim, class MemorySpace>
678PolarSpline<DDim, MemorySpace> integrals(
679 ExecSpace const& execution_space,
680 PolarSpline<DDim, MemorySpace> int_vals)
681{
682 static_assert(
683 Kokkos::SpaceAccessibility<ExecSpace, MemorySpace>::accessible,
684 "MemorySpace has to be accessible for ExecutionSpace.");
685 using BSplinesR = typename DDim::BSplinesR_tag;
686 using BSplinesTheta = typename DDim::BSplinesTheta_tag;
687 using tensor_product_idx_range_type = IdxRange<BSplinesR, BSplinesTheta>;
688 using tensor_product_idx_type = Idx<BSplinesR, BSplinesTheta>;
689 using IdxR = Idx<BSplinesR>;
690 using IdxTheta = Idx<BSplinesTheta>;
691
692 auto r_bspl_space = ddc::discrete_space<BSplinesR>();
693 auto theta_bspl_space = ddc::discrete_space<BSplinesTheta>();
694
695 assert(get_idx_range(int_vals.singular_spline_coef).extents() == DDim::n_singular_basis());
696 assert(get_idx_range(int_vals.spline_coef).front().template uid<BSplinesR>()
697 == DDim::continuity + 1);
698 assert(get_idx_range(int_vals.spline_coef).back().template uid<BSplinesR>()
699 == r_bspl_space.nbasis() - 1);
700 assert(get_idx_range(int_vals.spline_coef).template extent<BSplinesTheta>()
701 == theta_bspl_space.nbasis()
702 || get_idx_range(int_vals.spline_coef).template extent<BSplinesTheta>()
703 == theta_bspl_space.size());
704
705 DFieldMem<IdxRange<BSplinesR>, MemorySpace> r_integrals_alloc(
706 r_bspl_space.full_domain().take_first(IdxStep<BSplinesR> {r_bspl_space.nbasis()}));
707 DFieldMem<IdxRange<BSplinesTheta>, MemorySpace> theta_integrals_alloc(
708 theta_bspl_space.full_domain().take_first(
709 IdxStep<BSplinesTheta> {theta_bspl_space.size()}));
710 DField<IdxRange<BSplinesR>, MemorySpace> r_integrals = get_field(r_integrals_alloc);
711 DField<IdxRange<BSplinesTheta>, MemorySpace> theta_integrals = get_field(theta_integrals_alloc);
712
713 ddc::integrals(execution_space, r_integrals);
714 ddc::integrals(execution_space, theta_integrals);
715
716 IdxRange<BSplinesR, BSplinesTheta> singular_2d_idx_range(
717 get_idx_range(ddc::discrete_space<DDim>().m_singular_basis_elements));
718 DField<IdxRange<DDim>> singular_spline_integrals = get_field(int_vals.singular_spline_coef);
719
720 IdxRange<DDim> singular_idx_range = DDim::template singular_idx_range<DDim>();
721 Kokkos::parallel_for(
722 Kokkos::TeamPolicy<>(execution_space, singular_idx_range.size(), Kokkos::AUTO),
723 KOKKOS_LAMBDA(const Kokkos::TeamPolicy<>::member_type& team) {
724 const int idx = team.league_rank();
725 Idx<DDim> k(idx);
726
727 // Sum over quadrature dimensions
728 double teamSum = 0;
729 Kokkos::parallel_reduce(
730 Kokkos::TeamThreadMDRange(
731 team,
732 singular_2d_idx_range.template extent<BSplinesR>().value(),
733 singular_2d_idx_range.template extent<BSplinesTheta>().value()),
734 [&](int r_thread_index, int theta_thread_index, double& sum) {
735 IdxR i(r_thread_index);
736 IdxTheta j(theta_thread_index);
737 sum += ddc::discrete_space<DDim>().m_singular_basis_elements(k, i, j)
738 * r_integrals(i) * theta_integrals(j);
739 },
740 teamSum);
741 singular_spline_integrals(k) = teamSum;
742 });
743
744
745 IdxRange<BSplinesR> r_tensor_product_dom(get_idx_range(int_vals.spline_coef));
746 tensor_product_idx_range_type
747 tensor_bspline_idx_range(r_tensor_product_dom, get_idx_range(theta_integrals));
748 DField<IdxRange<BSplinesR, BSplinesTheta>> spline_integrals = get_field(int_vals.spline_coef);
749
750 ddc::parallel_for_each(
751 execution_space,
752 tensor_bspline_idx_range,
753 KOKKOS_LAMBDA(tensor_product_idx_type idx) {
754 int_vals.spline_coef(idx) = r_integrals(ddc::select<BSplinesR>(idx))
755 * theta_integrals(ddc::select<BSplinesTheta>(idx));
756 });
757
758 if (get_idx_range(int_vals.spline_coef).template extent<BSplinesTheta>()
759 == theta_bspl_space.size()) {
760 IdxRange<BSplinesTheta> periodic_points(
761 get_idx_range(theta_integrals)
762 .take_last(IdxStep<BSplinesTheta> {BSplinesTheta::degree()}));
763 tensor_product_idx_range_type repeat_idx_range(r_tensor_product_dom, periodic_points);
764 ddc::parallel_fill(execution_space, int_vals.spline_coef[repeat_idx_range], 0.0);
765 }
766 return int_vals;
767}
