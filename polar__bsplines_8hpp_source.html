242  template <class DiscreteMapping>
243  Impl(const DiscreteMapping& curvilinear_to_cartesian)
244  {
245  using DimX = typename DiscreteMapping::cartesian_tag_x;
246  using DimY = typename DiscreteMapping::cartesian_tag_y;
247  using mapping_tensor_product_discrete_element_type = ddc::DiscreteElement<
248  typename DiscreteMapping::BSplineR,
249  typename DiscreteMapping::BSplineP>;
250  if constexpr (C > -1) {
251  const ddc::Coordinate<DimX, DimY> pole
252  = curvilinear_to_cartesian(ddc::Coordinate<DimR, DimP>(0.0, 0.0));
253  const double x0 = ddc::get<DimX>(pole);
254  const double y0 = ddc::get<DimY>(pole);
255  double tau = 0.0;
256  for (std::size_t i(0); i < ddc::discrete_space<BSplinesP>().size(); ++i) {
257  const ddc::Coordinate<DimX, DimY> point
258  = curvilinear_to_cartesian.control_point(
259  mapping_tensor_product_discrete_element_type(1, i));
260 
261  const double c_x = ddc::get<DimX>(point);
262  const double c_y = ddc::get<DimY>(point);
263 
264  double tau1 = -2.0 * (c_x - x0);
265  double tau2 = c_x - x0 - sqrt(3.0) * (c_y - y0);
266  double tau3 = c_x - x0 + sqrt(3.0) * (c_y - y0);
267  tau = tau > tau1 ? tau : tau1;
268  tau = tau > tau2 ? tau : tau2;
269  tau = tau > tau3 ? tau : tau3;
270  }
271  // Determine the corners for the barycentric coordinates
272  const ddc::Coordinate<DimX, DimY> corner1(x0 + tau, y0);
273  const ddc::Coordinate<DimX, DimY>
274  corner2(x0 - 0.5 * tau, y0 + 0.5 * tau * sqrt(3.0));
275  const ddc::Coordinate<DimX, DimY>
276  corner3(x0 - 0.5 * tau, y0 - 0.5 * tau * sqrt(3.0));
277 
278  const CartesianToBarycentricCoordinates<
279  DimX,
280  DimY,
281  Corner1Tag,
282  Corner2Tag,
283  Corner3Tag>
284  barycentric_coordinate_converter(corner1, corner2, corner3);
285 
286  using BernsteinBasis = IntermediateBernsteinBasis<DiscreteMapping>;

288  ddc::init_discrete_space<BernsteinBasis>(barycentric_coordinate_converter);

290  // The number of radial bases used to construct the bsplines traversing the singular point.

292  assert(nr_in_singular.value() < int(ddc::discrete_space<BSplinesR>().size()));

295  const LengthP np_in_singular(ddc::discrete_space<BSplinesP>().nbasis());

316  ddc::DiscreteVector<BernsteinBasis> {n_singular_basis()});

318  ddc::DiscreteDomain<BSplinesP> poloidal_spline_domain

321  for (IndexR const ir : ddc::DiscreteDomain<BSplinesR>(IndexR(0), LengthR(C + 1))) {
322  for (IndexP const ip : poloidal_spline_domain.take_first(np_in_singular)) {

327  bernstein_domain);
328  ddc::discrete_space<BernsteinBasis>().eval_basis(bernstein_vals, point);
329  // Fill spline coefficients
330  for (auto k : bernstein_domain) {
331  m_singular_basis_elements(discrete_element_type {k.uid()}, ir, ip)

333  }
334  }
335  for (discrete_element_type k : singular_domain<DDim>()) {
336  for (IndexP const ip :
337  poloidal_spline_domain.take_first(LengthP {BSplinesP::degree()})) {
338  m_singular_basis_elements(k, ir, ip + np_in_singular)

340  }
341  }
342  }

588 template <class BSplinesR, class BSplinesP, int C>
589 template <class DDim, class MemorySpace>
590 template <class EvalTypeR, class EvalTypeP>
591 ddc::DiscreteElement<BSplinesR, BSplinesP> PolarBSplines<BSplinesR, BSplinesP, C>::
592  Impl<DDim, MemorySpace>::eval(
593  DSpan1D singular_values,
594  DSpan2D values,
595  ddc::Coordinate<DimR, DimP> coord_eval,
596  EvalTypeR const,
597  EvalTypeP const) const
598 {
599  assert(singular_values.extent(0) == n_singular_basis());
600  assert(values.extent(0) == BSplinesR::degree() + 1);
601  assert(values.extent(1) == BSplinesP::degree() + 1);
602  static_assert(
603  std::is_same_v<EvalTypeR, eval_type> || std::is_same_v<EvalTypeR, eval_deriv_type>);
604  static_assert(
605  std::is_same_v<EvalTypeP, eval_type> || std::is_same_v<EvalTypeP, eval_deriv_type>);
606 
607  ddc::DiscreteElement<BSplinesR> jmin_r;
608  ddc::DiscreteElement<BSplinesP> jmin_p;
609 
610  std::size_t constexpr nr = BSplinesR::degree() + 1;
611  std::size_t constexpr np = BSplinesP::degree() + 1;
612 
613  std::array<double, nr> vals_r_ptr;
614  std::array<double, np> vals_p_ptr;
615  DSpan1D const vals_r(vals_r_ptr.data(), nr);
616  DSpan1D const vals_p(vals_p_ptr.data(), np);
617 
618  if constexpr (std::is_same_v<EvalTypeR, eval_type>) {
619  jmin_r = ddc::discrete_space<BSplinesR>().eval_basis(vals_r, ddc::select<DimR>(coord_eval));
620  } else if constexpr (std::is_same_v<EvalTypeR, eval_deriv_type>) {
621  jmin_r = ddc::discrete_space<BSplinesR>().eval_deriv(vals_r, ddc::select<DimR>(coord_eval));
622  }
623  if constexpr (std::is_same_v<EvalTypeP, eval_type>) {
624  jmin_p = ddc::discrete_space<BSplinesP>().eval_basis(vals_p, ddc::select<DimP>(coord_eval));
625  } else if constexpr (std::is_same_v<EvalTypeP, eval_deriv_type>) {
626  jmin_p = ddc::discrete_space<BSplinesP>().eval_deriv(vals_p, ddc::select<DimP>(coord_eval));
627  }
628 
629  std::size_t nr_done = 0;
630 
631  if (jmin_r.uid() < C + 1) {
632  nr_done = C + 1 - jmin_r.uid();
633  for (discrete_element_type k : singular_domain<DDim>()) {
634  singular_values(k.uid()) = 0.0;
635  for (std::size_t i(0); i < nr_done; ++i) {
636  for (std::size_t j(0); j < np; ++j) {
637  singular_values(k.uid()) += m_singular_basis_elements(k, jmin_r + i, jmin_p + j)
638  * vals_r[i] * vals_p[j];
639  }
640  }
641  }
642  } else {
643  for (std::size_t k(0); k < n_singular_basis(); ++k) {
644  singular_values(k) = 0.0;
645  }
646  }
647 
648  for (std::size_t i(0); i < nr - nr_done; ++i) {
649  for (std::size_t j(0); j < np; ++j) {
650  values(i, j) = vals_r[i + nr_done] * vals_p[j];
651  }
652  }
653  for (std::size_t i(nr - nr_done); i < nr; ++i) {
654  for (std::size_t j(0); j < np; ++j) {
655  values(i, j) = 0.0;
656  }
657  }
658  return ddc::DiscreteElement<BSplinesR, BSplinesP>(jmin_r, jmin_p);
659 }

663 void PolarBSplines<BSplinesR, BSplinesP, C>::Impl<DDim, MemorySpace>::integrals(
664  PolarSplineSpan<DDim> int_vals) const
665 {
666  auto r_bspl_space = ddc::discrete_space<BSplinesR>();
667  auto p_bspl_space = ddc::discrete_space<BSplinesP>();
668 
669  assert(int_vals.singular_spline_coef.domain().extents() == n_singular_basis());
670  assert(int_vals.spline_coef.domain().front().template uid<BSplinesR>() == C + 1);
671  assert(int_vals.spline_coef.domain().back().template uid<BSplinesR>()
672  == r_bspl_space.nbasis() - 1);
673  assert(int_vals.spline_coef.domain().template extent<BSplinesP>() == p_bspl_space.nbasis()
674  || int_vals.spline_coef.domain().template extent<BSplinesP>() == p_bspl_space.size());
675 
676  ddc::Chunk<double, ddc::DiscreteDomain<BSplinesR>> r_integrals_alloc(
677  r_bspl_space.full_domain().take_first(
678  ddc::DiscreteVector<BSplinesR> {r_bspl_space.nbasis()}));
679  ddc::Chunk<double, ddc::DiscreteDomain<BSplinesP>> p_integrals_alloc(
680  p_bspl_space.full_domain().take_first(
681  ddc::DiscreteVector<BSplinesP> {p_bspl_space.size()}));
682  ddc::ChunkSpan r_integrals = r_integrals_alloc.span_view();
683  ddc::ChunkSpan p_integrals = p_integrals_alloc.span_view();
684 
685  r_bspl_space.integrals(r_integrals);
686  p_bspl_space.integrals(p_integrals);
687 
688  ddc::for_each(singular_domain<DDim>(), [&](auto k) {
689  int_vals.singular_spline_coef(k) = ddc::transform_reduce(
690  ddc::select<BSplinesR, BSplinesP>(m_singular_basis_elements.domain()),
691  0.0,
692  ddc::reducer::sum<double>(),
693  [&](tensor_product_discrete_element_type const idx) {
694  IndexR i = ddc::select<BSplinesR>(idx);
695  IndexP j = ddc::select<BSplinesP>(idx);
696  return m_singular_basis_elements(k, i, j) * r_integrals(i) * p_integrals(j);
697  });
698  });
699 
700  ddc::DiscreteDomain<BSplinesR> r_tensor_product_dom(
701  ddc::select<BSplinesR>(int_vals.spline_coef.domain()));
702 
703  tensor_product_discrete_domain_type
704  tensor_bspline_domain(r_tensor_product_dom, p_integrals.domain());
705 
706  ddc::for_each(tensor_bspline_domain, [&](auto idx) {
707  int_vals.spline_coef(idx) = r_integrals(ddc::select<BSplinesR>(idx))
708  * p_integrals(ddc::select<BSplinesP>(idx));
709  });
710 
711  if (int_vals.spline_coef.domain().template extent<BSplinesP>() == p_bspl_space.size()) {
712  ddc::DiscreteDomain<BSplinesP> periodic_points(p_integrals.domain().take_last(
713  ddc::DiscreteVector<BSplinesP> {BSplinesP::degree()}));
714  tensor_product_discrete_domain_type repeat_domain(r_tensor_product_dom, periodic_points);
715  ddc::for_each(repeat_domain, [&](auto idx) { int_vals.spline_coef(idx) = 0.0; });
716  }
