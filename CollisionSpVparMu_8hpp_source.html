119 public:
128  void deepcopy_poloidal_plane(DFieldThetaR dst, InputDFieldThetaR src)
129  {
130  if constexpr ((collisions_dimensions::is_spoofed_dim_v<GridR>)&&(
131  collisions_dimensions::is_spoofed_dim_v<GridTheta>)) {
132  ddc::parallel_fill(Kokkos::DefaultExecutionSpace(), dst, src);
133  } else if constexpr ((!collisions_dimensions::is_spoofed_dim_v<GridR>)&&(
134  !collisions_dimensions::is_spoofed_dim_v<GridTheta>)) {
135  ddc::parallel_deepcopy(dst, src);
136  } else {
137  using NonSpoofDim = std::
138  conditional_t<collisions_dimensions::is_spoofed_dim_v<GridR>, GridTheta, GridR>;
139  ddc::parallel_for_each(
140  Kokkos::DefaultExecutionSpace(),
141  get_idx_range(dst),
142  KOKKOS_LAMBDA(Idx<GridTheta, GridR> idx) {
143  dst(idx) = src(ddc::select<NonSpoofDim>(idx));
144  });
145  }
