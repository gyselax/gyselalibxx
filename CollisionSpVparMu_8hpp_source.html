127  // Return a Kokkos view from B_norm of type DFieldRtheta. Enable to treat the case when n_r or/and ntheta equal to 1.
128  template <class DFieldRTheta>
129  koliop_interface::MDL<double**> rthetaprof_to_koliop(
130  DFieldRTheta B_norm,
131  std::size_t const n_r,
132  std::size_t const n_theta)
133  {
134  koliop_interface::MDL<double**> kop_B_norm;
135 
136  if constexpr (std::is_same_v<DFieldRTheta, double>) {
137  // If DFieldRTheta is a double create an array and copy the double to GPU
138  kop_B_norm = koliop_interface::MDL<double**>(
139  Kokkos::view_alloc(Kokkos::WithoutInitializing, "B_norm"),
140  n_r,
141  n_theta);
142  Kokkos::deep_copy(kop_B_norm, B_norm);
143  } else {
144  // If DFieldRTheta is a chunk
145  static_assert(
146  std::is_same_v<
147  typename DFieldRTheta::memory_space,
148  typename Kokkos::DefaultExecutionSpace::memory_space>,
149  "The data should be provided on GPU");
150  auto B_norm_kokkos_view = B_norm.allocation_kokkos_view();
151  if constexpr (DFieldRTheta::rank() == 1) {
152  // If a 1D chunk then a reference to the data can be packed directly into a 2D chunk
153  kop_B_norm = koliop_interface::MDL<double**>(B_norm.data_handle(), n_r, n_theta);
154  } else if constexpr (
155  std::is_same_v<
156  typename decltype(B_norm_kokkos_view)::traits::array_layout,
157  typename koliop_interface::MDL<const double**>::traits::array_layout>) {
158  // If a 2D chunk with the right layout we can use the kokkos view directly
159  kop_B_norm = B_norm_kokkos_view;
160  } else {
161  // If a 2D chunk with the wrong layout then a copy is required
162  static_assert(DFieldRTheta::rank() == 2);
163  kop_B_norm = koliop_interface::MDL<double**>(
164  Kokkos::view_alloc(Kokkos::WithoutInitializing, "B_norm"),
165  n_r,
166  n_theta);
167  // A copy is required to change the layout
168  Kokkos::deep_copy(kop_B_norm, B_norm_kokkos_view);
169  }
170  }
171  return kop_B_norm;
172  }
