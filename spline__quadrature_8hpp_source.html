59template <class Grid1D, class SplineBuilder>
60host_t<DFieldMem<IdxRange<Grid1D>>> spline_quadrature_coefficients_1d(
61 IdxRange<Grid1D> const& idx_range,
62 SplineBuilder const& builder)
63{
64 static_assert(
65 SplineBuilder::s_nbc_xmin == 0,
66 "The spline quadrature requires a builder which can construct the coefficients using "
67 "only the values at the interpolation points.");
68 static_assert(
69 SplineBuilder::s_nbc_xmax == 0,
70 "The spline quadrature requires a builder which can construct the coefficients using "
71 "only the values at the interpolation points.");
72 // Since spline builder quadrature coeffs are not available on device, need host allocated builder.
73 // See https://github.com/CExA-project/ddc/issues/598
74 static_assert(
75 (std::is_same_v<typename SplineBuilder::memory_space, Kokkos::HostSpace>),
76 "SplineBuilder must be host allocated.");
77
78 DFieldMem<IdxRange<Grid1D>, ddc::KokkosAllocator<double, typename SplineBuilder::memory_space>>
79 quadrature_coefficients(builder.interpolation_domain());
80 std::tie(std::ignore, quadrature_coefficients, std::ignore) = builder.quadrature_coefficients();
81 return ddc::create_mirror_and_copy(quadrature_coefficients[idx_range]);
82}

98template <class ExecSpace, class... DDims, class... SplineBuilders>
99DFieldMem<IdxRange<DDims...>, ddc::KokkosAllocator<double, typename ExecSpace::memory_space>>
100spline_quadrature_coefficients(
101 IdxRange<DDims...> const& idx_range,
102 SplineBuilders const&... builders)
103{
104 assert((std::is_same_v<
105 typename DDims::continuous_dimension_type,
106 typename SplineBuilders::continuous_dimension_type> and ...));
107
108 // Get coefficients for each dimension
109 std::tuple<CoefficientFieldMem1D_host<DDims>...> current_dim_coeffs(
110 spline_quadrature_coefficients_1d(ddc::select<DDims>(idx_range), builders)...);
111
112 // Allocate ND coefficients
113 DFieldMem<IdxRange<DDims...>, ddc::KokkosAllocator<double, typename ExecSpace::memory_space>>
114 coefficients(idx_range);
115 auto coefficients_host = ddc::create_mirror(get_field(coefficients));
116 // Serial loop is used due to nvcc bug concerning functions with variadic template arguments
117 // (see https://github.com/kokkos/kokkos/pull/7059)
118 ddc::for_each(idx_range, [&](Idx<DDims...> const idim) {
119 // multiply the 1D coefficients by one another
120 coefficients_host(idim)
121 = (std::get<CoefficientFieldMem1D_host<DDims>>(current_dim_coeffs)(
122 ddc::select<DDims>(idim))
123 * ... * 1);
124 });
125 ddc::parallel_deepcopy(coefficients, coefficients_host);
126 return coefficients;
127}
