216 template <class Mapping>
217 PolarSplineFEMPoissonLikeSolver(
218 host_t<ConstSpline2D> coeff_alpha,
219 host_t<ConstSpline2D> coeff_beta,
220 Mapping const& mapping)
221 : nbasis_r(ddc::discrete_space<BSplinesR_Polar>().nbasis() - n_overlap_cells - 1)
222 , nbasis_theta(ddc::discrete_space<BSplinesTheta_Polar>().nbasis())
223 , fem_non_singular_idx_range(
224 ddc::discrete_space<PolarBSplinesRTheta>().tensor_bspline_idx_range().remove_last(
225 IdxStep<PolarBSplinesRTheta> {nbasis_theta}))
226 , radial_bsplines(ddc::discrete_space<BSplinesR_Polar>().full_domain().remove_first(
227 IdxStep<BSplinesR_Polar> {n_overlap_cells}))
228 , polar_bsplines(ddc::discrete_space<BSplinesTheta_Polar>().full_domain().take_first(
229 IdxStep<BSplinesTheta_Polar> {nbasis_theta}))
230 , quadrature_idx_range_r(
231 Idx<QDimRMesh>(0),
232 IdxStep<QDimRMesh>(
233 n_gauss_legendre_r * ddc::discrete_space<BSplinesR_Polar>().ncells()))
234 , quadrature_idx_range_theta(
235 Idx<QDimThetaMesh>(0),
236 IdxStep<QDimThetaMesh>(
237 n_gauss_legendre_theta
238 * ddc::discrete_space<BSplinesTheta_Polar>().ncells()))
239 , quadrature_idx_range_singular(
240 quadrature_idx_range_r.take_first(
241 IdxStep<QDimRMesh> {n_overlap_cells * n_gauss_legendre_r}),
242 quadrature_idx_range_theta)
243 , points_r(quadrature_idx_range_r)
244 , points_theta(quadrature_idx_range_theta)
245 , weights_r(quadrature_idx_range_r)
246 , weights_theta(quadrature_idx_range_theta)
247 , singular_basis_vals_and_derivs(IdxRange<PolarBSplinesRTheta, QDimRMesh, QDimThetaMesh>(
248 PolarBSplinesRTheta::singular_idx_range<PolarBSplinesRTheta>(),
249 ddc::select<QDimRMesh>(quadrature_idx_range_singular),
250 ddc::select<QDimThetaMesh>(quadrature_idx_range_singular)))
251 , r_basis_vals_and_derivs(
252 IdxRange<RBasisSubset, QDimRMesh>(non_zero_bases_r, quadrature_idx_range_r))
253 , theta_basis_vals_and_derivs(
254 IdxRange<
255 ThetaBasisSubset,
256 QDimThetaMesh>(non_zero_bases_theta, quadrature_idx_range_theta))
257 , int_volume(IdxRangeQuadratureRTheta(quadrature_idx_range_r, quadrature_idx_range_theta))
258 , m_polar_spline_evaluator(ddc::NullExtrapolationRule())
259 {
260 // Get break points
261 IdxRange<KnotsR> idx_range_r_edges
262 = ddc::discrete_space<BSplinesR_Polar>().break_point_domain();
263 IdxRange<KnotsTheta> idx_range_theta_edges
264 = ddc::discrete_space<BSplinesTheta_Polar>().break_point_domain();
265 host_t<FieldMem<Coord<R>, IdxRange<KnotsR>>> breaks_r(idx_range_r_edges);
266 host_t<FieldMem<Coord<Theta>, IdxRange<KnotsTheta>>> breaks_theta(idx_range_theta_edges);
267
268 ddc::for_each(idx_range_r_edges, [&](Idx<KnotsR> i) { breaks_r(i) = ddc::coordinate(i); });
269 ddc::for_each(idx_range_theta_edges, [&](Idx<KnotsTheta> i) {
270 breaks_theta(i) = ddc::coordinate(i);
271 });

273 // Define quadrature points and weights
274 GaussLegendre<R> gl_coeffs_r(n_gauss_legendre_r);
275 GaussLegendre<Theta> gl_coeffs_theta(n_gauss_legendre_theta);
276 gl_coeffs_r.compute_points_and_weights_on_mesh(
277 get_field(points_r),
278 get_field(weights_r),
279 get_const_field(breaks_r));
280 gl_coeffs_theta.compute_points_and_weights_on_mesh(
281 get_field(points_theta),
282 get_field(weights_theta),
283 get_const_field(breaks_theta));

285 std::vector<double> vect_points_r(points_r.size());
286 for (IdxQuadratureR i : quadrature_idx_range_r) {
287 vect_points_r[i - quadrature_idx_range_r.front()] = points_r(i);
288 }
289 std::vector<double> vect_points_theta(points_theta.size());
290 for (IdxQuadratureTheta i : quadrature_idx_range_theta) {
291 vect_points_theta[i - quadrature_idx_range_theta.front()] = points_theta(i);

298 // Find value and derivative of 1D bsplines in radial direction
299 ddc::for_each(quadrature_idx_range_r, [&](IdxQuadratureR const ir) {
300 std::array<double, 2 * n_non_zero_bases_r> data;
301 DSpan2D vals(data.data(), n_non_zero_bases_r, 2);
302 ddc::discrete_space<BSplinesR_Polar>()
303 .eval_basis_and_n_derivs(vals, ddc::coordinate(ir), 1);
304 for (auto ib : non_zero_bases_r) {
305 const int ib_idx = ib - non_zero_bases_r.front();
306 r_basis_vals_and_derivs(ib, ir).value = vals(ib_idx, 0);
307 r_basis_vals_and_derivs(ib, ir).derivative = vals(ib_idx, 1);
308 }
309 });
310
311 // Find value and derivative of 1D bsplines in poloidal direction
312 ddc::for_each(quadrature_idx_range_theta, [&](IdxQuadratureTheta const ip) {
313 std::array<double, 2 * n_non_zero_bases_theta> data;
314 DSpan2D vals(data.data(), n_non_zero_bases_theta, 2);
315 ddc::discrete_space<BSplinesTheta_Polar>()
316 .eval_basis_and_n_derivs(vals, ddc::coordinate(ip), 1);
317 for (auto ib : non_zero_bases_theta) {
318 const int ib_idx = ib - non_zero_bases_theta.front();
319 theta_basis_vals_and_derivs(ib, ip).value = vals(ib_idx, 0);
320 theta_basis_vals_and_derivs(ib, ip).derivative = vals(ib_idx, 1);
321 }
322 });

325 = PolarBSplinesRTheta::singular_idx_range<PolarBSplinesRTheta>();

327 // Find value and derivative of 2D bsplines covering the singular point
328 ddc::for_each(quadrature_idx_range_singular, [&](IdxQuadratureRTheta const irp) {
329 std::array<double, PolarBSplinesRTheta::n_singular_basis()> singular_data;
330 std::array<double, n_non_zero_bases_r * n_non_zero_bases_theta> data;
331 // Values of the polar basis splines around the singular point
332 // at a given coordinate
333 DSpan1D singular_vals(singular_data.data(), PolarBSplinesRTheta::n_singular_basis());
334 // Values of the polar basis splines, that do not cover the singular point,
335 // at a given coordinate
336 DSpan2D vals(data.data(), n_non_zero_bases_r, n_non_zero_bases_theta);
337 IdxQuadratureR ir = ddc::select<QDimRMesh>(irp);
338 IdxQuadratureTheta ip = ddc::select<QDimThetaMesh>(irp);
339
340 const CoordRTheta coord(ddc::coordinate(irp));
341
342 // Calculate the value
343 ddc::discrete_space<PolarBSplinesRTheta>().eval_basis(singular_vals, vals, coord);
344 for (auto ib : singular_idx_range) {
345 singular_basis_vals_and_derivs(ib, ir, ip).value
346 = singular_vals[ib - singular_idx_range.front()];
347 }
348
349 // Calculate the radial derivative
350 ddc::discrete_space<PolarBSplinesRTheta>().eval_deriv_r(singular_vals, vals, coord);
351 for (auto ib : singular_idx_range) {
352 singular_basis_vals_and_derivs(ib, ir, ip).radial_derivative
353 = singular_vals[ib - singular_idx_range.front()];
354 }
355
356 // Calculate the poloidal derivative
357 ddc::discrete_space<PolarBSplinesRTheta>().eval_deriv_theta(singular_vals, vals, coord);
358 for (auto ib : singular_idx_range) {
359 singular_basis_vals_and_derivs(ib, ir, ip).poloidal_derivative
360 = singular_vals[ib - singular_idx_range.front()];
361 }
362 });

364 // Find the integral volume associated with each point used in the quadrature scheme
365 IdxRangeQuadratureRTheta
366 all_quad_points(quadrature_idx_range_r, quadrature_idx_range_theta);
367 ddc::for_each(all_quad_points, [&](IdxQuadratureRTheta const irp) {

371 int_volume(ir, ip) = abs(mapping.jacobian(coord)) * weights_r(ir) * weights_theta(ip);

382 // Number of elements in the matrix that correspond to the splines
383 // that cover the singular point
384 constexpr int n_elements_singular
385 = PolarBSplinesRTheta::n_singular_basis() * PolarBSplinesRTheta::n_singular_basis();
386 // Number of non-zero elements in the matrix corresponding to the inner product of
387 // polar splines at the singular point and the other splines
388 const int n_elements_overlap = 2
389 * (PolarBSplinesRTheta::n_singular_basis()
390 * BSplinesR_Polar::degree() * nbasis_theta);
391 const int n_stencil_theta
392 = nbasis_theta * min(int(1 + 2 * BSplinesTheta_Polar::degree()), nbasis_theta);
393 const int n_stencil_r = nbasis_r * (1 + 2 * BSplinesR_Polar::degree())
394 - (1 + BSplinesR_Polar::degree()) * BSplinesR_Polar::degree();
395 // Number of non-zero elements in the matrix corresponding to the inner product of
396 // non-central splines. These have a tensor product structure
397 const int n_elements_stencil = n_stencil_r * n_stencil_theta;
398
399 const int batch_size = 1;
400 // Matrix size is equal to the number Polar bspline
401 const int matrix_size = ddc::discrete_space<PolarBSplinesRTheta>().nbasis() - nbasis_theta;
402 const int n_matrix_elements = n_elements_singular + n_elements_overlap + n_elements_stencil;
403

408 col_idx_csr_host("idx_csr", n_matrix_elements);
409 Kokkos::View<int*, Kokkos::LayoutRight, Kokkos::DefaultHostExecutionSpace>

414 // Calculate the matrix elements corresponding to the bsplines which cover the singular point
415 ddc::for_each(singular_idx_range, [&](IdxPolarBspl const idx_test) {
416 ddc::for_each(singular_idx_range, [&](IdxPolarBspl const idx_trial) {
417 // Calculate the weak integral
418 double const element = ddc::transform_reduce(
419 quadrature_idx_range_singular,
420 0.0,
421 ddc::reducer::sum<double>(),
422 [&](IdxQuadratureRTheta const quad_idx) {
423 IdxQuadratureR const ir = ddc::select<QDimRMesh>(quad_idx);
424 IdxQuadratureTheta const ip = ddc::select<QDimThetaMesh>(quad_idx);
425 return weak_integral_element(
426 ir,
427 ip,
428 singular_basis_vals_and_derivs(idx_test, ir, ip),
429 singular_basis_vals_and_derivs(idx_trial, ir, ip),
430 coeff_alpha,
431 coeff_beta,
432 spline_evaluator,
433 mapping);
434 });
435 const int row_idx = idx_test - singular_idx_range.front();
436 const int col_idx = idx_trial - singular_idx_range.front();
437 const int csr_idx_singular_area = nnz_per_row_csr_host(row_idx + 1);
438 //Fill the C matrix corresponding to the splines on the singular point
439 col_idx_csr_host(csr_idx_singular_area) = col_idx;

441 nnz_per_row_csr_host(row_idx + 1)++;
442 });

453 // Calculate the matrix elements where bspline products overlap the bsplines which cover the singular point

456 non_singular_idx_range_near_centre,

458 const IdxPolarBspl polar_idx_trial(

466 // Find the index range covering the cells where both the test and trial functions are non-zero
467 const Idx<RCellDim> first_overlap_element_r(
468 r_idx_trial.uid() < BSplinesR_Polar::degree()
469 ? 0
470 : r_idx_trial.uid() - BSplinesR_Polar::degree());
471 const Idx<ThetaCellDim> first_overlap_element_theta(
472 theta_mod(theta_idx_trial.uid() - BSplinesTheta_Polar::degree()));
473
474 const IdxStep<RCellDim> n_overlap_r(
475 n_overlap_cells - first_overlap_element_r.uid());
476 const IdxStep<ThetaCellDim> n_overlap_theta(
477 BSplinesTheta_Polar::degree() + 1);
478
479 const IdxRange<RCellDim> r_cells(first_overlap_element_r, n_overlap_r);
480 const IdxRange<ThetaCellDim>
481 theta_cells(first_overlap_element_theta, n_overlap_theta);
482 const IdxRange<RCellDim, ThetaCellDim> non_zero_cells(r_cells, theta_cells);
483
484 if (n_overlap_r > 0) {
485 double element = 0.0;
486
487 ddc::for_each(non_zero_cells, [&](IdxCell const cell_idx) {
488 const int cell_idx_r(ddc::select<RCellDim>(cell_idx).uid());
489 const int cell_idx_theta(
490 theta_mod(ddc::select<ThetaCellDim>(cell_idx).uid()));
491
492 const IdxRangeQuadratureRTheta cell_quad_points(
493 get_quadrature_points_in_cell(cell_idx_r, cell_idx_theta));
494 // Find the column where the non-zero data is stored
495 Idx<RBasisSubset> ib_trial_r(r_idx_trial.uid() - cell_idx_r);
496 Idx<ThetaBasisSubset> ib_trial_theta(
497 theta_mod(theta_idx_trial.uid() - cell_idx_theta));
498 // Calculate the weak integral
499 element += ddc::transform_reduce(
500 cell_quad_points,
501 0.0,
502 ddc::reducer::sum<double>(),
503 [&](IdxQuadratureRTheta const quad_idx) {
504 IdxQuadratureR const ir
505 = ddc::select<QDimRMesh>(quad_idx);
506 IdxQuadratureTheta const ip
507 = ddc::select<QDimThetaMesh>(quad_idx);
508 return weak_integral_element<Mapping>(
509 ir,
510 ip,
511 singular_basis_vals_and_derivs(
512 idx_test,
513 ir,
514 ip),
515 r_basis_vals_and_derivs(ib_trial_r, ir),
516 theta_basis_vals_and_derivs(ib_trial_theta, ip),
517 coeff_alpha,
518 coeff_beta,
519 spline_evaluator,
520 mapping);
521 });
522 });
523
524 int const row_idx = idx_test - singular_idx_range.front();
525 int const col_idx = polar_idx_trial - singular_idx_range.front();
526 //a_ij
527 col_idx_csr_host(nnz_per_row_csr_host(row_idx + 1)) = col_idx;
528 values_csr_host(batch_idx, nnz_per_row_csr_host(row_idx + 1)) = element;
529 nnz_per_row_csr_host(row_idx + 1)++;
530 //a_ji
531 col_idx_csr_host(nnz_per_row_csr_host(col_idx + 1)) = row_idx;
532 values_csr_host(batch_idx, nnz_per_row_csr_host(col_idx + 1)) = element;
533 nnz_per_row_csr_host(col_idx + 1)++;
534 }
535 });
536 });

538 // Calculate the matrix elements following a stencil
539 ddc::for_each(fem_non_singular_idx_range, [&](IdxPolarBspl const polar_idx_test) {
540 const IdxBSpline2D_Polar idx_test(PolarBSplinesRTheta::get_2d_index(polar_idx_test));
541 const std::size_t r_idx_test(
542 ddc::select<PolarBSplinesRTheta::BSplinesR_tag>(idx_test).uid());
543 const std::size_t theta_idx_test(
544 ddc::select<PolarBSplinesRTheta::BSplinesTheta_tag>(idx_test).uid());
545
546 // Calculate the index of the elements that are already filled
547 IdxRangeBSTheta_Polar remaining_theta(
548 Idx<BSplinesTheta_Polar> {theta_idx_test},
549 IdxStep<BSplinesTheta_Polar> {BSplinesTheta_Polar::degree() + 1});
550 ddc::for_each(remaining_theta, [&](auto const theta_idx_trial) {
551 IdxBSpline2D_Polar idx_trial(Idx<BSplinesR_Polar>(r_idx_test), theta_idx_trial);
552 IdxPolarBspl polar_idx_trial(
553 PolarBSplinesRTheta::get_polar_index<PolarBSplinesRTheta>(
554 IdxBSpline2D_Polar(r_idx_test, theta_mod(theta_idx_trial.uid()))));
555 double element = get_matrix_stencil_element(
556 idx_test,
557 idx_trial,
558 coeff_alpha,
559 coeff_beta,
560 spline_evaluator,
561 mapping);
562 int const int_polar_idx_test = polar_idx_test - singular_idx_range.front();
563 if (polar_idx_test == polar_idx_trial) {
564 const int idx = nnz_per_row_csr_host(int_polar_idx_test + 1);
565 col_idx_csr_host(idx) = int_polar_idx_test;
566 values_csr_host(batch_idx, idx) = element;
567 nnz_per_row_csr_host(int_polar_idx_test + 1)++;
568 } else {
569 int const int_polar_idx_trial = polar_idx_trial - singular_idx_range.front();
570
571 const int aij_idx = nnz_per_row_csr_host(int_polar_idx_test + 1);
572 col_idx_csr_host(aij_idx) = int_polar_idx_trial;
573 values_csr_host(batch_idx, aij_idx) = element;
574 nnz_per_row_csr_host(int_polar_idx_test + 1)++;
575
576 const int aji_idx = nnz_per_row_csr_host(int_polar_idx_trial + 1);
577 col_idx_csr_host(aji_idx) = int_polar_idx_test;
578 values_csr_host(batch_idx, aji_idx) = element;
579 nnz_per_row_csr_host(int_polar_idx_trial + 1)++;
580 }
581 });
582 IdxRangeBSR remaining_r(
583 ddc::select<BSplinesR_Polar>(idx_test) + 1,
584 IdxStep<BSplinesR_Polar> {
585 min(BSplinesR_Polar::degree(),
586 ddc::discrete_space<BSplinesR_Polar>().nbasis() - 2 - r_idx_test)});
587 IdxRangeBSTheta_Polar relevant_theta(
588 Idx<BSplinesTheta_Polar> {
589 theta_idx_test + ddc::discrete_space<BSplinesTheta_Polar>().nbasis()
590 - BSplinesTheta_Polar::degree()},
591 IdxStep<BSplinesTheta_Polar> {2 * BSplinesTheta_Polar::degree() + 1});
592
593 IdxRangeBSRTheta trial_idx_range(remaining_r, relevant_theta);
594
595 ddc::for_each(trial_idx_range, [&](IdxBSpline2D_Polar const idx_trial) {
596 const int r_idx_trial(
597 ddc::select<PolarBSplinesRTheta::BSplinesR_tag>(idx_trial).uid());
598 const int theta_idx_trial(
599 ddc::select<PolarBSplinesRTheta::BSplinesTheta_tag>(idx_trial).uid());
600 IdxPolarBspl polar_idx_trial(
601 PolarBSplinesRTheta::get_polar_index<PolarBSplinesRTheta>(
602 IdxBSpline2D_Polar(r_idx_trial, theta_mod(theta_idx_trial))));
603 double element = get_matrix_stencil_element(
604 idx_test,
605 idx_trial,
606 coeff_alpha,
607 coeff_beta,
608 spline_evaluator,
609 mapping);
610 int const int_polar_idx_test = polar_idx_test - singular_idx_range.front();
611 if (polar_idx_test == polar_idx_trial) {
612 const int idx = nnz_per_row_csr_host(int_polar_idx_test + 1);
613 col_idx_csr_host(idx) = int_polar_idx_test;
614 values_csr_host(batch_idx, idx) = element;
615 nnz_per_row_csr_host(int_polar_idx_test + 1)++;
616 } else {
617 int const int_polar_idx_trial = polar_idx_trial - singular_idx_range.front();
618 const int aij_idx = nnz_per_row_csr_host(int_polar_idx_test + 1);
619 col_idx_csr_host(aij_idx) = int_polar_idx_trial;
620 values_csr_host(batch_idx, aij_idx) = element;
621 nnz_per_row_csr_host(int_polar_idx_test + 1)++;
622
623 const int aji_idx = nnz_per_row_csr_host(int_polar_idx_trial + 1);
624 col_idx_csr_host(aji_idx) = int_polar_idx_test;
625 values_csr_host(batch_idx, aji_idx) = element;
626 nnz_per_row_csr_host(int_polar_idx_trial + 1)++;
627 }
628 });
629 });
630 assert(nnz_per_row_csr_host(matrix_size) == n_matrix_elements);
631 m_gko_matrix = std::make_unique<MatrixBatchCsr<
632 Kokkos::DefaultExecutionSpace,
633 MatrixBatchCsrSolver::CG>>(1, matrix_size, n_matrix_elements);
634
635 auto [values, col_idx, nnz_per_row] = m_gko_matrix->get_batch_csr();
636 Kokkos::deep_copy(values, values_csr_host);
637 Kokkos::deep_copy(col_idx, col_idx_csr_host);
638 Kokkos::deep_copy(nnz_per_row, nnz_per_row_csr_host);
639 m_gko_matrix->setup_solver();
640 }

655 template <class RHSFunction>
656 void operator()(RHSFunction const& rhs, SplinePolar& spline) const
657 {
658 const int b_size = ddc::discrete_space<PolarBSplinesRTheta>().nbasis()
659 - ddc::discrete_space<BSplinesTheta_Polar>().nbasis();
660 const int batch_size = 1;
661 Kokkos::View<double**, Kokkos::LayoutRight, Kokkos::DefaultHostExecutionSpace>
662 b_host("b_host", batch_size, b_size);
663
664 // Fill b
665 ddc::for_each(
666 PolarBSplinesRTheta::singular_idx_range<PolarBSplinesRTheta>(),
667 [&](IdxPolarBspl const idx) {
668 b_host(0, idx.uid()) = ddc::transform_reduce(
669 quadrature_idx_range_singular,
670 0.0,
671 ddc::reducer::sum<double>(),
672 [&](IdxQuadratureRTheta const quad_idx) {
673 IdxQuadratureR const ir = ddc::select<QDimRMesh>(quad_idx);
674 IdxQuadratureTheta const ip = ddc::select<QDimThetaMesh>(quad_idx);
675 CoordRTheta coord(ddc::coordinate(quad_idx));
676 return rhs(coord)
677 * singular_basis_vals_and_derivs(idx, ir, ip).value
678 * int_volume(ir, ip);
679 });
680 });
681 const std::size_t ncells_r = ddc::discrete_space<BSplinesR_Polar>().ncells();
682 ddc::for_each(fem_non_singular_idx_range, [&](IdxPolarBspl const idx) {
683 const IdxBSpline2D_Polar idx_2d(PolarBSplinesRTheta::get_2d_index(idx));
684 const std::size_t r_idx(ddc::select<PolarBSplinesRTheta::BSplinesR_tag>(idx_2d).uid());
685 const std::size_t theta_idx(
686 ddc::select<PolarBSplinesRTheta::BSplinesTheta_tag>(idx_2d).uid());
687
688 // Find the cells on which the bspline is non-zero
689 int first_cell_r(r_idx - BSplinesR_Polar::degree());
690 int first_cell_theta(theta_idx - BSplinesTheta_Polar::degree());
691 std::size_t last_cell_r(r_idx + 1);
692 if (first_cell_r < 0)
693 first_cell_r = 0;
694 if (last_cell_r > ncells_r)
695 last_cell_r = ncells_r;
696 IdxStep<RCellDim> const r_length(last_cell_r - first_cell_r);
697 IdxStep<ThetaCellDim> const theta_length(BSplinesTheta_Polar::degree() + 1);
698
699
700 Idx<RCellDim> const start_r(first_cell_r);
701 Idx<ThetaCellDim> const start_theta(theta_mod(first_cell_theta));
702 const IdxRange<RCellDim> r_cells(start_r, r_length);
703 const IdxRange<ThetaCellDim> theta_cells(start_theta, theta_length);
704 const IdxRange<RCellDim, ThetaCellDim> non_zero_cells(r_cells, theta_cells);
705 assert(r_length * theta_length > 0);
706 double element = 0.0;
707 ddc::for_each(non_zero_cells, [&](IdxCell const cell_idx) {
708 const int cell_idx_r(ddc::select<RCellDim>(cell_idx).uid());
709 const int cell_idx_theta(theta_mod(ddc::select<ThetaCellDim>(cell_idx).uid()));
710
711 const IdxRangeQuadratureRTheta cell_quad_points(
712 get_quadrature_points_in_cell(cell_idx_r, cell_idx_theta));
713
714 // Find the column where the non-zero data is stored
715 Idx<RBasisSubset> ib_r(r_idx - cell_idx_r);
716 Idx<ThetaBasisSubset> ib_theta(theta_mod(theta_idx - cell_idx_theta));
717
718 // Calculate the weak integral
719 element += ddc::transform_reduce(
720 cell_quad_points,
721 0.0,
722 ddc::reducer::sum<double>(),
723 [&](IdxQuadratureRTheta const quad_idx) {
724 IdxQuadratureR const ir = ddc::select<QDimRMesh>(quad_idx);
725 IdxQuadratureTheta const ip = ddc::select<QDimThetaMesh>(quad_idx);
726 CoordRTheta coord(ddc::coordinate(quad_idx));
727 double rb = r_basis_vals_and_derivs(ib_r, ir).value;
728 double pb = theta_basis_vals_and_derivs(ib_theta, ip).value;
729 return rhs(coord) * rb * pb * int_volume(ir, ip);
730 });
731 });
732 b_host(0, idx.uid()) = element;
733 });
734
735 // Solve the matrix equation
736 Kokkos::View<double**, Kokkos::LayoutRight, Kokkos::DefaultExecutionSpace>
737 b("b", batch_size, b_size);
738 Kokkos::deep_copy(b, b_host);
739
740 m_gko_matrix->solve(b);
741
742 Kokkos::deep_copy(b_host, b);
743 //-----------------
744 IdxRangeBSRTheta dirichlet_boundary_idx_range(
745 radial_bsplines.take_last(IdxStep<BSplinesR_Polar> {1}),
746 polar_bsplines);
747 IdxRangeBSTheta_Polar polar_idx_range(
748 ddc::discrete_space<BSplinesTheta_Polar>().full_domain());
749
750
751 // Fill the spline
752 ddc::for_each(
753 PolarBSplinesRTheta::singular_idx_range<PolarBSplinesRTheta>(),
754 [&](IdxPolarBspl const idx) {
755 const int bspl_idx
756 = idx
757 - PolarBSplinesRTheta::singular_idx_range<PolarBSplinesRTheta>()
758 .front();
759 spline.singular_spline_coef(idx) = b_host(0, bspl_idx);
760 });
761 ddc::for_each(fem_non_singular_idx_range, [&](IdxPolarBspl const idx) {
762 const IdxBSpline2D_Polar idx_2d(PolarBSplinesRTheta::get_2d_index(idx));
763 spline.spline_coef(idx_2d) = b_host(0, idx.uid());
764 });
765 ddc::for_each(dirichlet_boundary_idx_range, [&](IdxBSpline2D_Polar const idx) {
766 spline.spline_coef(idx) = 0.0;
767 });
768
769 // Copy the periodic elements
770 IdxRangeBSRTheta copy_idx_range(
771 radial_bsplines,
772 polar_idx_range.remove_first(IdxStep<BSplinesTheta_Polar>(
773 ddc::discrete_space<BSplinesTheta_Polar>().nbasis())));
774 ddc::for_each(copy_idx_range, [&](IdxBSpline2D_Polar const idx_2d) {
775 spline.spline_coef(
776 ddc::select<BSplinesR_Polar>(idx_2d),
777 ddc::select<BSplinesTheta_Polar>(idx_2d))
778 = spline.spline_coef(
779 ddc::select<BSplinesR_Polar>(idx_2d),
780 ddc::select<BSplinesTheta_Polar>(idx_2d)
781 - ddc::discrete_space<BSplinesTheta_Polar>().nbasis());
782 });
783 }

961 template <class Mapping, class TestValDerivType, class TrialValDerivType>
962 double templated_weak_integral_element(
963 IdxQuadratureR ir,
964 IdxQuadratureTheta ip,
965 TestValDerivType const& test_bspline_val_and_deriv,
966 TrialValDerivType const& trial_bspline_val_and_deriv,
967 TestValDerivType const& test_bspline_val_and_deriv_theta,
968 TrialValDerivType const& trial_bspline_val_and_deriv_theta,
969 host_t<ConstSpline2D> coeff_alpha,
970 host_t<ConstSpline2D> coeff_beta,
971 SplineRThetaEvaluatorNullBound const& spline_evaluator,
972 Mapping const& mapping)
973 {
974 static_assert(
975 std::is_same_v<
976 TestValDerivType,
977 EvalDeriv1DType> || std::is_same_v<TestValDerivType, EvalDeriv2DType>);
978 static_assert(
979 std::is_same_v<
980 TrialValDerivType,
981 EvalDeriv1DType> || std::is_same_v<TrialValDerivType, EvalDeriv2DType>);
982
983 // Calculate coefficients at quadrature point
984 CoordRTheta coord(ddc::coordinate(ir), ddc::coordinate(ip));
985 const double alpha = spline_evaluator(coord, coeff_alpha);
986 const double beta = spline_evaluator(coord, coeff_beta);
987
988 // Define the value and gradient of the test and trial basis functions
989 double basis_val_test_space;
990 double basis_val_trial_space;
991 std::array<double, 2> basis_gradient_test_space;
992 std::array<double, 2> basis_gradient_trial_space;
993 get_value_and_gradient(
994 basis_val_test_space,
995 basis_gradient_test_space,
996 test_bspline_val_and_deriv,
997 test_bspline_val_and_deriv_theta);
998 get_value_and_gradient(
999 basis_val_trial_space,
1000 basis_gradient_trial_space,
1001 trial_bspline_val_and_deriv,
1002 trial_bspline_val_and_deriv_theta);
1003
1004 MetricTensor<Mapping, CoordRTheta> metric_tensor(mapping);
1005
1006 // Assemble the weak integral element
1007 return int_volume(ir, ip)
1008 * (alpha
1009 * dot_product(
1010 basis_gradient_test_space,
1011 metric_tensor.to_covariant(basis_gradient_trial_space, coord))
1012 + beta * basis_val_test_space * basis_val_trial_space);
1013 }

1019 template <class Mapping>
1020 double get_matrix_stencil_element(
1021 IdxBSpline2D_Polar idx_test,
1022 IdxBSpline2D_Polar idx_trial,
1023 host_t<ConstSpline2D> coeff_alpha,
1024 host_t<ConstSpline2D> coeff_beta,
1025 SplineRThetaEvaluatorNullBound const& evaluator,
1026 Mapping const& mapping)
1027 {
1028 // 0 <= r_idx_test < 8
1029 // 0 <= r_idx_trial < 8
1030 // r_idx_test < r_idx_trial
1031 const int r_idx_test(ddc::select<BSplinesR_Polar>(idx_test).uid());
1032 const int r_idx_trial(ddc::select<BSplinesR_Polar>(idx_trial).uid());
1033 // 0 <= theta_idx_test < 8
1034 // 0 <= theta_idx_trial < 8
1035 int theta_idx_test(theta_mod(ddc::select<BSplinesTheta_Polar>(idx_test).uid()));
1036 int theta_idx_trial(theta_mod(ddc::select<BSplinesTheta_Polar>(idx_trial).uid()));
1037
1038 const std::size_t ncells_r = ddc::discrete_space<BSplinesR_Polar>().ncells();
1039
1040 // 0<= r_offset <= degree_r
1041 // -degree_theta <= theta_offset <= degree_theta
1042 const int r_offset = r_idx_trial - r_idx_test;
1043 int theta_offset = theta_mod(theta_idx_trial - theta_idx_test);
1044 if (theta_offset >= int(nbasis_theta - BSplinesTheta_Polar::degree())) {
1045 theta_offset -= nbasis_theta;
1046 }
1047 assert(r_offset >= 0);
1048 assert(r_offset <= int(BSplinesR_Polar::degree()));
1049 assert(theta_offset >= -int(BSplinesTheta_Polar::degree()));
1050 assert(theta_offset <= int(BSplinesTheta_Polar::degree()));
1051
1052 // Find the index range covering the cells where both the test and trial functions are non-zero
1053 int n_overlap_stencil_r(BSplinesR_Polar::degree() + 1 - r_offset);
1054 int first_overlap_r(r_idx_trial - BSplinesR_Polar::degree());
1055
1056 int first_overlap_theta;
1057 int n_overlap_stencil_theta;
1058 if (theta_offset > 0) {
1059 n_overlap_stencil_theta = BSplinesTheta_Polar::degree() + 1 - theta_offset;
1060 first_overlap_theta = theta_mod(theta_idx_trial - BSplinesTheta_Polar::degree());
1061 } else {
1062 n_overlap_stencil_theta = BSplinesTheta_Polar::degree() + 1 + theta_offset;
1063 first_overlap_theta = theta_mod(theta_idx_test - BSplinesTheta_Polar::degree());
1064 }
1065
1066 if (first_overlap_r < 0) {
1067 const int n_compact = first_overlap_r;
1068 first_overlap_r = 0;
1069 n_overlap_stencil_r += n_compact;
1070 }
1071
1072 const int n_to_edge_r(ncells_r - first_overlap_r);
1073
1074 const IdxStep<RCellDim> n_overlap_r(min(n_overlap_stencil_r, n_to_edge_r));
1075 const IdxStep<ThetaCellDim> n_overlap_theta(n_overlap_stencil_theta);
1076
1077 const Idx<RCellDim> first_overlap_element_r(first_overlap_r);
1078 const Idx<ThetaCellDim> first_overlap_element_theta(first_overlap_theta);
1079
1080 const IdxRange<RCellDim> r_cells(first_overlap_element_r, n_overlap_r);
1081 const IdxRange<ThetaCellDim> theta_cells(first_overlap_element_theta, n_overlap_theta);
1082 const IdxRange<RCellDim, ThetaCellDim> non_zero_cells(r_cells, theta_cells);
1083
1084 assert(n_overlap_r * n_overlap_theta > 0);
1085 return ddc::transform_reduce(
1086 non_zero_cells,
1087 0.0,
1088 ddc::reducer::sum<double>(),
1089 [&](IdxCell const cell_idx) {
1090 const int cell_idx_r(ddc::select<RCellDim>(cell_idx).uid());
1091 const int cell_idx_theta(theta_mod(ddc::select<ThetaCellDim>(cell_idx).uid()));
1092
1093 const IdxRangeQuadratureRTheta cell_quad_points(
1094 get_quadrature_points_in_cell(cell_idx_r, cell_idx_theta));
1095
1096 int ib_test_theta_idx = theta_idx_test - cell_idx_theta;
1097 int ib_trial_theta_idx = theta_idx_trial - cell_idx_theta;
1098
1099 // Find the column where the non-zero data is stored
1100 Idx<RBasisSubset> ib_test_r(r_idx_test - cell_idx_r);
1101 Idx<ThetaBasisSubset> ib_test_theta(theta_mod(ib_test_theta_idx));
1102 Idx<RBasisSubset> ib_trial_r(r_idx_trial - cell_idx_r);
1103 Idx<ThetaBasisSubset> ib_trial_theta(theta_mod(ib_trial_theta_idx));
1104
1105 assert(ib_test_r.uid() < BSplinesR_Polar::degree() + 1);
1106 assert(ib_test_theta.uid() < BSplinesTheta_Polar::degree() + 1);
1107 assert(ib_trial_r.uid() < BSplinesR_Polar::degree() + 1);
1108 assert(ib_trial_theta.uid() < BSplinesTheta_Polar::degree() + 1);
1109
1110 // Calculate the weak integral
1111 return ddc::transform_reduce(
1112 cell_quad_points,
1113 0.0,
1114 ddc::reducer::sum<double>(),
1115 [&](IdxQuadratureRTheta const quad_idx) {
1116 IdxQuadratureR const r_idx = ddc::select<QDimRMesh>(quad_idx);
1117 IdxQuadratureTheta const theta_idx
1118 = ddc::select<QDimThetaMesh>(quad_idx);
1119 return weak_integral_element(
1120 r_idx,
1121 theta_idx,
1122 r_basis_vals_and_derivs(ib_test_r, r_idx),
1123 r_basis_vals_and_derivs(ib_trial_r, r_idx),
1124 theta_basis_vals_and_derivs(ib_test_theta, theta_idx),
1125 theta_basis_vals_and_derivs(ib_trial_theta, theta_idx),
1126 coeff_alpha,
1127 coeff_beta,
1128 evaluator,
1129 mapping);
1130 });
1131 });
