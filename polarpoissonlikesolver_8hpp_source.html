695  void operator()(RHSFunction const& rhs, host_t<SplinePolar>& spline) const
696  {
697  static_assert(
698  std::is_invocable_r_v<double, RHSFunction, CoordRTheta>,
699  "RHSFunction must have an operator() which takes a coordinate and returns a "
700  "double");
701  const int b_size = ddc::discrete_space<PolarBSplinesRTheta>().nbasis()
702  - ddc::discrete_space<BSplinesTheta>().nbasis();
703  const int batch_size = 1;
704  Kokkos::View<double**, Kokkos::LayoutRight, Kokkos::DefaultHostExecutionSpace>
705  b_host("b_host", batch_size, b_size);
706 
707  // Fill b
708  ddc::for_each(
709  PolarBSplinesRTheta::template singular_idx_range<PolarBSplinesRTheta>(),
710  [&](IdxBSPolar const idx) {
711  b_host(0, idx.uid()) = ddc::transform_reduce(
712  m_idxrange_quadrature_singular,
713  0.0,
714  ddc::reducer::sum<double>(),
715  [&](IdxQuadratureRTheta const idx_quad) {
716  IdxQuadratureR const idx_r = ddc::select<QDimRMesh>(idx_quad);
717  IdxQuadratureTheta const idx_theta
718  = ddc::select<QDimThetaMesh>(idx_quad);
719  CoordRTheta coord(ddc::coordinate(idx_quad));
720  return rhs(coord)
721  * m_singular_basis_vals_and_derivs(idx, idx_r, idx_theta)
722  .value
723  * m_int_volume(idx_r, idx_theta);
724  });
725  });
726  const std::size_t ncells_r = ddc::discrete_space<BSplinesR>().ncells();
727  ddc::for_each(m_idxrange_fem_non_singular, [&](IdxBSPolar const idx) {
728  const IdxBSRTheta idx_2d(PolarBSplinesRTheta::get_2d_index(idx));
729  const std::size_t idx_r(ddc::select<BSplinesR>(idx_2d).uid());
730  const std::size_t idx_theta(ddc::select<BSplinesTheta>(idx_2d).uid());
731 
732  // Find the cells on which the bspline is non-zero
733  int first_cell_r(idx_r - BSplinesR::degree());
734  int first_cell_theta(idx_theta - BSplinesTheta::degree());
735  std::size_t last_cell_r(idx_r + 1);
736  if (first_cell_r < 0)
737  first_cell_r = 0;
738  if (last_cell_r > ncells_r)
739  last_cell_r = ncells_r;
740  IdxStep<RCellDim> const r_length(last_cell_r - first_cell_r);
741  IdxStep<ThetaCellDim> const theta_length(BSplinesTheta::degree() + 1);
742 
743 
744  Idx<RCellDim> const start_r(first_cell_r);
745  Idx<ThetaCellDim> const start_theta(theta_mod(first_cell_theta));
746  const IdxRange<RCellDim> r_cells(start_r, r_length);
747  const IdxRange<ThetaCellDim> theta_cells(start_theta, theta_length);
748  const IdxRange<RCellDim, ThetaCellDim> non_zero_cells(r_cells, theta_cells);
749  assert(r_length * theta_length > 0);
750  double element = 0.0;
751  ddc::for_each(non_zero_cells, [&](IdxCell const cell_idx) {
752  const int cell_idx_r(ddc::select<RCellDim>(cell_idx).uid());
753  const int cell_idx_theta(theta_mod(ddc::select<ThetaCellDim>(cell_idx).uid()));
754 
755  const IdxRangeQuadratureRTheta cell_quad_points(
756  get_quadrature_points_in_cell(cell_idx_r, cell_idx_theta));
757 
758  // Find the column where the non-zero data is stored
759  Idx<RBasisSubset> ib_r(idx_r - cell_idx_r);
760  Idx<ThetaBasisSubset> ib_theta(theta_mod(idx_theta - cell_idx_theta));
761 
762  // Calculate the weak integral
763  element += ddc::transform_reduce(
764  cell_quad_points,
765  0.0,
766  ddc::reducer::sum<double>(),
767  [&](IdxQuadratureRTheta const idx_quad) {

769  IdxQuadratureTheta const idx_theta
770  = ddc::select<QDimThetaMesh>(idx_quad);
771  CoordRTheta coord(ddc::coordinate(idx_quad));
772  double rb = r_basis_vals_and_derivs(ib_r, idx_r).value;
773  double pb = m_theta_basis_vals_and_derivs(ib_theta, idx_theta).value;
774  return rhs(coord) * rb * pb * m_int_volume(idx_r, idx_theta);
775  });
776  });
777  b_host(0, idx.uid()) = element;
778  });
779

781  Kokkos::View<double**, Kokkos::LayoutRight, Kokkos::DefaultExecutionSpace>
782  b("b", batch_size, b_size);
783  Kokkos::deep_copy(b, b_host);
784 
785  m_gko_matrix->solve(b);
786 
787  Kokkos::deep_copy(b_host, b);
788  //-----------------
789  IdxRangeBSRTheta dirichlet_boundary_idx_range(
790  m_idxrange_bsplines_r.take_last(IdxStep<BSplinesR> {1}),
791  m_idxrange_bsplines_theta);
792  IdxRangeBSTheta idxrange_polar(ddc::discrete_space<BSplinesTheta>().full_domain());
793 
794

797  PolarBSplinesRTheta::template singular_idx_range<PolarBSplinesRTheta>(),
798  [&](IdxBSPolar const idx) {
799  const int bspl_idx = idx
800  - PolarBSplinesRTheta::template singular_idx_range<
801  PolarBSplinesRTheta>()
802  .front();

804  });
805  ddc::for_each(m_idxrange_fem_non_singular, [&](IdxBSPolar const idx) {
806  const IdxBSRTheta idx_2d(PolarBSplinesRTheta::get_2d_index(idx));
807  spline.spline_coef(idx_2d) = b_host(0, idx.uid());
808  });
809  ddc::for_each(dirichlet_boundary_idx_range, [&](IdxBSRTheta const idx) {
810  spline.spline_coef(idx) = 0.0;
811  });

815  m_idxrange_bsplines_r,
816  idxrange_polar.remove_first(
817  IdxStep<BSplinesTheta>(ddc::discrete_space<BSplinesTheta>().nbasis())));

820  = spline.spline_coef(
821  ddc::select<BSplinesR>(idx_2d),
822  ddc::select<BSplinesTheta>(idx_2d)
823  - ddc::discrete_space<BSplinesTheta>().nbasis());
824  });

1004  template <class Mapping, class TestValDerivType, class TrialValDerivType>
1005  double templated_weak_integral_element(
1006  IdxQuadratureR idx_r,
1007  IdxQuadratureTheta idx_theta,
1008  TestValDerivType const& test_bspline_val_and_deriv,
1009  TrialValDerivType const& trial_bspline_val_and_deriv,
1010  TestValDerivType const& test_bspline_val_and_deriv_theta,
1011  TrialValDerivType const& trial_bspline_val_and_deriv_theta,
1012  host_t<ConstSpline2D> coeff_alpha,
1013  host_t<ConstSpline2D> coeff_beta,
1014  SplineRThetaEvaluatorNullBound const& spline_evaluator,
1015  Mapping const& mapping)
1016  {
1017  static_assert(
1018  std::is_same_v<
1019  TestValDerivType,
1020  EvalDeriv1DType> || std::is_same_v<TestValDerivType, EvalDeriv2DType>);
1021  static_assert(
1022  std::is_same_v<
1023  TrialValDerivType,
1024  EvalDeriv1DType> || std::is_same_v<TrialValDerivType, EvalDeriv2DType>);
1025 
1026  // Calculate coefficients at quadrature point
1027  CoordRTheta coord(ddc::coordinate(idx_r), ddc::coordinate(idx_theta));
1028  const double alpha = spline_evaluator(coord, coeff_alpha);
1029  const double beta = spline_evaluator(coord, coeff_beta);
1030 
1031  // Define the value and gradient of the test and trial basis functions
1032  double basis_val_test_space;
1033  double basis_val_trial_space;
1034  std::array<double, 2> basis_gradient_test_space;
1035  std::array<double, 2> basis_gradient_trial_space;
1036  get_value_and_gradient(
1037  basis_val_test_space,
1038  basis_gradient_test_space,
1039  test_bspline_val_and_deriv,
1040  test_bspline_val_and_deriv_theta);
1041  get_value_and_gradient(
1042  basis_val_trial_space,
1043  basis_gradient_trial_space,
1044  trial_bspline_val_and_deriv,
1045  trial_bspline_val_and_deriv_theta);
1046 
1047  MetricTensor<Mapping, CoordRTheta> metric_tensor(mapping);
1048 
1049  // Assemble the weak integral element
1050  return m_int_volume(idx_r, idx_theta)
1051  * (alpha
1052  * dot_product(
1053  basis_gradient_test_space,
1054  metric_tensor.to_covariant(basis_gradient_trial_space, coord))
1055  + beta * basis_val_test_space * basis_val_trial_space);
1056  }

1062  template <class Mapping>
1063  double get_matrix_stencil_element(
1064  IdxBSRTheta idx_test,
1065  IdxBSRTheta idx_trial,
1066  host_t<ConstSpline2D> coeff_alpha,
1067  host_t<ConstSpline2D> coeff_beta,
1068  SplineRThetaEvaluatorNullBound const& evaluator,
1069  Mapping const& mapping)
1070  {
1071  // 0 <= idx_test_r < 8
1072  // 0 <= idx_trial_r < 8
1073  // idx_test_r < idx_trial_r
1074  const int idx_test_r(ddc::select<BSplinesR>(idx_test).uid());
1075  const int idx_trial_r(ddc::select<BSplinesR>(idx_trial).uid());
1076  // 0 <= idx_test_theta < 8
