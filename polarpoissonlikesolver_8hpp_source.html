272 template <class Mapping>
273 PolarSplineFEMPoissonLikeSolver(
274 ConstSpline2D coeff_alpha,
275 ConstSpline2D coeff_beta,
276 Mapping const& mapping,
277 SplineRThetaEvaluatorNullBound const& spline_evaluator)
278 : m_nbasis_r(ddc::discrete_space<BSplinesR>().nbasis() - m_n_overlap_cells - 1)
279 , m_nbasis_theta(ddc::discrete_space<BSplinesTheta>().nbasis())
280 , m_matrix_size(ddc::discrete_space<PolarBSplinesRTheta>().nbasis() - m_nbasis_theta)
281 , m_idxrange_fem_non_singular(
282 ddc::discrete_space<PolarBSplinesRTheta>().tensor_bspline_idx_range().remove_last(
283 IdxStep<PolarBSplinesRTheta> {m_nbasis_theta}))
284 , m_idxrange_bsplines_r(ddc::discrete_space<BSplinesR>().full_domain().remove_first(
285 IdxStep<BSplinesR> {m_n_overlap_cells}))
286 , m_idxrange_bsplines_theta(ddc::discrete_space<BSplinesTheta>().full_domain().take_first(
287 IdxStep<BSplinesTheta> {m_nbasis_theta}))
288 , m_idxrange_quadrature_r(
289 Idx<QDimRMesh>(0),
290 IdxStep<QDimRMesh>(
291 s_n_gauss_legendre_r * ddc::discrete_space<BSplinesR>().ncells()))
292 , m_idxrange_quadrature_theta(
293 Idx<QDimThetaMesh>(0),
294 IdxStep<QDimThetaMesh>(
295 s_n_gauss_legendre_theta * ddc::discrete_space<BSplinesTheta>().ncells()))
296 , m_idxrange_quadrature_singular(
297 m_idxrange_quadrature_r.take_first(
298 IdxStep<QDimRMesh> {m_n_overlap_cells * s_n_gauss_legendre_r}),
299 m_idxrange_quadrature_theta)
300 , m_weights_r(m_idxrange_quadrature_r)
301 , m_weights_theta(m_idxrange_quadrature_theta)
302 , m_singular_basis_vals_and_derivs(IdxRange<PolarBSplinesRTheta, QDimRMesh, QDimThetaMesh>(
303 PolarBSplinesRTheta::template singular_idx_range<PolarBSplinesRTheta>(),
304 ddc::select<QDimRMesh>(m_idxrange_quadrature_singular),
305 ddc::select<QDimThetaMesh>(m_idxrange_quadrature_singular)))
306 , m_r_basis_vals_and_derivs(
307 IdxRange<RBasisSubset, QDimRMesh>(m_non_zero_bases_r, m_idxrange_quadrature_r))
308 , m_theta_basis_vals_and_derivs(
309 IdxRange<
310 ThetaBasisSubset,
311 QDimThetaMesh>(m_non_zero_bases_theta, m_idxrange_quadrature_theta))
312 , m_polar_spline_evaluator(ddc::NullExtrapolationRule())
313 , m_phi_spline_coef(
314 PolarBSplinesRTheta::template singular_idx_range<PolarBSplinesRTheta>(),
315 IdxRangeBSRTheta(
316 m_idxrange_bsplines_r,
317 ddc::discrete_space<BSplinesTheta>().full_domain()))
318 , m_x_init(
319 "x_init",
320 1,
321 ddc::discrete_space<PolarBSplinesRTheta>().nbasis()
322 - ddc::discrete_space<BSplinesTheta>().nbasis())
323 {
324 static_assert(has_2d_jacobian_v<Mapping, CoordRTheta>);
325 //initialize x_init
326 Kokkos::deep_copy(m_x_init, 0);
327 // Get break points
328 IdxRange<KnotsR> idxrange_r_edges = ddc::discrete_space<BSplinesR>().break_point_domain();
329 IdxRange<KnotsTheta> idxrange_theta_edges
330 = ddc::discrete_space<BSplinesTheta>().break_point_domain();
331 host_t<FieldMem<Coord<R>, IdxRange<KnotsR>>> breaks_r(idxrange_r_edges);
332 host_t<FieldMem<Coord<Theta>, IdxRange<KnotsTheta>>> breaks_theta(idxrange_theta_edges);
333
334 ddcHelper::dump_coordinates(Kokkos::DefaultHostExecutionSpace(), get_field(breaks_r));
335 ddcHelper::dump_coordinates(Kokkos::DefaultHostExecutionSpace(), get_field(breaks_theta));
336
337 // Define quadrature points and weights
338 GaussLegendre<QDimRMesh, s_n_gauss_legendre_r> gl_coeffs_r(get_const_field(breaks_r));
339 GaussLegendre<QDimThetaMesh, s_n_gauss_legendre_theta> gl_coeffs_theta(
340 get_const_field(breaks_theta));
341 m_int_volume = compute_coeffs_on_mapping(
342 Kokkos::DefaultExecutionSpace(),
343 mapping,
344 gauss_legendre_quadrature_coefficients<
345 Kokkos::DefaultExecutionSpace>(gl_coeffs_r, gl_coeffs_theta));
346
347 // Find value and derivative of 1D bsplines in radial direction
348 ddc::for_each(m_idxrange_quadrature_r, [&](IdxQuadratureR const idx_r) {
349 std::array<double, 2 * m_n_non_zero_bases_r> data;
350 DSpan2D vals(data.data(), m_n_non_zero_bases_r, 2);
351 ddc::discrete_space<BSplinesR>()
352 .eval_basis_and_n_derivs(vals, ddc::coordinate(idx_r), 1);
353 for (auto ib : m_non_zero_bases_r) {
354 const int ib_idx = ib - m_non_zero_bases_r.front();
355 m_r_basis_vals_and_derivs(ib, idx_r).value = vals(ib_idx, 0);
356 m_r_basis_vals_and_derivs(ib, idx_r).derivative = vals(ib_idx, 1);
357 }
358 });
359
360 // Find value and derivative of 1D bsplines in poloidal direction
361 ddc::for_each(m_idxrange_quadrature_theta, [&](IdxQuadratureTheta const idx_theta) {
362 std::array<double, 2 * m_n_non_zero_bases_theta> data;
363 DSpan2D vals(data.data(), m_n_non_zero_bases_theta, 2);
364 ddc::discrete_space<BSplinesTheta>()
365 .eval_basis_and_n_derivs(vals, ddc::coordinate(idx_theta), 1);
366 for (auto ib : m_non_zero_bases_theta) {
367 const int ib_idx = ib - m_non_zero_bases_theta.front();
368 m_theta_basis_vals_and_derivs(ib, idx_theta).value = vals(ib_idx, 0);
369 m_theta_basis_vals_and_derivs(ib, idx_theta).derivative = vals(ib_idx, 1);
370 }
371 });
372
373 IdxRangeBSPolar idxrange_singular
374 = PolarBSplinesRTheta::template singular_idx_range<PolarBSplinesRTheta>();
375
376 // Find value and derivative of 2D bsplines covering the singular point
377 ddc::for_each(m_idxrange_quadrature_singular, [&](IdxQuadratureRTheta const irp) {
378 std::array<double, PolarBSplinesRTheta::n_singular_basis()> singular_data;
379 std::array<double, m_n_non_zero_bases_r * m_n_non_zero_bases_theta> data;
380 // Values of the polar basis splines around the singular point
381 // at a given coordinate
382 DSpan1D singular_vals(singular_data.data(), PolarBSplinesRTheta::n_singular_basis());
383 // Values of the polar basis splines, that do not cover the singular point,
384 // at a given coordinate
385 DSpan2D vals(data.data(), m_n_non_zero_bases_r, m_n_non_zero_bases_theta);
386 IdxQuadratureR const idx_r(irp);
387 IdxQuadratureTheta const idx_theta(irp);
388
389 const CoordRTheta coord(ddc::coordinate(irp));
390
391 // Calculate the value
392 ddc::discrete_space<PolarBSplinesRTheta>().eval_basis(singular_vals, vals, coord);
393 for (IdxBSPolar ib : idxrange_singular) {
394 m_singular_basis_vals_and_derivs(ib, idx_r, idx_theta).value
395 = singular_vals[ib - idxrange_singular.front()];
396 }
397
398 // Calculate the radial derivative
399 ddc::discrete_space<PolarBSplinesRTheta>().eval_deriv_r(singular_vals, vals, coord);
400 for (IdxBSPolar ib : idxrange_singular) {
401 m_singular_basis_vals_and_derivs(ib, idx_r, idx_theta).radial_derivative
402 = singular_vals[ib - idxrange_singular.front()];
403 }
404
405 // Calculate the poloidal derivative
406 ddc::discrete_space<PolarBSplinesRTheta>().eval_deriv_theta(singular_vals, vals, coord);
407 for (IdxBSPolar ib : idxrange_singular) {
408 m_singular_basis_vals_and_derivs(ib, idx_r, idx_theta).poloidal_derivative
409 = singular_vals[ib - idxrange_singular.front()];
410 }
411 });
412
413 // Number of elements in the matrix that correspond to the splines
414 // that cover the singular point
415 constexpr int n_elements_singular
416 = PolarBSplinesRTheta::n_singular_basis() * PolarBSplinesRTheta::n_singular_basis();
417 // Number of non-zero elements in the matrix corresponding to the inner product of
418 // polar splines at the singular point and the other splines
419 const int n_elements_overlap = 2
420 * (PolarBSplinesRTheta::n_singular_basis()
421 * BSplinesR::degree() * m_nbasis_theta);
422 const int n_stencil_theta
423 = m_nbasis_theta * min(int(1 + 2 * BSplinesTheta::degree()), m_nbasis_theta);
424 const int n_stencil_r = m_nbasis_r * (1 + 2 * BSplinesR::degree())
425 - (1 + BSplinesR::degree()) * BSplinesR::degree();
426 // Number of non-zero elements in the matrix corresponding to the inner product of
427 // non-central splines. These have a tensor product structure
428 const int n_elements_stencil = n_stencil_r * n_stencil_theta;
429
430 const int batch_size = 1;
431
432 const int n_matrix_elements = n_elements_singular + n_elements_overlap + n_elements_stencil;
433
434 //CSR data storage
435 Kokkos::View<double**, Kokkos::LayoutRight, Kokkos::HostSpace>
436 values_csr_host("values_csr", batch_size, n_matrix_elements);
437 Kokkos::View<int*, Kokkos::LayoutRight, Kokkos::HostSpace>
438 col_idx_csr_host("idx_csr", n_matrix_elements);
439 Kokkos::View<int*, Kokkos::LayoutRight, Kokkos::DefaultExecutionSpace>
440 nnz_per_row_csr("nnz_per_row_csr", m_matrix_size + 1);
441 Kokkos::View<int*, Kokkos::LayoutRight, Kokkos::HostSpace>
442 nnz_per_row_csr_host("nnz_per_row_csr", m_matrix_size + 1);
443
444 m_gko_matrix = std::make_unique<MatrixBatchCsr<
445 Kokkos::DefaultExecutionSpace,
446 MatrixBatchCsrSolver::CG>>(1, m_matrix_size, n_matrix_elements);
447 auto [values, col_idx, nnz_per_row] = m_gko_matrix->get_batch_csr();
448 init_nnz_per_line(nnz_per_row);
449 Kokkos::deep_copy(nnz_per_row_csr_host, nnz_per_row);
450
451 compute_singular_elements(
452 coeff_alpha,
453 coeff_beta,
454 mapping,
455 spline_evaluator,
456 values_csr_host,
457 col_idx_csr_host,
458 nnz_per_row_csr_host);
459 compute_overlapping_singular_elements(
460 coeff_alpha,
461 coeff_beta,
462 mapping,
463 spline_evaluator,
464 values_csr_host,
465 col_idx_csr_host,
466 nnz_per_row_csr_host);
467 compute_stencil_elements(
468 coeff_alpha,
469 coeff_beta,
470 mapping,
471 spline_evaluator,
472 values_csr_host,
473 col_idx_csr_host,
474 nnz_per_row_csr_host);
475
476 assert(nnz_per_row_csr_host(m_matrix_size) == n_matrix_elements);
477 Kokkos::deep_copy(values, values_csr_host);
478 Kokkos::deep_copy(col_idx, col_idx_csr_host);
479 Kokkos::deep_copy(nnz_per_row, nnz_per_row_csr_host);
480 m_gko_matrix->setup_solver();
481 }

505 template <class Mapping>
506 void compute_singular_elements(
507 ConstSpline2D coeff_alpha,
508 ConstSpline2D coeff_beta,
509 Mapping const& mapping,
510 SplineRThetaEvaluatorNullBound const& spline_evaluator,
511 Kokkos::View<double**, Kokkos::LayoutRight, Kokkos::HostSpace> const values_csr_host,
512 Kokkos::View<int*, Kokkos::LayoutRight, Kokkos::HostSpace> const col_idx_csr_host,
513 Kokkos::View<int*, Kokkos::LayoutRight, Kokkos::HostSpace> const nnz_per_row_csr_host)
514 {
515 IdxRangeBSPolar idxrange_singular
516 = PolarBSplinesRTheta::template singular_idx_range<PolarBSplinesRTheta>();
517 IdxRangeQuadratureRTheta idxrange_quadrature_singular = m_idxrange_quadrature_singular;
518
519 auto singular_basis_vals_and_derivs_alloc = ddc::create_mirror_view_and_copy(
520 Kokkos::DefaultExecutionSpace(),
521 get_field(m_singular_basis_vals_and_derivs));
522 auto r_basis_vals_and_derivs_alloc = ddc::create_mirror_view_and_copy(
523 Kokkos::DefaultExecutionSpace(),
524 get_field(m_r_basis_vals_and_derivs));
525 Field<EvalDeriv2DType, IdxRange<PolarBSplinesRTheta, QDimRMesh, QDimThetaMesh>>
526 singular_basis_vals_and_derivs = get_field(singular_basis_vals_and_derivs_alloc);
527 DField<IdxRangeQuadratureRTheta> int_volume_proxy = get_field(m_int_volume);
528
529 Kokkos::Profiling::pushRegion("PolarPoissonFillFemMatrix");
530 // Calculate the matrix elements corresponding to the bsplines which cover the singular point
531 ddc::for_each(idxrange_singular, [&](IdxBSPolar const idx_test) {
532 ddc::for_each(idxrange_singular, [&](IdxBSPolar const idx_trial) {
533 // Calculate the weak integral
534 double const element = ddc::parallel_transform_reduce(
535 Kokkos::DefaultExecutionSpace(),
536 idxrange_quadrature_singular,
537 0.0,
538 ddc::reducer::sum<double>(),
539 KOKKOS_LAMBDA(Idx<QDimRMesh, QDimThetaMesh> const& idx_quad) {
540 Idx<QDimRMesh> const idx_r(idx_quad);
541 Idx<QDimThetaMesh> const idx_theta(idx_quad);
542 return weak_integral_element(
543 idx_r,
544 idx_theta,
545 singular_basis_vals_and_derivs(idx_test, idx_r, idx_theta),
546 singular_basis_vals_and_derivs(idx_trial, idx_r, idx_theta),
547 coeff_alpha,
548 coeff_beta,
549 spline_evaluator,
550 mapping,
551 int_volume_proxy);
552 });
553 const int row_idx = idx_test - idxrange_singular.front();
554 const int col_idx = idx_trial - idxrange_singular.front();
555 const int csr_idx_singular_area = nnz_per_row_csr_host(row_idx + 1);
556 //Fill the C matrix corresponding to the splines on the singular point
557 col_idx_csr_host(csr_idx_singular_area) = col_idx;
558 values_csr_host(m_batch_idx, csr_idx_singular_area) = element;
559 nnz_per_row_csr_host(row_idx + 1)++;
560 });
561 });
562 }

586 template <class Mapping>
587 void compute_overlapping_singular_elements(
588 ConstSpline2D coeff_alpha,
589 ConstSpline2D coeff_beta,
590 Mapping const& mapping,
591 SplineRThetaEvaluatorNullBound const& spline_evaluator,
592 Kokkos::View<double**, Kokkos::LayoutRight, Kokkos::HostSpace> const values_csr_host,
593 Kokkos::View<int*, Kokkos::LayoutRight, Kokkos::HostSpace> const col_idx_csr_host,
594 Kokkos::View<int*, Kokkos::LayoutRight, Kokkos::HostSpace> const nnz_per_row_csr_host)
595 {
596 // Create index ranges associated with the 2D splines
597 IdxRangeBSPolar idxrange_singular
598 = PolarBSplinesRTheta::template singular_idx_range<PolarBSplinesRTheta>();
599 IdxRangeBSR central_radial_bspline_idx_range(
600 m_idxrange_bsplines_r.take_first(IdxStep<BSplinesR> {BSplinesR::degree()}));
601
602 IdxRangeBSRTheta idxrange_non_singular_near_centre(
603 central_radial_bspline_idx_range,
604 m_idxrange_bsplines_theta);
605
606 auto singular_basis_vals_and_derivs_alloc = ddc::create_mirror_view_and_copy(
607 Kokkos::DefaultExecutionSpace(),
608 get_field(m_singular_basis_vals_and_derivs));
609 auto r_basis_vals_and_derivs_alloc = ddc::create_mirror_view_and_copy(
610 Kokkos::DefaultExecutionSpace(),
611 get_field(m_r_basis_vals_and_derivs));
612 auto theta_basis_vals_and_derivs_alloc = ddc::create_mirror_view_and_copy(
613 Kokkos::DefaultExecutionSpace(),
614 get_field(m_theta_basis_vals_and_derivs));
615 DField<IdxRangeQuadratureRTheta> int_volume_proxy = get_field(m_int_volume);
616 Field<EvalDeriv2DType, IdxRange<PolarBSplinesRTheta, QDimRMesh, QDimThetaMesh>>
617 singular_basis_vals_and_derivs = get_field(singular_basis_vals_and_derivs_alloc);
618 Field<EvalDeriv1DType, IdxRange<RBasisSubset, QDimRMesh>> r_basis_vals_and_derivs
619 = get_field(r_basis_vals_and_derivs_alloc);
620 Field<EvalDeriv1DType, IdxRange<ThetaBasisSubset, QDimThetaMesh>>
621 theta_basis_vals_and_derivs = get_field(theta_basis_vals_and_derivs_alloc);
622 // Calculate the matrix elements where bspline products overlap the bsplines which cover the singular point
623 ddc::for_each(idxrange_singular, [&](IdxBSPolar const idx_test) {
624 ddc::for_each(idxrange_non_singular_near_centre, [&](IdxBSRTheta const idx_trial) {

848 std::is_invocable_r_v<double, RHSFunction, CoordRTheta>,

857 //Create an initial guess
858 Kokkos::View<double**, Kokkos::LayoutRight, Kokkos::HostSpace>

863 PolarBSplinesRTheta::template singular_idx_range<PolarBSplinesRTheta>(),

867 PolarBSplinesRTheta>()

880 * int_volume_host(idx_r, idx_theta);

894 if (first_cell_r < 0)
895 first_cell_r = 0;

897 last_cell_r = ncells_r;

903 Idx<ThetaCellDim> const start_theta(theta_mod(first_cell_theta));
