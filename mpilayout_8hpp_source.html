72 template <class HeadTag>
73 IdxRange<HeadTag> internal_distribute_idx_range(
74 IdxRange<HeadTag> global_idx_range,
75 int comm_size,
76 int rank)
77 {
78 if constexpr (ddc::in_tags_v<HeadTag, distributed_type_seq>) {
79 assert(global_idx_range.size() % comm_size == 0);
80 IdxStep<HeadTag> elems_on_dim(global_idx_range.size() / comm_size);
81 IdxRange<HeadTag>
82 local_idx_range(global_idx_range.front() + rank * elems_on_dim, elems_on_dim);
83 return local_idx_range;
84 } else {
85 // Data is not actually distributed as it handles the case of an index range which is not defined on a distributed dimension.
86 assert(comm_size == 1);

104 template <class HeadTag, class... Tags, std::enable_if_t<(sizeof...(Tags) > 0), bool> = true>
105 IdxRange<HeadTag, Tags...> internal_distribute_idx_range(
106 IdxRange<HeadTag, Tags...> idx_range,
107 int comm_size,
108 int rank)
109 {
110 IdxRange<HeadTag> global_idx_range_along_dim = ddc::select<HeadTag>(idx_range);
111 IdxRange<HeadTag> local_idx_range_along_dim;
112 IdxRange<Tags...> remaining_idx_range;
113
114 if constexpr (ddc::in_tags_v<HeadTag, distributed_type_seq>) {
115 // The number of MPI processes along this dimension
116 int n_ranks_along_dim = std::gcd(comm_size, global_idx_range_along_dim.size());
117 // The number of MPI processes along all subsequent dimensions
118 int n_elems_lower_dims = comm_size / n_ranks_along_dim;
119 // The rank index for the MPI process along this dimension
120 int rank_along_dim = rank / n_elems_lower_dims;
121 // The rank index for the MPI process along all subsequent dimensions
122 int remaining_rank = rank % n_elems_lower_dims;
123 // Calculate the local index range
124 IdxStep<HeadTag> elems_on_dim(global_idx_range_along_dim.size() / n_ranks_along_dim);
125 Idx<HeadTag> distrib_start(
126 global_idx_range_along_dim.front() + rank_along_dim * elems_on_dim);
127 local_idx_range_along_dim = IdxRange<HeadTag>(distrib_start, elems_on_dim);
128 // Calculate the index range for the subsequent dimensions
129 IdxRange<Tags...> remaining_dims = ddc::select<Tags...>(idx_range);
130 remaining_idx_range = internal_distribute_idx_range(
131 remaining_dims,
132 n_elems_lower_dims,
133 remaining_rank);
134 } else {
135 // Calculate the local index range
136 local_idx_range_along_dim = global_idx_range_along_dim;
137 // Calculate the index range for the subsequent dimensions
138 remaining_idx_range = internal_distribute_idx_range(
139 ddc::select<Tags...>(idx_range),
140 comm_size,
141 rank);
142 }
143 return IdxRange<HeadTag, Tags...>(local_idx_range_along_dim, remaining_idx_range);
144 }
