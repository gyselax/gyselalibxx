106  template <class HeadTag, class... Tags, std::enable_if_t<(sizeof...(Tags) > 0), bool> = true>
107  ddc::DiscreteDomain<HeadTag, Tags...> internal_distribute_domain(
108  ddc::DiscreteDomain<HeadTag, Tags...> domain,
109  int comm_size,
110  int rank)
111  {
112  ddc::DiscreteDomain<HeadTag> global_domain_along_dim = ddc::select<HeadTag>(domain);
113  ddc::DiscreteDomain<HeadTag> local_domain_along_dim;
114  ddc::DiscreteDomain<Tags...> remaining_domain;
115 
116  if constexpr (ddc::in_tags_v<HeadTag, distributed_type_seq>) {
117  // The number of MPI processes along this dimension
118  int n_ranks_along_dim = std::gcd(comm_size, global_domain_along_dim.size());
119  // The number of MPI processes along all subsequent dimensions
120  int n_elems_lower_dims = comm_size / n_ranks_along_dim;
121  // The rank index for the MPI process along this dimension
122  int rank_along_dim = rank / n_elems_lower_dims;
123  // The rank index for the MPI process along all subsequent dimensions
124  int remaining_rank = rank % n_elems_lower_dims;
125  // Calculate the local domain
126  ddc::DiscreteVector<HeadTag> elems_on_dim(
127  global_domain_along_dim.size() / n_ranks_along_dim);
128  ddc::DiscreteElement<HeadTag> distrib_start(
129  global_domain_along_dim.front() + rank_along_dim * elems_on_dim);
130  local_domain_along_dim = ddc::DiscreteDomain<HeadTag>(distrib_start, elems_on_dim);
131  // Calculate the domain for the subsequent dimensions
132  ddc::DiscreteDomain<Tags...> remaining_dims = ddc::select<Tags...>(domain);
133  remaining_domain = internal_distribute_domain(
134  remaining_dims,
135  n_elems_lower_dims,
136  remaining_rank);
137  } else {
138  // Calculate the local domain
139  local_domain_along_dim = global_domain_along_dim;
140  // Calculate the domain for the subsequent dimensions
141  remaining_domain
142  = internal_distribute_domain(ddc::select<Tags...>(domain), comm_size, rank);
143  }
144  return ddc::DiscreteDomain<HeadTag, Tags...>(local_domain_along_dim, remaining_domain);
