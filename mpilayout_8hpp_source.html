105  template <class HeadTag, class... Tags, std::enable_if_t<(sizeof...(Tags) > 0), bool> = true>
106  IdxRange<HeadTag, Tags...> internal_distribute_idx_range(
107  IdxRange<HeadTag, Tags...> idx_range,
108  int comm_size,
109  int rank)
110  {
111  IdxRange<HeadTag> global_idx_range_along_dim = ddc::select<HeadTag>(idx_range);
112  IdxRange<HeadTag> local_idx_range_along_dim;
113  IdxRange<Tags...> remaining_idx_range;
114 
115  if constexpr (ddc::in_tags_v<HeadTag, distributed_type_seq>) {
116  // The number of MPI processes along this dimension
117  int n_ranks_along_dim = std::gcd(comm_size, global_idx_range_along_dim.size());
118  // The number of MPI processes along all subsequent dimensions
119  int n_elems_lower_dims = comm_size / n_ranks_along_dim;
120  // The rank index for the MPI process along this dimension
121  int rank_along_dim = rank / n_elems_lower_dims;
122  // The rank index for the MPI process along all subsequent dimensions
123  int remaining_rank = rank % n_elems_lower_dims;
124  // Calculate the local index range
125  IdxStep<HeadTag> elems_on_dim(global_idx_range_along_dim.size() / n_ranks_along_dim);
126  Idx<HeadTag> distrib_start(
127  global_idx_range_along_dim.front() + rank_along_dim * elems_on_dim);
128  local_idx_range_along_dim = IdxRange<HeadTag>(distrib_start, elems_on_dim);
129  // Calculate the index range for the subsequent dimensions
130  IdxRange<Tags...> remaining_dims = ddc::select<Tags...>(idx_range);
131  remaining_idx_range = internal_distribute_idx_range(
132  remaining_dims,
133  n_elems_lower_dims,
134  remaining_rank);
135  } else {
136  // Calculate the local index range
137  local_idx_range_along_dim = global_idx_range_along_dim;
138  // Calculate the index range for the subsequent dimensions
139  remaining_idx_range = internal_distribute_idx_range(
140  ddc::select<Tags...>(idx_range),
141  comm_size,
142  rank);
143  }
144  return IdxRange<HeadTag, Tags...>(local_idx_range_along_dim, remaining_idx_range);
145  }
