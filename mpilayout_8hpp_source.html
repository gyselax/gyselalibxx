78 template <class HeadTag>
79 static IdxRange<HeadTag> internal_distribute_idx_range(
80 IdxRange<HeadTag> global_idx_range,
81 int comm_size,
82 int rank)
83 {
84 if constexpr (ddc::in_tags_v<HeadTag, distributed_type_seq>) {
85 if (global_idx_range.size() % comm_size != 0) {
86 throw std::runtime_error("The provided index range cannot be split equally over "
87 "the specified number of MPI ranks.");
88 }
89 IdxStep<HeadTag> elems_on_dim(global_idx_range.size() / comm_size);
90 IdxRange<HeadTag>
91 local_idx_range(global_idx_range.front() + rank * elems_on_dim, elems_on_dim);
92 return local_idx_range;
93 } else {
94 // Data is not actually distributed as it handles the case of an index range which is not defined on a distributed dimension.
95 assert(comm_size == 1);
96 return global_idx_range;
97 }

113 template <class HeadTag, class... Tags, std::enable_if_t<(sizeof...(Tags) > 0), bool> = true>
114 static IdxRange<HeadTag, Tags...> internal_distribute_idx_range(
115 IdxRange<HeadTag, Tags...> idx_range,
116 int comm_size,
117 int rank)
118 {
119 IdxRange<HeadTag> global_idx_range_along_dim = ddc::select<HeadTag>(idx_range);
120 IdxRange<HeadTag> local_idx_range_along_dim;
121 IdxRange<Tags...> remaining_idx_range;
122
123 if constexpr (ddc::in_tags_v<HeadTag, distributed_type_seq>) {
124 // The number of MPI processes along this dimension
125 int n_ranks_along_dim = std::gcd(comm_size, global_idx_range_along_dim.size());
126 // The number of MPI processes along all subsequent dimensions
127 int n_elems_lower_dims = comm_size / n_ranks_along_dim;
128 // The rank index for the MPI process along this dimension
129 int rank_along_dim = rank / n_elems_lower_dims;
130 // The rank index for the MPI process along all subsequent dimensions
131 int remaining_rank = rank % n_elems_lower_dims;
132 // Calculate the local index range
133 IdxStep<HeadTag> elems_on_dim(global_idx_range_along_dim.size() / n_ranks_along_dim);
134 Idx<HeadTag> distrib_start(
135 global_idx_range_along_dim.front() + rank_along_dim * elems_on_dim);
136 local_idx_range_along_dim = IdxRange<HeadTag>(distrib_start, elems_on_dim);
137 // Calculate the index range for the subsequent dimensions
138 IdxRange<Tags...> remaining_dims = ddc::select<Tags...>(idx_range);
139 remaining_idx_range = internal_distribute_idx_range(
140 remaining_dims,
141 n_elems_lower_dims,
142 remaining_rank);
143 } else {
144 // Calculate the local index range
145 local_idx_range_along_dim = global_idx_range_along_dim;
146 // Calculate the index range for the subsequent dimensions
147 remaining_idx_range = internal_distribute_idx_range(
148 ddc::select<Tags...>(idx_range),
149 comm_size,
150 rank);
151 }
152 return IdxRange<HeadTag, Tags...>(local_idx_range_along_dim, remaining_idx_range);
153 }
154};
