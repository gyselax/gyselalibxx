58  template <class ExecutionSpace, class IntegratorFunction>
59  double operator()(ExecutionSpace exec_space, IntegratorFunction integrated_function) const
60  {
61  static_assert(
62  Kokkos::SpaceAccessibility<ExecutionSpace, MemorySpace>::accessible,
63  "Execution space is not compatible with memory space where coefficients are found");
64  static_assert(
65  std::is_invocable_v<IntegratorFunction, QuadratureIndex>,
66  "The object passed to Quadrature::operator() is not defined on the quadrature "
67  "domain.");
68 
69  ddc::ChunkSpan const coeff_proxy = m_coefficients;
70 
71  // This fence helps avoid a CPU seg fault. See #290 for more details
72  exec_space.fence();
73  return ddc::parallel_transform_reduce(
74  exec_space,
75  coeff_proxy.domain(),
76  0.0,

98  template <class ExecutionSpace, class BatchDomain, class IntegratorFunction>
99  void operator()(
100  ExecutionSpace exec_space,
101  ddc::ChunkSpan<double, BatchDomain, std::experimental::layout_right, MemorySpace> const
102  result,
103  IntegratorFunction integrated_function) const
104  {
105  static_assert(
106  Kokkos::SpaceAccessibility<ExecutionSpace, MemorySpace>::accessible,
107  "Execution space is not compatible with memory space where coefficients are found");
108  static_assert(
109  std::is_same_v<ExecutionSpace, Kokkos::DefaultExecutionSpace>,
110  "Kokkos::TeamPolicy only works with the default execution space. Please use "
111  "DefaultExecutionSpace to call this batched operator.");
112  using ExpectedBatchDims = ddc::type_seq_remove_t<
113  ddc::to_type_seq_t<TotalDomain>,
114  ddc::to_type_seq_t<QuadratureDomain>>;
115  static_assert(
116  ddc::type_seq_same_v<ddc::to_type_seq_t<BatchDomain>, ExpectedBatchDims>,
117  "The batch domain deduced from the type of result does not match the class "
118  "template parameters.");
119 
120  // Get useful index types
121  using TotalIndex = typename TotalDomain::discrete_element_type;
122  using BatchIndex = typename BatchDomain::discrete_element_type;
123 
124  static_assert(
125  std::is_invocable_v<IntegratorFunction, TotalIndex>,
126  "The object passed to Quadrature::operator() is not defined on the total domain.");
127 
128  // Get domains
129  QuadratureDomain quad_domain(m_coefficients.domain());
130  BatchDomain batch_domain(result.domain());
131 
132  ddc::ChunkSpan const coeff_proxy = m_coefficients;
133  // Loop over batch dimensions
134  Kokkos::parallel_for(
135  Kokkos::TeamPolicy<>(exec_space, batch_domain.size(), Kokkos::AUTO),
136  KOKKOS_LAMBDA(const Kokkos::TeamPolicy<>::member_type& team) {
137  const int idx = team.league_rank();
138  BatchIndex ib = to_discrete_element(idx, batch_domain);
139 
140  // Sum over quadrature dimensions
141  double teamSum = 0;
142  Kokkos::parallel_reduce(
143  Kokkos::TeamThreadRange(team, quad_domain.size()),
144  [&](int const& thread_index, double& sum) {
145  QuadratureIndex iq = to_discrete_element(thread_index, quad_domain);
146  TotalIndex it(ib, iq);
147  sum += coeff_proxy(iq) * integrated_function(it);
148  },
149  teamSum);
150  result(ib) = teamSum;
151  });
152  }
153 
154 private:
165  template <class HeadDim, class... DDim>
166  KOKKOS_FUNCTION static ddc::DiscreteElement<HeadDim, DDim...> to_discrete_element(
167  int idx,
168  ddc::DiscreteDomain<HeadDim, DDim...> dom)
169  {
170  ddc::DiscreteDomain<DDim...> subdomain(dom);
171  ddc::DiscreteElement<HeadDim> head_idx(
172  ddc::select<HeadDim>(dom).front() + idx / subdomain.size());
173  if constexpr (sizeof...(DDim) == 0) {
174  return head_idx;
175  } else {
176  ddc::DiscreteElement<DDim...> tail_idx
177  = to_discrete_element(idx % subdomain.size(), subdomain);
178  return ddc::DiscreteElement<HeadDim, DDim...>(head_idx, tail_idx);
179  }
180  }
