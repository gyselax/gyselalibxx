60 template <class ExecutionSpace, class IntegratorFunction>
61 double operator()(ExecutionSpace exec_space, IntegratorFunction integrated_function) const
62 {
63 static_assert(
64 Kokkos::SpaceAccessibility<ExecutionSpace, MemorySpace>::accessible,
65 "Execution space is not compatible with memory space where coefficients are found");
66 static_assert(
67 std::is_invocable_v<IntegratorFunction, IdxQuadrature>,
68 "The object passed to Quadrature::operator() is not defined on the quadrature "
69 "idx_range.");
70
71 QuadConstField const coeff_proxy = m_coefficients;
72
73 // This fence helps avoid a CPU seg fault. See #290 for more details
74 exec_space.fence();
75 // This condition is necessary to execute in serial, even in a device activated build.
76 // Without it a seg fault appears
77 if constexpr (std::is_same_v<ExecutionSpace, Kokkos::DefaultHostExecutionSpace>) {
78 return ddc::transform_reduce(
79 get_idx_range(coeff_proxy),
80 0.0,
81 ddc::reducer::sum<double>(),
82 KOKKOS_LAMBDA(IdxQuadrature const ix) {
83 return coeff_proxy(ix) * integrated_function(ix);
84 });
85 } else {
86 return ddc::parallel_transform_reduce(
87 exec_space,
88 get_idx_range(coeff_proxy),
89 0.0,
90 ddc::reducer::sum<double>(),
91 KOKKOS_LAMBDA(IdxQuadrature const ix) {
92 return coeff_proxy(ix) * integrated_function(ix);
93 });
94 }
95 }

112 template <class ExecutionSpace, class BatchIdxRange, class IntegratorFunction>
113 void operator()(
114 ExecutionSpace exec_space,
115 Field<double, BatchIdxRange, std::experimental::layout_right, MemorySpace> const result,
116 IntegratorFunction integrated_function) const
117 {
118 static_assert(
119 Kokkos::SpaceAccessibility<ExecutionSpace, MemorySpace>::accessible,
120 "Execution space is not compatible with memory space where coefficients are found");
121 static_assert(
122 std::is_same_v<ExecutionSpace, Kokkos::DefaultExecutionSpace>,
123 "Kokkos::TeamPolicy only works with the default execution space. Please use "
124 "DefaultExecutionSpace to call this batched operator.");
125 using ExpectedBatchDims = ddc::type_seq_remove_t<
126 ddc::to_type_seq_t<IdxRangeTotal>,
127 ddc::to_type_seq_t<IdxRangeQuadrature>>;
128 static_assert(
129 ddc::type_seq_same_v<ddc::to_type_seq_t<BatchIdxRange>, ExpectedBatchDims>,
130 "The batch idx_range deduced from the type of result does not match the class "
131 "template parameters.");
132
133 // Get useful index types
134 using IdxTotal = typename IdxRangeTotal::discrete_element_type;
135 using IdxBatch = typename BatchIdxRange::discrete_element_type;
136
137 static_assert(
138 std::is_invocable_v<IntegratorFunction, IdxTotal>,
139 "The object passed to Quadrature::operator() is not defined on the total "
140 "idx_range.");
141
142 // Get index ranges
143 IdxRangeQuadrature quad_idx_range(get_idx_range(m_coefficients));
144 BatchIdxRange batch_idx_range(get_idx_range(result));
145
146 QuadConstField const coeff_proxy = m_coefficients;
147 // Loop over batch dimensions
148 Kokkos::parallel_for(
149 Kokkos::TeamPolicy<>(exec_space, batch_idx_range.size(), Kokkos::AUTO),
150 KOKKOS_LAMBDA(const Kokkos::TeamPolicy<>::member_type& team) {
151 const int idx = team.league_rank();
152 IdxBatch ib = to_discrete_element(idx, batch_idx_range);
153
154 // Sum over quadrature dimensions
155 double teamSum = 0;
156 Kokkos::parallel_reduce(
157 Kokkos::TeamThreadRange(team, quad_idx_range.size()),
158 [&](int const& thread_index, double& sum) {
159 IdxQuadrature iq
160 = to_discrete_element(thread_index, quad_idx_range);
161 IdxTotal it(ib, iq);
162 sum += coeff_proxy(iq) * integrated_function(it);
163 },
164 teamSum);
165 result(ib) = teamSum;
166 });
167 }
168
169private:
180 template <class HeadDim, class... Grid1D>
181 KOKKOS_FUNCTION static Idx<HeadDim, Grid1D...> to_discrete_element(
182 int idx,
183 IdxRange<HeadDim, Grid1D...> idx_range)
184 {
185 IdxRange<Grid1D...> subidx_range(idx_range);
186 Idx<HeadDim> head_idx(ddc::select<HeadDim>(idx_range).front() + idx / subidx_range.size());
187 if constexpr (sizeof...(Grid1D) == 0) {
188 return head_idx;
189 } else {
190 Idx<Grid1D...> tail_idx = to_discrete_element(idx % subidx_range.size(), subidx_range);
191 return Idx<HeadDim, Grid1D...>(head_idx, tail_idx);
192 }
193 }
