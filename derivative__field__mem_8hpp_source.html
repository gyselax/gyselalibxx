144  template <std::size_t ArrayIndex, class Tag>
145  IdxRange<Tag> get_idx_range(
146  physical_idx_range_type val_idx_range,
147  physical_deriv_idx_range_type deriv_idx_ranges)
148  {
149  // If the Tag describes a dimension for which a derivative is defined
150  if constexpr (ddc::in_tags_v<Tag, physical_deriv_grids>) {
151  // If the FieldMem at this index contains the derivatives of this dimension
152  if constexpr (ArrayIndex & (1 << ddc::type_seq_rank_v<Tag, physical_deriv_grids>)) {
153  return ddc::select<Tag>(deriv_idx_ranges);
154  }
155  // If the FieldMem at this index doesn't contain derivatives of this dimension
156  else {
157  return ddc::select<Tag>(val_idx_range);
158  }
159  }
160  // If the Tag describes a derivative
161  else if constexpr (ddc::in_tags_v<Tag, deriv_tags>) {
162  // If the FieldMem at this index contains the derivatives of this dimension
163  if constexpr (ArrayIndex & (1 << ddc::type_seq_rank_v<Tag, deriv_tags>)) {
164  return IdxRange<Tag>(Idx<Tag>(1), IdxStep<Tag>(NDerivs));
165  }
166  // If the FieldMem at this index doesn't contain derivatives of this dimension
167  else {
168  return IdxRange<Tag>(Idx<Tag>(0), IdxStep<Tag>(1));
169  }
170  }
171  // If the Tag describes a dimension for which derivatives are not defined.
172  else {
173  return ddc::select<Tag>(val_idx_range);
174  }
175  }

188  template <class QueryDDim, std::size_t ArrayIndex>
189  std::size_t get_mdspan_size()
190  {
191  if constexpr (ddc::in_tags_v<QueryDDim, deriv_tags>) {
192  if (ArrayIndex & (1 << ddc::type_seq_rank_v<QueryDDim, deriv_tags>)) {
193  return NDerivs;
194  } else {
195  return 1;
196  }
197  } else if constexpr (ddc::in_tags_v<QueryDDim, physical_deriv_grids>) {
198  if constexpr (
199  ArrayIndex & (1 << ddc::type_seq_rank_v<ddc::Deriv<QueryDDim>, deriv_tags>)) {
200  IdxRangeSlice<QueryDDim> idx_range_local(base_type::m_cross_derivative_idx_range);
201  return idx_range_local.extents().value();
202  } else {
203  IdxRange<QueryDDim> idx_range_local(base_type::m_physical_idx_range);
204  return idx_range_local.extents().value();
205  }
206  } else {
207  IdxRange<QueryDDim> idx_range_local(base_type::m_physical_idx_range);
208  return idx_range_local.extents().value();
209  }
