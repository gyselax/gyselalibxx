139  template <std::size_t ArrayIndex, class Tag>
140  IdxRange<Tag> get_idx_range(
141  physical_idx_range_type val_idx_range,
142  physical_deriv_idx_range_type deriv_idx_ranges)
143  {
144  // If the Tag describes a dimension for which a derivative is defined
145  if constexpr (ddc::in_tags_v<Tag, physical_deriv_grids>) {
146  // If the FieldMem at this index contains the derivatives of this dimension
147  if constexpr (ArrayIndex & (1 << ddc::type_seq_rank_v<Tag, physical_deriv_grids>)) {
148  return ddc::select<Tag>(deriv_idx_ranges);
149  }
150  // If the FieldMem at this index doesn't contain derivatives of this dimension
151  else {
152  return ddc::select<Tag>(val_idx_range);
153  }
154  }
155  // If the Tag describes a derivative
156  else if constexpr (ddc::in_tags_v<Tag, deriv_tags>) {
157  // If the FieldMem at this index contains the derivatives of this dimension
158  if constexpr (ArrayIndex & (1 << ddc::type_seq_rank_v<Tag, deriv_tags>)) {
159  return IdxRange<Tag>(Idx<Tag>(1), IdxStep<Tag>(NDerivs));
160  }
161  // If the FieldMem at this index doesn't contain derivatives of this dimension
162  else {
163  return IdxRange<Tag>(Idx<Tag>(0), IdxStep<Tag>(1));
164  }
165  }
166  // If the Tag describes a dimension for which derivatives are not defined.
167  else {
168  return ddc::select<Tag>(val_idx_range);
169  }
170  }

183  template <class QueryDDim, std::size_t ArrayIndex>
184  std::size_t get_mdspan_size()
185  {
186  if constexpr (ddc::in_tags_v<QueryDDim, deriv_tags>) {
187  if (ArrayIndex & (1 << ddc::type_seq_rank_v<QueryDDim, deriv_tags>)) {
188  return NDerivs;

190  return 1;
191  }
192  } else if constexpr (ddc::in_tags_v<QueryDDim, physical_deriv_grids>) {
193  if constexpr (

195  IdxRangeSlice<QueryDDim> idx_range_local(base_type::m_cross_derivative_idx_range);
196  return idx_range_local.extents().value();
