148  template <std::size_t ArrayIndex, class Tag>
149  IdxRange<Tag> get_idx_range(
150  physical_idx_range_type val_idx_range,
151  physical_deriv_idx_range_type deriv_idx_ranges)
152  {
153  // If the Tag describes a dimension for which a derivative is defined
154  if constexpr (ddc::in_tags_v<Tag, physical_deriv_grids>) {
155  // If the FieldMem at this index contains the derivatives of this dimension
156  if constexpr (ArrayIndex & (1 << ddc::type_seq_rank_v<Tag, physical_deriv_grids>)) {
157  return ddc::select<Tag>(deriv_idx_ranges);
158  }
159  // If the FieldMem at this index doesn't contain derivatives of this dimension
160  else {
161  return ddc::select<Tag>(val_idx_range);
162  }
163  }
164  // If the Tag describes a derivative
165  else if constexpr (ddc::in_tags_v<Tag, deriv_tags>) {
166  // If the FieldMem at this index contains the derivatives of this dimension
167  if constexpr (ArrayIndex & (1 << ddc::type_seq_rank_v<Tag, deriv_tags>)) {
168  return IdxRange<Tag>(Idx<Tag>(1), IdxStep<Tag>(NDerivs));
169  }
170  // If the FieldMem at this index doesn't contain derivatives of this dimension
171  else {
172  return IdxRange<Tag>(Idx<Tag>(0), IdxStep<Tag>(1));
173  }
174  }
175  // If the Tag describes a dimension for which derivatives are not defined.
176  else {
177  return ddc::select<Tag>(val_idx_range);
178  }
179  }

194  {
195  if constexpr (ddc::in_tags_v<QueryDDim, deriv_tags>) {
196  if (ArrayIndex & (1 << ddc::type_seq_rank_v<QueryDDim, deriv_tags>)) {
197  return NDerivs;

204  IdxRangeSlice<QueryDDim> idx_range_local(base_type::m_cross_derivative_idx_range);
