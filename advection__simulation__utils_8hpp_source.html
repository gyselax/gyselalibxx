33std::string to_lower(std::string s)
34{
35 std::transform(s.begin(), s.end(), s.begin(), [](unsigned char c) {
36 if (c == ' ') {
37 return '_';
38 } else {
39 return static_cast<char>(std::tolower(c));
40 }
41 });
42 return s;
43}

57template <class LogicalToPhysicalMapping>

89template <class LogicalToPhysicalMapping>
90void output_feet(
91 LogicalToPhysicalMapping const& to_physical_mapping,
92 IdxRangeRTheta const& idx_range_rp,
93 host_t<FieldRTheta<CoordRTheta>> const& feet_coords_rp,
94 std::string const& name)
95{
96 std::ofstream file_feet(name, std::ofstream::out);
97 file_feet << std::fixed << std::setprecision(16);
98
99 IdxRangeR idx_range_r(idx_range_rp);

105 IdxR ir(irp);
106 IdxTheta ip(irp);
107 file_feet << std::setw(15) << (ir - ir_start).value() << std::setw(15)
108 << (ip - ip_start).value();

110 print_coordinate(file_feet, feet_coords_rp(irp), to_physical_mapping, idx_range_p);
111 file_feet << std::endl;
112 });

173template <class AdvectionField, class LogicalToPhysicalMapping, class PhysicalToLogicalMapping>
174host_t<FieldMemRTheta<CoordRTheta>> compute_exact_feet_rp(
175 IdxRangeRTheta const& idx_range_rp,
176 LogicalToPhysicalMapping const& logical_to_physical_mapping,
177 PhysicalToLogicalMapping const& physical_to_logical_mapping,
178 AdvectionField const& advection_field,
179 double const time)
180{
181 static_assert(!std::is_same_v<
182 LogicalToPhysicalMapping,
183 DiscreteToCartesian<X, Y, SplineRThetaEvaluatorConstBound_host>>);
184
185 host_t<FieldMemRTheta<CoordRTheta>> feet_coords_rp(idx_range_rp);
186 CoordXY const coord_xy_center = CoordXY(logical_to_physical_mapping(CoordRTheta(0, 0)));
187 ddc::for_each(idx_range_rp, [&](IdxRTheta const irp) {
188 CoordRTheta const coord_rp = ddc::coordinate(irp);
189 CoordXY const coord_xy
190 = advection_field.exact_feet(logical_to_physical_mapping(coord_rp), time);
191
192 CoordXY const coord_diff = coord_xy - coord_xy_center;
193 if (norm_inf(coord_diff) < 1e-15) {
194 feet_coords_rp(irp) = CoordRTheta(0, 0);
195 } else {
196 feet_coords_rp(irp) = physical_to_logical_mapping(coord_xy);
197 }
198 });
199
200 return feet_coords_rp;
201}

224template <class LogicalToPhysicalMapping, class Function>
225double compute_difference_L2_norm(
226 LogicalToPhysicalMapping const& to_physical_mapping,
227 IdxRangeRTheta const& grid,
228 host_t<DFieldRTheta> allfdistribu_advected,
229 Function& function_to_be_advected,
230 host_t<FieldRTheta<CoordRTheta>> const& feet_coord)
231{
232 host_t<DFieldMemRTheta> exact_function(grid);
233 host_t<DFieldMemRTheta> difference_function(grid);
234 ddc::for_each(grid, [&](IdxRTheta const irp) {
235 exact_function(irp) = function_to_be_advected(feet_coord(irp));
236 difference_function(irp) = exact_function(irp) - allfdistribu_advected(irp);
237 });
238
239 host_t<DFieldMemRTheta> const quadrature_coeffs = compute_coeffs_on_mapping(
240 Kokkos::DefaultHostExecutionSpace(),
241 to_physical_mapping,
242 trapezoid_quadrature_coefficients<Kokkos::DefaultHostExecutionSpace>(grid));
243 host_t<Quadrature<IdxRangeRTheta>> quadrature(get_const_field(quadrature_coeffs));
244
245 double const normL2_exact_function
246 = norm_L2(Kokkos::DefaultHostExecutionSpace(), quadrature, get_field(exact_function));
247 double const normL2_difference_function = norm_L2(
248 Kokkos::DefaultHostExecutionSpace(),
249 quadrature,
250 get_field(difference_function));
251
252 return normL2_difference_function / normL2_exact_function;
253}

337template <
338 class LogicalToPhysicalMapping,
339 class PhysicalToLogicalMapping,
340 class AnalyticalMapping,
341 class TimeStepper,
342 class AdvectionDomain,
343 class Simulation>
344void simulate(
345 LogicalToPhysicalMapping const& to_physical_mapping,
346 PhysicalToLogicalMapping const& to_logical_mapping,
347 AnalyticalMapping const& analytical_mapping,
348 IdxRangeRTheta const& grid,
349 TimeStepper const& time_stepper,
350 AdvectionDomain& advection_domain,
351 Simulation& simulation,
352 PreallocatableSplineInterpolatorRTheta<ddc::NullExtrapolationRule> const&
353 function_interpolator,
354 SplineRThetaBuilder_host const& advection_builder,
355 SplineRThetaEvaluatorConstBound_host& advection_evaluator,
356 double const final_time,
357 double const dt,
358 bool save_curves,
359 bool save_feet,
360 std::string const& output_folder)
361{
362 SplineFootFinder<TimeStepper, AdvectionDomain, LogicalToPhysicalMapping> const foot_finder(
363 time_stepper,
364 advection_domain,
365 to_physical_mapping,
366 advection_builder,
367 advection_evaluator);
368
369 BslAdvectionRTheta advection_operator(function_interpolator, foot_finder, to_physical_mapping);
370 auto function_to_be_advected_test = simulation.get_test_function();
371 auto advection_field_test = simulation.get_advection_field();
372
373
374
375 // TO CLOCK THE SIMULATION ------------------------------------------------------------------
376 std::chrono::time_point<std::chrono::system_clock> start_simulation;
377 std::chrono::time_point<std::chrono::system_clock> end_simulation;
378
379
380
381 // PARAMETERS OF THE TEST -------------------------------------------------------------------
382 int const iteration_number = int(final_time / dt);
383 double const end_time = dt * iteration_number;
384
385
386 host_t<DFieldMemRTheta> allfdistribu_test(grid);
387 host_t<DFieldRTheta> allfdistribu_advected_test;
388
389 host_t<DVectorFieldMemRTheta<X, Y>> advection_field_test_vec(grid);
390
391
392 // START TEST -------------------------------------------------------------------------------
393 start_simulation = std::chrono::system_clock::now();
394
395 // Initialization of the advected function:
396 ddc::for_each(grid, [&](IdxRTheta const irp) {
397 CoordRTheta coord = coordinate(irp);
398 if (ddc::get<R>(coord) <= 1e-15) {
399 ddc::get<Theta>(coord) = 0;
400 }
401 allfdistribu_test(irp) = function_to_be_advected_test(coord);
402 });
403
404
405
406 // Definition of advection field:
407 ddc::for_each(grid, [&](IdxRTheta const irp) {
408 // Moving the coordinates in the physical index range:
409 CoordXY const coord_xy = to_physical_mapping(ddc::coordinate(irp));
410 CoordXY const advection_field = advection_field_test(coord_xy, 0.);
411
412 // Define the advection field on the physical index range:
413 ddcHelper::get<X>(advection_field_test_vec)(irp) = ddc::get<X>(advection_field);
414 ddcHelper::get<Y>(advection_field_test_vec)(irp) = ddc::get<Y>(advection_field);
415 });
416
417
418 // SIMULATION -------------------------------------------------------------------------------
419 // Advect "iteration_number" times:
420 for (int i(0); i < iteration_number; ++i) {
421 allfdistribu_advected_test = advection_operator(
422 get_field(allfdistribu_test),
423 get_const_field(advection_field_test_vec),
424 dt);
425
426 // Save the advected function for each iteration:
427 if (save_curves) {
428 std::string const name = output_folder + "/after_" + std::to_string(i + 1) + ".txt";
429 saving_computed(to_physical_mapping, get_field(allfdistribu_advected_test), name);
430 }
431 }
432
433
434
435 // TREATMENT OF DATA ------------------------------------------------------------------------
436 // Compute the exact characteristic feet:
437 host_t<FieldMemRTheta<CoordRTheta>> feet_coords_rp_end_time(grid);
438 host_t<FieldMemRTheta<CoordRTheta>> feet_coords_rp_dt(grid);
439 feet_coords_rp_end_time = compute_exact_feet_rp(
440 grid,
441 analytical_mapping,
442 to_logical_mapping,
443 advection_field_test,
444 end_time);
445 feet_coords_rp_dt = compute_exact_feet_rp(
446 grid,
447 analytical_mapping,
448 to_logical_mapping,
449 advection_field_test,
450 dt);
451
452
453 // Compute the maximal absolute error on the space at the end of the simulation:
454 double max_err = 0.;
455 ddc::for_each(grid, [&](IdxRTheta const irp) {
456 double const err
457 = fabs(allfdistribu_advected_test(irp)
458 - function_to_be_advected_test(feet_coords_rp_end_time(irp)));
459 max_err = max_err > err ? max_err : err;
460 });
461
462
463 std::cout << " - [" << iteration_number << " iterations] Max absolute error : " << max_err;
464 std::cout << std::endl;
465
466
467 // Print the density difference [density conservation expected]:
468 std::cout << " ... "
469 << "Relative L2 norm error: "
470 << compute_difference_L2_norm(
471 to_physical_mapping,
472 grid,
473 allfdistribu_advected_test,
474 function_to_be_advected_test,
475 get_field(feet_coords_rp_end_time))
476 << std::endl;
477
478
479
480 end_simulation = std::chrono::system_clock::now();
481
482 // Control simulation time
483 display_time(start_simulation, end_simulation, " ... Complete simulation time: ");
484
485
486 // SAVE DATA --------------------------------------------------------------------------------
487 // Save the computed characteristic feet:
488 if (save_feet) {
489 host_t<FieldMemRTheta<CoordRTheta>> feet(grid);
490 ddc::for_each(grid, [&](const IdxRTheta irp) { feet(irp) = ddc::coordinate(irp); });
491 foot_finder(get_field(feet), get_const_field(advection_field_test_vec), dt);
492 std::string const name = output_folder + "/feet_computed.txt";
493 output_feet(to_physical_mapping, grid, get_field(feet), name);
494 }
495
496 // Save the values of the exact function at the initial and final states:
497 if (save_curves) {
498 std::string const name_0 = output_folder + "/after_" + std::to_string(0) + ".txt";
499 std::string const name_1
500 = output_folder + "/after_" + std::to_string(iteration_number) + "_exact.txt";
501
502 host_t<DFieldMemRTheta> initial_function(grid);
503 host_t<DFieldMemRTheta> end_function(grid);
504 ddc::for_each(grid, [&](const IdxRTheta irp) {
505 initial_function(irp) = function_to_be_advected_test(ddc::coordinate(irp));
506
507 // Exact final state
508 end_function(irp) = function_to_be_advected_test(feet_coords_rp_end_time(irp));
509 });
510 saving_computed(to_physical_mapping, get_field(initial_function), name_0);
511 saving_computed(to_physical_mapping, get_field(end_function), name_1);
512 }
513
514
515
516 // Save the exact characteristic feet for a displacement on dt:
517 if (save_feet) {
518 std::string const name = output_folder + "/feet_exact.txt";
519 output_feet(to_physical_mapping, grid, get_field(feet_coords_rp_dt), name);
520 }
521
522
523 std::cout << "--------------------------------------------------------------------------------"
524 << std::endl;
525}
