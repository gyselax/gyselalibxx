32std::string to_lower(std::string s)
33{
34 std::transform(s.begin(), s.end(), s.begin(), [](unsigned char c) {
35 if (c == ' ') {
36 return '_';
37 } else {
38 return static_cast<char>(std::tolower(c));
39 }
40 });
41 return s;
42}

159template <class AdvectionField, class Mapping>
160host_t<FieldMemRTheta<CoordRTheta>> compute_exact_feet_rp(
161 IdxRangeRTheta const& idx_range_rp,
162 Mapping const& mapping,
163 AdvectionField const& advection_field,
164 double const time)
165{
166 static_assert(
167 !std::is_same_v<Mapping, DiscreteToCartesian<X, Y, SplineRThetaEvaluatorConstBound>>);
168
169 host_t<FieldMemRTheta<CoordRTheta>> feet_coords_rp(idx_range_rp);
170 CoordXY const coord_xy_center = CoordXY(mapping(CoordRTheta(0, 0)));
171
172 ddc::for_each(idx_range_rp, [&](IdxRTheta const irp) {
173 CoordRTheta const coord_rp = ddc::coordinate(irp);
174 CoordXY const coord_xy = advection_field.exact_feet(mapping(coord_rp), time);
175
176 CoordXY const coord_diff = coord_xy - coord_xy_center;
177 if (norm_inf(coord_diff) < 1e-15) {
178 feet_coords_rp(irp) = CoordRTheta(0, 0);
179 } else {
180 feet_coords_rp(irp) = mapping(coord_xy);
181 }
182 });
183
184 return feet_coords_rp;
185}

208template <class Mapping, class Function>
209double compute_difference_L2_norm(
210 Mapping const& mapping,
211 IdxRangeRTheta const& grid,
212 host_t<DFieldRTheta> allfdistribu_advected,
213 Function& function_to_be_advected,
214 host_t<FieldRTheta<CoordRTheta>> const& feet_coord)
215{
216 host_t<DFieldMemRTheta> exact_function(grid);
217 host_t<DFieldMemRTheta> difference_function(grid);
218 ddc::for_each(grid, [&](IdxRTheta const irp) {
219 exact_function(irp) = function_to_be_advected(feet_coord(irp));
220 difference_function(irp) = exact_function(irp) - allfdistribu_advected(irp);
221 });
222
223 host_t<DFieldMemRTheta> const quadrature_coeffs = compute_coeffs_on_mapping(
224 mapping,
225 trapezoid_quadrature_coefficients<Kokkos::DefaultHostExecutionSpace>(grid));
226 host_t<Quadrature<IdxRangeRTheta>> quadrature(get_const_field(quadrature_coeffs));
227
228 double const normL2_exact_function = compute_L2_norm(quadrature, get_field(exact_function));
229 double const normL2_difference_function
230 = compute_L2_norm(quadrature, get_field(difference_function));
231
232 return normL2_difference_function / normL2_exact_function;
233}

317template <
318 class Mapping,
319 class AnalyticalMapping,
320 class TimeStepper,
321 class AdvectionDomain,
322 class Simulation>
323void simulate(
324 Mapping const& mapping,
325 AnalyticalMapping const& analytical_mapping,
326 IdxRangeRTheta const& grid,
327 TimeStepper const& time_stepper,
328 AdvectionDomain& advection_domain,
329 Simulation& simulation,
330 PreallocatableSplineInterpolatorRTheta<ddc::NullExtrapolationRule> const&
331 function_interpolator,
332 SplineRThetaBuilder const& advection_builder,
333 SplineRThetaEvaluatorConstBound& advection_evaluator,
334 double const final_time,
335 double const dt,
336 bool if_save_curves,
337 bool if_save_feet,
338 std::string const& output_folder)
339{
340 SplineFootFinder<TimeStepper, AdvectionDomain> const
341 foot_finder(time_stepper, advection_domain, advection_builder, advection_evaluator);
342
343 BslAdvectionRTheta advection_operator(function_interpolator, foot_finder, mapping);
344 auto function_to_be_advected_test = simulation.get_test_function();
345 auto advection_field_test = simulation.get_advection_field();
346
347
348
349 // TO CLOCK THE SIMULATION ------------------------------------------------------------------
350 std::chrono::time_point<std::chrono::system_clock> start_simulation;
351 std::chrono::time_point<std::chrono::system_clock> end_simulation;
352
353
354
355 // PARAMETERS OF THE TEST -------------------------------------------------------------------
356 int const iteration_number = int(final_time / dt);
357 double const end_time = dt * iteration_number;
358
359
360 host_t<DFieldMemRTheta> allfdistribu_test(grid);
361 host_t<DFieldRTheta> allfdistribu_advected_test;
362
363 host_t<DVectorFieldMemRTheta<X, Y>> advection_field_test_vec(grid);
364
365
366 // START TEST -------------------------------------------------------------------------------
367 start_simulation = std::chrono::system_clock::now();
368
369 // Initialization of the advected function:
370 ddc::for_each(grid, [&](IdxRTheta const irp) {
371 CoordRTheta coord = coordinate(irp);
372 if (ddc::get<R>(coord) <= 1e-15) {
373 ddc::get<Theta>(coord) = 0;
374 }
375 allfdistribu_test(irp) = function_to_be_advected_test(coord);
376 });
377
378
379
380 // Definition of advection field:
381 ddc::for_each(grid, [&](IdxRTheta const irp) {
382 // Moving the coordinates in the physical index range:
383 CoordXY const coord_xy = mapping(ddc::coordinate(irp));
384 CoordXY const advection_field = advection_field_test(coord_xy, 0.);
385
386 // Define the advection field on the physical index range:
387 ddcHelper::get<X>(advection_field_test_vec)(irp) = ddc::get<X>(advection_field);
388 ddcHelper::get<Y>(advection_field_test_vec)(irp) = ddc::get<Y>(advection_field);
389 });
390
391
392 // SIMULATION -------------------------------------------------------------------------------
393 // Advect "iteration_number" times:
394 for (int i(0); i < iteration_number; ++i) {
395 allfdistribu_advected_test = advection_operator(
396 allfdistribu_test,
397 get_const_field(advection_field_test_vec),
398 dt);
399
400 // Save the advected function for each iteration:
401 if (if_save_curves) {
402 std::string const name = output_folder + "/after_" + std::to_string(i + 1) + ".txt";
403 saving_computed(mapping, get_field(allfdistribu_advected_test), name);
404 }
405 }
406
407
408
409 // TREATMENT OF DATA ------------------------------------------------------------------------
410 // Compute the exact characteristic feet:
411 host_t<FieldMemRTheta<CoordRTheta>> feet_coords_rp_end_time(grid);
412 host_t<FieldMemRTheta<CoordRTheta>> feet_coords_rp_dt(grid);
413 feet_coords_rp_end_time
414 = compute_exact_feet_rp(grid, analytical_mapping, advection_field_test, end_time);
415 feet_coords_rp_dt = compute_exact_feet_rp(grid, analytical_mapping, advection_field_test, dt);
416
417
418 // Compute the maximal absolute error on the space at the end of the simulation:
419 double max_err = 0.;
420 ddc::for_each(grid, [&](IdxRTheta const irp) {
421 double const err
422 = fabs(allfdistribu_advected_test(irp)
423 - function_to_be_advected_test(feet_coords_rp_end_time(irp)));
424 max_err = max_err > err ? max_err : err;
425 });
426
427
428 std::cout << " - [" << iteration_number << " iterations] Max absolute error : " << max_err;
429 std::cout << std::endl;
430
431
432 // Print the density difference [density conservation expected]:
433 std::cout << " ... "
434 << "Relative L2 norm error: "
435 << compute_difference_L2_norm(
436 mapping,
437 grid,
438 allfdistribu_advected_test,
439 function_to_be_advected_test,
440 get_field(feet_coords_rp_end_time))
441 << std::endl;
442
443
444
445 end_simulation = std::chrono::system_clock::now();
446
447 // Control simulation time
448 display_time(start_simulation, end_simulation, " ... Complete simulation time: ");
449
450
451 // SAVE DATA --------------------------------------------------------------------------------
452 // Save the computed characteristic feet:
453 if (if_save_feet) {
454 host_t<FieldMemRTheta<CoordRTheta>> feet(grid);
455 ddc::for_each(grid, [&](const IdxRTheta irp) { feet(irp) = ddc::coordinate(irp); });
456 foot_finder(get_field(feet), advection_field_test_vec, dt);
457 std::string const name = output_folder + "/feet_computed.txt";
458 save_feet(mapping, grid, get_field(feet), name);
459 }
460
461 // Save the values of the exact function at the initial and final states:
462 if (if_save_curves) {
463 std::string const name_0 = output_folder + "/after_" + std::to_string(0) + ".txt";
464 std::string const name_1
465 = output_folder + "/after_" + std::to_string(iteration_number) + "_exact.txt";
466
467 host_t<DFieldMemRTheta> initial_function(grid);
468 host_t<DFieldMemRTheta> end_function(grid);
469 ddc::for_each(grid, [&](const IdxRTheta irp) {
470 initial_function(irp) = function_to_be_advected_test(ddc::coordinate(irp));
471
472 // Exact final state
473 end_function(irp) = function_to_be_advected_test(feet_coords_rp_end_time(irp));
474 });
475 saving_computed(mapping, get_field(initial_function), name_0);
476 saving_computed(mapping, get_field(end_function), name_1);
477 }
478
479
480
481 // Save the exact characteristic feet for a displacement on dt:
482 if (if_save_feet) {
483 std::string const name = output_folder + "/feet_exact.txt";
484 save_feet(mapping, grid, get_field(feet_coords_rp_dt), name);
485 }
486
487
488 std::cout << "--------------------------------------------------------------------------------"
489 << std::endl;
490}
