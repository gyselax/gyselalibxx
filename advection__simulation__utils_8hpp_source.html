170 template <class AdvectionField, class Mapping>
171 host_t<FieldMemRTheta<CoordRTheta>> compute_exact_feet_rp(
172  IdxRangeRTheta const& idx_range_rp,
173  Mapping const& mapping,
174  AdvectionField const& advection_field,
175  double const time)
176 {
177  static_assert(
178  !std::is_same_v<Mapping, DiscreteToCartesian<X, Y, SplineRThetaEvaluatorConstBound>>);
179 
180  host_t<FieldMemRTheta<CoordRTheta>> feet_coords_rp(idx_range_rp);
181  CoordXY const coord_xy_center = CoordXY(mapping(CoordRTheta(0, 0)));
182 
183  ddc::for_each(idx_range_rp, [&](IdxRTheta const irp) {
184  CoordRTheta const coord_rp = ddc::coordinate(irp);
185  CoordXY const coord_xy = advection_field.exact_feet(mapping(coord_rp), time);
186 
187  CoordXY const coord_diff = coord_xy - coord_xy_center;
188  if (norm_inf(coord_diff) < 1e-15) {
189  feet_coords_rp(irp) = CoordRTheta(0, 0);
190  } else {
191  feet_coords_rp(irp) = mapping(coord_xy);
192  }
193  });
194 
195  return feet_coords_rp;
196 }

219 template <class Mapping, class Function>
220 double compute_difference_L2_norm(
221  Mapping const& mapping,
222  IdxRangeRTheta const& grid,
223  host_t<DFieldRTheta> allfdistribu_advected,
224  Function& function_to_be_advected,
225  host_t<FieldRTheta<CoordRTheta>> const& feet_coord)
226 {
227  host_t<DFieldMemRTheta> exact_function(grid);
228  host_t<DFieldMemRTheta> difference_function(grid);
229  ddc::for_each(grid, [&](IdxRTheta const irp) {
230  exact_function(irp) = function_to_be_advected(feet_coord(irp));
231  difference_function(irp) = exact_function(irp) - allfdistribu_advected(irp);
232  });
233 
234  host_t<DFieldMemRTheta> const quadrature_coeffs = compute_coeffs_on_mapping(
235  Kokkos::DefaultHostExecutionSpace(),
236  mapping,
237  trapezoid_quadrature_coefficients<Kokkos::DefaultHostExecutionSpace>(grid));
238  host_t<Quadrature<IdxRangeRTheta>> quadrature(get_const_field(quadrature_coeffs));
239 
240  double const normL2_exact_function
241  = norm_L2(Kokkos::DefaultHostExecutionSpace(), quadrature, get_field(exact_function));
242  double const normL2_difference_function = norm_L2(
243  Kokkos::DefaultHostExecutionSpace(),
244  quadrature,
245  get_field(difference_function));
246 
247  return normL2_difference_function / normL2_exact_function;
248 }

332 template <
333  class Mapping,
334  class AnalyticalMapping,
335  class TimeStepper,
336  class AdvectionDomain,
337  class Simulation>
338 void simulate(
339  Mapping const& mapping,
340  AnalyticalMapping const& analytical_mapping,
341  IdxRangeRTheta const& grid,
342  TimeStepper const& time_stepper,
343  AdvectionDomain& advection_domain,
344  Simulation& simulation,
345  PreallocatableSplineInterpolatorRTheta<ddc::NullExtrapolationRule> const&
346  function_interpolator,
347  SplineRThetaBuilder const& advection_builder,
348  SplineRThetaEvaluatorConstBound& advection_evaluator,
349  double const final_time,
350  double const dt,
351  bool if_save_curves,
352  bool if_save_feet,
353  std::string const& output_folder)
354 {
355  SplineFootFinder<TimeStepper, AdvectionDomain, Mapping> const foot_finder(
356  time_stepper,
357  advection_domain,
358  mapping,
359  advection_builder,
360  advection_evaluator);
361 
362  BslAdvectionRTheta advection_operator(function_interpolator, foot_finder, mapping);
363  auto function_to_be_advected_test = simulation.get_test_function();
364  auto advection_field_test = simulation.get_advection_field();
365 
366 
367 
368  // TO CLOCK THE SIMULATION ------------------------------------------------------------------
369  std::chrono::time_point<std::chrono::system_clock> start_simulation;
370  std::chrono::time_point<std::chrono::system_clock> end_simulation;
371 
372 
373 
374  // PARAMETERS OF THE TEST -------------------------------------------------------------------
375  int const iteration_number = int(final_time / dt);
376  double const end_time = dt * iteration_number;
377 
378 
379  host_t<DFieldMemRTheta> allfdistribu_test(grid);
380  host_t<DFieldRTheta> allfdistribu_advected_test;
381 
382  host_t<DVectorFieldMemRTheta<X, Y>> advection_field_test_vec(grid);
383 
384 
385  // START TEST -------------------------------------------------------------------------------
386  start_simulation = std::chrono::system_clock::now();
387 
388  // Initialization of the advected function:
389  ddc::for_each(grid, [&](IdxRTheta const irp) {
390  CoordRTheta coord = coordinate(irp);
391  if (ddc::get<R>(coord) <= 1e-15) {
392  ddc::get<Theta>(coord) = 0;
393  }
394  allfdistribu_test(irp) = function_to_be_advected_test(coord);
395  });
396 
397 
398 
399  // Definition of advection field:
400  ddc::for_each(grid, [&](IdxRTheta const irp) {
401  // Moving the coordinates in the physical index range:
402  CoordXY const coord_xy = mapping(ddc::coordinate(irp));
403  CoordXY const advection_field = advection_field_test(coord_xy, 0.);
404 
405  // Define the advection field on the physical index range:
406  ddcHelper::get<X>(advection_field_test_vec)(irp) = ddc::get<X>(advection_field);
407  ddcHelper::get<Y>(advection_field_test_vec)(irp) = ddc::get<Y>(advection_field);
408  });
409 
410 
411  // SIMULATION -------------------------------------------------------------------------------
412  // Advect "iteration_number" times:
413  for (int i(0); i < iteration_number; ++i) {
414  allfdistribu_advected_test = advection_operator(
415  allfdistribu_test,
416  get_const_field(advection_field_test_vec),
417  dt);
418 
419  // Save the advected function for each iteration:
420  if (if_save_curves) {
421  std::string const name = output_folder + "/after_" + std::to_string(i + 1) + ".txt";
422  saving_computed(mapping, get_field(allfdistribu_advected_test), name);
423  }
424  }
425 
426 
427 
428  // TREATMENT OF DATA ------------------------------------------------------------------------
429  // Compute the exact characteristic feet:
430  host_t<FieldMemRTheta<CoordRTheta>> feet_coords_rp_end_time(grid);
431  host_t<FieldMemRTheta<CoordRTheta>> feet_coords_rp_dt(grid);
432  feet_coords_rp_end_time
433  = compute_exact_feet_rp(grid, analytical_mapping, advection_field_test, end_time);
434  feet_coords_rp_dt = compute_exact_feet_rp(grid, analytical_mapping, advection_field_test, dt);
435 
436 
437  // Compute the maximal absolute error on the space at the end of the simulation:
438  double max_err = 0.;
439  ddc::for_each(grid, [&](IdxRTheta const irp) {
440  double const err
441  = fabs(allfdistribu_advected_test(irp)
442  - function_to_be_advected_test(feet_coords_rp_end_time(irp)));
443  max_err = max_err > err ? max_err : err;
444  });
445 
446 
447  std::cout << " - [" << iteration_number << " iterations] Max absolute error : " << max_err;
448  std::cout << std::endl;
449 
450 
451  // Print the density difference [density conservation expected]:
452  std::cout << " ... "
453  << "Relative L2 norm error: "
454  << compute_difference_L2_norm(
455  mapping,
456  grid,
457  allfdistribu_advected_test,
458  function_to_be_advected_test,
459  get_field(feet_coords_rp_end_time))
460  << std::endl;
461 
462 
463 
464  end_simulation = std::chrono::system_clock::now();
465 
466  // Control simulation time
467  display_time(start_simulation, end_simulation, " ... Complete simulation time: ");
468 
469 
470  // SAVE DATA --------------------------------------------------------------------------------
471  // Save the computed characteristic feet:
472  if (if_save_feet) {
473  host_t<FieldMemRTheta<CoordRTheta>> feet(grid);
474  ddc::for_each(grid, [&](const IdxRTheta irp) { feet(irp) = ddc::coordinate(irp); });
475  foot_finder(get_field(feet), get_const_field(advection_field_test_vec), dt);
476  std::string const name = output_folder + "/feet_computed.txt";
477  save_feet(mapping, grid, get_field(feet), name);
478  }
479 
480  // Save the values of the exact function at the initial and final states:
481  if (if_save_curves) {
482  std::string const name_0 = output_folder + "/after_" + std::to_string(0) + ".txt";
483  std::string const name_1
484  = output_folder + "/after_" + std::to_string(iteration_number) + "_exact.txt";
485 
486  host_t<DFieldMemRTheta> initial_function(grid);
487  host_t<DFieldMemRTheta> end_function(grid);
488  ddc::for_each(grid, [&](const IdxRTheta irp) {
489  initial_function(irp) = function_to_be_advected_test(ddc::coordinate(irp));
490 
491  // Exact final state
492  end_function(irp) = function_to_be_advected_test(feet_coords_rp_end_time(irp));
493  });
494  saving_computed(mapping, get_field(initial_function), name_0);
495  saving_computed(mapping, get_field(end_function), name_1);
496  }
497 
498 
499 
500  // Save the exact characteristic feet for a displacement on dt:
501  if (if_save_feet) {
502  std::string const name = output_folder + "/feet_exact.txt";
503  save_feet(mapping, grid, get_field(feet_coords_rp_dt), name);
504  }
505 
506 
507  std::cout << "--------------------------------------------------------------------------------"
508  << std::endl;
509 }
