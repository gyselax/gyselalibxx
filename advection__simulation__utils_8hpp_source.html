181 template <class AdvectionField, class Mapping>
182 FieldRP<CoordRP> compute_exact_feet_rp(
183  IDomainRP const& rp_dom,
184  Mapping const& mapping,
185  AdvectionField const& advection_field,
186  double const time)
187 {
188  static_assert(!std::is_same_v<Mapping, DiscreteToCartesian<RDimX, RDimY, SplineRPBuilder>>);
189 
190  FieldRP<CoordRP> feet_coords_rp(rp_dom);
191  CoordXY const coord_xy_center = CoordXY(mapping(CoordRP(0, 0)));
192 
193  ddc::for_each(rp_dom, [&](IndexRP const irp) {
194  CoordRP const coord_rp = ddc::coordinate(irp);
195  CoordXY const coord_xy = advection_field.exact_feet(mapping(coord_rp), time);
196 
197  CoordXY const coord_diff = coord_xy - coord_xy_center;
198  if (norm_inf(coord_diff) < 1e-15) {
199  feet_coords_rp(irp) = CoordRP(0, 0);
200  } else {
201  feet_coords_rp(irp) = mapping(coord_xy);
202  }
203  });
204 
205  return feet_coords_rp;

229 template <class Mapping, class Function>
230 double compute_difference_L2_norm(
231  Mapping const& mapping,
232  IDomainRP const& grid,
233  DSpanRP allfdistribu_advected,
234  Function& function_to_be_advected,
235  SpanRP<CoordRP> const& feet_coord)
236 {
237  DFieldRP exact_function(grid);
238  ddc::for_each(grid, [&](IndexRP const irp) {
239  exact_function(irp) = function_to_be_advected(feet_coord(irp));
240  });
241 
242  DFieldRP quadrature_coeffs
243  = compute_coeffs_on_mapping(mapping, trapezoid_quadrature_coefficients(grid));
244  Quadrature<IDimR, IDimP> quadrature(quadrature_coeffs);
245 
246  double const normL2_exact_function = compute_L2_norm(quadrature, exact_function.span_view());
247  double const normL2_computed_function = compute_L2_norm(quadrature, allfdistribu_advected);
248 
249  return (normL2_exact_function - normL2_computed_function) / normL2_exact_function;
250 }

341  Mapping const& mapping,
342  AnalyticalMapping const& analytical_mapping,
343  IDomainRP const& grid,
344  TimeStepper const& time_stepper,
345  AdvectionDomain& advection_domain,
346  Simulation& simulation,
347  PreallocatableSplineInterpolatorRP const& function_interpolator,
348  SplineRPBuilder const& advection_builder,
349  SplineRPEvaluator& advection_evaluator,
350  double const final_time,
351  double const dt,
352  bool if_save_curves,
353  bool if_save_feet,
354  std::string const& output_folder)
355 {
356  SplineFootFinder<TimeStepper, AdvectionDomain> const
357  foot_finder(time_stepper, advection_domain, advection_builder, advection_evaluator);
358 
359  BslAdvectionRP advection_operator(function_interpolator, foot_finder, mapping);
360  auto function_to_be_advected_test = simulation.get_test_function();
361  auto advection_field_test = simulation.get_advection_field();
362 
363 
364 
365  // TO CLOCK THE SIMULATION ------------------------------------------------------------------
366  std::chrono::time_point<std::chrono::system_clock> start_simulation;
367  std::chrono::time_point<std::chrono::system_clock> end_simulation;
368 
369 
370 
371  // PARAMETERS OF THE TEST -------------------------------------------------------------------
372  int const iteration_number = int(final_time / dt);
373  double const end_time = dt * iteration_number;
374 
375 
376  DFieldRP allfdistribu_test(grid);
377  DSpanRP allfdistribu_advected_test;
378 
379  VectorDFieldRP<RDimX, RDimY> advection_field_test_vec(grid);
380 
381 
382  // START TEST -------------------------------------------------------------------------------
383  start_simulation = std::chrono::system_clock::now();
384 
385  // Initialization of the advected function:
386  for_each(grid, [&](IndexRP const irp) {
387  CoordRP coord = coordinate(irp);
388  if (ddc::get<RDimR>(coord) <= 1e-15) {
389  ddc::get<RDimP>(coord) = 0;
390  }
391  allfdistribu_test(irp) = function_to_be_advected_test(coord);
392  });
393 
394 
395 
396  // Definition of advection field:
397  for_each(grid, [&](IndexRP const irp) {
398  // Moving the coordinates in the physical domain:
399  CoordXY const coord_xy = mapping(ddc::coordinate(irp));
400  CoordXY const advection_field = advection_field_test(coord_xy, 0.);
401 
402  // Define the advection field on the physical domain:
403  ddcHelper::get<RDimX>(advection_field_test_vec)(irp) = ddc::get<RDimX>(advection_field);
404  ddcHelper::get<RDimY>(advection_field_test_vec)(irp) = ddc::get<RDimY>(advection_field);
405  });
406 
407 
408  // SIMULATION -------------------------------------------------------------------------------
409  // Advect "iteration_number" times:
410  for (int i(0); i < iteration_number; ++i) {
411  allfdistribu_advected_test
412  = advection_operator(allfdistribu_test, advection_field_test_vec.span_cview(), dt);
413 
414  // Save the advected function for each iteration:
415  if (if_save_curves) {
416  std::string const name = output_folder + "/after_" + std::to_string(i + 1) + ".txt";
417  saving_computed(mapping, allfdistribu_advected_test.span_view(), name);
418  }
419  }
420 
421 
422 
423  // TREATMENT OF DATA ------------------------------------------------------------------------
424  // Compute the exact characteristic feet:
425  FieldRP<CoordRP> feet_coords_rp_end_time(grid);
426  FieldRP<CoordRP> feet_coords_rp_dt(grid);
427  feet_coords_rp_end_time
428  = compute_exact_feet_rp(grid, analytical_mapping, advection_field_test, end_time);
429  feet_coords_rp_dt = compute_exact_feet_rp(grid, analytical_mapping, advection_field_test, dt);
430 
431 
432  // Compute the maximal absolute error on the space at the end of the simulation:
433  double max_err = 0.;
434  for_each(grid, [&](IndexRP const irp) {
435  double const err
436  = fabs(allfdistribu_advected_test(irp)
437  - function_to_be_advected_test(feet_coords_rp_end_time(irp)));
438  max_err = max_err > err ? max_err : err;
439  });
440 
441 
442  std::cout << " - [" << iteration_number << " iterations] Max absolute error : " << max_err;
443  std::cout << std::endl;
444 
445 
446  // Print the density difference [density conservation expected]:
447  std::cout << " ... "
448  << "Relative L2 norm error: "
449  << compute_difference_L2_norm(
450  mapping,
451  grid,
452  allfdistribu_advected_test,
453  function_to_be_advected_test,
454  feet_coords_rp_end_time.span_view())
455  << std::endl;
456 
457 
458 
459  end_simulation = std::chrono::system_clock::now();
460 
461  // Control simulation time
462  display_time(start_simulation, end_simulation, " ... Complete simulation time: ");
463 
464 
465  // SAVE DATA --------------------------------------------------------------------------------
466  // Save the computed characteristic feet:
467  if (if_save_feet) {
468  FieldRP<CoordRP> feet(grid);
469  for_each(grid, [&](const IndexRP irp) { feet(irp) = ddc::coordinate(irp); });
470  foot_finder(feet.span_view(), advection_field_test_vec, dt);
471  std::string const name = output_folder + "/feet_computed.txt";
472  save_feet(mapping, grid, feet.span_view(), name);
473  }
474 
475  // Save the values of the exact function at the initial and final states:
476  if (if_save_curves) {
477  std::string const name_0 = output_folder + "/after_" + std::to_string(0) + ".txt";
478  std::string const name_1
479  = output_folder + "/after_" + std::to_string(iteration_number) + "_exact.txt";
480 
481  DFieldRP initial_function(grid);
482  DFieldRP end_function(grid);
483  for_each(grid, [&](const IndexRP irp) {
484  initial_function(irp) = function_to_be_advected_test(ddc::coordinate(irp));
485 
486  // Exact final state
487  end_function(irp) = function_to_be_advected_test(feet_coords_rp_end_time(irp));
488  });
489  saving_computed(mapping, initial_function.span_view(), name_0);
490  saving_computed(mapping, end_function.span_view(), name_1);
491  }
492 
493 
494 
495  // Save the exact characteristic feet for a displacement on dt:
496  if (if_save_feet) {
497  std::string const name = output_folder + "/feet_exact.txt";
498  save_feet(mapping, grid, feet_coords_rp_dt.span_view(), name);
499  }
500 
501 
502  std::cout << "--------------------------------------------------------------------------------"
503  << std::endl;
504 }
