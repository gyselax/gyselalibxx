192  // Initialise the characteristics on the mesh points .....................................
193  /*
194  For the time integration solver, the function we advect (here the characteristics)
195  need to be defined on the same domain as the advection field. We then work on space
196  slices of the characteristic feet.
197  */
198  FeetChunk slice_feet_alloc(advection_dom);
199  FeetSpan slice_feet = slice_feet_alloc.span_view();
200  ddc::parallel_for_each(
201  Kokkos::DefaultExecutionSpace(),
202  advection_dom,
203  KOKKOS_LAMBDA(AdvectionIndex const idx) {
204  slice_feet(idx) = ddc::coordinate(IndexInterest(idx));

206 
207 
208  // Compute the characteristic feet .......................................................
209  /*
210  We use a time stepper method to solve the charateristic equation.
211  A TimeStepper needs a function to compute the updated advection field and a function to
212  compute the updated feet.
213  * update_adv_field: evaluate the advection field spline at the updated feet.
214  */
215  // The function describing how the derivative of the evolve function is calculated.
216  std::function<void(AdvecFieldSpan, FeetView)> update_adv_field
217  = [&](AdvecFieldSpan updated_advection_field, FeetView slice_feet) {
218  m_adv_field_evaluator(
219  updated_advection_field,
220  slice_feet,
221  advection_field_coefs.span_cview());
222  };
223 
224 
225  // Solve the characteristic equation with a time integration method
226  m_time_stepper
227  .update(Kokkos::DefaultExecutionSpace(),
228  slice_feet.span_view(),
229  -dt,
230  update_adv_field);
231 
232 
233  // Interpolate the function ..............................................................
234  /*
235  To interpolate the function we want to advect, we build for the feet a Chunk defined
236  on the domain where the function is defined.
237  */

239  ddc::ChunkSpan feet = feet_alloc.span_view();
240  ddc::parallel_for_each(
241  Kokkos::DefaultExecutionSpace(),
242  function_dom,
243  KOKKOS_LAMBDA(FunctionIndex const idx) {
244  AdvectionIndex slice_foot_index(idx);
245  feet(idx) = slice_feet(slice_foot_index);
246  });
247 
248 
249  // Interpolate the function at the characteristic feet
250  function_interpolator(allfdistribu, feet);
