204  // Initialise the characteristics on the mesh points .....................................
205  /*
206  For the time integration solver, the function we advect (here the characteristics)

221  /*

223  A TimeStepper needs a function to compute the updated advection field and a function to

229  = [&](AdvecField updated_advection_field, FeetConstField slice_feet) {
230  m_adv_field_evaluator(
231  updated_advection_field,
232  slice_feet,
233  get_const_field(advection_field_coefs));
234  };
235 
236 
237  // Solve the characteristic equation with a time integration method
238  m_time_stepper

241  -dt,
242  update_adv_field);
243 
244

258  });
