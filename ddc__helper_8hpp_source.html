87template <class IDim>
88constexpr std::enable_if_t<
89 IDim::continuous_dimension_type::PERIODIC,
90 typename IDim::continuous_element_type>
91restrict_to_idx_range(
92 typename IDim::continuous_element_type coord,
93 ddc::DiscreteDomain<IDim> const& idx_range)
94{
95 using Coord = typename IDim::continuous_element_type;
96 double const x_min = ddc::rmin(idx_range);
97 double const length = total_interval_length(idx_range);
98 double const x_max = x_min + length;
99
100 assert(length > 0);
101 coord -= x_min;
102 if (fabs(coord) > 10 * length) {
103 double periodic_factor = 2 * M_PI / length;
104 double coord_2pi = double(coord) * periodic_factor;
105 coord_2pi = std::copysign(std::acos(std::cos(coord_2pi)), std::sin(coord_2pi));
106 coord = coord_2pi < 0 ? Coord((coord_2pi + 2 * M_PI) / periodic_factor)
107 : Coord((coord_2pi) / periodic_factor);
108 }
109 coord += x_min;
110 while (coord < x_min)
111 coord += length;
112 while (coord >= x_max)
113 coord -= length;
114 return coord;
115}

130template <class BSpline>
131KOKKOS_INLINE_FUNCTION void restrict_to_bspline_domain(
132 Coord<typename BSpline::continuous_dimension_type>& coord)
133{
134 if constexpr (BSpline::is_periodic()) {
135 if (coord < ddc::discrete_space<BSpline>().rmin()
136 || coord > ddc::discrete_space<BSpline>().rmax()) {
137 coord -= Kokkos::floor(

139 / ddc::discrete_space<BSpline>().length())
140 * ddc::discrete_space<BSpline>().length();

172template <
173 class TargetDomain,
174 class ElementType,
175 class Domain,
176 class ChunkLayoutType,
177 class ExecSpace,
178 class MemSpace>
179auto create_transpose_mirror_view_and_copy(
180 ExecSpace const& execution_space,
181 ddc::ChunkSpan<ElementType, Domain, ChunkLayoutType, MemSpace> src)
182{
183 static_assert(
184 ddc::type_seq_same_v<ddc::to_type_seq_t<Domain>, ddc::to_type_seq_t<TargetDomain>>);
185 if constexpr (std::is_same_v<TargetDomain, Domain>) {
186 return src.span_view();
187 } else {
188 TargetDomain transposed_domain(src.domain());
189 using ElemType = std::remove_const_t<ElementType>;
190 ddc::Chunk<ElemType, TargetDomain, ddc::KokkosAllocator<ElemType, MemSpace>> chunk(
191 transposed_domain);
192 transpose_layout(execution_space, chunk.span_view(), src.span_cview());
193 return chunk;
194 }
195}
