88template <class IDim>
89constexpr std::enable_if_t<
90 IDim::continuous_dimension_type::PERIODIC,
91 typename IDim::continuous_element_type>
92restrict_to_idx_range(typename IDim::continuous_element_type coord, IdxRange<IDim> const& idx_range)
93{
94 using Coord = typename IDim::continuous_element_type;
95 double const x_min = ddc::rmin(idx_range);
96 double const length = total_interval_length(idx_range);
97 double const x_max = x_min + length;
98
99 assert(length > 0);
100 coord -= x_min;
101 if (fabs(coord) > 10 * length) {
102 double periodic_factor = 2 * M_PI / length;
103 double coord_2pi = double(coord) * periodic_factor;
104 coord_2pi = std::copysign(std::acos(std::cos(coord_2pi)), std::sin(coord_2pi));
105 coord = coord_2pi < 0 ? Coord((coord_2pi + 2 * M_PI) / periodic_factor)
106 : Coord((coord_2pi) / periodic_factor);
107 }
108 coord += x_min;
109 while (coord < x_min)
110 coord += length;
111 while (coord >= x_max)
112 coord -= length;
113 return coord;
114}

129template <class BSpline>
130KOKKOS_INLINE_FUNCTION void restrict_to_bspline_domain(
131 Coord<typename BSpline::continuous_dimension_type>& coord)
132{
133 if constexpr (BSpline::is_periodic()) {
134 if (coord < ddc::discrete_space<BSpline>().rmin()
135 || coord > ddc::discrete_space<BSpline>().rmax()) {
136 coord -= Kokkos::floor(
137 (coord - ddc::discrete_space<BSpline>().rmin())
138 / ddc::discrete_space<BSpline>().length())
139 * ddc::discrete_space<BSpline>().length();
140 }
141 }

191template <
192 class TargetDomain,
193 class ElementType,
194 class Domain,

196 class MemSpace,
197 class FieldLayoutType>
198auto create_transpose_mirror_view_and_copy(
199 ExecSpace const& execution_space,
200 Field<ElementType, Domain, MemSpace, FieldLayoutType> src)
201{
202 static_assert(
203 ddc::type_seq_same_v<ddc::to_type_seq_t<Domain>, ddc::to_type_seq_t<TargetDomain>>);
204 if constexpr (std::is_same_v<TargetDomain, Domain>) {
205 return get_field(src);
206 } else {
207 TargetDomain transposed_domain(src.domain());
208 using ElemType = std::remove_const_t<ElementType>;
209 FieldMem<ElemType, TargetDomain, MemSpace> field_alloc(transposed_domain);
210 transpose_layout(execution_space, get_field(field_alloc), get_const_field(src));
211 return field_alloc;
212 }
213}
