86 template <class IDim>
87 constexpr std::enable_if_t<
88  IDim::continuous_dimension_type::PERIODIC,
89  typename IDim::continuous_element_type>
90 restrict_to_domain(
91  typename IDim::continuous_element_type coord,
92  ddc::DiscreteDomain<IDim> const& dom)
93 {
94  using Coord = typename IDim::continuous_element_type;
95  double const x_min = ddc::rmin(dom);
96  double const length = total_interval_length(dom);
97  double const x_max = x_min + length;
98 
99  assert(length > 0);
100  coord -= x_min;
101  if (fabs(coord) > 10 * length) {
102  double periodic_factor = 2 * M_PI / length;
103  double coord_2pi = double(coord) * periodic_factor;
104  coord_2pi = std::copysign(std::acos(std::cos(coord_2pi)), std::sin(coord_2pi));
105  coord = coord_2pi < 0 ? Coord((coord_2pi + 2 * M_PI) / periodic_factor)
106  : Coord((coord_2pi) / periodic_factor);
107  }
108  coord += x_min;
109  while (coord < x_min)
110  coord += length;
111  while (coord >= x_max)
112  coord -= length;
113  return coord;
114 }
