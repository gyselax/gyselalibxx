90template <class IDim>
91constexpr std::enable_if_t<
92 IDim::continuous_dimension_type::PERIODIC,
93 typename IDim::continuous_element_type>
94restrict_to_idx_range(typename IDim::continuous_element_type coord, IdxRange<IDim> const& idx_range)
95{
96 using Coord = typename IDim::continuous_element_type;
97 double const x_min = ddc::rmin(idx_range);
98 double const length = total_interval_length(idx_range);
99 double const x_max = x_min + length;
100
101 assert(length > 0);
102 coord -= x_min;
103 if (fabs(coord) > 10 * length) {
104 double periodic_factor = 2 * M_PI / length;
105 double coord_2pi = double(coord) * periodic_factor;
106 coord_2pi = std::copysign(std::acos(std::cos(coord_2pi)), std::sin(coord_2pi));
107 coord = coord_2pi < 0 ? Coord((coord_2pi + 2 * M_PI) / periodic_factor)
108 : Coord((coord_2pi) / periodic_factor);
109 }
110 coord += x_min;
111 while (coord < x_min)
112 coord += length;
113 while (coord >= x_max)
114 coord -= length;
115 return coord;
116}

131template <class BSpline>
132KOKKOS_INLINE_FUNCTION void restrict_to_bspline_domain(
133 Coord<typename BSpline::continuous_dimension_type>& coord)
134{
135 if constexpr (BSpline::is_periodic()) {
136 if (coord < ddc::discrete_space<BSpline>().rmin()
137 || coord > ddc::discrete_space<BSpline>().rmax()) {
138 coord -= Kokkos::floor(
139 (coord - ddc::discrete_space<BSpline>().rmin())
140 / ddc::discrete_space<BSpline>().length())
141 * ddc::discrete_space<BSpline>().length();

173template <
174 class TargetDomain,
175 class ElementType,
176 class Domain,
177 class ExecSpace,
178 class MemSpace,
179 class FieldLayoutType>
180auto create_transpose_mirror_view_and_copy(
181 ExecSpace const& execution_space,
182 Field<ElementType, Domain, MemSpace, FieldLayoutType> src)
183{
184 static_assert(
185 ddc::type_seq_same_v<ddc::to_type_seq_t<Domain>, ddc::to_type_seq_t<TargetDomain>>);
186 if constexpr (std::is_same_v<TargetDomain, Domain>) {
187 return get_field(src);
188 } else {
189 TargetDomain transposed_domain(src.domain());
190 using ElemType = std::remove_const_t<ElementType>;
191 FieldMem<ElemType, TargetDomain, MemSpace> field_alloc(transposed_domain);
192 transpose_layout(execution_space, get_field(field_alloc), get_const_field(src));
193 return field_alloc;
194 }
195}
