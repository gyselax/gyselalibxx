321  template <class Domain>
322  RefinedDiscreteToCartesian(
323  ddc::Chunk<double, spline_domain>&& curvilinear_to_x,
324  ddc::Chunk<double, spline_domain>&& curvilinear_to_y,
325  SplineEvaluator2D<BSplineR, BSplineP> const& evaluator,
326  Domain const& domain)
327  {
328  const double rmin = ddc::coordinate(ddc::get<IDimR>(domain).front());
329  const double rmax = ddc::coordinate(ddc::get<IDimR>(domain).back());
330 
331  const double pmin = ddc::coordinate(ddc::get<IDimP>(domain).front());
332  const double pmax = ddc::coordinate(ddc::get<IDimP>(domain).back());
333 
334  // Creation of a refined grid:
335  CoordRRefined const r_min(rmin);
336  CoordRRefined const r_max(rmax);
337  IVectRRefined const r_size(Nr);
338 
339  CoordPRefined const p_min(pmin);
340  CoordPRefined const p_max(pmax);
341  IVectPRefined const p_size(Nt);
342 
343  double const dr((r_max - r_min) / r_size);
344  double const dp((p_max - p_min) / p_size);
345 
346  std::vector<CoordRRefined> r_knots(r_size + 1);
347  std::vector<CoordPRefined> p_knots(p_size + 1);
348 
349  for (int i(0); i < r_size + 1; ++i) {
350  r_knots[i] = CoordRRefined(r_min + i * dr);
351  }
352  for (int i(0); i < p_size + 1; ++i) {
353  p_knots[i] = CoordPRefined(p_min + i * dp);
354  }
355 
356 
357  ddc::init_discrete_space<BSplineRRefined>(r_knots);
358  ddc::init_discrete_space<BSplinePRefined>(p_knots);
359 
360  ddc::init_discrete_space<IDimR>(SplineInterpPointsRRefined::get_sampling());
361  ddc::init_discrete_space<IDimP>(SplineInterpPointsPRefined::get_sampling());
362 
363  IDomainRRefined const interpolation_domain_R(SplineInterpPointsRRefined::get_domain());
364  IDomainPRefined const interpolation_domain_P(SplineInterpPointsPRefined::get_domain());
365  IDomainRPRefined const refined_domain(interpolation_domain_R, interpolation_domain_P);
366 
367 
368  // Operators in the refined domain
369  SplineRPBuilderRefined refined_builder(refined_domain);
370 
371  ConstantExtrapolationBoundaryValue2D<BSplineRRefined, BSplinePRefined, RDimRRefined>
372  boundary_condition_r_left(r_min);
373  ConstantExtrapolationBoundaryValue2D<BSplineRRefined, BSplinePRefined, RDimRRefined>
374  boundary_condition_r_right(r_max);
375 
376  SplineRPEvaluatorRefined refined_evaluator(
377  boundary_condition_r_left,
378  boundary_condition_r_right,
379  g_null_boundary_2d<BSplineRRefined, BSplinePRefined>,
380  g_null_boundary_2d<BSplineRRefined, BSplinePRefined>);
381 
382 
383  // Computation of B-splines coefficients in the refined domain
384  ddc::Chunk<double, IDomainRPRefined> refined_evaluated_x;
385  ddc::Chunk<double, IDomainRPRefined> refined_evaluated_y;
386 
387  ddc::for_each(refined_domain, [&](IndexRPRefined irp) {
388  ddc::Coordinate<RDimR, RDimP> coord(from_refined(ddc::coordinate(irp)));
389  refined_evaluated_x(irp) = evaluator(coord, curvilinear_to_x);
390  refined_evaluated_y(irp) = evaluator(coord, curvilinear_to_y);
391  });
392 
393  ddc::Chunk<double, spline_domain_refined> refined_curvilinear_to_x;
394  ddc::Chunk<double, spline_domain_refined> refined_curvilinear_to_y;
395  refined_builder(refined_curvilinear_to_x, refined_evaluated_x);
396  refined_builder(refined_curvilinear_to_y, refined_evaluated_y);
397 
398 
399  m_mapping(refined_curvilinear_to_x, refined_curvilinear_to_y, refined_evaluator);
400  }

569  template <class Mapping>
570  static RefinedDiscreteToCartesian analytical_to_refined(

572  ddc::DiscreteDomain<IDimR, IDimP> const& domain)
573  {
574  const double rmin = ddc::discrete_space<BSplineR>().rmin();
575  const double rmax = ddc::discrete_space<BSplineR>().rmax();
576 
577  const double pmin = ddc::discrete_space<BSplineP>().rmin();
578  const double pmax = ddc::discrete_space<BSplineP>().rmax();
579 
580  // Create refined grid
581  CoordRRefined const r_min(rmin);
582  CoordRRefined const r_max(rmax);
583  IVectRRefined const r_size(Nr);
584 
585  CoordPRefined const p_min(pmin);
586  CoordPRefined const p_max(pmax);
587  IVectPRefined const p_size(Nt);
588 
589  double const dr((r_max - r_min) / r_size);
590  double const dp((p_max - p_min) / p_size);
591 
592  std::vector<CoordRRefined> r_knots(r_size + 1);
593  std::vector<CoordPRefined> p_knots(p_size + 1);
594 
595  for (int i(0); i < r_size + 1; ++i) {
596  r_knots[i] = CoordRRefined(r_min + i * dr);
597  }
598  r_knots[0] = CoordRRefined(r_min);
599  r_knots[r_size] = CoordRRefined(r_max);
600  for (int i(0); i < p_size + 1; ++i) {
601  p_knots[i] = CoordPRefined(p_min + i * dp);
602  }
603  p_knots[0] = CoordPRefined(p_min);
604  p_knots[p_size] = CoordPRefined(p_max);
605 
606 
607  ddc::init_discrete_space<BSplineRRefined>(r_knots);
608  ddc::init_discrete_space<BSplinePRefined>(p_knots);
609 
610  ddc::init_discrete_space<IDimRRefined>(SplineInterpPointsRRefined::get_sampling());
611  ddc::init_discrete_space<IDimPRefined>(SplineInterpPointsPRefined::get_sampling());
612 
613  IDomainRRefined interpolation_domain_R(SplineInterpPointsRRefined::get_domain());
614  IDomainPRefined interpolation_domain_P(SplineInterpPointsPRefined::get_domain());
615  IDomainRPRefined refined_domain(interpolation_domain_R, interpolation_domain_P);
616 
617 
618  // Operators on the refined grid
619  SplineRPBuilderRefined refined_builder(refined_domain);
620 
621  ConstantExtrapolationBoundaryValue2D<BSplineRRefined, BSplinePRefined, RDimRRefined>
622  boundary_condition_r_left(r_min);
623  ConstantExtrapolationBoundaryValue2D<BSplineRRefined, BSplinePRefined, RDimRRefined>
624  boundary_condition_r_right(r_max);
625 
626  SplineRPEvaluatorRefined refined_evaluator(
627  boundary_condition_r_left,
628  boundary_condition_r_right,
629  g_null_boundary_2d<BSplineRRefined, BSplinePRefined>,
630  g_null_boundary_2d<BSplineRRefined, BSplinePRefined>);
631 
632 
633  // Compute the B-splines coefficients of the analytical mapping
634  ddc::Chunk<double, spline_domain_refined> curvilinear_to_x_spline(
635  refined_builder.spline_domain());
636  ddc::Chunk<double, spline_domain_refined> curvilinear_to_y_spline(
637  refined_builder.spline_domain());
638  ddc::Chunk<double, ddc::DiscreteDomain<IDimRRefined, IDimPRefined>> curvilinear_to_x_vals(
639  refined_builder.interpolation_domain());
640  ddc::Chunk<double, ddc::DiscreteDomain<IDimRRefined, IDimPRefined>> curvilinear_to_y_vals(
641  refined_builder.interpolation_domain());
642  ddc::for_each(
643  refined_builder.interpolation_domain(),
644  [&](typename ddc::DiscreteDomain<IDimRRefined, IDimPRefined>::
645  discrete_element_type const& el) {
646  ddc::Coordinate<RDimR, RDimP> polar_coord(from_refined(ddc::coordinate(el)));
647  ddc::Coordinate<RDimXRefined, RDimYRefined> cart_coord
648  = to_refined(analytical_mapping(polar_coord));
649 
650  curvilinear_to_x_vals(el) = ddc::select<RDimXRefined>(cart_coord);
651  curvilinear_to_y_vals(el) = ddc::select<RDimYRefined>(cart_coord);
652  });
653  refined_builder(curvilinear_to_x_spline, curvilinear_to_x_vals);
654  refined_builder(curvilinear_to_y_spline, curvilinear_to_y_vals);
655 
656  return RefinedDiscreteToCartesian(
657  std::move(curvilinear_to_x_spline),
658  std::move(curvilinear_to_y_spline),
659  refined_evaluator);
660  }
