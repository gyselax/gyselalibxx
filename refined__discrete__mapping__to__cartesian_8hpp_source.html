604  template <class Mapping>
605  static RefinedDiscreteToCartesian analytical_to_refined(
606  Mapping const& analytical_mapping,
607  ddc::DiscreteDomain<GridR, GridTheta> const& domain)
608  {
609  const double rmin = ddc::discrete_space<BSplineR>().rmin();
610  const double rmax = ddc::discrete_space<BSplineR>().rmax();
611 
612  const double pmin = ddc::discrete_space<BSplineTheta>().rmin();
613  const double pmax = ddc::discrete_space<BSplineTheta>().rmax();
614 
615  // Create refined grid
616  CoordRRefined const r_min(rmin);
617  CoordRRefined const r_max(rmax);
618  IdxStepRRefined const r_size(Nr);
619 
620  CoordThetaRefined const theta_min(pmin);
621  CoordThetaRefined const theta_max(pmax);
622  IdxStepThetaRefined const theta_size(Nt);
623 
624  if constexpr (BSplineRRefined::is_uniform()) {
625  ddc::init_discrete_space<BSplineRRefined>(r_min, r_max, r_size);
626  ddc::init_discrete_space<BSplineThetaRefined>(theta_min, theta_max, theta_size);
627  } else {
628  double const dr((r_max - r_min) / r_size);
629  double const dp((theta_max - theta_min) / theta_size);
630 
631  std::vector<CoordRRefined> r_knots(r_size + 1);
632  std::vector<CoordThetaRefined> theta_knots(theta_size + 1);
633 
634  for (int i(0); i < r_size + 1; ++i) {
635  r_knots[i] = CoordRRefined(r_min + i * dr);
636  }
637  for (int i(0); i < theta_size + 1; ++i) {
638  theta_knots[i] = CoordThetaRefined(theta_min + i * dp);
639  }
640 
641  ddc::init_discrete_space<BSplineRRefined>(r_knots);
642  ddc::init_discrete_space<BSplineThetaRefined>(theta_knots);
643  }
644 
645  ddc::init_discrete_space<GridRRefined>(
646  SplineInterpThetaointsRRefined::template get_sampling<GridRRefined>());
647  ddc::init_discrete_space<GridThetaRefined>(
648  SplineInterpThetaointsThetaRefined::template get_sampling<GridThetaRefined>());
649 
650  IdxRangeRRefined const interpolation_domain_R(
651  SplineInterpThetaointsRRefined::template get_domain<GridRRefined>());
652  IdxRangeThetaRefined const interpolation_domain_P(
653  SplineInterpThetaointsThetaRefined::template get_domain<GridThetaRefined>());
654  IdxRangeRThetaRefined const refined_domain(interpolation_domain_R, interpolation_domain_P);
655 
656  // Operators on the refined grid
657  SplineRThetaBuilderRefined const refined_builder(refined_domain);
658 
659  BSIdxRangeRThetaRefined const spline_domain = refined_builder.spline_domain();
660 
661  // Compute the B-splines coefficients of the analytical mapping
662  ddc::Chunk<double, BSIdxRangeRThetaRefined> curvilinear_to_x_spline(spline_domain);
663  ddc::Chunk<double, BSIdxRangeRThetaRefined> curvilinear_to_y_spline(spline_domain);
664  ddc::Chunk<double, IdxRangeRThetaRefined> curvilinear_to_x_vals(refined_domain);
665  ddc::Chunk<double, IdxRangeRThetaRefined> curvilinear_to_y_vals(refined_domain);
666  ddc::for_each(
667  refined_builder.interpolation_domain(),
668  [&](typename ddc::DiscreteDomain<GridRRefined, GridThetaRefined>::
669  discrete_element_type const& el) {
670  ddc::Coordinate<R, Theta> polar_coord(from_refined(ddc::coordinate(el)));
671  ddc::Coordinate<XRefined, YRefined> cart_coord
672  = to_refined(analytical_mapping(polar_coord));
673 
674  curvilinear_to_x_vals(el) = ddc::select<XRefined>(cart_coord);
675  curvilinear_to_y_vals(el) = ddc::select<YRefined>(cart_coord);
676  });
677  refined_builder(curvilinear_to_x_spline.span_view(), curvilinear_to_x_vals.span_cview());
678  refined_builder(curvilinear_to_y_spline.span_view(), curvilinear_to_y_vals.span_cview());
679 
680  return RefinedDiscreteToCartesian(
681  refined_domain,
682  std::move(curvilinear_to_x_spline),
683  std::move(curvilinear_to_y_spline),
684  r_min,
685  r_max);
686  }
