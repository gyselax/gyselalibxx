601  template <class Mapping>
602  static RefinedDiscreteToCartesian analytical_to_refined(
603  Mapping const& analytical_mapping,
604  ddc::DiscreteDomain<IDimR, IDimP> const& domain)
605  {
606  const double rmin = ddc::discrete_space<BSplineR>().rmin();
607  const double rmax = ddc::discrete_space<BSplineR>().rmax();
608 
609  const double pmin = ddc::discrete_space<BSplineP>().rmin();
610  const double pmax = ddc::discrete_space<BSplineP>().rmax();
611 
612  // Create refined grid
613  CoordRRefined const r_min(rmin);
614  CoordRRefined const r_max(rmax);
615  IVectRRefined const r_size(Nr);
616 
617  CoordPRefined const p_min(pmin);
618  CoordPRefined const p_max(pmax);
619  IVectPRefined const p_size(Nt);
620 
621  if constexpr (BSplineRRefined::is_uniform()) {
622  ddc::init_discrete_space<BSplineRRefined>(r_min, r_max, r_size);
623  ddc::init_discrete_space<BSplinePRefined>(p_min, p_max, p_size);
624  } else {
625  double const dr((r_max - r_min) / r_size);
626  double const dp((p_max - p_min) / p_size);
627 
628  std::vector<CoordRRefined> r_knots(r_size + 1);
629  std::vector<CoordPRefined> p_knots(p_size + 1);
630 
631  for (int i(0); i < r_size + 1; ++i) {
632  r_knots[i] = CoordRRefined(r_min + i * dr);
633  }
634  for (int i(0); i < p_size + 1; ++i) {
635  p_knots[i] = CoordPRefined(p_min + i * dp);
636  }
637 
638  ddc::init_discrete_space<BSplineRRefined>(r_knots);
639  ddc::init_discrete_space<BSplinePRefined>(p_knots);
640  }
641 
642  ddc::init_discrete_space<IDimRRefined>(
643  SplineInterpPointsRRefined::template get_sampling<IDimRRefined>());
644  ddc::init_discrete_space<IDimPRefined>(
645  SplineInterpPointsPRefined::template get_sampling<IDimPRefined>());
646 
647  IDomainRRefined const interpolation_domain_R(
648  SplineInterpPointsRRefined::template get_domain<IDimRRefined>());
649  IDomainPRefined const interpolation_domain_P(
650  SplineInterpPointsPRefined::template get_domain<IDimPRefined>());
651  IDomainRPRefined const refined_domain(interpolation_domain_R, interpolation_domain_P);
652 
653  // Operators on the refined grid
654  SplineRPBuilderRefined const refined_builder(refined_domain);
655 
656  BSDomainRPRefined const spline_domain = refined_builder.spline_domain();
657 
658  // Compute the B-splines coefficients of the analytical mapping
659  ddc::Chunk<double, BSDomainRPRefined> curvilinear_to_x_spline(spline_domain);
660  ddc::Chunk<double, BSDomainRPRefined> curvilinear_to_y_spline(spline_domain);
661  ddc::Chunk<double, IDomainRPRefined> curvilinear_to_x_vals(refined_domain);
662  ddc::Chunk<double, IDomainRPRefined> curvilinear_to_y_vals(refined_domain);
663  ddc::for_each(
664  refined_builder.interpolation_domain(),
665  [&](typename ddc::DiscreteDomain<IDimRRefined, IDimPRefined>::
666  discrete_element_type const& el) {
667  ddc::Coordinate<RDimR, RDimP> polar_coord(from_refined(ddc::coordinate(el)));
668  ddc::Coordinate<RDimXRefined, RDimYRefined> cart_coord
669  = to_refined(analytical_mapping(polar_coord));
670 
671  curvilinear_to_x_vals(el) = ddc::select<RDimXRefined>(cart_coord);
672  curvilinear_to_y_vals(el) = ddc::select<RDimYRefined>(cart_coord);
673  });
674  refined_builder(curvilinear_to_x_spline.span_view(), curvilinear_to_x_vals.span_cview());
675  refined_builder(curvilinear_to_y_spline.span_view(), curvilinear_to_y_vals.span_cview());
676 
677  return RefinedDiscreteToCartesian(
678  refined_domain,
679  std::move(curvilinear_to_x_spline),
680  std::move(curvilinear_to_y_spline),
681  r_min,
682  r_max);
683  }
