213  template <class CurrentGrid, class TargetGrid>
214  bool const search_for_match(Idx<TargetGrid>& target_idx, Idx<CurrentGrid> const& current_idx)
215  const
216  {
217  static_assert(
218  std::is_same_v<CurrentGrid, EdgeGrid1> || std::is_same_v<CurrentGrid, EdgeGrid2>,
219  "Current index must be associated with one of the edges.");
220  static_assert(
221  std::is_same_v<TargetGrid, EdgeGrid1> || std::is_same_v<TargetGrid, EdgeGrid2>,
222  "Target index must be associated with one of the edges.");
223  static_assert(
224  !std::is_same_v<TargetGrid, CurrentGrid>,
225  "The types of the indices should be different");
226 
227  // Index range of CurrentIdx.
228  using CurrentIdxRange = IdxRange<CurrentGrid>;
229  // Index range of other dimension.
230  using TargetIdxRange = IdxRange<TargetGrid>;
231  // Coordinate on the discontinuous dimension of the current grid.
232  using CurrentCoord = Coord<typename CurrentGrid::continuous_dimension_type>;
233  // Index on the target grid
234  using TargetIdx = typename TargetIdxRange::discrete_element_type;
235  // Index step on the target grid
236  using TargetIdxStep = typename TargetIdxRange::mlength_type;
237 
238 
239  bool is_equivalent_idx = false;
240 
241  // Get the 1D index range corresponding to the current and target domains.
242  IdxRange<EdgeGrid1, EdgeGrid2> const
243  combined_idx_range(m_idx_range_patch_1, m_idx_range_patch_2);
244  CurrentIdxRange const current_idx_range(combined_idx_range);
245  TargetIdxRange const target_idx_range(combined_idx_range);
246 
247  int const n_cells_current = current_idx_range.size() - 1;
248  int const n_cells_target = target_idx_range.size() - 1;
249 
250  if constexpr ( // Uniform case
251  ddc::is_uniform_point_sampling_v<
252  EdgeGrid1> && ddc::is_uniform_point_sampling_v<EdgeGrid2>) {
253  // Get greatest common divisor
254  int const gcd_cells = std::gcd(n_cells_current, n_cells_target);
255 
256  int const current_idx_value = (current_idx - current_idx_range.front()).value();
257  /*
258  There is an equivalent index if the current index is a multiple of
259  (number of current cells / gcd). If gcd == 1 then this condition is
260  only true if the current index represents an extremity
261  (current_idx_value == 0 or current_idx_value == n_cells_current).
262  */
263  is_equivalent_idx = (current_idx_value % (n_cells_current / gcd_cells) == 0);
264 
265  // If there is an equivalent index, update target_idx.
266  if (is_equivalent_idx) {
267  double const rescaling_factor = double(n_cells_current) / double(n_cells_target);
268  TargetIdxStep target_idx_step_rescaled(int(current_idx_value / rescaling_factor));
269  target_idx = TargetIdx(target_idx_range.front());
270  target_idx += target_idx_step_rescaled;
271 
272  if constexpr (!Interface::orientations_agree) {
273  TargetIdxStep target_idx_step = TargetIdx(target_idx_range.back()) - target_idx;
274  target_idx = TargetIdx(target_idx_range.front());
275  target_idx += target_idx_step;
276  }
277  }
278  } else { // Non uniform case
279  // Dichotomy method comparing the coordinates of indexes of the target edge.
280  target_idx = get_target_idx(target_idx_range, current_idx);
281  CurrentCoord const current_coord(ddc::coordinate(current_idx));
282  CurrentCoord const target_equivalent_coord((*this)(ddc::coordinate(target_idx)));
283  is_equivalent_idx = (abs(current_coord - target_equivalent_coord) < 1e-14);
284  }
285 
286  return is_equivalent_idx;
287  }

300  template <class TargetGrid, class CurrentGrid>
301  Idx<TargetGrid> const get_target_idx(
302  IdxRange<TargetGrid> const& target_idx_range,
303  Idx<CurrentGrid> const& current_idx) const
304  {
305  // Coordinate on the discontinuous dimension of the current grid.
306  using CurrentCoord = Coord<typename CurrentGrid::continuous_dimension_type>;
307  // Index on the target grid
308  using TargetIdx = Idx<TargetGrid>;
309  // Index step on the target grid
310  using TargetIdxStep = IdxStep<TargetGrid>;
311 
312  // Dichotomy method
313  CurrentCoord const current_coord(ddc::coordinate(current_idx));
314 
315  TargetIdx target_idx_min(target_idx_range.front());
316  TargetIdx target_idx_max(target_idx_range.back());
317  TargetIdx target_idx_mid = get_mid(target_idx_min, target_idx_max);
318 
319  TargetIdxStep target_idx_step_diff = target_idx_max - target_idx_min;
320  while (target_idx_step_diff != TargetIdxStep(1)) {
321  CurrentCoord target_equivalent_coord_mid((*this)(ddc::coordinate(target_idx_mid)));
322 
323  if ((current_coord > target_equivalent_coord_mid && Interface::orientations_agree)
324  || (current_coord <= target_equivalent_coord_mid
325  && !Interface::orientations_agree)) {
326  target_idx_min = target_idx_mid;
327  } else {
328  target_idx_max = target_idx_mid;
329  }
330 
331  target_idx_mid = get_mid(target_idx_min, target_idx_max);
332  target_idx_step_diff = target_idx_max - target_idx_min;
333  }
334 
335  CurrentCoord target_coord_min((*this)(ddc::coordinate(target_idx_min)));
336  CurrentCoord target_coord_max((*this)(ddc::coordinate(target_idx_max)));
337  if (abs(current_coord - target_coord_min) < abs(current_coord - target_coord_max)) {
338  return target_idx_min;
339  } else {
340  return target_idx_max;
341  }
342  }
