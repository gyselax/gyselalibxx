86 template <class CurrentDim>
87 Coord<std::conditional_t<std::is_same_v<CurrentDim, EdgeDim1>, EdgeDim2, EdgeDim1>> operator()(
88 Coord<CurrentDim> const& current_coord) const
89 {
90 // The other continuous dimension
91 using TargetDim
92 = std::conditional_t<std::is_same_v<CurrentDim, EdgeDim1>, EdgeDim2, EdgeDim1>;
93
94 // The index range of CurrentDim
95 using IdxRangeCurrent = std::
96 conditional_t<std::is_same_v<CurrentDim, EdgeDim1>, IdxRangeEdge1, IdxRangeEdge2>;
97 // The index range of other dimension
98 using IdxRangeTarget = std::
99 conditional_t<std::is_same_v<CurrentDim, EdgeDim1>, IdxRangeEdge2, IdxRangeEdge1>;
100
101 // Get min and length on each 1D index ranges:
102 IdxRange<EdgeGrid1, EdgeGrid2> const
103 combined_idx_range(m_idx_range_patch_1, m_idx_range_patch_2);
104 IdxRangeCurrent const current_idx_range(combined_idx_range);
105 IdxRangeTarget const target_idx_range(combined_idx_range);
106
107 Coord<CurrentDim> const current_min = ddc::coordinate(current_idx_range.front());
108 double const current_length = ddcHelper::total_interval_length(current_idx_range);
109
110 Coord<TargetDim> const target_min = ddc::coordinate(target_idx_range.front());
111 double const target_length = ddcHelper::total_interval_length(target_idx_range);
112
113 double rescale_x = (current_coord - current_min) / current_length * target_length;
114
115 bool constexpr orientations_agree = Interface::orientations_agree;
116 if constexpr (!orientations_agree) {
117 rescale_x = target_length - rescale_x;
118 }
119 return target_min + rescale_x;
120 }

211 template <class CurrentGrid, class TargetGrid>
212 bool search_for_match(Idx<TargetGrid>& target_idx, Idx<CurrentGrid> const& current_idx) const
213 {
214 static_assert(
215 std::is_same_v<CurrentGrid, EdgeGrid1> || std::is_same_v<CurrentGrid, EdgeGrid2>,
216 "Current index must be associated with one of the edges.");
217 static_assert(
218 std::is_same_v<TargetGrid, EdgeGrid1> || std::is_same_v<TargetGrid, EdgeGrid2>,
219 "Target index must be associated with one of the edges.");
220 static_assert(
221 !std::is_same_v<TargetGrid, CurrentGrid>,
222 "The types of the indices should be different");
223
224 // Index range of CurrentIdx.
225 using IdxRangeCurrent = IdxRange<CurrentGrid>;
226 // Index range of other dimension.
227 using IdxRangeTarget = IdxRange<TargetGrid>;
228 // Coordinate on the discontinuous dimension of the current grid.
229 using CurrentCoord = Coord<typename CurrentGrid::continuous_dimension_type>;
230 // Index on the target grid
231 using IdxTarget = typename IdxRangeTarget::discrete_element_type;
232 // Index step on the target grid
233 using IdxStepTarget = typename IdxRangeTarget::mlength_type;
234
235
236 bool is_equivalent_idx = false;
237
238 // Get the 1D index range corresponding to the current and target domains.
239 IdxRange<EdgeGrid1, EdgeGrid2> const
240 combined_idx_range(m_idx_range_patch_1, m_idx_range_patch_2);
241 IdxRangeCurrent const current_idx_range(combined_idx_range);
242 IdxRangeTarget const target_idx_range(combined_idx_range);
243
244 int const n_cells_current = current_idx_range.size() - 1;
245 int const n_cells_target = target_idx_range.size() - 1;
246
247 if constexpr ( // Uniform case
248 ddc::is_uniform_point_sampling_v<
249 EdgeGrid1> && ddc::is_uniform_point_sampling_v<EdgeGrid2>) {
250 // Get greatest common divisor
251 int const gcd_cells = std::gcd(n_cells_current, n_cells_target);
252
253 int const current_idx_value = (current_idx - current_idx_range.front()).value();
254 /*
255 There is an equivalent index if the current index is a multiple of
256 (number of current cells / gcd). If gcd == 1 then this condition is
257 only true if the current index represents an extremity
258 (current_idx_value == 0 or current_idx_value == n_cells_current).
259 */
260 is_equivalent_idx = (current_idx_value % (n_cells_current / gcd_cells) == 0);
261
262 // If there is an equivalent index, update target_idx.
263 if (is_equivalent_idx) {
264 double const rescaling_factor = double(n_cells_current) / double(n_cells_target);
265 IdxStepTarget target_idx_step_rescaled(int(current_idx_value / rescaling_factor));
266 target_idx = IdxTarget(target_idx_range.front());
267 target_idx += target_idx_step_rescaled;
268
269 if constexpr (!Interface::orientations_agree) {
270 IdxStepTarget target_idx_step = IdxTarget(target_idx_range.back()) - target_idx;
271 target_idx = IdxTarget(target_idx_range.front());
272 target_idx += target_idx_step;
273 }
274 }
275 } else { // Non uniform case
276 // Dichotomy method comparing the coordinates of indexes of the target edge.
277 target_idx = get_target_idx(target_idx_range, current_idx);
278 CurrentCoord const current_coord(ddc::coordinate(current_idx));
279 CurrentCoord const target_equivalent_coord((*this)(ddc::coordinate(target_idx)));
280 is_equivalent_idx = (abs(current_coord - target_equivalent_coord) < 1e-14);
281 }
282
283 return is_equivalent_idx;
284 }

297 template <class TargetGrid, class CurrentGrid>
298 Idx<TargetGrid> const get_target_idx(
299 IdxRange<TargetGrid> const& target_idx_range,
300 Idx<CurrentGrid> const& current_idx) const
301 {
302 // Coordinate on the discontinuous dimension of the current grid.
303 using CurrentCoord = Coord<typename CurrentGrid::continuous_dimension_type>;
304 // Index on the target grid
305 using IdxTarget = Idx<TargetGrid>;
306 // Index step on the target grid
307 using IdxStepTarget = IdxStep<TargetGrid>;
308
309 // Dichotomy method
310 CurrentCoord const current_coord(ddc::coordinate(current_idx));
311
312 IdxTarget target_idx_min(target_idx_range.front());
313 IdxTarget target_idx_max(target_idx_range.back());
314 IdxTarget target_idx_mid = get_mid(target_idx_min, target_idx_max);
315
316 IdxStepTarget target_idx_step_diff = target_idx_max - target_idx_min;
317 while (target_idx_step_diff != IdxStepTarget(1)) {
318 CurrentCoord target_equivalent_coord_mid((*this)(ddc::coordinate(target_idx_mid)));
319
320 if ((current_coord > target_equivalent_coord_mid && Interface::orientations_agree)
321 || (current_coord <= target_equivalent_coord_mid
322 && !Interface::orientations_agree)) {
323 target_idx_min = target_idx_mid;
324 } else {
325 target_idx_max = target_idx_mid;
326 }
327
328 target_idx_mid = get_mid(target_idx_min, target_idx_max);
329 target_idx_step_diff = target_idx_max - target_idx_min;
330 }
331
332 CurrentCoord target_coord_min((*this)(ddc::coordinate(target_idx_min)));
333 CurrentCoord target_coord_max((*this)(ddc::coordinate(target_idx_max)));
334 if (abs(current_coord - target_coord_min) < abs(current_coord - target_coord_max)) {
335 return target_idx_min;
336 } else {
337 return target_idx_max;
338 }
339 }
