230 KOKKOS_FUNCTION InvJacobianMatrixType inv_jacobian_matrix(CoordJacobian const& coord) const
231 {
232 if constexpr (
233 (has_singular_o_point_inv_jacobian_v<Mapping1>)
234 || (has_singular_o_point_inv_jacobian_v<InverseMapping2>)) {
235 using R = ddc::type_seq_element_t<0, ddc::to_type_seq_t<CoordJacobian>>;
236 using Theta = ddc::type_seq_element_t<1, ddc::to_type_seq_t<CoordJacobian>>;
237 double r = ddc::get<R>(coord);
238 if (r < m_epsilon) {
239 InvJacobianOPoint<CombinedMapping<Mapping1, Mapping2>, CoordJacobian> o_point_val(
240 *this);
241 CoordJacobian coord_eps(m_epsilon, ddc::get<Theta>(coord));
242 Tensor J_0 = o_point_val();
243 Tensor J_eps = non_singular_inverse_jacobian_matrix(coord_eps);
244 return (1 - r / m_epsilon) * J_0 + r / m_epsilon * J_eps;
245 } else {
246 return non_singular_inverse_jacobian_matrix(coord);
247 }
248 } else {
249 return non_singular_inverse_jacobian_matrix(coord);
250 }
251 }
