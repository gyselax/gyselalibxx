227  KOKKOS_FUNCTION void inv_jacobian_matrix(CoordJacobian const& coord, Matrix_2x2& matrix) const
228  {
229  if constexpr (
230  (has_singular_o_point_inv_jacobian_v<Mapping1>)
231  || (has_singular_o_point_inv_jacobian_v<InverseMapping2>)) {
232  using R = ddc::type_seq_element_t<0, ddc::to_type_seq_t<CoordJacobian>>;
233  double r = ddc::get<R>(coord);
234  if (r < m_epsilon) {
235  InvJacobianOPoint<CombinedMapping<Mapping1, Mapping2>, CoordJacobian> o_point_val(
236  *this);
237  CoordJacobian coord_eps(m_epsilon, ddc::get<Theta>(coord));
238  Matrix_2x2 J_0 = o_point_val();
239  Matrix_2x2 J_eps;
240  non_singular_inverse_jacobian_matrix(coord_eps, J_eps);
241  matrix[0][0] = (1 - r / m_epsilon) * J_0[0][0] + r / m_epsilon * J_eps[0][0];
242  matrix[0][1] = (1 - r / m_epsilon) * J_0[0][1] + r / m_epsilon * J_eps[0][1];
243  matrix[1][0] = (1 - r / m_epsilon) * J_0[1][0] + r / m_epsilon * J_eps[1][0];
244  matrix[1][1] = (1 - r / m_epsilon) * J_0[1][1] + r / m_epsilon * J_eps[1][1];
245  } else {
246  non_singular_inverse_jacobian_matrix(coord, matrix);
247  }
248  } else {
249  non_singular_inverse_jacobian_matrix(coord, matrix);
250  }
251  }
