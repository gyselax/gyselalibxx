22template <
23 class ExecSpace,
24 class ElementType,
25 class IdxRangeOut,
26 class MemorySpace,
27 class IdxRangeIn,
28 class LayoutStridedPolicyIn,
29 class LayoutStridedPolicyOut>
30Field<ElementType, IdxRangeIn, MemorySpace, LayoutStridedPolicyOut> transpose_layout(
31 ExecSpace const& execution_space,
32 Field<ElementType, IdxRangeIn, MemorySpace, LayoutStridedPolicyOut> transposed_field,
33 ConstField<ElementType, IdxRangeOut, MemorySpace, LayoutStridedPolicyIn> field_to_transpose)
34{
35 static_assert(
36 Kokkos::SpaceAccessibility<ExecSpace, MemorySpace>::accessible,
37 "MemorySpace has to be accessible for ExecutionSpace.");
38 // assert that IdxRange<Dims...> is a transposed discrete domain by
39 // checking that it is a subset of IdxRangeOut and that IdxRangeOut is a subset
40 // of it.
41 static_assert(ddc::type_seq_contains_v<
42 ddc::to_type_seq_t<IdxRangeOut>,
43 ddc::to_type_seq_t<IdxRangeIn>>);
44 static_assert(ddc::type_seq_contains_v<
45 ddc::to_type_seq_t<IdxRangeIn>,
46 ddc::to_type_seq_t<IdxRangeOut>>);
47
48 // Check that both views have the same domain (just reordered)
49 assert(get_idx_range(field_to_transpose) == get_idx_range(transposed_field));
50
51 IdxRangeOut idx_range(field_to_transpose.domain());
52
53 constexpr std::size_t n_dims(ddc::type_seq_size_v<ddc::to_type_seq_t<IdxRangeOut>>);
54
55 if constexpr (n_dims < 7) {
56 using ToTransposeIndex = typename IdxRangeOut::discrete_element_type;
57 ddc::parallel_for_each(
58 execution_space,
59 idx_range,
60 KOKKOS_LAMBDA(ToTransposeIndex idx) {
61 transposed_field(idx) = field_to_transpose(idx);
62 });
63 } else {
64 using IdxRangeParallel = ddc::detail::convert_type_seq_to_discrete_domain_t<
65 type_seq_range_t<ddc::to_type_seq_t<IdxRangeOut>, 0, 6>>;
66 using IdxRangeSerial = ddc::detail::convert_type_seq_to_discrete_domain_t<
67 type_seq_range_t<ddc::to_type_seq_t<IdxRangeOut>, 6, n_dims>>;
68 using IdxParallel = typename IdxRangeParallel::discrete_element_type;
69 using IdxSerial = typename IdxRangeSerial::discrete_element_type;
70 IdxRangeParallel parallel_idx_range(idx_range);
71 IdxRangeSerial serial_idx_range(idx_range);
72 ddc::parallel_for_each(
73 execution_space,
74 parallel_idx_range,
75 KOKKOS_LAMBDA(IdxParallel p_idx) {
76 for (IdxSerial s_idx : serial_idx_range) {
77 transposed_field(p_idx, s_idx) = field_to_transpose(p_idx, s_idx);
78 }
79 });
80 }
81 return transposed_field;
82}

96template <
97 class TargetDomain,
98 class ElementType,
99 class Domain,
100 class ExecSpace,
101 class MemSpace,
102 class FieldLayoutType>
103auto create_transpose_mirror_view_and_copy(
104 ExecSpace const& execution_space,
105 Field<ElementType, Domain, MemSpace, FieldLayoutType> src)
106{
107 static_assert(
108 ddc::type_seq_same_v<ddc::to_type_seq_t<Domain>, ddc::to_type_seq_t<TargetDomain>>);
109 if constexpr (std::is_same_v<TargetDomain, Domain>) {
110 return get_field(src);
111 } else {
112 TargetDomain transposed_domain(src.domain());
113 using ElemType = std::remove_const_t<ElementType>;
114 FieldMem<ElemType, TargetDomain, MemSpace> field_alloc(transposed_domain);
115 transpose_layout(execution_space, get_field(field_alloc), get_const_field(src));
116 return field_alloc;
117 }
118}
