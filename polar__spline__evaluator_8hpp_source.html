312 template <class Mapping>
313 double integrate(
314 host_t<ConstPolarSpline<PolarBSplinesType>> const spline_coef,
315 Mapping const mapping) const
316 {
317 int constexpr nr = ddc::discrete_space<BSplinesR>().ncells() + BSplinesR::degree() - 2;
318 int constexpr ntheta
319 = ddc::discrete_space<BSplinesTheta>().ncells() + BSplinesTheta::degree();
320 std::array<double, PolarBSplinesType::eval_size()> singular_values;
321 DSpan1D singular_vals(singular_values.data(), PolarBSplinesType::n_singular_basis());
322 std::array<double, nr * ntheta> values;
323 DSpan2D vals(values.data(), nr, ntheta);
324
325 ddc::discrete_space<PolarBSplinesType>().integrals(singular_vals, vals);
326
327 double y = 0.;
328 ddc::for_each(
329 get_idx_range(spline_coef.singular_spline_coef),
330 [=](Idx<PolarBSplinesType> const i) {
331 y += spline_coef.singular_spline_coef(i) * singular_vals(i)
332 * mapping.determinant(i);
333 });
334 ddc::for_each(
335 get_idx_range(spline_coef.spline_coef),
336 [=](Idx<BSplinesR, BSplinesTheta> const i) {
337 y += spline_coef.spline_coef(i)
338 * vals(ddc::select<BSplinesR>(i), ddc::select<BSplinesTheta>(i))
339 * mapping.determinant(i);
340 });
341 return y;
342 }

345 double eval(
346 ddc::Coordinate<DimR, DimTheta> coord_eval,
347 host_t<ConstPolarSpline<PolarBSplinesType>> const spline_coef) const
348 {
349 const double coord_eval1 = ddc::get<DimR>(coord_eval);
350 double coord_eval2 = ddc::get<DimTheta>(coord_eval);
351 if (coord_eval1 > ddc::discrete_space<BSplinesR>().rmax()) {
352 return m_outer_bc(coord_eval, spline_coef);
353 }
354 if (coord_eval2 < ddc::discrete_space<BSplinesTheta>().rmin()
355 || coord_eval2 > ddc::discrete_space<BSplinesTheta>().rmax()) {
356 coord_eval2 -= std::floor(
357 (coord_eval2 - ddc::discrete_space<BSplinesTheta>().rmin())
358 / ddc::discrete_space<BSplinesTheta>().length())
359 * ddc::discrete_space<BSplinesTheta>().length();
360 }
361 ddc::Coordinate<DimR, DimTheta> coord_eval_new(coord_eval1, coord_eval2);
362 return eval_no_bc(coord_eval_new, spline_coef, eval_type());
363 }

365 template <class EvalType>
366 double eval_no_bc(
367 ddc::Coordinate<DimR, DimTheta> coord_eval,
368 host_t<ConstPolarSpline<PolarBSplinesType>> const spline_coef,
369 EvalType const) const
370 {
371 static_assert(
372 std::is_same_v<
373 EvalType,
374 eval_type> || std::is_same_v<EvalType, eval_deriv_r_type> || std::is_same_v<EvalType, eval_deriv_theta_type> || std::is_same_v<EvalType, eval_deriv_r_theta_type>);
375
376 std::array<double, PolarBSplinesType::n_singular_basis()> singular_data;
377 DSpan1D singular_vals(singular_data.data(), PolarBSplinesType::n_singular_basis());
378 std::array<double, (BSplinesR::degree() + 1) * (BSplinesTheta::degree() + 1)> data;
379 DSpan2D vals(data.data(), BSplinesR::degree() + 1, BSplinesTheta::degree() + 1);
380
381 Idx<BSplinesR, BSplinesTheta> jmin;
382
383 if constexpr (std::is_same_v<EvalType, eval_type>) {
384 jmin = ddc::discrete_space<PolarBSplinesType>()
385 .eval_basis(singular_vals, vals, coord_eval);
386 } else if constexpr (std::is_same_v<EvalType, eval_deriv_r_type>) {
387 jmin = ddc::discrete_space<PolarBSplinesType>()
388 .eval_deriv_r(singular_vals, vals, coord_eval);
389 } else if constexpr (std::is_same_v<EvalType, eval_deriv_theta_type>) {
390 jmin = ddc::discrete_space<PolarBSplinesType>()
391 .eval_deriv_theta(singular_vals, vals, coord_eval);
392 } else if constexpr (std::is_same_v<EvalType, eval_deriv_r_theta_type>) {
393 jmin = ddc::discrete_space<PolarBSplinesType>()
394 .eval_deriv_r_and_theta(singular_vals, vals, coord_eval);
395 }
396
397 double y = 0.0;
398 for (std::size_t i = 0; i < PolarBSplinesType::n_singular_basis(); ++i) {
399 y += spline_coef.singular_spline_coef(Idx<PolarBSplinesType>(i)) * singular_vals(i);
400 }
401 Idx<BSplinesR> jmin_r = ddc::select<BSplinesR>(jmin);
402 Idx<BSplinesTheta> jmin_theta = ddc::select<BSplinesTheta>(jmin);
403 int nr = BSplinesR::degree() + 1;
404 if (jmin_r < Idx<BSplinesR>(continuity + 1)) {
405 nr = nr - (Idx<BSplinesR>(continuity + 1) - jmin_r);
406 jmin_r = Idx<BSplinesR>(continuity + 1);
407 }
408 for (int i = 0; i < nr; ++i) {
409 for (std::size_t j = 0; j < BSplinesTheta::degree() + 1; ++j) {
410 y += spline_coef.spline_coef(jmin_r + i, jmin_theta + j) * vals(i, j);
411 }
412 }
413 return y;
414 }
