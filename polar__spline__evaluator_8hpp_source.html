307  template <class Mapping>
308  double integrate(
309  PolarSplineView<PolarBSplinesType, MemSpace> const spline_coef,
310  Mapping const mapping) const
311  {
312  int constexpr nr = ddc::discrete_space<BSplinesR>().ncells() + BSplinesR::degree() - 2;
313  int constexpr ntheta
314  = ddc::discrete_space<BSplinesTheta>().ncells() + BSplinesTheta::degree();
315  std::array<double, PolarBSplinesType::eval_size()> singular_values;
316  DSpan1D singular_vals(singular_values.data(), PolarBSplinesType::n_singular_basis());
317  std::array<double, nr * ntheta> values;
318  DSpan2D vals(values.data(), nr, ntheta);
319 
320  ddc::discrete_space<PolarBSplinesType>().integrals(singular_vals, vals);
321 
322  double y = 0.;
323  ddc::for_each(
324  spline_coef.singular_spline_coef.domain(),
325  [=](ddc::DiscreteElement<PolarBSplinesType> const i) {
326  y += spline_coef.singular_spline_coef(i) * singular_vals(i)
327  * mapping.determinant(i);
328  });
329  ddc::for_each(
330  spline_coef.spline_coef.domain(),
331  [=](ddc::DiscreteElement<BSplinesR, BSplinesTheta> const i) {
332  y += spline_coef.spline_coef(i)
333  * vals(ddc::select<BSplinesR>(i), ddc::select<BSplinesTheta>(i))
334  * mapping.determinant(i);
335  });
336  return y;
337  }

340  double eval(
341  ddc::Coordinate<DimR, DimTheta> coord_eval,
342  PolarSplineView<PolarBSplinesType, MemSpace> const spline_coef) const
343  {
344  const double coord_eval1 = ddc::get<DimR>(coord_eval);
345  double coord_eval2 = ddc::get<DimTheta>(coord_eval);
346  if (coord_eval1 > ddc::discrete_space<BSplinesR>().rmax()) {
347  return m_outer_bc(coord_eval, spline_coef);
348  }
349  if (coord_eval2 < ddc::discrete_space<BSplinesTheta>().rmin()
350  || coord_eval2 > ddc::discrete_space<BSplinesTheta>().rmax()) {
351  coord_eval2 -= std::floor(
352  (coord_eval2 - ddc::discrete_space<BSplinesTheta>().rmin())
353  / ddc::discrete_space<BSplinesTheta>().length())
354  * ddc::discrete_space<BSplinesTheta>().length();
355  }
356  return eval_no_bc(coord_eval, spline_coef, eval_type());
357  }

359  template <class EvalType>
360  double eval_no_bc(
361  ddc::Coordinate<DimR, DimTheta> coord_eval,
362  PolarSplineView<PolarBSplinesType, MemSpace> const spline_coef,
363  EvalType const) const
364  {
365  static_assert(
366  std::is_same_v<
367  EvalType,
368  eval_type> || std::is_same_v<EvalType, eval_deriv_r_type> || std::is_same_v<EvalType, eval_deriv_theta_type> || std::is_same_v<EvalType, eval_deriv_r_theta_type>);
369 
370  std::array<double, PolarBSplinesType::n_singular_basis()> singular_data;
371  DSpan1D singular_vals(singular_data.data(), PolarBSplinesType::n_singular_basis());
372  std::array<double, (BSplinesR::degree() + 1) * (BSplinesTheta::degree() + 1)> data;
373  DSpan2D vals(data.data(), BSplinesR::degree() + 1, BSplinesTheta::degree() + 1);
374 
375  ddc::DiscreteElement<BSplinesR, BSplinesTheta> jmin;
376 
377  if constexpr (std::is_same_v<EvalType, eval_type>) {
378  jmin = ddc::discrete_space<PolarBSplinesType>()
379  .eval_basis(singular_vals, vals, coord_eval);
380  } else if constexpr (std::is_same_v<EvalType, eval_deriv_r_type>) {
381  jmin = ddc::discrete_space<PolarBSplinesType>()
382  .eval_deriv_r(singular_vals, vals, coord_eval);
383  } else if constexpr (std::is_same_v<EvalType, eval_deriv_theta_type>) {
384  jmin = ddc::discrete_space<PolarBSplinesType>()
385  .eval_deriv_theta(singular_vals, vals, coord_eval);
386  } else if constexpr (std::is_same_v<EvalType, eval_deriv_r_theta_type>) {
387  jmin = ddc::discrete_space<PolarBSplinesType>()
388  .eval_deriv_r_and_theta(singular_vals, vals, coord_eval);
389  }
390 
391  double y = 0.0;
392  for (std::size_t i = 0; i < PolarBSplinesType::n_singular_basis(); ++i) {
393  y += spline_coef.singular_spline_coef(ddc::DiscreteElement<PolarBSplinesType>(i))
394  * singular_vals(i);
395  }
396  ddc::DiscreteElement<BSplinesR> jmin_r = ddc::select<BSplinesR>(jmin);
397  ddc::DiscreteElement<BSplinesTheta> jmin_theta = ddc::select<BSplinesTheta>(jmin);
398  int nr = BSplinesR::degree() + 1;
399  if (jmin_r.uid() < continuity + 1) {
400  nr = nr - (continuity + 1 - jmin_r.uid());
401  jmin_r = ddc::DiscreteElement<BSplinesR>(continuity + 1);
402  }
403  for (int i = 0; i < nr; ++i) {
404  for (std::size_t j = 0; j < BSplinesTheta::degree() + 1; ++j) {
405  y += spline_coef.spline_coef(jmin_r + i, jmin_theta + j) * vals(i, j);
406  }
407  }
408  return y;
409  }
