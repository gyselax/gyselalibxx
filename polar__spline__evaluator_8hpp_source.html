302 template <class Mapping>
303 double integrate(PolarSplineView<PolarBSplinesType> const spline_coef, Mapping const mapping)
304 const
305 {
306 int constexpr nr = ddc::discrete_space<BSplinesR>().ncells() + BSplinesR::degree() - 2;
307 int constexpr ntheta
308 = ddc::discrete_space<BSplinesTheta>().ncells() + BSplinesTheta::degree();
309 std::array<double, PolarBSplinesType::eval_size()> singular_values;
310 DSpan1D singular_vals(singular_values.data(), PolarBSplinesType::n_singular_basis());
311 std::array<double, nr * ntheta> values;
312 DSpan2D vals(values.data(), nr, ntheta);
313
314 ddc::discrete_space<PolarBSplinesType>().integrals(singular_vals, vals);
315
316 double y = 0.;
317 ddc::for_each(
318 spline_coef.singular_spline_coef.domain(),
319 [=](ddc::DiscreteElement<PolarBSplinesType> const i) {
320 y += spline_coef.singular_spline_coef(i) * singular_vals(i)
321 * mapping.determinant(i);
322 });
323 ddc::for_each(
324 spline_coef.spline_coef.domain(),
325 [=](ddc::DiscreteElement<BSplinesR, BSplinesTheta> const i) {
326 y += spline_coef.spline_coef(i)
327 * vals(ddc::select<BSplinesR>(i), ddc::select<BSplinesTheta>(i))
328 * mapping.determinant(i);
329 });
330 return y;
331 }

334 double eval(
335 ddc::Coordinate<DimR, DimTheta> coord_eval,
336 PolarSplineView<PolarBSplinesType> const spline_coef) const
337 {
338 const double coord_eval1 = ddc::get<DimR>(coord_eval);
339 double coord_eval2 = ddc::get<DimTheta>(coord_eval);
340 if (coord_eval1 > ddc::discrete_space<BSplinesR>().rmax()) {
341 return m_outer_bc(coord_eval, spline_coef);
342 }
343 if (coord_eval2 < ddc::discrete_space<BSplinesTheta>().rmin()
344 || coord_eval2 > ddc::discrete_space<BSplinesTheta>().rmax()) {
345 coord_eval2 -= std::floor(
346 (coord_eval2 - ddc::discrete_space<BSplinesTheta>().rmin())
347 / ddc::discrete_space<BSplinesTheta>().length())
348 * ddc::discrete_space<BSplinesTheta>().length();
349 }
350 return eval_no_bc(coord_eval, spline_coef, eval_type());
351 }

353 template <class EvalType>
354 double eval_no_bc(
355 ddc::Coordinate<DimR, DimTheta> coord_eval,
356 PolarSplineView<PolarBSplinesType> const spline_coef,
357 EvalType const) const
358 {
359 static_assert(
360 std::is_same_v<
361 EvalType,
362 eval_type> || std::is_same_v<EvalType, eval_deriv_r_type> || std::is_same_v<EvalType, eval_deriv_theta_type> || std::is_same_v<EvalType, eval_deriv_r_theta_type>);
363
364 std::array<double, PolarBSplinesType::n_singular_basis()> singular_data;
365 DSpan1D singular_vals(singular_data.data(), PolarBSplinesType::n_singular_basis());
366 std::array<double, (BSplinesR::degree() + 1) * (BSplinesTheta::degree() + 1)> data;
367 DSpan2D vals(data.data(), BSplinesR::degree() + 1, BSplinesTheta::degree() + 1);
368
369 ddc::DiscreteElement<BSplinesR, BSplinesTheta> jmin;
370
371 if constexpr (std::is_same_v<EvalType, eval_type>) {
372 jmin = ddc::discrete_space<PolarBSplinesType>()
373 .eval_basis(singular_vals, vals, coord_eval);
374 } else if constexpr (std::is_same_v<EvalType, eval_deriv_r_type>) {
375 jmin = ddc::discrete_space<PolarBSplinesType>()
376 .eval_deriv_r(singular_vals, vals, coord_eval);
377 } else if constexpr (std::is_same_v<EvalType, eval_deriv_theta_type>) {
378 jmin = ddc::discrete_space<PolarBSplinesType>()
379 .eval_deriv_theta(singular_vals, vals, coord_eval);
380 } else if constexpr (std::is_same_v<EvalType, eval_deriv_r_theta_type>) {
381 jmin = ddc::discrete_space<PolarBSplinesType>()
382 .eval_deriv_r_and_theta(singular_vals, vals, coord_eval);
383 }
384
385 double y = 0.0;
386 for (std::size_t i = 0; i < PolarBSplinesType::n_singular_basis(); ++i) {
387 y += spline_coef.singular_spline_coef(ddc::DiscreteElement<PolarBSplinesType>(i))
388 * singular_vals(i);
389 }
390 ddc::DiscreteElement<BSplinesR> jmin_r = ddc::select<BSplinesR>(jmin);
391 ddc::DiscreteElement<BSplinesTheta> jmin_theta = ddc::select<BSplinesTheta>(jmin);
392 int nr = BSplinesR::degree() + 1;
393 if (jmin_r.uid() < continuity + 1) {
394 nr = nr - (continuity + 1 - jmin_r.uid());
395 jmin_r = ddc::DiscreteElement<BSplinesR>(continuity + 1);
396 }
397 for (int i = 0; i < nr; ++i) {
398 for (std::size_t j = 0; j < BSplinesTheta::degree() + 1; ++j) {
399 y += spline_coef.spline_coef(jmin_r + i, jmin_theta + j) * vals(i, j);
400 }
401 }
402 return y;
403 }
