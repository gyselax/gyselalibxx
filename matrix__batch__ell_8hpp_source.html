239  vals_view(vals_proxy, values_layout);
240 
241  double result = 0;
242  Kokkos::parallel_reduce(
243  "L-infinitty norm",
244  Kokkos::RangePolicy<ExecSpace>(0, tmp_mat_size),
245  KOKKOS_LAMBDA(int i, double& res) {
246  double row_sum = 0.;
247  for (int k = 0; k < non_zeros; k++) {
248  row_sum += Kokkos::abs(vals_view(batch_idx, i, k));
249  }
250  if (row_sum > res) {
251  res = row_sum;
252  }
253  },
254  Kokkos::Max<double>(result));
255  return result;
