22 void compute_Dcoll(
23  ddc::ChunkSpan<double, ddc::DiscreteDomain<IDimSp, IDimX, IDimension>> Dcoll,
24  DViewSpX collfreq,
25  DViewSpX density,
26  DViewSpX temperature)
27 {
28  ddc::for_each(Dcoll.domain(), [&](auto const ispxvx) {
29  double const vT(std::sqrt(2. * temperature(ddc::select<IDimSp, IDimX>(ispxvx))));
30  double const v_norm(std::fabs(ddc::coordinate(ddc::select<IDimension>(ispxvx))) / vT);
31  double const tol = 1.e-15;
32  if (v_norm > tol) {
33  double const coeff(2. / std::sqrt(M_PI));
34  double const AD(
35  3. * std::sqrt(2. * M_PI) / 4. * temperature(ddc::select<IDimSp, IDimX>(ispxvx))
36  * collfreq(ddc::select<IDimSp, IDimX>(ispxvx)));
37  double const inv_v_norm(1. / v_norm);
38  double const phi(std::erf(v_norm));
39  double const phi_prime(coeff * std::exp(-v_norm * v_norm));
40  double const psi((phi - v_norm * phi_prime) * 0.5 * inv_v_norm * inv_v_norm);
41 
42  Dcoll(ispxvx) = AD * (phi - psi) * inv_v_norm;
43 
44  } else {
45  Dcoll(ispxvx) = std::sqrt(2) * temperature(ddc::select<IDimSp, IDimX>(ispxvx))
46  * collfreq(ddc::select<IDimSp, IDimX>(ispxvx));
47  }
48  });
49 }

61  double const tol = 1.e-15;
62  if (v_norm > tol) {
63  double const coeff(2. / std::sqrt(M_PI));
64  double const AD(
65  3. * std::sqrt(2. * M_PI) / 4. * temperature(ddc::select<IDimSp, IDimX>(ispxvx))

67  double const inv_v_norm(1. / v_norm);
68  double const phi(std::erf(v_norm));

77  / std::sqrt(2 * temperature(ddc::select<IDimSp, IDimX>(ispxvx)))

80  } else {

101 template <class IDimension>
102 void compute_Vcoll_Tcoll(
103  DSpanSpX Vcoll,
104  DSpanSpX Tcoll,
105  DViewSpXVx allfdistribu,
106  ddc::ChunkSpan<double const, ddc::DiscreteDomain<IDimSp, IDimX, IDimension>> Dcoll,
107  ddc::ChunkSpan<double const, ddc::DiscreteDomain<IDimSp, IDimX, IDimension>> dvDcoll)
108 {
109  Quadrature<IDimVx> const integrate_v(
110  trapezoid_quadrature_coefficients(ddc::get_domain<IDimVx>(allfdistribu)));
111 
112  // computation of the integrands
113  DFieldSpXVx I0mean_integrand(allfdistribu.domain());
114  DFieldSpXVx I1mean_integrand(allfdistribu.domain());
115  DFieldSpXVx I2mean_integrand(allfdistribu.domain());
116  DFieldSpXVx I3mean_integrand(allfdistribu.domain());
117  DFieldSpXVx I4mean_integrand(allfdistribu.domain());
118  ddc::for_each(allfdistribu.domain(), [&](IndexSpXVx const ispxvx) {
119  ddc::DiscreteElement<IDimension> const idimx(ddc::select<IDimVx>(ispxvx).uid() + 1);
120  ddc::DiscreteElement<IDimSp, IDimX, IDimension>
121  ispxdimx(ddc::select<IDimSp>(ispxvx), ddc::select<IDimX>(ispxvx), idimx);
122  CoordVx const coordv = ddc::coordinate(ddc::select<IDimVx>(ispxvx));
123  I0mean_integrand(ispxvx) = Dcoll(ispxdimx) * allfdistribu(ispxvx);
124  I1mean_integrand(ispxvx) = I0mean_integrand(ispxvx) * coordv;
125  I2mean_integrand(ispxvx) = I1mean_integrand(ispxvx) * coordv;
126  I3mean_integrand(ispxvx) = dvDcoll(ispxdimx) * allfdistribu(ispxvx);
127  I4mean_integrand(ispxvx) = I0mean_integrand(ispxvx) + I3mean_integrand(ispxvx) * coordv;
128  });
129 
130  // computation of the integrals over the Vx direction
131  DFieldSpX I0mean(ddc::get_domain<IDimSp, IDimX>(allfdistribu));
132  DFieldSpX I1mean(ddc::get_domain<IDimSp, IDimX>(allfdistribu));
133  DFieldSpX I2mean(ddc::get_domain<IDimSp, IDimX>(allfdistribu));
134  DFieldSpX I3mean(ddc::get_domain<IDimSp, IDimX>(allfdistribu));
135  DFieldSpX I4mean(ddc::get_domain<IDimSp, IDimX>(allfdistribu));
136  ddc::for_each(ddc::get_domain<IDimSp, IDimX>(allfdistribu), [&](IndexSpX const ispx) {
137  I0mean(ispx) = integrate_v(I0mean_integrand[ispx]);
138  I1mean(ispx) = integrate_v(I1mean_integrand[ispx]);
139  I2mean(ispx) = integrate_v(I2mean_integrand[ispx]);
140  I3mean(ispx) = integrate_v(I3mean_integrand[ispx]);
141  I4mean(ispx) = integrate_v(I4mean_integrand[ispx]);
142 
143  double const inv_Pcoll(1. / (I0mean(ispx) * I4mean(ispx) - I1mean(ispx) * I3mean(ispx)));
144  Vcoll(ispx) = inv_Pcoll * (I1mean(ispx) * I4mean(ispx) - I2mean(ispx) * I3mean(ispx));
145  Tcoll(ispx) = inv_Pcoll * (I0mean(ispx) * I2mean(ispx) - I1mean(ispx) * I1mean(ispx));
146  });
147 }
