39template <class LocalGridVx>
40void compute_Dcoll(
41 DField<IdxRange<Species, GridX, LocalGridVx>> Dcoll,
42 DConstFieldSpX collfreq,
43 DConstFieldSpX density,
44 DConstFieldSpX temperature)
45{
46 ddc::parallel_for_each(
47 Kokkos::DefaultExecutionSpace(),
48 get_idx_range(Dcoll),
49 KOKKOS_LAMBDA(Idx<Species, GridX, LocalGridVx> const ispxdimvx) {
50 double const vT(
51 Kokkos::sqrt(2. * temperature(ddc::select<Species, GridX>(ispxdimvx))));
52 double const v_norm(
53 Kokkos::fabs(ddc::coordinate(ddc::select<LocalGridVx>(ispxdimvx))) / vT);
54 double const tol = 1.e-15;
55 if (v_norm > tol) {
56 double const coeff(2. / Kokkos::sqrt(M_PI));
57 double const AD(
58 3. * Kokkos::sqrt(2. * M_PI) / 4.
59 * temperature(ddc::select<Species, GridX>(ispxdimvx))
60 * collfreq(ddc::select<Species, GridX>(ispxdimvx)));
61 double const inv_v_norm(1. / v_norm);
62 double const phi(Kokkos::erf(v_norm));
63 double const phi_prime(coeff * Kokkos::exp(-v_norm * v_norm));
64 double const psi((phi - v_norm * phi_prime) * 0.5 * inv_v_norm * inv_v_norm);
65
66 Dcoll(ispxdimvx) = AD * (phi - psi) * inv_v_norm;
67
68 } else {
69 Dcoll(ispxdimvx) = Kokkos::sqrt(2)
70 * temperature(ddc::select<Species, GridX>(ispxdimvx))
71 * collfreq(ddc::select<Species, GridX>(ispxdimvx));
72 }
73 });
74}

83template <class LocalGridVx>
84void compute_dvDcoll(
85 DField<IdxRange<Species, GridX, LocalGridVx>> dvDcoll,
86 DConstFieldSpX collfreq,
87 DConstFieldSpX density,
88 DConstFieldSpX temperature)
89{
90 ddc::parallel_for_each(
91 Kokkos::DefaultExecutionSpace(),
92 get_idx_range(dvDcoll),
93 KOKKOS_LAMBDA(Idx<Species, GridX, LocalGridVx> const ispxdimvx) {
94 double const vT(
95 Kokkos::sqrt(2. * temperature(ddc::select<Species, GridX>(ispxdimvx))));
96 double const v_norm(
97 Kokkos::fabs(ddc::coordinate(ddc::select<LocalGridVx>(ispxdimvx))) / vT);
98 double const tol = 1.e-15;
99 if (v_norm > tol) {
100 double const coeff(2. / Kokkos::sqrt(M_PI));
101 double const AD(
102 3. * Kokkos::sqrt(2. * M_PI) / 4.
103 * temperature(ddc::select<Species, GridX>(ispxdimvx))
104 * collfreq(ddc::select<Species, GridX>(ispxdimvx)));
105 double const inv_v_norm(1. / v_norm);
106 double const phi(Kokkos::erf(v_norm));
107 double const phi_prime(coeff * Kokkos::exp(-v_norm * v_norm));
108 double const psi((phi - v_norm * phi_prime) * 0.5 * inv_v_norm * inv_v_norm);
109
110 double const sign(
111 ddc::coordinate(ddc::select<LocalGridVx>(ispxdimvx))
112 / Kokkos::fabs(ddc::coordinate(ddc::select<LocalGridVx>(ispxdimvx))));
113
114 dvDcoll(ispxdimvx)
115 = sign * AD
116 / Kokkos::sqrt(
117 2 * temperature(ddc::select<Species, GridX>(ispxdimvx)))
118 * inv_v_norm * inv_v_norm * (3 * psi - phi);
119
120 } else {
121 dvDcoll(ispxdimvx) = 0.;
122 }
123 });
124}

149template <class LocalGridVx>
150void compute_Vcoll_Tcoll(
151 DFieldSpX Vcoll,
152 DFieldSpX Tcoll,
153 DConstFieldSpXVx allfdistribu,
154 DField<IdxRange<Species, GridX, LocalGridVx>> Dcoll,
155 DField<IdxRange<Species, GridX, LocalGridVx>> dvDcoll)
156{
157 DFieldMemVx const quadrature_coeffs_alloc(
158 trapezoid_quadrature_coefficients<Kokkos::DefaultExecutionSpace>(
159 get_idx_range<GridVx>(allfdistribu)));
160 DConstFieldVx const quadrature_coeffs = get_const_field(quadrature_coeffs_alloc);
161
162 // computation of the integrands
163 DFieldMemSpXVx I0mean_integrand_alloc(get_idx_range(allfdistribu));
164 DFieldMemSpXVx I1mean_integrand_alloc(get_idx_range(allfdistribu));
165 DFieldMemSpXVx I2mean_integrand_alloc(get_idx_range(allfdistribu));
166 DFieldMemSpXVx I3mean_integrand_alloc(get_idx_range(allfdistribu));
167 DFieldMemSpXVx I4mean_integrand_alloc(get_idx_range(allfdistribu));
168 auto I0mean_integrand = get_field(I0mean_integrand_alloc);
169 auto I1mean_integrand = get_field(I1mean_integrand_alloc);
170 auto I2mean_integrand = get_field(I2mean_integrand_alloc);
171 auto I3mean_integrand = get_field(I3mean_integrand_alloc);
172 auto I4mean_integrand = get_field(I4mean_integrand_alloc);
173
174 ddc::parallel_for_each(
175 Kokkos::DefaultExecutionSpace(),
176 get_idx_range(allfdistribu),
177 KOKKOS_LAMBDA(IdxSpXVx const ispxvx) {
178 Idx<LocalGridVx> const idimvx(ddc::select<GridVx>(ispxvx).uid() + 1);
179 Idx<Species, GridX, LocalGridVx>
180 ispxdimvx(ddc::select<Species>(ispxvx), ddc::select<GridX>(ispxvx), idimvx);
181 CoordVx const coordv = ddc::coordinate(ddc::select<GridVx>(ispxvx));
182 I0mean_integrand(ispxvx) = Dcoll(ispxdimvx) * allfdistribu(ispxvx);
183 I1mean_integrand(ispxvx) = I0mean_integrand(ispxvx) * coordv;
184 I2mean_integrand(ispxvx) = I1mean_integrand(ispxvx) * coordv;
185 I3mean_integrand(ispxvx) = dvDcoll(ispxdimvx) * allfdistribu(ispxvx);
186 I4mean_integrand(ispxvx)
187 = I0mean_integrand(ispxvx) + I3mean_integrand(ispxvx) * coordv;
188 });
189
190
191 // computation of the integrals over the Vx direction
192 IdxRangeSpX grid_sp_x(get_idx_range<Species, GridX>(allfdistribu));
193 DFieldMemSpX I0mean_alloc(grid_sp_x);
194 DFieldMemSpX I1mean_alloc(grid_sp_x);
195 DFieldMemSpX I2mean_alloc(grid_sp_x);
196 DFieldMemSpX I3mean_alloc(grid_sp_x);
197 DFieldMemSpX I4mean_alloc(grid_sp_x);
198 auto I0mean = get_field(I0mean_alloc);
199 auto I1mean = get_field(I1mean_alloc);
200 auto I2mean = get_field(I2mean_alloc);
201 auto I3mean = get_field(I3mean_alloc);
202 auto I4mean = get_field(I4mean_alloc);
203 ddc::parallel_fill(I0mean, 0.);
204 ddc::parallel_fill(I1mean, 0.);
205 ddc::parallel_fill(I2mean, 0.);
206 ddc::parallel_fill(I3mean, 0.);
207 ddc::parallel_fill(I4mean, 0.);
208
209 IdxRangeVx const idx_range_vx(get_idx_range<GridVx>(allfdistribu));
210
211 ddc::parallel_for_each(
212 Kokkos::DefaultExecutionSpace(),
213 grid_sp_x,
214 KOKKOS_LAMBDA(IdxSpX const ispx) {
215 for (IdxVx const ivx : idx_range_vx) {
216 I0mean(ispx) += quadrature_coeffs(ivx) * I0mean_integrand(ispx, ivx);
217 I1mean(ispx) += quadrature_coeffs(ivx) * I1mean_integrand(ispx, ivx);
218 I2mean(ispx) += quadrature_coeffs(ivx) * I2mean_integrand(ispx, ivx);
219 I3mean(ispx) += quadrature_coeffs(ivx) * I3mean_integrand(ispx, ivx);
220 I4mean(ispx) += quadrature_coeffs(ivx) * I4mean_integrand(ispx, ivx);
221 }
222
223 double const inv_Pcoll(
224 1. / (I0mean(ispx) * I4mean(ispx) - I1mean(ispx) * I3mean(ispx)));
225 Vcoll(ispx)
226 = inv_Pcoll * (I1mean(ispx) * I4mean(ispx) - I2mean(ispx) * I3mean(ispx));
227 Tcoll(ispx)
228 = inv_Pcoll * (I0mean(ispx) * I2mean(ispx) - I1mean(ispx) * I1mean(ispx));
229 });
230}

239template <class LocalGridVx>
240void compute_Nucoll(
241 DField<IdxRange<Species, GridX, LocalGridVx>> Nucoll,
242 DField<IdxRange<Species, GridX, LocalGridVx>> Dcoll,
243 DConstFieldSpX Vcoll,
244 DConstFieldSpX Tcoll)
245{
246 ddc::parallel_for_each(
247 Kokkos::DefaultExecutionSpace(),
248 get_idx_range(Dcoll),
249 KOKKOS_LAMBDA(Idx<Species, GridX, LocalGridVx> const ispxdimvx) {
250 double const coordv(ddc::coordinate(ddc::select<LocalGridVx>(ispxdimvx)));
251 Nucoll(ispxdimvx) = -Dcoll(ispxdimvx)
252 * (coordv - Vcoll(ddc::select<Species, GridX>(ispxdimvx)))
253 / Tcoll(ddc::select<Species, GridX>(ispxdimvx));
254 });
255}
