36 template <class LocalGridVx>
37 void compute_Dcoll(
38  DField<IdxRange<Species, GridX, LocalGridVx>> Dcoll,
39  DConstFieldSpX collfreq,
40  DConstFieldSpX density,
41  DConstFieldSpX temperature)
42 {
43  ddc::parallel_for_each(
44  Kokkos::DefaultExecutionSpace(),
45  get_idx_range(Dcoll),
46  KOKKOS_LAMBDA(Idx<Species, GridX, LocalGridVx> const ispxdimvx) {
47  double const vT(
48  Kokkos::sqrt(2. * temperature(ddc::select<Species, GridX>(ispxdimvx))));
49  double const v_norm(
50  Kokkos::fabs(ddc::coordinate(ddc::select<LocalGridVx>(ispxdimvx))) / vT);
51  double const tol = 1.e-15;
52  if (v_norm > tol) {
53  double const coeff(2. / Kokkos::sqrt(M_PI));
54  double const AD(
55  3. * Kokkos::sqrt(2. * M_PI) / 4.
56  * temperature(ddc::select<Species, GridX>(ispxdimvx))
57  * collfreq(ddc::select<Species, GridX>(ispxdimvx)));
58  double const inv_v_norm(1. / v_norm);
59  double const phi(Kokkos::erf(v_norm));
60  double const phi_prime(coeff * Kokkos::exp(-v_norm * v_norm));
61  double const psi((phi - v_norm * phi_prime) * 0.5 * inv_v_norm * inv_v_norm);
62 
63  Dcoll(ispxdimvx) = AD * (phi - psi) * inv_v_norm;
64 
65  } else {
66  Dcoll(ispxdimvx) = Kokkos::sqrt(2)
67  * temperature(ddc::select<Species, GridX>(ispxdimvx))
68  * collfreq(ddc::select<Species, GridX>(ispxdimvx));
69  }
70  });
71 }

166  auto I1mean_integrand = get_field(I1mean_integrand_alloc);
167  auto I2mean_integrand = get_field(I2mean_integrand_alloc);
168  auto I3mean_integrand = get_field(I3mean_integrand_alloc);
169  auto I4mean_integrand = get_field(I4mean_integrand_alloc);
170 
171  ddc::parallel_for_each(
172  Kokkos::DefaultExecutionSpace(),
173  get_idx_range(allfdistribu),
174  KOKKOS_LAMBDA(IdxSpXVx const ispxvx) {
175  Idx<LocalGridVx> const idimvx(ddc::select<GridVx>(ispxvx).uid() + 1);
176  Idx<Species, GridX, LocalGridVx>
177  ispxdimvx(ddc::select<Species>(ispxvx), ddc::select<GridX>(ispxvx), idimvx);
178  CoordVx const coordv = ddc::coordinate(ddc::select<GridVx>(ispxvx));
179  I0mean_integrand(ispxvx) = Dcoll(ispxdimvx) * allfdistribu(ispxvx);
180  I1mean_integrand(ispxvx) = I0mean_integrand(ispxvx) * coordv;
181  I2mean_integrand(ispxvx) = I1mean_integrand(ispxvx) * coordv;
182  I3mean_integrand(ispxvx) = dvDcoll(ispxdimvx) * allfdistribu(ispxvx);
183  I4mean_integrand(ispxvx)
184  = I0mean_integrand(ispxvx) + I3mean_integrand(ispxvx) * coordv;
185  });
186 
187 
188  // computation of the integrals over the Vx direction
189  IdxRangeSpX grid_sp_x(get_idx_range<Species, GridX>(allfdistribu));
190  DFieldMemSpX I0mean_alloc(grid_sp_x);
191  DFieldMemSpX I1mean_alloc(grid_sp_x);
192  DFieldMemSpX I2mean_alloc(grid_sp_x);
193  DFieldMemSpX I3mean_alloc(grid_sp_x);
194  DFieldMemSpX I4mean_alloc(grid_sp_x);

196  auto I1mean = get_field(I1mean_alloc);
197  auto I2mean = get_field(I2mean_alloc);
198  auto I3mean = get_field(I3mean_alloc);
199  auto I4mean = get_field(I4mean_alloc);
200  ddc::parallel_fill(I0mean, 0.);
201  ddc::parallel_fill(I1mean, 0.);
202  ddc::parallel_fill(I2mean, 0.);
203  ddc::parallel_fill(I3mean, 0.);
204  ddc::parallel_fill(I4mean, 0.);
205 
206  ddc::parallel_for_each(
207  Kokkos::DefaultExecutionSpace(),
208  grid_sp_x,
209  KOKKOS_LAMBDA(IdxSpX const ispx) {
210  for (IdxVx const ivx : get_idx_range<GridVx>(allfdistribu)) {
211  I0mean(ispx) += quadrature_coeffs(ivx) * I0mean_integrand(ispx, ivx);
212  I1mean(ispx) += quadrature_coeffs(ivx) * I1mean_integrand(ispx, ivx);
213  I2mean(ispx) += quadrature_coeffs(ivx) * I2mean_integrand(ispx, ivx);
214  I3mean(ispx) += quadrature_coeffs(ivx) * I3mean_integrand(ispx, ivx);
215  I4mean(ispx) += quadrature_coeffs(ivx) * I4mean_integrand(ispx, ivx);
216  }
217 
218  double const inv_Pcoll(
219  1. / (I0mean(ispx) * I4mean(ispx) - I1mean(ispx) * I3mean(ispx)));
220  Vcoll(ispx)
221  = inv_Pcoll * (I1mean(ispx) * I4mean(ispx) - I2mean(ispx) * I3mean(ispx));
222  Tcoll(ispx)
223  = inv_Pcoll * (I0mean(ispx) * I2mean(ispx) - I1mean(ispx) * I1mean(ispx));
224  });
