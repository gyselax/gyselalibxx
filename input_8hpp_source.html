60 template <class Grid1D, class BSplines, class InterpPointInitMethod>
61 inline IdxRange<Grid1D> init_spline_dependent_idx_range(
62  PC_tree_t const& conf_voicexx,
63  std::string const& mesh_identifier)
64 {
65  using Dim = typename Grid1D::continuous_dimension_type;
66  using Coord1D = ddc::Coordinate<Dim>;
67 
68  std::vector<Coord1D> breakpoints;
69 
70  if constexpr (BSplines::is_uniform()) {
71  // If uniform BSplines are used and interpolation points are calculated from them
72  Coord1D min(PCpp_double(conf_voicexx, ".SplineMesh." + mesh_identifier + "_min"));
73  Coord1D max(PCpp_double(conf_voicexx, ".SplineMesh." + mesh_identifier + "_max"));
74  IdxStep<Grid1D> ncells(
75  PCpp_int(conf_voicexx, ".SplineMesh." + mesh_identifier + "_ncells"));
76  ddc::init_discrete_space<BSplines>(min, max, ncells);
77  } else if constexpr (!ddcHelper::is_non_uniform_interpolation_points_v<InterpPointInitMethod>) {
78  PDI_get_arrays("read_" + mesh_identifier, "breakpoints_" + mesh_identifier, breakpoints);
79  ddc::init_discrete_space<BSplines>(breakpoints);
80  }
81 
82  if constexpr (ddcHelper::is_non_uniform_interpolation_points_v<InterpPointInitMethod>) {
83  // If uniform BSplines are used but the interpolation points are provided by the user
84  // This may be the case if you want to test a new choice of interpolation points or
85  // if you want to ensure that the interpolation points used match exactly the points
86  // used to initialise values passed into the simulation.
87  std::vector<Coord1D> mesh;
88  if constexpr (BSplines::is_uniform()) {
89  PDI_get_arrays("read_" + mesh_identifier, "grid_" + mesh_identifier, mesh);
90  } else {
91  PDI_get_arrays(
92  "read_" + mesh_identifier,
93  "breakpoints_" + mesh_identifier,
94  breakpoints,
95  "grid_" + mesh_identifier,
96  mesh);
97  ddc::init_discrete_space<BSplines>(breakpoints);
98  }
99  ddc::init_discrete_space<Grid1D>(
100  InterpPointInitMethod::template get_sampling<Grid1D>(mesh));
101  } else {
102  ddc::init_discrete_space<Grid1D>(InterpPointInitMethod::template get_sampling<Grid1D>());
103  }
104  IdxRange<Grid1D> interpolation_idx_range(InterpPointInitMethod::template get_domain<Grid1D>());
105  return interpolation_idx_range;
106 }
