92 template <class IDim, class BSplines, class InterpPointInitMethod>
93 inline ddc::DiscreteDomain<IDim> init_pseudo_uniform_spline_dependent_domain(
94  PC_tree_t const& conf_voicexx,
95  std::string mesh_identifier)
96 {
97  static_assert(!BSplines::is_uniform());
98  using RDim = typename IDim::continuous_dimension_type;
99  using Coord = ddc::Coordinate<RDim>;
100 
101  Coord min(PCpp_double(conf_voicexx, ".SplineMesh." + mesh_identifier + "_min"));
102  Coord max(PCpp_double(conf_voicexx, ".SplineMesh." + mesh_identifier + "_max"));
103  ddc::DiscreteVector<IDim> ncells(
104  PCpp_int(conf_voicexx, ".SplineMesh." + mesh_identifier + "_ncells"));
105 
106  std::vector<Coord> break_points(ncells + 1);
107 
108  double const delta(double(max - min) / ncells);
109 
110  break_points[0] = min;
111  for (int i(1); i < ncells; ++i) {
112  break_points[i] = min + i * delta;
113  }
114  break_points[ncells] = max;
115 
116  ddc::init_discrete_space<BSplines>(break_points);
117  ddc::init_discrete_space<IDim>(InterpPointInitMethod::template get_sampling<IDim>());
118  ddc::DiscreteDomain<IDim> interpolation_domain(
119  InterpPointInitMethod::template get_domain<IDim>());
120  return interpolation_domain;
121 }
