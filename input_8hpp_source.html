63 template <class IDim, class BSplines, class InterpPointInitMethod>
64 inline ddc::DiscreteDomain<IDim> init_pseudo_uniform_spline_dependent_domain(
65  PC_tree_t const& conf_voicexx,
66  std::string mesh_identifier)
67 {
68  static_assert(!BSplines::is_uniform());
69  using RDim = typename IDim::continuous_dimension_type;
70  using Coord = ddc::Coordinate<RDim>;
71 
72  Coord min(PCpp_double(conf_voicexx, ".SplineMesh." + mesh_identifier + "_min"));
73  Coord max(PCpp_double(conf_voicexx, ".SplineMesh." + mesh_identifier + "_max"));
74  ddc::DiscreteVector<IDim> ncells(
75  PCpp_int(conf_voicexx, ".SplineMesh." + mesh_identifier + "_ncells"));
76 
77  std::vector<Coord> break_points(ncells + 1);
78 
79  double const delta(double(max - min) / ncells);
80 
81  break_points[0] = min;
82  for (int i(1); i < ncells; ++i) {
83  break_points[i] = min + i * delta;
84  }
85  break_points[ncells] = max;
86 
87  ddc::init_discrete_space<BSplines>(break_points);
88  ddc::init_discrete_space<IDim>(InterpPointInitMethod::template get_sampling<IDim>());
89  ddc::DiscreteDomain<IDim> interpolation_domain(
90  InterpPointInitMethod::template get_domain<IDim>());
91  return interpolation_domain;
92 }
