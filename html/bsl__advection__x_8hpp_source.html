12 template <class Geometry, class DDimX>
13 class BslAdvectionSpatial : public IAdvectionSpatial<Geometry, DDimX>
14 {
15  using DDimSp = typename Geometry::DDimSp;
16  using DDimV = typename Geometry::template velocity_dim_for<DDimX>;
17  using DDom = typename Geometry::FdistribuDDom;
18  static_assert(std::is_same_v<DDom, ddc::DiscreteDomain<DDimSp, DDimX, DDimV>>);
19  using DElemX = ddc::DiscreteElement<DDimX>;
20  using DElemV = ddc::DiscreteElement<DDimV>;
21  using DElemSp = ddc::DiscreteElement<DDimSp>;
22  using DElemSpV = ddc::DiscreteElement<DDimSp, DDimV>;
23  using CDimX = typename DDimX::continuous_dimension_type;
24 
25 private:
26  IPreallocatableInterpolator<DDimX> const& m_interpolator_x;
27 
28 public:
29  BslAdvectionSpatial(IPreallocatableInterpolator<DDimX> const& interpolator_x)
30  : m_interpolator_x(interpolator_x)
31  {
32  }
33 
34  ~BslAdvectionSpatial() override = default;
35 
36  ddc::ChunkSpan<double, DDom> operator()(
37  ddc::ChunkSpan<double, DDom> const allfdistribu,
38  double const dt) const override
39  {
40  DDom const dom = allfdistribu.domain();
41  ddc::DiscreteDomain<DDimX> const x_dom = ddc::select<DDimX>(dom);
42  ddc::DiscreteDomain<DDimV> const v_dom = ddc::select<DDimV>(dom);
43  ddc::DiscreteDomain<DDimSp> const sp_dom = ddc::select<DDimSp>(dom);
44 
45  // pre-allocate some memory to prevent allocation later in loop
46  ddc::Chunk<ddc::Coordinate<CDimX>, ddc::DiscreteDomain<DDimX>> feet_coords(x_dom);
47  ddc::Chunk<double, ddc::DiscreteDomain<DDimX>> contiguous_slice(x_dom);
48  InterpolatorProxy<DDimX> const interpolator_x = m_interpolator_x.preallocate();
49 
50  auto c_dom = ddc::remove_dims_of(
51  dom,
52  ddc::DiscreteDomain<DDimSp, DDimX, DDimV>(sp_dom, x_dom, v_dom));
53  using DElemC = typename decltype(c_dom)::discrete_element_type;
54 
55  ddc::for_each(c_dom, [&](DElemC const ic) {
56  ddc::for_each(sp_dom, [&](DElemSp const isp) {
57  double const sqrt_me_on_mspecies = std::sqrt(mass(ielec()) / mass(isp));
58 
59  ddc::for_each(v_dom, [&](DElemV const iv) {
60  // compute the displacement
61  double const dx = sqrt_me_on_mspecies * dt * ddc::coordinate(iv);
62 
63  // compute the coordinates of the feet
64  ddc::for_each(x_dom, [&](DElemX const ix) {
65  feet_coords(ix) = ddc::Coordinate<CDimX>(ddc::coordinate(ix) - dx);
66  });
67 
68  // copy the slice in contiguous memory
69  ddc::deepcopy(contiguous_slice, allfdistribu[ic][isp][iv]);
70 
71  // interpolate the function at the feet using the provided interpolator
72  interpolator_x(contiguous_slice, feet_coords.span_cview());
73 
74  // copy back
75  ddc::deepcopy(allfdistribu[ic][isp][iv], contiguous_slice);
76  });
77  });
78  });
79 
80  return allfdistribu;
81  }
82 };
