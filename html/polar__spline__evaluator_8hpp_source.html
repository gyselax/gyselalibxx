131  template <class Mapping>
132  double integrate(PolarSplineView<PolarBSplinesType> const spline_coef, Mapping const mapping)
133  const
134  {
135  int constexpr nr = ddc::discrete_space<BSplinesR>().ncells() + BSplinesR::degree() - 2;
136  int constexpr np = ddc::discrete_space<BSplinesP>().ncells() + BSplinesP::degree();
137  std::array<double, PolarBSplinesType::eval_size()> singular_values;
138  DSpan1D singular_vals(singular_values.data(), PolarBSplinesType::n_singular_basis());
139  std::array<double, nr * np> values;
140  DSpan2D vals(values.data(), nr, np);
141 
142  ddc::discrete_space<PolarBSplinesType>().integrals(singular_vals, vals);
143 
144  double y = 0.;
145  ddc::for_each(
146  spline_coef.singular_spline_coef.domain(),
147  [=](ddc::DiscreteElement<PolarBSplinesType> const i) {
148  y += spline_coef.singular_spline_coef(i) * singular_vals(i)
149  * mapping.determinant(i);
150  });
151  ddc::for_each(
152  spline_coef.spline_coef.domain(),
153  [=](ddc::DiscreteElement<BSplinesR, BSplinesP> const i) {
154  y += spline_coef.spline_coef(i)
155  * vals(ddc::select<BSplinesR>(i), ddc::select<BSplinesP>(i))
156  * mapping.determinant(i);
157  });
158  return y;
159  }

162  double eval(
163  ddc::Coordinate<DimR, DimP> coord_eval,
164  PolarSplineView<PolarBSplinesType> const spline_coef) const
165  {
166  const double coord_eval1 = ddc::get<DimR>(coord_eval);
167  double coord_eval2 = ddc::get<DimP>(coord_eval);
168  if (coord_eval1 > ddc::discrete_space<BSplinesR>().rmax()) {
169  return m_outer_bc(coord_eval1, coord_eval2, spline_coef);
170  }
171  if (coord_eval2 < ddc::discrete_space<BSplinesP>().rmin()
172  || coord_eval2 > ddc::discrete_space<BSplinesP>().rmax()) {
173  coord_eval2 -= std::floor(
174  (coord_eval2 - ddc::discrete_space<BSplinesP>().rmin())
175  / ddc::discrete_space<BSplinesP>().length())
176  * ddc::discrete_space<BSplinesP>().length();
177  }
178  return eval_no_bc(
179  ddc::Coordinate<DimR, DimP>(coord_eval1, coord_eval2),
180  spline_coef,

184  template <class EvalType>
185  double eval_no_bc(
186  ddc::Coordinate<DimR, DimP> coord_eval,
187  PolarSplineView<PolarBSplinesType> const spline_coef,
188  EvalType const) const
189  {
190  static_assert(
191  std::is_same_v<
192  EvalType,
193  eval_type> || std::is_same_v<EvalType, eval_deriv_r_type> || std::is_same_v<EvalType, eval_deriv_p_type> || std::is_same_v<EvalType, eval_deriv_r_p_type>);
194 
195  std::array<double, PolarBSplinesType::n_singular_basis()> singular_data;
196  DSpan1D singular_vals(singular_data.data(), PolarBSplinesType::n_singular_basis());
197  std::array<double, (BSplinesR::degree() + 1) * (BSplinesP::degree() + 1)> data;
198  DSpan2D vals(data.data(), BSplinesR::degree() + 1, BSplinesP::degree() + 1);
199 
200  ddc::DiscreteElement<BSplinesR, BSplinesP> jmin;
201 
202  if constexpr (std::is_same_v<EvalType, eval_type>) {
203  jmin = ddc::discrete_space<PolarBSplinesType>()
204  .eval_basis(singular_vals, vals, coord_eval);
205  } else if constexpr (std::is_same_v<EvalType, eval_deriv_r_type>) {
206  jmin = ddc::discrete_space<PolarBSplinesType>()
207  .eval_deriv_r(singular_vals, vals, coord_eval);
208  } else if constexpr (std::is_same_v<EvalType, eval_deriv_p_type>) {
209  jmin = ddc::discrete_space<PolarBSplinesType>()
210  .eval_deriv_p(singular_vals, vals, coord_eval);
211  } else if constexpr (std::is_same_v<EvalType, eval_deriv_r_p_type>) {
212  jmin = ddc::discrete_space<PolarBSplinesType>()
213  .eval_deriv_r_and_p(singular_vals, vals, coord_eval);
214  }
215 
216  double y = 0.0;
217  for (std::size_t i = 0; i < PolarBSplinesType::n_singular_basis(); ++i) {
218  y += spline_coef.singular_spline_coef(ddc::DiscreteElement<PolarBSplinesType>(i))
219  * singular_vals(i);
220  }
221  ddc::DiscreteElement<BSplinesR> jmin_r = ddc::select<BSplinesR>(jmin);
222  ddc::DiscreteElement<BSplinesP> jmin_p = ddc::select<BSplinesP>(jmin);
223  int nr = BSplinesR::degree() + 1;
224  if (jmin_r.uid() < continuity + 1) {
225  nr = nr - (continuity + 1 - jmin_r.uid());
226  jmin_r = ddc::DiscreteElement<BSplinesR>(continuity + 1);
227  }
228  for (int i = 0; i < nr; ++i) {
229  for (std::size_t j = 0; j < BSplinesP::degree() + 1; ++j) {
230  y += spline_coef.spline_coef(jmin_r + i, jmin_p + j) * vals(i, j);
231  }
232  }
233  return y;
234  }
