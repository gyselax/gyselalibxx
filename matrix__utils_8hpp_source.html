16 template <class KokkosViewType>
17 auto to_gko_multivector(
18  std::shared_ptr<const gko::Executor> const& gko_exec,
19  KokkosViewType const& view)
20 {
21  static_assert((Kokkos::is_view_v<KokkosViewType> && KokkosViewType::rank == 2));
22  using value_type = typename KokkosViewType::traits::value_type;
23 
24  assert(view.stride_1() == 1);

27  create(gko_exec,
28  gko::batch_dim<2>(view.extent(0), gko::dim<2>(view.extent(1), 1)),

41 inline void check_conv(
42  int const batch_size,
43  double const tol,
44  std::shared_ptr<const gko::Executor> gko_exec,
45  std::shared_ptr<const gko::batch::log::BatchConvergence<double>> logger)
46 {
47  auto logger_residual_host
48  = gko::make_temporary_clone(gko_exec->get_master(), &logger->get_residual_norm());
49  bool has_converged = false;
50  Kokkos::parallel_reduce(
51  "convergence",
52  Kokkos::RangePolicy<Kokkos::DefaultHostExecutionSpace>(0, batch_size),
53  [&](int batch_idx, bool& check_tol) {
54  check_tol = check_tol && (logger_residual_host->get_const_data()[batch_idx] <= tol);
55  },
56  Kokkos::LAnd<bool>(has_converged));
57  if (!has_converged) {
58  throw ::std::runtime_error("Residual tolerance is not reached");
59  }
60 }

102 template <class batch_sparse_type>
103 void save_logger(
104  std::shared_ptr<batch_sparse_type> batch_matrix,
105  Kokkos::View<double**, Kokkos::LayoutRight, Kokkos::DefaultExecutionSpace> const x_view,
106  Kokkos::View<double**, Kokkos::LayoutRight, Kokkos::DefaultExecutionSpace> const b_view,
107  std::shared_ptr<const gko::batch::log::BatchConvergence<double>> logger,
108  double const tol,
109  std::string const& filename)
110 {
111  int const batch_size = x_view.extent(0);
112  std::shared_ptr const gko_exec = batch_matrix->get_executor();
113  // allocate the residual
114  auto res = gko::batch::MultiVector<double>::
115  create(gko_exec, gko::batch_dim<2>(batch_size, gko::dim<2>(x_view.extent(1), 1)));
116  res->copy_from(to_gko_multivector(gko_exec, b_view));
117 
118  auto norm_dim = gko::batch_dim<2>(batch_size, gko::dim<2>(1, 1));
119  // allocate rhs norm on host.
120  auto b_norm_host = gko::batch::MultiVector<double>::create(gko_exec->get_master(), norm_dim);
121  b_norm_host->fill(0.0);
122  // allocate the residual norm on host.
123  auto res_norm_host = gko::batch::MultiVector<double>::create(gko_exec->get_master(), norm_dim);
124  res_norm_host->fill(0.0);
125  // compute rhs norm.
126  to_gko_multivector(gko_exec, b_view)->compute_norm2(b_norm_host);
127  // we need constants on the device
128  auto one = gko::batch::MultiVector<double>::create(gko_exec, norm_dim);
129  one->fill(1.0);
130  auto neg_one = gko::batch::MultiVector<double>::create(gko_exec, norm_dim);
131  neg_one->fill(-1.0);
132  //to estimate the "true" residual, the apply function below computes Ax-res, and stores the result in res.
133  batch_matrix->apply(one, to_gko_multivector(gko_exec, x_view), neg_one, res);
134  //compute residual norm.
135  res->compute_norm2(res_norm_host);
136 
137  auto log_resid_host
138  = gko::make_temporary_clone(gko_exec->get_master(), &logger->get_residual_norm());
139  auto log_iters_host
140  = gko::make_temporary_clone(gko_exec->get_master(), &logger->get_num_iterations());
141 
142  std::fstream log_file(filename + "_logger.txt", std::ios::out | std::ios::app);
143  // "unbatch" converts a batch object into a vector
144  // of objects of the corresponding single type.
145  for (int i = 0; i < batch_size; ++i) {
146  write_log(
147  log_file,
148  i,
149  log_iters_host->get_const_data()[i],
150  log_resid_host->get_const_data()[i],
151  res_norm_host->create_const_view_for_item(i)->at(0, 0),
152  b_norm_host->create_const_view_for_item(i)->at(0, 0),
153  tol);
154  }
155  log_file.close();
156 }
