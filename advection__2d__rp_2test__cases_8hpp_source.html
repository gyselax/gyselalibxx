65template <class Mapping>
66class FunctionToBeAdvected_cos_4_elipse : public FunctionToBeAdvected
67{
68private:
69 Mapping const& m_mapping;
70
71public:
78 explicit FunctionToBeAdvected_cos_4_elipse(Mapping const& mapping) : m_mapping(mapping) {}
79
80 double operator()(CoordRTheta coord_rp) final
81 {
82 CoordXY const coord_xy(m_mapping(coord_rp));
83 double const x = ddc::get<X>(coord_xy);
84 double const y = ddc::get<Y>(coord_xy);
85
86 double const a = 0.3;
87 double const x_bar = 0.;
88 double const y_bar = 0.;
89
90 double const r_1 = std::sqrt((x - x_bar) * (x - x_bar) + 8 * (y - y_bar) * (y - y_bar));
91 double const r_2 = std::sqrt(8 * (x - x_bar) * (x - x_bar) + (y - y_bar) * (y - y_bar));
92
93 double const G_1 = std::pow(std::cos(M_PI * r_1 / 2. / a), 4) * (r_1 < a);
94 double const G_2 = std::pow(std::cos(M_PI * r_2 / 2. / a), 4) * (r_2 < a);
95
96 return 1. / 2. * (G_1 + G_2);
97 }
98};

109template <class Mapping>
110class FunctionToBeAdvected_gaussian : public FunctionToBeAdvected
111{
112private:
113 Mapping const& m_mapping;
114 double const m_constant;
115 double const m_x0;
116 double const m_y0;
117 double const m_sig_x;
118 double const m_sig_y;
119 double const m_rmin;
120 double const m_rmax;
121
122public:
143 FunctionToBeAdvected_gaussian(
144 Mapping const& mapping,
145 double const constant,
146 double const x0,
147 double const y0,
148 double const sig_x,
149 double const sig_y,
150 double const rmin,
151 double const rmax)
152 : m_mapping(mapping)
153 , m_constant(constant)
154 , m_x0(x0)
155 , m_y0(y0)
156 , m_sig_x(sig_x)
157 , m_sig_y(sig_y)
158 , m_rmin(rmin)
159 , m_rmax(rmax)
160 {
161 }
162
163 double operator()(CoordRTheta coord_rp) final
164 {
165 // Gaussian centered in (x0, y0):
166 CoordXY const coord_xy(m_mapping(coord_rp));
167 double const x = ddc::get<X>(coord_xy);
168 double const y = ddc::get<Y>(coord_xy);
169 double const r = ddc::get<R>(coord_rp);
170 if ((m_rmin <= r) and (r <= m_rmax)) {
171 return m_constant
172 * std::exp(
173 -pow(x - m_x0, 2) / (2 * m_sig_x * m_sig_x)
174 - pow(y - m_y0, 2) / (2 * m_sig_y * m_sig_y));
175 } else {
176 return 0.0;
177 }
178 }
179};

236class AdvectionField_decentred_rotation : public AdvectionField
237{
238private:
239 double const m_omega;
240 double const m_xc;
241 double const m_yc;
242 double const m_x_bar;
243 double const m_y_bar;
244
245public:
250 AdvectionField_decentred_rotation()
251 : m_omega(2 * M_PI)
252 , m_xc(0.25)
253 , m_yc(0.)
254 , m_x_bar(0.)
255 , m_y_bar(0.)
256 {
257 }
258
259 CoordXY operator()(CoordXY const coord, double const t) const final
260 {
261 double const x = m_omega * (m_yc - ddc::get<Y>(coord));
262 double const y = m_omega * (ddc::get<X>(coord) - m_xc);
263 return CoordXY(x, y);
264 }
265
266 CoordXY exact_feet(CoordXY coord, double const t) const final
267 {
268 double const x = ddc::get<X>(coord);
269 double const y = ddc::get<Y>(coord);
270 double const foot_x
271 = m_xc + (x - m_xc) * std::cos(m_omega * -t) - (y - m_yc) * std::sin(m_omega * -t);
272 double const foot_y
273 = m_yc + (x - m_xc) * std::sin(m_omega * -t) + (y - m_yc) * std::cos(m_omega * -t);
274 return CoordXY(foot_x, foot_y);
275 }
276};

346class AdvectionField_rotation : public AdvectionField
347{
348private:
349 double const m_vr;
350 double const m_vtheta;
351 CartesianToCircular<X, Y, R, Theta> const m_physical_to_logical_mapping;
352 CircularToCartesian<R, Theta, X, Y> const m_logical_to_physical_mapping;
353
354
355public:
364 AdvectionField_rotation(CoordVr vr, CoordVtheta vtheta)
365 : m_vr(vr)
366 , m_vtheta(vtheta)
367 , m_physical_to_logical_mapping()
368 , m_logical_to_physical_mapping()
369 {
370 }
371
372 CoordXY operator()(CoordXY const coord, double const t) const final
373 {
374 CoordRTheta const coord_rp(m_physical_to_logical_mapping(coord));
375 std::array<std::array<double, 2>, 2> jacobian;
376 m_logical_to_physical_mapping.jacobian_matrix(coord_rp, jacobian);
377 double const vx = m_vr * jacobian[0][0] + m_vtheta * jacobian[0][1];
378 double const vy = m_vr * jacobian[1][0] + m_vtheta * jacobian[1][1];
379 return CoordXY(vx, vy);
380 }
381
382 CoordXY exact_feet(CoordXY coord_xy, double const t) const final
383 {
384 CoordRTheta const coord_rp(m_physical_to_logical_mapping(coord_xy));
385 CoordRTheta const velocity(m_vr, m_vtheta);
386 return m_logical_to_physical_mapping(coord_rp - t * velocity);
387 }
388};
