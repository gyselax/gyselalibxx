42template <class Mapping>
43class FunctionToBeAdvected_cos_4_elipse
44{
45private:
46 Mapping const& m_mapping;
47
48public:
55 explicit FunctionToBeAdvected_cos_4_elipse(Mapping const& mapping) : m_mapping(mapping) {}
56
58 explicit KOKKOS_DEFAULTED_FUNCTION FunctionToBeAdvected_cos_4_elipse(
59 FunctionToBeAdvected_cos_4_elipse const&)
60 = default;
61
70 KOKKOS_FUNCTION double operator()(CoordRTheta coord_rp) const
71 {
72 CoordXY const coord_xy(m_mapping(coord_rp));
73 double const x = ddc::get<X>(coord_xy);
74 double const y = ddc::get<Y>(coord_xy);
75
76 double const a = 0.3;
77 double const x_bar = 0.;
78 double const y_bar = 0.;
79
80 double const r_1 = Kokkos::sqrt((x - x_bar) * (x - x_bar) + 8 * (y - y_bar) * (y - y_bar));
81 double const r_2 = Kokkos::sqrt(8 * (x - x_bar) * (x - x_bar) + (y - y_bar) * (y - y_bar));
82
83 double const G_1 = ipow(Kokkos::cos(M_PI * r_1 / 2. / a), 4) * (r_1 < a);
84 double const G_2 = ipow(Kokkos::cos(M_PI * r_2 / 2. / a), 4) * (r_2 < a);
85
86 return 1. / 2. * (G_1 + G_2);
87 }
88};

99template <class Mapping>
100class FunctionToBeAdvected_gaussian
101{
102private:
103 Mapping const& m_mapping;
104 double const m_constant;
105 double const m_x0;
106 double const m_y0;
107 double const m_sig_x;
108 double const m_sig_y;
109 double const m_rmin;
110 double const m_rmax;
111
112public:
133 FunctionToBeAdvected_gaussian(
134 Mapping const& mapping,
135 double const constant,
136 double const x0,
137 double const y0,
138 double const sig_x,
139 double const sig_y,
140 double const rmin,
141 double const rmax)
142 : m_mapping(mapping)
143 , m_constant(constant)
144 , m_x0(x0)
145 , m_y0(y0)
146 , m_sig_x(sig_x)
147 , m_sig_y(sig_y)
148 , m_rmin(rmin)
149 , m_rmax(rmax)
150 {
151 }
152
154 explicit KOKKOS_DEFAULTED_FUNCTION FunctionToBeAdvected_gaussian(
155 FunctionToBeAdvected_gaussian const&)
156 = default;
157
166 KOKKOS_FUNCTION double operator()(CoordRTheta coord_rp) const
167 {
168 // Gaussian centered in (x0, y0):
169 CoordXY const coord_xy(m_mapping(coord_rp));
170 double const x = ddc::get<X>(coord_xy);
171 double const y = ddc::get<Y>(coord_xy);
172 double const r = ddc::get<R>(coord_rp);
173 if ((m_rmin <= r) and (r <= m_rmax)) {
174 return m_constant
175 * Kokkos::exp(
176 -ipow(x - m_x0, 2) / (2 * m_sig_x * m_sig_x)
177 - ipow(y - m_y0, 2) / (2 * m_sig_y * m_sig_y));
178 } else {
179 return 0.0;
180 }
181 }
182};

204class AdvectionField_decentred_rotation
205{
206private:
207 double const m_omega;
208 double const m_xc;
209 double const m_yc;
210 double const m_x_bar;
211 double const m_y_bar;
212
213public:
218 AdvectionField_decentred_rotation()
219 : m_omega(2 * M_PI)
220 , m_xc(0.25)
221 , m_yc(0.)
222 , m_x_bar(0.)
223 , m_y_bar(0.)
224 {
225 }
226
228 explicit KOKKOS_DEFAULTED_FUNCTION AdvectionField_decentred_rotation(
229 AdvectionField_decentred_rotation const&)
230 = default;
231
242 KOKKOS_FUNCTION CoordXY operator()(CoordXY const coord, double const t) const
243 {
244 double const x = m_omega * (m_yc - ddc::get<Y>(coord));
245 double const y = m_omega * (ddc::get<X>(coord) - m_xc);
246 return CoordXY(x, y);
247 }
248
259 KOKKOS_FUNCTION CoordXY exact_feet(CoordXY coord, double const t) const
260 {
261 double const x = ddc::get<X>(coord);
262 double const y = ddc::get<Y>(coord);
263 double const foot_x = m_xc + (x - m_xc) * Kokkos::cos(m_omega * -t)
264 - (y - m_yc) * Kokkos::sin(m_omega * -t);
265 double const foot_y = m_yc + (x - m_xc) * Kokkos::sin(m_omega * -t)
266 + (y - m_yc) * Kokkos::cos(m_omega * -t);
267 return CoordXY(foot_x, foot_y);
268 }
269};

363class AdvectionField_rotation
364{
365private:
366 double const m_vr;
367 double const m_vtheta;
368 CartesianToCircular<X, Y, R, Theta> const m_physical_to_logical_mapping;
369 CircularToCartesian<R, Theta, X, Y> const m_logical_to_physical_mapping;
370
371
372public:
381 AdvectionField_rotation(CoordVr vr, CoordVtheta vtheta)
382 : m_vr(vr)
383 , m_vtheta(vtheta)
384 , m_physical_to_logical_mapping()
385 , m_logical_to_physical_mapping()
386 {
387 }
388
390 KOKKOS_DEFAULTED_FUNCTION AdvectionField_rotation(AdvectionField_rotation const&) = default;
391
402 KOKKOS_FUNCTION CoordXY operator()(CoordXY const coord, double const t) const
403 {
404 CoordRTheta const coord_rp(m_physical_to_logical_mapping(coord));
405 std::array<std::array<double, 2>, 2> jacobian;
406 m_logical_to_physical_mapping.jacobian_matrix(coord_rp, jacobian);
407 double const vx = m_vr * jacobian[0][0] + m_vtheta * jacobian[0][1];
408 double const vy = m_vr * jacobian[1][0] + m_vtheta * jacobian[1][1];
409 return CoordXY(vx, vy);
410 }
411
422 KOKKOS_FUNCTION CoordXY exact_feet(CoordXY coord_xy, double const t) const
423 {
424 CoordRTheta const coord_rp(m_physical_to_logical_mapping(coord_xy));
425 CoordRTheta const velocity(m_vr, m_vtheta);
426 return m_logical_to_physical_mapping(coord_rp - t * velocity);
427 }
428};
