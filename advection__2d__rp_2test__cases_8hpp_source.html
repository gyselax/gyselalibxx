68 template <class Mapping>
69 class FunctionToBeAdvected_cos_4_elipse : public FunctionToBeAdvected
70 {
71 private:
72  Mapping const& m_mapping;
73 
74 public:
81  FunctionToBeAdvected_cos_4_elipse(Mapping const& mapping) : m_mapping(mapping) {};
82  ~FunctionToBeAdvected_cos_4_elipse() {};
83 
84  double operator()(double r, double th) final
85  {
86  CoordXY const coord_xy(m_mapping(CoordRP(r, th)));
87  double const x = ddc::get<RDimX>(coord_xy);
88  double const y = ddc::get<RDimY>(coord_xy);
89 
90  double const a = 0.3;
91  double const x_bar = 0.;
92  double const y_bar = 0.;
93 
94  double const r_1 = std::sqrt((x - x_bar) * (x - x_bar) + 8 * (y - y_bar) * (y - y_bar));
95  double const r_2 = std::sqrt(8 * (x - x_bar) * (x - x_bar) + (y - y_bar) * (y - y_bar));
96 
97  double const G_1 = std::pow(std::cos(M_PI * r_1 / 2. / a), 4) * (r_1 < a);
98  double const G_2 = std::pow(std::cos(M_PI * r_2 / 2. / a), 4) * (r_2 < a);
99 
100  return 1. / 2. * (G_1 + G_2);
101  };
102 };

126 public:
147  FunctionToBeAdvected_gaussian(
148  Mapping const& mapping,
149  double const constant,
150  double const x0,
151  double const y0,
152  double const sig_x,
153  double const sig_y,
154  double const rmin,
155  double const rmax)
156  : m_mapping(mapping)
157  , m_constant(constant)
158  , m_x0(x0)
159  , m_y0(y0)
160  , m_sig_x(sig_x)
161  , m_sig_y(sig_y)
162  , m_rmin(rmin)
163  , m_rmax(rmax) {};
164  ~FunctionToBeAdvected_gaussian() {};
165 
166  double operator()(double r, double th) final
167  {
168  // Gaussian centered in (x0, y0):
169  CoordXY const coord_xy(m_mapping(CoordRP(r, th)));
170  double const x = ddc::get<RDimX>(coord_xy);
171  double const y = ddc::get<RDimY>(coord_xy);
172  if ((m_rmin <= r) and (r <= m_rmax)) {
173  return m_constant
174  * std::exp(
175  -pow(x - m_x0, 2) / (2 * m_sig_x * m_sig_x)
176  - pow(y - m_y0, 2) / (2 * m_sig_y * m_sig_y));
177  } else {
178  return 0.0;
179  }
180  };
