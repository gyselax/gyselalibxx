62template <class... Query, class ContainerTypeSeq>
63struct DimensionsAreAdjacent<ddc::detail::TypeSeq<Query...>, ContainerTypeSeq>
64{
65 static constexpr bool check_adjacent()
66 {
67 static_assert((ddc::in_tags_v<Query, ContainerTypeSeq> && ...));
68 std::array<std::size_t, sizeof...(Query)> ranks(
69 {ddc::type_seq_rank_v<Query, ContainerTypeSeq>...});
70 bool ordered = true;
71 for (int i(1); i < sizeof...(Query); ++i) {
72 ordered &= ((ranks[i] - ranks[i - 1]) == 1);
73 }
74 return ordered;
75 }
76 static constexpr bool value = check_adjacent();
77};
