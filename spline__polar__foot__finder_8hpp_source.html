213 void operator()(
214 FieldRTheta<CoordRTheta> feet,
215 DVectorConstFieldRTheta<X, Y> advection_field,
216 double dt) const final
217 {
218 VectorSplineCoeffsMem2D<X_adv, Y_adv> advection_field_in_adv_domain_coefs(
219 get_spline_idx_range(m_builder_advection_field));
220
221 // Compute the advection field in the advection domain.
222 auto advection_field_in_adv_domain = create_geometry_mirror_view(
223 ExecSpace(),
224 advection_field,
225 m_pseudo_physical_to_physical);
226
227 // Get the coefficients of the advection field in the advection domain.
228 m_builder_advection_field(
229 ddcHelper::get<X_adv>(advection_field_in_adv_domain_coefs),
230 ddcHelper::get<X_adv>(get_const_field(advection_field_in_adv_domain)));
231 m_builder_advection_field(
232 ddcHelper::get<Y_adv>(advection_field_in_adv_domain_coefs),
233 ddcHelper::get<Y_adv>(get_const_field(advection_field_in_adv_domain)));
234
235
236 // The function describing how the derivative of the evolve function is calculated.
237 std::function<void(DVectorFieldRTheta<X_adv, Y_adv>, ConstFieldRTheta<CoordRTheta>)> dy
238 = [&](DVectorFieldRTheta<X_adv, Y_adv> updated_advection_field,
239 ConstFieldRTheta<CoordRTheta> feet) {
240 m_evaluator_advection_field(
241 get_field(ddcHelper::get<X_adv>(updated_advection_field)),
242 get_const_field(feet),
243 get_const_field(
244 ddcHelper::get<X_adv>(advection_field_in_adv_domain_coefs)));
245 m_evaluator_advection_field(
246 get_field(ddcHelper::get<Y_adv>(updated_advection_field)),
247 get_const_field(feet),
248 get_const_field(
249 ddcHelper::get<Y_adv>(advection_field_in_adv_domain_coefs)));
250 };
251
252 IdxRangeRTheta const idx_range_rp = get_idx_range<GridR, GridTheta>(feet);
253
254 CoordXY_adv coord_centre(m_logical_to_pseudo_physical(CoordRTheta(0, 0)));
255 LogicalToPseudoPhysicalMapping logical_to_pseudo_physical_proxy
256 = m_logical_to_pseudo_physical;
257 PseudoPhysicalToLogicalMapping pseudo_physical_to_logical_proxy
258 = m_pseudo_physical_to_logical;
259
260 // The function describing how the value(s) are updated using the derivative.
261 std::function<void(FieldRTheta<CoordRTheta>, DVectorConstFieldRTheta<X_adv, Y_adv>, double)>
262 update_function = [&](FieldRTheta<CoordRTheta> feet,
263 DVectorConstFieldRTheta<X_adv, Y_adv> advection_field,
264 double dt) {
265 // Compute the characteristic feet at t^n:
266 ddc::parallel_for_each(
267 ExecSpace(),
268 idx_range_rp,
269 KOKKOS_LAMBDA(IdxRTheta const irtheta) {
270 CoordRTheta const coord_rtheta(feet(irtheta));
271 CoordXY_adv const coord_xy
272 = logical_to_pseudo_physical_proxy(coord_rtheta);
273
274 CoordXY_adv const feet_xy
275 = coord_xy - dt * advection_field(irtheta);
276
277 if (norm_inf(feet_xy - coord_centre) < 1e-15) {
278 feet(irtheta) = CoordRTheta(0, 0);
279 } else {
280 feet(irtheta) = pseudo_physical_to_logical_proxy(feet_xy);
281 ddc::select<Theta>(feet(irtheta))
282 = ddcHelper::restrict_to_idx_range(
283 ddc::select<Theta>(feet(irtheta)),
284 IdxRangeTheta(idx_range_rp));
285 }
286 });
287
288 // Treatment to conserve the C0 property of the advected function:
289 unify_value_at_centre_pt(feet);
290 // Test if the values are the same at the centre point
291 is_unified(feet);
292 };
293
294
295 // Solve the characteristic equation
296 m_time_stepper.update(ExecSpace(), feet, dt, dy, update_function);
297
298 is_unified(feet);
299 }

315 template <class T>
316 void is_unified(Field<T, IdxRangeRTheta, memory_space> const& values) const
317 {
318 IdxRangeR const r_idx_range = get_idx_range<GridR>(values);
319 IdxRangeTheta const theta_idx_range = get_idx_range<GridTheta>(values);
320 IdxR r0_idx = r_idx_range.front();
321 if (Kokkos::fabs(ddc::coordinate(r0_idx)) < 1e-15) {
322 ddc::parallel_for_each(
323 ExecSpace(),
324 theta_idx_range,
325 KOKKOS_LAMBDA(const IdxTheta itheta) {
326 if (norm_inf(
327 values(r0_idx, itheta)
328 - values(r0_idx, theta_idx_range.front()))
329 > 1e-15) {
330 Kokkos::printf("WARNING ! -> Discontinuous at the centre point.");
331 }
332 KOKKOS_ASSERT(
333 values(r0_idx, itheta) == values(r0_idx, theta_idx_range.front()));
334 });
335 }
336 }
