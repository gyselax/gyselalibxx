210  template <class SplineType, class Evaluator>
211  void compute_advection_field_XY(
212  Evaluator evaluator,
213  SplineType& electrostatic_potential_coef,
214  DVectorFieldRTheta<X, Y> advection_field_xy) const
215  {
216  static_assert(
217  (std::is_same_v<
218  Evaluator,
219  SplineRThetaEvaluatorNullBound> && std::is_same_v<SplineType, Spline2DField>)
220  || (std::is_same_v<
221  Evaluator,
222  PolarSplineEvaluator<
223  PolarBSplinesRTheta,
224  ddc::NullExtrapolationRule>> && std::is_same_v<SplineType, SplinePolar>));
225 
226  IdxRangeRTheta const grid = get_idx_range(advection_field_xy);
227  DVectorFieldMemRTheta<X, Y> electric_field(grid);
228 
229  FieldMemRTheta<CoordRTheta> coords(grid);
230  ddc::for_each(grid, [&](IdxRTheta const irp) { coords(irp) = ddc::coordinate(irp); });
231 
232  // > computation of the phi derivatives
233  DFieldMemRTheta deriv_r_phi(grid);
234  DFieldMemRTheta deriv_p_phi(grid);
235 
236  evaluator.deriv_dim_1(
237  get_field(deriv_r_phi),
238  get_const_field(coords),
239  get_const_field(electrostatic_potential_coef));
240  evaluator.deriv_dim_2(
241  get_field(deriv_p_phi),
242  get_const_field(coords),
243  get_const_field(electrostatic_potential_coef));
244 
245  // > computation of the electric field
246  ddc::for_each(grid, [&](IdxRTheta const irp) {
247  double const r = ddc::coordinate(ddc::select<GridR>(irp));
248  double const th = ddc::coordinate(ddc::select<GridTheta>(irp));
249 
250  if (r > m_epsilon) {
251  CoordRTheta const coord_rp(r, th);
252 
253  Matrix_2x2 inv_J; // Inverse Jacobian matrix
254  m_mapping.inv_jacobian_matrix(coord_rp, inv_J);
255 
256  // Gradiant of phi in the physical index range (Cartesian index range)
257  double const deriv_x_phi
258  = deriv_r_phi(irp) * inv_J[0][0] + deriv_p_phi(irp) * inv_J[1][0];
259  double const deriv_y_phi
260  = deriv_r_phi(irp) * inv_J[0][1] + deriv_p_phi(irp) * inv_J[1][1];
261 
262  // E = -grad phi
263  ddcHelper::get<X>(electric_field)(irp) = -deriv_x_phi;
264  ddcHelper::get<Y>(electric_field)(irp) = -deriv_y_phi;
265 
266  } else {
267  // Linearisation of the electric field
268  double const th1 = M_PI / 4.;
269  double const th2 = -M_PI / 4. + 2 * M_PI;
270 
271  // --- Value at r = 0:
272  CoordRTheta const coord_1_0(0, th1);
273  CoordRTheta const coord_2_0(0, th2);
274 
275  double const dr_x_1 = m_mapping.jacobian_11(coord_1_0); // dr_x (0, th1)
276  double const dr_y_1 = m_mapping.jacobian_21(coord_1_0); // dr_y (0, th1)
277 
278  double const dr_x_2 = m_mapping.jacobian_11(coord_2_0); // dr_x (0, th2)
279  double const dr_y_2 = m_mapping.jacobian_21(coord_2_0); // dr_y (0, th2)
280 
281  double deriv_r_phi_1 = evaluator.deriv_dim_1(
282  coord_1_0,
283  get_const_field(electrostatic_potential_coef));
284  double deriv_r_phi_2 = evaluator.deriv_dim_1(
285  coord_2_0,
286  get_const_field(electrostatic_potential_coef));
287 
288  double const determinant = dr_x_1 * dr_y_2 - dr_x_2 * dr_y_1;
289 
290  double const deriv_x_phi_0
291  = (dr_y_2 * deriv_r_phi_1 - dr_y_1 * deriv_r_phi_2) / determinant;
292  double const deriv_y_phi_0
293  = (-dr_x_2 * deriv_r_phi_1 + dr_x_1 * deriv_r_phi_2) / determinant;
294 
295  // E = -grad phi
296  double const electric_field_x_0 = -deriv_x_phi_0;
297  double const electric_field_y_0 = -deriv_y_phi_0;
298 
299 
300 
301  // --- Value at r = m_epsilon:
302  CoordRTheta const coord_rp_epsilon(m_epsilon, th);
303 
304  Matrix_2x2 inv_J_eps; // Jacobian matrix
305  m_mapping.inv_jacobian_matrix(coord_rp_epsilon, inv_J_eps);
306 
307  double const deriv_r_phi_epsilon = evaluator.deriv_dim_1(
308  coord_rp_epsilon,
309  get_const_field(electrostatic_potential_coef));
310  double const deriv_p_phi_epsilon = evaluator.deriv_dim_2(
311  coord_rp_epsilon,
312  get_const_field(electrostatic_potential_coef));
313 
314  // Gradiant of phi in the physical index range (Cartesian index range)
315  double const deriv_x_phi_epsilon = deriv_r_phi_epsilon * inv_J_eps[0][0]
316  + deriv_p_phi_epsilon * inv_J_eps[1][0];
317  double const deriv_y_phi_epsilon = deriv_r_phi_epsilon * inv_J_eps[0][1]
318  + deriv_p_phi_epsilon * inv_J_eps[1][1];
319 
320  // E = -grad phi
321  double const electric_field_x_epsilon = -deriv_x_phi_epsilon;
322  double const electric_field_y_epsilon = -deriv_y_phi_epsilon;
323 
324 
325  // --- Linearisation:
326  ddcHelper::get<X>(electric_field)(irp) = electric_field_x_0 * (1 - r / m_epsilon)
327  + electric_field_x_epsilon * r / m_epsilon;
328  ddcHelper::get<Y>(electric_field)(irp) = electric_field_y_0 * (1 - r / m_epsilon)
329  + electric_field_y_epsilon * r / m_epsilon;
330  }
331 
332  // > computation of the advection field
333  ddcHelper::get<X>(advection_field_xy)(irp) = -ddcHelper::get<Y>(electric_field)(irp);
334  ddcHelper::get<Y>(advection_field_xy)(irp) = ddcHelper::get<X>(electric_field)(irp);
335  });
336  }

440  void compute_advection_field_RTheta(
441  Evaluator evaluator,
442  SplineType& electrostatic_potential_coef,
443  DVectorFieldRTheta<R, Theta> advection_field_rp,
444  CoordXY& advection_field_xy_center) const

446  static_assert(
447  (std::is_same_v<
448  Evaluator,

450  || (std::is_same_v<
451  Evaluator,
452  PolarSplineEvaluator<
453  PolarBSplinesRTheta,
454  ddc::NullExtrapolationRule>> && std::is_same_v<SplineType, SplinePolar>));
455 
456  IdxRangeRTheta const grid_without_Opoint = get_idx_range(advection_field_rp);
457 
458  FieldMemRTheta<CoordRTheta> coords(grid_without_Opoint);
459  ddc::for_each(grid_without_Opoint, [&](IdxRTheta const irp) {
460  coords(irp) = ddc::coordinate(irp);
461  });
462 
463  // > computation of the phi derivatives
464  DFieldMemRTheta deriv_r_phi(grid_without_Opoint);
465  DFieldMemRTheta deriv_p_phi(grid_without_Opoint);
466 
467  evaluator.deriv_dim_1(
468  get_field(deriv_r_phi),
469  get_const_field(coords),
470  get_const_field(electrostatic_potential_coef));
471  evaluator.deriv_dim_2(
472  get_field(deriv_p_phi),
473  get_const_field(coords),
474  get_const_field(electrostatic_potential_coef));
475 
476  // > computation of the advection field
477  ddc::for_each(grid_without_Opoint, [&](IdxRTheta const irp) {
478  CoordRTheta const coord_rp(ddc::coordinate(irp));
479 
480  Matrix_2x2 J; // Jacobian matrix
481  m_mapping.jacobian_matrix(coord_rp, J);
482  Matrix_2x2 inv_G; // Inverse metric tensor
483  m_mapping.inverse_metric_tensor(coord_rp, inv_G);
484  Matrix_2x2 G; // Metric tensor
485  m_mapping.metric_tensor(coord_rp, G);
486 
487  // E = -grad phi
488  double const electric_field_r
489  = (-deriv_r_phi(irp) * inv_G[0][0] - deriv_p_phi(irp) * inv_G[1][0])
490  * std::sqrt(G[0][0]);
491  double const electric_field_p
492  = (-deriv_r_phi(irp) * inv_G[0][1] - deriv_p_phi(irp) * inv_G[1][1])
493  * std::sqrt(G[1][1]);
494 
495  // A = E \wedge e_z
496  ddcHelper::get<R>(advection_field_rp)(irp) = -electric_field_p;
497  ddcHelper::get<Theta>(advection_field_rp)(irp) = electric_field_r;
498  });
499 
500  // SPECIAL TREATMENT FOR THE O-POINT =====================================================
501  // Linearisation of the electric field
502  double const th1 = M_PI / 4.;
503  double const th2 = -M_PI / 4. + 2 * M_PI;
504 
505  // --- Value at r = 0:
506  CoordRTheta const coord_1_0(0, th1);
507  CoordRTheta const coord_2_0(0, th2);
508 
509  double const dr_x_1 = m_mapping.jacobian_11(coord_1_0); // dr_x (0, th1)
510  double const dr_y_1 = m_mapping.jacobian_21(coord_1_0); // dr_y (0, th1)
511 
512  double const dr_x_2 = m_mapping.jacobian_11(coord_2_0); // dr_x (0, th2)
513  double const dr_y_2 = m_mapping.jacobian_21(coord_2_0); // dr_y (0, th2)
514 
515  double const deriv_r_phi_1
516  = evaluator.deriv_dim_1(coord_1_0, get_const_field(electrostatic_potential_coef));
517  double const deriv_r_phi_2
518  = evaluator.deriv_dim_1(coord_2_0, get_const_field(electrostatic_potential_coef));
519 
520  double const determinant = dr_x_1 * dr_y_2 - dr_x_2 * dr_y_1;
521 
522  double const deriv_x_phi_0
523  = (dr_y_2 * deriv_r_phi_1 - dr_y_1 * deriv_r_phi_2) / determinant;
524  double const deriv_y_phi_0
525  = (-dr_x_2 * deriv_r_phi_1 + dr_x_1 * deriv_r_phi_2) / determinant;
526 
527  advection_field_xy_center = CoordXY(deriv_y_phi_0, -deriv_x_phi_0);
528  }
