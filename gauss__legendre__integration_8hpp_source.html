130 template <class ExecSpace>
131 DFieldMem<IdxRange<GLGrid>, typename ExecSpace::memory_space> gauss_legendre_coefficients()
132 const
133 {
134 DFieldMem<IdxRange<GLGrid>, typename ExecSpace::memory_space> coefficients_alloc(
135 m_valid_idx_range);
136 auto coefficients_host = ddc::create_mirror_view(get_field(coefficients_alloc));
137 ddc::for_each(m_valid_idx_range, [&](Idx<GLGrid> ix) {
138 int i = (ix - m_valid_idx_range.front()) / NPoints;
139 int j = (ix - m_valid_idx_range.front()) % NPoints;
140 coefficients_host(ix) = m_cell_lengths[i] * m_glc.weight[j];
141 });
142 ddc::parallel_deepcopy(get_field(coefficients_alloc), coefficients_host);
143 return coefficients_alloc;
144 }

146private:
147 void get_sampling_on_cell(
148 std::vector<Coord<Dim>>& grid,
149 Coord<Dim> x0,
150 Coord<Dim> x1,
151 int i,
152 int& k)
153 {
154 double const l = 0.5 * (ddc::get<Dim>(x1) - ddc::get<Dim>(x0));
155 m_cell_lengths[i] = l;
156 Coord<Dim> const c(0.5 * (x0 + x1));
157 for (int j(0); j < NPoints; ++k, ++j) {
158 grid[k] = c + double(l * m_glc.pos[j]);
159 }
160 }
