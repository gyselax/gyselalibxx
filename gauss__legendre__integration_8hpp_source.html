135 template <class ExecSpace>
136 DFieldMem<IdxRange<GLGrid>, typename ExecSpace::memory_space> gauss_legendre_coefficients()
137 const
138 {
139 DFieldMem<IdxRange<GLGrid>, typename ExecSpace::memory_space> coefficients_alloc(
140 m_valid_idx_range);
141 auto coefficients_host = ddc::create_mirror_view(get_field(coefficients_alloc));
142 ddc::for_each(m_valid_idx_range, [&](Idx<GLGrid> ix) {
143 int i = (ix - m_valid_idx_range.front()) / NPoints;
144 int j = (ix - m_valid_idx_range.front()) % NPoints;
145 coefficients_host(ix) = m_cell_lengths[i] * m_glc.weight[j];
146 });
147 ddc::parallel_deepcopy(get_field(coefficients_alloc), coefficients_host);
148 return coefficients_alloc;
149 }
150
151private:
152 void get_sampling_on_cell(
153 std::vector<Coord<Dim>>& grid,
154 Coord<Dim> x0,
155 Coord<Dim> x1,
156 int i,
157 int& k)
158 {
159 double const l = 0.5 * (ddc::get<Dim>(x1) - ddc::get<Dim>(x0));
160 m_cell_lengths[i] = l;
161 Coord<Dim> const c(0.5 * (x0 + x1));
162 for (int j(0); j < NPoints; ++k, ++j) {
163 grid[k] = c + double(l * m_glc.pos[j]);
164 }
165 }

167 template <class InputIt>
168 std::vector<Coord<Dim>> get_sampling(InputIt mesh_edges_begin, InputIt mesh_edges_end)
169 {
170 std::vector<Coord<Dim>> grid(m_nbcells * NPoints);
171
172 Coord<Dim> x0;
173 Coord<Dim> x1(*mesh_edges_begin);
174
175 int k(0);
176 for (int i(0); i < m_nbcells; ++i) {
177 x0 = x1;
178 x1 = *(++mesh_edges_begin);
179 get_sampling_on_cell(grid, x0, x1, i, k);
180 }
181 return grid;
182 }
183
184 template <class Grid1D>
185 std::vector<Coord<Dim>> get_sampling(
186 ConstField<Coord<Dim>, IdxRange<Grid1D>, Kokkos::HostSpace> mesh_edges)
187 {
188 std::vector<Coord<Dim>> grid(m_nbcells * NPoints);
189
190 IdxRange<Grid1D> idx_range = ::get_idx_range(mesh_edges);
191 Idx<Grid1D> mesh_idx(idx_range.front());
192
193 Coord<Dim> x0;
194 Coord<Dim> x1 = mesh_edges(mesh_idx);
195
196 int k(0);
197 for (int i(0); i < m_nbcells; ++i) {
198 x0 = x1;
199 x1 = mesh_edges(++mesh_idx);
200 get_sampling_on_cell(grid, x0, x1, i, k);
201 }
202 return grid;
203 }

216template <class ExecSpace, class... GaussLegendreQuad>
217DFieldMem<IdxRange<typename GaussLegendreQuad::Grid1D...>, typename ExecSpace::memory_space>
218gauss_legendre_quadrature_coefficients(GaussLegendreQuad const&... gl)
219{
220 // Get coefficients for each dimension
221 std::tuple<host_t<DFieldMem<IdxRange<typename GaussLegendreQuad::Grid1D>>>...>
222 current_dim_coeffs(gl.template gauss_legendre_coefficients<Kokkos::HostSpace>()...);
223
224 IdxRange<typename GaussLegendreQuad::Grid1D...> idx_range(gl.get_idx_range()...);
225
226 // Allocate ND coefficients
227 DFieldMem<IdxRange<typename GaussLegendreQuad::Grid1D...>, typename ExecSpace::memory_space>
228 coefficients(idx_range);
229 auto coefficients_host = ddc::create_mirror(get_field(coefficients));
230 // Serial loop is used due to nvcc bug concerning functions with variadic template arguments
231 // (see https://github.com/kokkos/kokkos/pull/7059)
232 ddc::for_each(idx_range, [&](Idx<typename GaussLegendreQuad::Grid1D...> const idim) {
233 // multiply the 1D coefficients by one another
234 coefficients_host(idim)
235 = (std::get<host_t<DFieldMem<IdxRange<typename GaussLegendreQuad::Grid1D>>>>(
236 current_dim_coeffs)(
237 ddc::select<typename GaussLegendreQuad::Grid1D>(idim))
238 * ... * 1);
239 });
240 ddc::parallel_deepcopy(coefficients, coefficients_host);
241 return coefficients;
242}
