145 double dt,
146 std::function<void(DerivFieldMem, ValConstField)> dy,
147 std::function<void(ValField, DerivConstField, double)> y_update) const
148 {
149 static_assert(
150 Kokkos::SpaceAccessibility<ExecSpace, typename FieldMemType::memory_space>::
151 accessible,
152 "MemorySpace has to be accessible for ExecutionSpace.");
153 static_assert(
154 Kokkos::SpaceAccessibility<ExecSpace, typename DerivFieldMemType::memory_space>::
155 accessible,
156 "MemorySpace has to be accessible for ExecutionSpace.");
157 FieldMemType m_y_prime_alloc(m_idx_range);
158 DerivFieldMemType m_k1_alloc(m_idx_range);
159 DerivFieldMemType m_k2_alloc(m_idx_range);
160 DerivFieldMemType m_k3_alloc(m_idx_range);
161 DerivFieldMemType m_k4_alloc(m_idx_range);
162 DerivFieldMemType m_k_total_alloc(m_idx_range);
163 ValField m_y_prime = get_field(m_y_prime_alloc);
164 DerivFieldMem m_k1 = get_field(m_k1_alloc);
165 DerivFieldMem m_k2 = get_field(m_k2_alloc);
166 DerivFieldMem m_k3 = get_field(m_k3_alloc);
167 DerivFieldMem m_k4 = get_field(m_k4_alloc);
168 DerivFieldMem m_k_total = get_field(m_k_total_alloc);
169
170
171 // Save initial conditions
172 copy(m_y_prime, y);
173
174 // --------- Calculate k1 ------------
175 // k1 = f(y)
176 dy(m_k1, y);
177
178 // --------- Calculate k2 ------------
179 // Calculate y_new := y_n + h/2*k_1
180 y_update(m_y_prime, m_k1, 0.5 * dt);
181
182 // Calculate k2 = f(y_new)
183 dy(m_k2, m_y_prime);
184
185 // --------- Calculate k3 ------------
186 // Collect initial conditions
187 copy(m_y_prime, y);
188
189 // Calculate y_new := y_n + h/2*k_2
190 y_update(m_y_prime, m_k2, 0.5 * dt);
191
192 // Calculate k3 = f(y_new)
193 dy(m_k3, m_y_prime);
194
195 // --------- Calculate k3 ------------
196 // Collect initial conditions
197 copy(m_y_prime, y);
198
199 // Calculate y_new := y_n + h*k_3
200 y_update(m_y_prime, m_k3, dt);
201
202 // Calculate k4 = f(y_new)
203 dy(m_k4, m_y_prime);
204
205 // --------- Update y ------------
206 // Calculation of step
207 if constexpr (is_field_v<DerivFieldMemType>) {
208 ddc::parallel_for_each(
209 exec_space,
210 get_idx_range(m_k_total),
211 KOKKOS_CLASS_LAMBDA(Idx const i) {
212 // k_total = k1 + 4 * k2 + k3
213 fill_k_total(i, m_k_total, m_k1(i) + 2 * m_k2(i) + 2 * m_k3(i) + m_k4(i));
214 });
215 } else {
216 ddc::parallel_for_each(
217 exec_space,
218 get_idx_range(m_k_total),
219 KOKKOS_LAMBDA(Idx const i) {
220 // k_total = k1 + 4 * k2 + k3
221 m_k_total(i) = m_k1(i) + 2 * m_k2(i) + 2 * m_k3(i) + m_k4(i);
222 });
223 }
224
225 // Calculate y_{n+1} := y_n + (k1 + 2 * k2 + 2 * k3 + k4) * h/6
226 y_update(y, m_k_total, dt / 6.);
227 }
