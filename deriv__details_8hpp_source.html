112template <
113 template <typename...>
114 class Container,
115 class HeadQueryTag,
116 class... QueryTags,
117 class... Tags>
118KOKKOS_FUNCTION Container<HeadQueryTag, QueryTags...> select_default(
119 Container<Tags...> const& known_values,
120 Container<HeadQueryTag, QueryTags...> const& default_values)
121{
122 if constexpr (sizeof...(QueryTags) == 0) {
123 if constexpr (ddc::in_tags_v<HeadQueryTag, ddc::detail::TypeSeq<Tags...>>) {
124 return ddc::select<HeadQueryTag>(known_values);
125 } else {
126 return ddc::select<HeadQueryTag>(default_values);
127 }
128 } else {
129 return Container<HeadQueryTag, QueryTags...>(
130 select_default(known_values, ddc::select<HeadQueryTag>(default_values)),
131 select_default(known_values, ddc::select<QueryTags...>(default_values)));
132 }
133}
