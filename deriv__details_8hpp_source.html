111 template <
112  template <typename...>
113  class Container,
114  class HeadQueryTag,
115  class... QueryTags,
116  class... Tags>
117 KOKKOS_FUNCTION Container<HeadQueryTag, QueryTags...> select_default(
118  Container<Tags...> const& known_values,
119  Container<HeadQueryTag, QueryTags...> const& default_values)
120 {
121  if constexpr (sizeof...(QueryTags) == 0) {
122  if constexpr (ddc::in_tags_v<HeadQueryTag, ddc::detail::TypeSeq<Tags...>>) {
123  return ddc::select<HeadQueryTag>(known_values);
124  } else {
125  return ddc::select<HeadQueryTag>(default_values);
126  }
127  } else {
128  return Container<HeadQueryTag, QueryTags...>(
129  select_default(known_values, ddc::select<HeadQueryTag>(default_values)),
130  select_default(known_values, ddc::select<QueryTags...>(default_values)));
131  }
132 }
