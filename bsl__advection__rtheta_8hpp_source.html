123 DFieldRTheta operator()(
124 DFieldRTheta allfdistribu,
125 DConstVectorFieldRTheta<X, Y> advection_field_xy,
126 double dt) const override
127 {
128 // Pre-allocate some memory to prevent allocation later in loop
129 std::unique_ptr<IInterpolator2D<IdxRangeRTheta, IdxRangeRTheta>> const interpolator_ptr
130 = m_interpolator.preallocate();
131
132 // Initialise the feet
133 FieldMemRTheta<CoordRTheta> feet_rtheta_alloc(get_idx_range(advection_field_xy));
134 FieldRTheta<CoordRTheta> feet_rtheta = get_field(feet_rtheta_alloc);
135 ddc::parallel_for_each(
136 Kokkos::DefaultExecutionSpace(),
137 get_idx_range(advection_field_xy),
138 KOKKOS_LAMBDA(IdxRTheta const irtheta) {
139 feet_rtheta(irtheta) = ddc::coordinate(irtheta);
140 });
141
142 // Compute the feet of the characteristics at tn -----------------------------------------
143 m_find_feet(feet_rtheta, get_const_field(advection_field_xy), dt);
144
145 // Interpolate the function on the feet of the characteristics. --------------------------
146 (*interpolator_ptr)(get_field(allfdistribu), get_const_field(feet_rtheta));
147
148 return allfdistribu;
149 }

169 DFieldRTheta operator()(
170 DFieldRTheta allfdistribu,
171 DConstVectorFieldRTheta<R, Theta> advection_field_rtheta,
172 CoordXY const& advection_field_xy_centre,
173 double dt) const override
174 {
175 Kokkos::Profiling::pushRegion("PolarAdvection");
176 IdxRangeRTheta grid(get_idx_range<GridR, GridTheta>(allfdistribu));
177
178 const int npoints_theta = IdxRangeTheta(grid).size();
179 IdxRangeRTheta const grid_without_Opoint(grid.remove_first(IdxStepRTheta(1, 0)));
180 IdxRangeRTheta const Opoint_grid(grid.take_first(IdxStepRTheta(1, npoints_theta)));
181
182
183 // Convert advection field on RTheta to advection field on XY
184 DVectorFieldMemRTheta<X, Y> advection_field_xy_alloc(grid);
185 DVectorFieldRTheta<X, Y> advection_field_xy = get_field(advection_field_xy_alloc);
186
187 Mapping const& mapping_proxy = m_mapping;
188
189 // (Ax, Ay) = J (Ar, Atheta)
190 ddc::parallel_for_each(
191 Kokkos::DefaultExecutionSpace(),
192 grid_without_Opoint,
193 KOKKOS_LAMBDA(IdxRTheta const irtheta) {
194 CoordRTheta const coord_rtheta(ddc::coordinate(irtheta));
195
196 Tensor J = mapping_proxy.jacobian_matrix(coord_rtheta);
197
198 DVector<X, Y> advec_field_xy = tensor_mul(
199 index<'i', 'j'>(J),
200 index<'j'>(advection_field_rtheta(irtheta)));
201 ddcHelper::get<X>(advection_field_xy)(irtheta)
202 = ddcHelper::get<X>(advec_field_xy);
203 ddcHelper::get<Y>(advection_field_xy)(irtheta)
204 = ddcHelper::get<Y>(advec_field_xy);
205 });
206
207 ddc::parallel_for_each(
208 Kokkos::DefaultExecutionSpace(),
209 Opoint_grid,
210 KOKKOS_LAMBDA(IdxRTheta const irtheta) {
211 ddcHelper::get<X>(advection_field_xy)(irtheta)
212 = CoordX(advection_field_xy_centre);
213 ddcHelper::get<Y>(advection_field_xy)(irtheta)
214 = CoordY(advection_field_xy_centre);
215 });
216
217 (*this)(get_field(allfdistribu), get_const_field(advection_field_xy), dt);
218
219 Kokkos::Profiling::popRegion();
220
221 return allfdistribu;
222 }
