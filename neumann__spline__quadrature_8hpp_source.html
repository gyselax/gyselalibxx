52 template <class ExecSpace, class Grid1D, class SplineBuilder>
53 DFieldMem<IdxRange<Grid1D>, typename ExecSpace::memory_space>
54 neumann_spline_quadrature_coefficients_1d(
55  IdxRange<Grid1D> const& idx_range,
56  SplineBuilder const& builder)
57 {
58  constexpr int nbc_xmin = SplineBuilder::s_nbc_xmin;
59  constexpr int nbc_xmax = SplineBuilder::s_nbc_xmax;
60  static_assert(
61  SplineBuilder::s_bc_xmin == ddc::BoundCond::HERMITE,
62  "The neumann spline quadrature requires a builder which uses Hermite boundary "
63  "conditions.");
64  static_assert(
65  SplineBuilder::s_bc_xmax == ddc::BoundCond::HERMITE,
66  "The neumann spline quadrature requires a builder which uses Hermite boundary "
67  "conditions.");
68  static_assert(
69  nbc_xmin == 1,
70  "The neumann spline quadrature requires a builder which uses the value of the "
71  "derivative.");
72  static_assert(
73  nbc_xmax == 1,
74  "The neumann spline quadrature requires a builder which uses the value of the "
75  "derivative.");
76  assert(idx_range.size()
77  == ddc::discrete_space<typename SplineBuilder::bsplines_type>().nbasis() - nbc_xmin
78  - nbc_xmax);
79 
80  DFieldMem<IdxRange<Grid1D>, typename SplineBuilder::memory_space> quadrature_coefficients(
81  builder.interpolation_domain());
82  // Even if derivatives coefficients on boundaries are eventually non-zero,
83  // they are ignored for 0-flux Neumann boundary condition because
84  // they would always be multiplied by f'(x)=0
85  std::tie(std::ignore, quadrature_coefficients, std::ignore) = builder.quadrature_coefficients();
86  DFieldMem<IdxRange<Grid1D>, typename ExecSpace::memory_space> output_quad_coefficients(
87  idx_range);
88  ddc::parallel_deepcopy(output_quad_coefficients, quadrature_coefficients[idx_range]);
89  return output_quad_coefficients;

108 template <class ExecSpace, class... DDims, class... SplineBuilders>
109 DFieldMem<IdxRange<DDims...>, typename ExecSpace::memory_space>
110 neumann_spline_quadrature_coefficients(
111  IdxRange<DDims...> const& idx_range,
112  SplineBuilders const&... builders)
113 {
114  assert((std::is_same_v<
115  typename DDims::continuous_dimension_type,
116  typename SplineBuilders::continuous_dimension_type> and ...));
117 
118  // Get coefficients for each dimension
119  std::tuple<CoefficientFieldMem1D<Kokkos::DefaultHostExecutionSpace, DDims>...>
120  current_dim_coeffs_alloc(
121  neumann_spline_quadrature_coefficients_1d<
122  Kokkos::DefaultHostExecutionSpace>(ddc::select<DDims>(idx_range), builders)...);
123  std::tuple<CoefficientField1D<Kokkos::DefaultHostExecutionSpace, DDims>...> current_dim_coeffs(
124  get_field(std::get<CoefficientFieldMem1D<Kokkos::DefaultHostExecutionSpace, DDims>>(
125  current_dim_coeffs_alloc))...);
126  // Allocate ND coefficients
127  DFieldMem<IdxRange<DDims...>, typename ExecSpace::memory_space> coefficients_alloc(idx_range);
128  auto coefficients_alloc_host = ddc::create_mirror(get_field(coefficients_alloc));
129  host_t<DField<IdxRange<DDims...>>> coefficients(get_field(coefficients_alloc_host));
130  // Serial loop is used due to nvcc bug concerning functions with variadic template arguments
131  // (see https://github.com/kokkos/kokkos/pull/7059)
132  ddc::for_each(idx_range, [&](Idx<DDims...> const idim) {
133  // multiply the 1D coefficients by one another
134  coefficients(idim)
135  = (std::get<CoefficientField1D<Kokkos::DefaultHostExecutionSpace, DDims>>(
136  current_dim_coeffs)(ddc::select<DDims>(idim))
137  * ... * 1);
138  });
139  ddc::parallel_deepcopy(coefficients_alloc, coefficients_alloc_host);
140  return coefficients_alloc;
141 }
