92 template <typename Execspace, class DDimI, BCond BcMin, BCond BcMax>
93 ddc::DiscreteElement<DDimI> Lagrange<Execspace, DDimI, BcMin, BcMax>::getclosest(
94  CoordDimI value) const
95 {
96  assert(value >= m_left_bound && value <= m_right_bound);
97  auto it = std::min_element(m_domain.begin(), m_domain.end(), [=](DElemI x, DElemI y) {
98  return std::abs(ddc::coordinate(x) - value) < std::abs(ddc::coordinate(y) - value);
99  });
100  return *it;
101 }
102 template <typename Execspace, class DDimI, BCond BcMin, BCond BcMax>
103 KOKKOS_INLINE_FUNCTION ddc::DiscreteElement<DDimI> Lagrange<Execspace, DDimI, BcMin, BcMax>::
104  getclosest_binsearch(CoordDimI x_interp) const
105 {
106  assert(x_interp >= m_left_bound && x_interp <= m_right_bound);
107  DElemI begin = m_domain.front();
108  DElemI end = m_domain.back();
109  DElemI elm_cell = begin + (end - begin) / 2;
110  while (x_interp < ddc::coordinate(elm_cell)
111  || x_interp > ddc::coordinate(elm_cell + DVectI(1))) {
112  if (x_interp < ddc::coordinate(elm_cell)) {
113  end = elm_cell;
114  } else {
115  begin = elm_cell;
116  }
117 
118  elm_cell = begin + (end - begin) / 2;
119  }
120  return elm_cell;
121 }

132 template <typename Execspace, class DDimI, BCond BcMin, BCond BcMax>
133 KOKKOS_INLINE_FUNCTION double Lagrange<Execspace, DDimI, BcMin, BcMax>::compute_basis(
134  CoordDimI x_interp,
135  DElemI j,
136  ddc::DiscreteDomain<DDimI> polynom_subdomain) const
137 {
138  double w(1);
139  CoordDimI coord_j = ddc::coordinate(j);
140  for (DElemI const i : polynom_subdomain) {
141  CoordDimI coord_i = ddc::coordinate(i);
142  if (coord_i != coord_j) {
143  w *= (x_interp - coord_i) / (coord_j - coord_i);
144  }
145  }
146  return w;
147 }

149 template <typename Execspace, class DDimI, BCond BcMin, BCond BcMax>
150 KOKKOS_INLINE_FUNCTION double Lagrange<Execspace, DDimI, BcMin, BcMax>::apply_bc(
151  CoordDimI x_interp) const
152 {
153  CoordDimI bc_val = x_interp;
154  const double d = m_right_bound - m_left_bound;
155  if constexpr (BcMin == BCond::PERIODIC) {
156  bc_val -= std::floor((x_interp - m_left_bound) / d) * d;
157  } else {
158  if (x_interp < m_left_bound && BcMin == BCond::DIRICHLET) {
159  bc_val = m_left_bound;
160  } else if (x_interp > m_right_bound && BcMax == BCond::DIRICHLET) {
161  bc_val = m_right_bound;
162  }
163  }
164  return evaluate_lagrange(bc_val);
165 }

178 template <typename Execspace, class DDimI, BCond BcMin, BCond BcMax>
179 KOKKOS_INLINE_FUNCTION double Lagrange<Execspace, DDimI, BcMin, BcMax>::evaluate_lagrange(
180  CoordDimI x_intern) const
181 {
182  assert(x_intern >= m_left_bound && x_intern <= m_right_bound);
183 
184  DElemI begin, end;
185  DElemI icell = getclosest_binsearch(x_intern);
186  DElemI mid = icell;
187  if (mid >= m_inner_domain.back() && BcMax == BCond::PERIODIC) {
188  begin = mid - m_poly_support / 2;
189  end = std::min(m_inner_domain.back(), begin + m_poly_support);
190  } else if (mid <= m_inner_domain.front() && BcMin == BCond::PERIODIC) {
191  begin = std::max(m_domain.front(), mid - m_poly_support / 2);
192  end = begin + m_poly_support;
193  } else {
194  if (m_inner_domain.front() + m_poly_support / 2 > mid) {
195  begin = m_inner_domain.front();
196  } else {
197  begin = mid - m_poly_support / 2;
198  }
199  end = std::min(m_domain.back() + DVectI(1), begin + m_poly_support);
200  }
201 
202  if (end == m_domain.back())
203  begin = end - m_poly_support;
204  DVectI npoints_subdomain(end - begin);
205  ddc::DiscreteDomain<DDimI> subdomain(begin, npoints_subdomain);
206  double p = 0;
207  for (DElemI const ix : subdomain) {
208  p += compute_basis(x_intern, ix, subdomain) * m_ChunkSpan(ix);
209  }
210  return p;
211 }
