355 void compute_advection_field(
356 host_t<DConstVectorFieldRTheta<X, Y>> advection_field,
357 host_t<DVectorFieldRTheta<X_adv, Y_adv>> advection_field_pseudo_Cart) const
358 {
359 static_assert(!std::is_same_v<Mapping, CircularToCartesian<X, Y, R, Theta>>);
360
361 IdxRangeRTheta const idx_range_rp = get_idx_range(advection_field);
362 CircularToCartesian<X_adv, Y_adv, R, Theta> const pseudo_Cartesian_mapping;
363
364 ddc::for_each(idx_range_rp, [&](IdxRTheta const irp) {
365 CoordRTheta const coord_rp(ddc::coordinate(irp));
366 double const r(ddc::get<R>(coord_rp));
367 double const th(ddc::get<Theta>(coord_rp));
368
369 Matrix2x2 J;
370
371 if (r > m_epsilon) {
372 Matrix2x2 pseudo_Cart_J;
373 Matrix2x2 map_J;
374 pseudo_Cartesian_mapping.jacobian_matrix(coord_rp, pseudo_Cart_J);
375 m_mapping.inv_jacobian_matrix(coord_rp, map_J);
376
377 J[0][0] = pseudo_Cart_J[0][0] * map_J[0][0] + pseudo_Cart_J[0][1] * map_J[1][0];
378 J[0][1] = pseudo_Cart_J[0][0] * map_J[0][1] + pseudo_Cart_J[0][1] * map_J[1][1];
379 J[1][0] = pseudo_Cart_J[1][0] * map_J[0][0] + pseudo_Cart_J[1][1] * map_J[1][0];
380 J[1][1] = pseudo_Cart_J[1][0] * map_J[0][1] + pseudo_Cart_J[1][1] * map_J[1][1];
381
382 } else {
383 Matrix2x2 J_0;
384 m_mapping.to_pseudo_cartesian_jacobian_center_matrix(J_0);
385
386 CoordRTheta const coord_rp_epsilon(m_epsilon, th);
387
388 Matrix2x2 pseudo_Cart_J_epsilon;
389 Matrix2x2 map_J_epsilon;
390 pseudo_Cartesian_mapping.jacobian_matrix(coord_rp_epsilon, pseudo_Cart_J_epsilon);
391 m_mapping.jacobian_matrix(coord_rp_epsilon, map_J_epsilon);
392
393 Matrix2x2 J_eps;
394 J_eps[0][0] = pseudo_Cart_J_epsilon[0][0] * map_J_epsilon[1][1]
395 - pseudo_Cart_J_epsilon[0][1] * map_J_epsilon[1][0];
396
397 J_eps[0][1] = pseudo_Cart_J_epsilon[0][1] * map_J_epsilon[0][0]
398 - pseudo_Cart_J_epsilon[0][0] * map_J_epsilon[0][1];
399
400 J_eps[1][0] = pseudo_Cart_J_epsilon[1][0] * map_J_epsilon[1][1]
401 - pseudo_Cart_J_epsilon[1][1] * map_J_epsilon[1][0];
402
403 J_eps[1][1] = pseudo_Cart_J_epsilon[1][1] * map_J_epsilon[0][0]
404 - pseudo_Cart_J_epsilon[1][0] * map_J_epsilon[0][1];
405
406 J[0][0] = (1 - r / m_epsilon) * J_0[0][0] + r / m_epsilon * J_eps[0][0];
407 J[0][1] = (1 - r / m_epsilon) * J_0[0][1] + r / m_epsilon * J_eps[0][1];
408 J[1][0] = (1 - r / m_epsilon) * J_0[1][0] + r / m_epsilon * J_eps[1][0];
409 J[1][1] = (1 - r / m_epsilon) * J_0[1][1] + r / m_epsilon * J_eps[1][1];
410 }
411
412 ddcHelper::get<X_adv>(advection_field_pseudo_Cart)(irp)
413 = ddcHelper::get<X>(advection_field)(irp) * J[0][0]
414 + ddcHelper::get<Y>(advection_field)(irp) * J[0][1];
415 ddcHelper::get<Y_adv>(advection_field_pseudo_Cart)(irp)
416 = ddcHelper::get<X>(advection_field)(irp) * J[1][0]
417 + ddcHelper::get<Y>(advection_field)(irp) * J[1][1];
418 });
419 }
