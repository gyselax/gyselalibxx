357  void compute_advection_field(
358  DConstVectorFieldRTheta<X, Y> advection_field,
359  DVectorFieldRTheta<X_adv, Y_adv> advection_field_pseudo_Cart) const
360  {
361  static_assert(!std::is_same_v<Mapping, CircularToCartesian<X, Y, R, Theta>>);
362 
363  IdxRangeRTheta const idx_range_rp = get_idx_range(advection_field);
364  CircularToCartesian<X_adv, Y_adv, R, Theta> const pseudo_Cartesian_mapping;
365 
366  ddc::for_each(idx_range_rp, [&](IdxRTheta const irp) {
367  CoordRTheta const coord_rp(ddc::coordinate(irp));
368  double const r(ddc::get<R>(coord_rp));
369  double const th(ddc::get<Theta>(coord_rp));
370 
371  Matrix2x2 J;
372 
373  if (r > m_epsilon) {
374  Matrix2x2 pseudo_Cart_J;
375  Matrix2x2 map_J;
376  pseudo_Cartesian_mapping.jacobian_matrix(coord_rp, pseudo_Cart_J);
377  m_mapping.inv_jacobian_matrix(coord_rp, map_J);
378 
379  J[0][0] = pseudo_Cart_J[0][0] * map_J[0][0] + pseudo_Cart_J[0][1] * map_J[1][0];
380  J[0][1] = pseudo_Cart_J[0][0] * map_J[0][1] + pseudo_Cart_J[0][1] * map_J[1][1];
381  J[1][0] = pseudo_Cart_J[1][0] * map_J[0][0] + pseudo_Cart_J[1][1] * map_J[1][0];
382  J[1][1] = pseudo_Cart_J[1][0] * map_J[0][1] + pseudo_Cart_J[1][1] * map_J[1][1];
383 
384  } else {
385  Matrix2x2 J_0;
386  m_mapping.to_pseudo_cartesian_jacobian_center_matrix(idx_range_rp, J_0);
387 
388  CoordRTheta const coord_rp_epsilon(m_epsilon, th);
389 
390  Matrix2x2 pseudo_Cart_J_epsilon;
391  Matrix2x2 map_J_epsilon;
392  pseudo_Cartesian_mapping.jacobian_matrix(coord_rp_epsilon, pseudo_Cart_J_epsilon);
393  m_mapping.jacobian_matrix(coord_rp_epsilon, map_J_epsilon);
394 
395  Matrix2x2 J_eps;
396  J_eps[0][0] = pseudo_Cart_J_epsilon[0][0] * map_J_epsilon[1][1]
397  - pseudo_Cart_J_epsilon[0][1] * map_J_epsilon[1][0];
398 
399  J_eps[0][1] = pseudo_Cart_J_epsilon[0][1] * map_J_epsilon[0][0]
400  - pseudo_Cart_J_epsilon[0][0] * map_J_epsilon[0][1];
401 
402  J_eps[1][0] = pseudo_Cart_J_epsilon[1][0] * map_J_epsilon[1][1]
403  - pseudo_Cart_J_epsilon[1][1] * map_J_epsilon[1][0];
404 
405  J_eps[1][1] = pseudo_Cart_J_epsilon[1][1] * map_J_epsilon[0][0]
406  - pseudo_Cart_J_epsilon[1][0] * map_J_epsilon[0][1];
407 
408  J[0][0] = (1 - r / m_epsilon) * J_0[0][0] + r / m_epsilon * J_eps[0][0];
409  J[0][1] = (1 - r / m_epsilon) * J_0[0][1] + r / m_epsilon * J_eps[0][1];
410  J[1][0] = (1 - r / m_epsilon) * J_0[1][0] + r / m_epsilon * J_eps[1][0];
411  J[1][1] = (1 - r / m_epsilon) * J_0[1][1] + r / m_epsilon * J_eps[1][1];
412  }
413 
414  ddcHelper::get<X_adv>(advection_field_pseudo_Cart)(irp)
415  = ddcHelper::get<X>(advection_field)(irp) * J[0][0]
416  + ddcHelper::get<Y>(advection_field)(irp) * J[0][1];
417  ddcHelper::get<Y_adv>(advection_field_pseudo_Cart)(irp)
418  = ddcHelper::get<X>(advection_field)(irp) * J[1][0]
419  + ddcHelper::get<Y>(advection_field)(irp) * J[1][1];
420  });
421  }
