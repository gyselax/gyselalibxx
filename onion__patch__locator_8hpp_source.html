157 KOKKOS_INLINE_FUNCTION int operator()(Coord<X, Y> const coord) const
158 {
159 int patch_index_min = 0;
160 int patch_index_max = n_patches - 1;
161
162 Coord<R> r_min = m_radii(patch_index_min);
163 Coord<R> r_max = m_radii(patch_index_max + 1);
164 Coord<R> r(m_to_curvilinear_mapping(coord));
165 KOKKOS_ASSERT(!Kokkos::isnan(double(r)));
166
167 if (r > r_max) {
168 return outside_rmax_domain;
169 }
170 if (r < r_min) {
171 return outside_rmin_domain;
172 }
173
174 while (patch_index_max >= patch_index_min) {
175 int patch_index_mid = (patch_index_min + patch_index_max) / 2;
176 r_min = m_radii(patch_index_mid);
177 r_max = m_radii(patch_index_mid + 1);
178
179 if (r_min <= r && r < r_max) {
180 return patch_index_mid;
181 } else if (r < r_min) {
182 patch_index_max = patch_index_mid - 1;
183 } else if (r_max <= r) {
184 patch_index_min = patch_index_mid + 1;
185 }
186 };
187
188 if (r != m_radii(n_patches)) {
189 Kokkos::abort("Dichotomy method failed to find the right patch.");
190 }
191 return n_patches - 1;
192 }

234 void set_and_check_radii()
235 {
236 Kokkos::View<Coord<R>*, Kokkos::HostSpace> radii_host("radii_host", n_patches + 1);
237
238 std::array<Coord<R>, n_patches> r_min {
239 (Coord<R>(ddc::coordinate(m_all_idx_ranges.template get<Patches>().front())))...};
240 std::array<Coord<R>, n_patches> r_max {
241 (Coord<R>(ddc::coordinate(m_all_idx_ranges.template get<Patches>().back())))...};
242
243 radii_host(0) = r_min[0];
244 for (std::size_t i(0); i < n_patches - 1; i++) {
245 if (abs(double(r_min[i + 1] - r_max[i])) > 1e-14) {
246 throw std::invalid_argument("The patches listed in PatchOrdering must be ordered.");
247 }
248 radii_host(i + 1) = r_max[i];
249 }
250 radii_host(n_patches) = r_max[n_patches - 1];
251
252 Kokkos::deep_copy(m_radii, radii_host);
253 }
