122  KOKKOS_INLINE_FUNCTION int operator()(Coord<X, Y> const coord) const
123  {
124  int patch_index_min = 0;
125  int patch_index_max = n_patches - 1;
126 
127  Coord<R> r_max = m_radii(patch_index_max + 1);
128  Coord<R> r(m_mapping(coord));
129  if (r == r_max) {
130  return patch_index_max;
131  }
132 
133  while (patch_index_max >= patch_index_min) {
134  int patch_index_mid = (patch_index_min + patch_index_max) / 2;
135  Coord<R> r_min = m_radii(patch_index_mid);
136  r_max = m_radii(patch_index_mid + 1);
137 
138  if (r_min <= r && r < r_max) {
139  return patch_index_mid;
140  } else if (r < r_min) {
141  patch_index_max = patch_index_mid - 1;
142  } else if (r_max < r) {
143  patch_index_min = patch_index_mid + 1;
144  }
145  };
146  return IPatchLocator::outside_domain;
147  }

171  void set_and_check_radii()
172  {
173  Kokkos::View<Coord<R>*, Kokkos::HostSpace> radii_host("radii_host", n_patches + 1);
174 
175  std::array<Coord<R>, n_patches> r_min {
176  (Coord<R>(ddc::coordinate(m_all_idx_ranges.template get<Patches>().front())))...};
177  std::array<Coord<R>, n_patches> r_max {
178  (Coord<R>(ddc::coordinate(m_all_idx_ranges.template get<Patches>().back())))...};
179 
180  radii_host(0) = r_min[0];
181  for (std::size_t i(0); i < n_patches - 1; i++) {
182  if (abs(double(r_min[i + 1] - r_max[i])) > 1e-14) {
183  throw std::invalid_argument("The patches listed in PatchOrdering must be ordered.");
184  }
185  radii_host(i + 1) = r_max[i];
186  }
187  radii_host(n_patches) = r_max[n_patches - 1];
188 
189  Kokkos::deep_copy(m_radii, radii_host);
190  }
