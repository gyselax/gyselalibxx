136  KOKKOS_INLINE_FUNCTION int operator()(Coord<X, Y> const coord) const
137  {
138  int patch_index_min = 0;
139  int patch_index_max = n_patches - 1;
140 
141  Coord<R> r_min = m_radii(patch_index_min);
142  Coord<R> r_max = m_radii(patch_index_max + 1);
143  Coord<R> r(m_mapping(coord));
144 
145  if (r > r_max) {
146  return outside_rmax_domain;
147  }
148  if (r < r_min) {
149  return outside_rmin_domain;
150  }
151 
152  while (patch_index_max >= patch_index_min) {
153  int patch_index_mid = (patch_index_min + patch_index_max) / 2;
154  r_min = m_radii(patch_index_mid);
155  r_max = m_radii(patch_index_mid + 1);
156 
157  if (r_min <= r && r < r_max) {
158  return patch_index_mid;
159  } else if (r < r_min) {
160  patch_index_max = patch_index_mid - 1;
161  } else if (r_max < r) {
162  patch_index_min = patch_index_mid + 1;
163  }
164  };
165 
166  if (r != m_radii(n_patches)) {
167  Kokkos::abort("Dichotomy method failed to find the right patch.");
168  }
169  return n_patches - 1;
170  }

212  void set_and_check_radii()
213  {
214  Kokkos::View<Coord<R>*, Kokkos::HostSpace> radii_host("radii_host", n_patches + 1);
215 
216  std::array<Coord<R>, n_patches> r_min {
217  (Coord<R>(ddc::coordinate(m_all_idx_ranges.template get<Patches>().front())))...};
218  std::array<Coord<R>, n_patches> r_max {
219  (Coord<R>(ddc::coordinate(m_all_idx_ranges.template get<Patches>().back())))...};
220 
221  radii_host(0) = r_min[0];
222  for (std::size_t i(0); i < n_patches - 1; i++) {
223  if (abs(double(r_min[i + 1] - r_max[i])) > 1e-14) {
224  throw std::invalid_argument("The patches listed in PatchOrdering must be ordered.");
225  }
226  radii_host(i + 1) = r_max[i];
227  }
228  radii_host(n_patches) = r_max[n_patches - 1];
229 
230  Kokkos::deep_copy(m_radii, radii_host);
231  }
