114  int operator()(Coord<X, Y> const& coord) const
115  {
116  // Dichotomy to find the right patch.
117  int patch_index_min = 0;
118  int patch_index_max = n_patches - 1;
119 
120  // Check if the physical coordinate corresponds to rmax of the domain.
121  Coord<R> r_max = m_radii[patch_index_max + 1];
122  Coord<R> r(m_mapping(coord));
123  if (r == r_max) {
124  return patch_index_max;
125  }
126 
127  while (patch_index_max >= patch_index_min) {
128  int patch_index_mid = (patch_index_min + patch_index_max) / 2;
129  Coord<R> r_min = m_radii[patch_index_mid];
130  r_max = m_radii[patch_index_mid + 1];
131 
132  if (r_min <= r && r < r_max) {
133  return patch_index_mid;
134  } else if (r < r_min) {
135  patch_index_max = patch_index_mid - 1;
136  } else if (r_max < r) {
137  patch_index_min = patch_index_mid + 1;
138  }
139  };
140  return IPatchLocator<X, Y>::outside_domain;
141  }

144 private:
148  void set_and_check_radii()
149  {
150  std::array<Coord<R>, n_patches> r_min {
151  (Coord<R>(ddc::coordinate(m_all_idx_ranges.template get<Patches>().front())))...};
152  std::array<Coord<R>, n_patches> r_max {
153  (Coord<R>(ddc::coordinate(m_all_idx_ranges.template get<Patches>().back())))...};
154 
155  m_radii[0] = r_min[0];
156  for (std::size_t i(0); i < n_patches - 1; i++) {
157  if (abs(double(r_min[i + 1] - r_max[i])) > 1e-14) {
158  throw std::invalid_argument("The patches listed in PatchOrdering must be ordered.");
159  }
160  m_radii[i + 1] = r_max[i];
161  }
162  m_radii[n_patches] = r_max[n_patches - 1];
163  }
