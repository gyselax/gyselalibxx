35  ddc::ChunkSpan<const double, SpatialDDom> const electric_field,
36  double const dt) const override
37  {
38  FdistribuDDom const dom = allfdistribu.domain();
39  ddc::DiscreteDomain<DDimV> const v_dom = ddc::select<DDimV>(dom);
40  ddc::DiscreteDomain<DDimSp> const sp_dom = ddc::select<DDimSp>(dom);
41 
42  // pre-allocate some memory to prevent allocation later in loop
43  ddc::Chunk<ddc::Coordinate<CDimV>, ddc::DiscreteDomain<DDimV>> feet_coords(v_dom);
44  ddc::Chunk<double, ddc::DiscreteDomain<DDimV>> contiguous_slice(v_dom);
45  std::unique_ptr<IInterpolator<DDimV>> const interpolator_v_ptr
46  = m_interpolator_v.preallocate();
47  IInterpolator<DDimV> const& interpolator_v = *interpolator_v_ptr;
48 
49  SpatialDDom const spatial_dom(allfdistribu.domain());
50 
51  auto c_dom = ddc::remove_dims_of(
52  ddc::remove_dims_of(ddc::remove_dims_of(dom, sp_dom), spatial_dom),
53  v_dom);
54  using DElemC = typename decltype(c_dom)::discrete_element_type;
55  using DElemSpatial = typename SpatialDDom::discrete_element_type;
56 
57  ddc::for_each(c_dom, [&](DElemC const ic) {
58  ddc::for_each(sp_dom, [&](DElemSp const isp) {
59  double const sqrt_me_on_mspecies = std::sqrt(mass(ielec()) / mass(isp));
60 
61  ddc::for_each(spatial_dom, [&](DElemSpatial const ix) {
62  // compute the displacement
63  double const dvx = charge(isp) * sqrt_me_on_mspecies * dt * electric_field(ix);
64 
65  // compute the coordinates of the feet
66  ddc::for_each(v_dom, [&](DElemV const iv) {
67  feet_coords(iv) = ddc::Coordinate<CDimV>(ddc::coordinate(iv) - dvx);
68  });
69 
70  // copy the slice in contiguous memory
71  ddc::deepcopy(contiguous_slice, allfdistribu[ic][isp][ix]);
72 
73  // build a spline representation of the data
74  interpolator_v(contiguous_slice, feet_coords.span_cview());
75 
76  // copy back
77  ddc::deepcopy(allfdistribu[ic][isp][ix], contiguous_slice);
78  });
79  });
80  });
81 
82  return allfdistribu;
83  }
