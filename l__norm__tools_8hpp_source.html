75// General implementation of the infinity norm. This function is in a namespace to avoid code duplication
76// without creating a function so general that it also captures multipatch types.
77template <class ExecSpace, class FuncType>
78double norm_inf(ExecSpace exec_space, FuncType function)
79{
80 static_assert(
81 Kokkos::SpaceAccessibility<ExecSpace, typename FuncType::memory_space>::accessible);
82 using IdxRangeFunc = typename FuncType::discrete_domain_type;
83 using IdxFunc = typename IdxRangeFunc::discrete_element_type;
84 IdxRangeFunc idx_range = get_idx_range(function);
85 return ddc::parallel_transform_reduce(
86 exec_space,
87 idx_range,
88 0.,
89 ddc::reducer::max<double>(),
90 KOKKOS_LAMBDA(IdxFunc const idx) { return ::norm_inf(function(idx)); });
91}

93// General implementation of the infinity norm of an error. This function is in a namespace to avoid code duplication
94// without creating a function so general that it also captures multipatch types.
95template <class ExecSpace, class FuncType>
96double error_norm_inf(ExecSpace exec_space, FuncType function, FuncType exact_function)
97{
98 static_assert(
99 Kokkos::SpaceAccessibility<ExecSpace, typename FuncType::memory_space>::accessible);
100 using IdxRangeFunc = typename FuncType::discrete_domain_type;
101 using IdxFunc = typename IdxRangeFunc::discrete_element_type;
102 IdxRangeFunc idx_range = get_idx_range(function);
103 return ddc::parallel_transform_reduce(
104 exec_space,
105 idx_range,
106 0.,
107 ddc::reducer::max<double>(),
108 KOKKOS_LAMBDA(IdxFunc const idx) {
109 return ::norm_inf(function(idx) - exact_function(idx));
