74// General implementation of the infinity norm. This function is in a namespace to avoid code duplication
75// without creating a function so general that it also captures multipatch types.
76template <class ExecSpace, class FuncType>
77double norm_inf(ExecSpace exec_space, FuncType function)
78{
79 static_assert(
80 Kokkos::SpaceAccessibility<ExecSpace, typename FuncType::memory_space>::accessible);
81 using IdxRangeFunc = typename FuncType::discrete_domain_type;
82 using IdxFunc = typename IdxRangeFunc::discrete_element_type;
83 IdxRangeFunc idx_range = get_idx_range(function);
84 return ddc::parallel_transform_reduce(
85 exec_space,
86 idx_range,
87 0.,
88 ddc::reducer::max<double>(),
89 KOKKOS_LAMBDA(IdxFunc const idx) { return ::norm_inf(function(idx)); });
90}

92// General implementation of the infinity norm of an error. This function is in a namespace to avoid code duplication
93// without creating a function so general that it also captures multipatch types.
94template <class ExecSpace, class FuncType>
95double error_norm_inf(ExecSpace exec_space, FuncType function, FuncType exact_function)
96{
97 static_assert(
98 Kokkos::SpaceAccessibility<ExecSpace, typename FuncType::memory_space>::accessible);
99 using IdxRangeFunc = typename FuncType::discrete_domain_type;
100 using IdxFunc = typename IdxRangeFunc::discrete_element_type;
101 IdxRangeFunc idx_range = get_idx_range(function);
102 return ddc::parallel_transform_reduce(
103 exec_space,
104 idx_range,
105 0.,
106 ddc::reducer::max<double>(),
107 KOKKOS_LAMBDA(IdxFunc const idx) {
108 return ::norm_inf(function(idx) - exact_function(idx));
