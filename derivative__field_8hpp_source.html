38template <
39 class FieldDst,
40 class FieldSrc,
41 std::enable_if_t<
42 is_borrowed_deriv_field_v<FieldDst> && is_borrowed_deriv_field_v<FieldSrc>,
43 bool> = true>
44auto deepcopy(FieldDst&& dst, FieldSrc&& src)
45{
46 assert(dst.get_values_field().domain().extents() == src.get_values_field().domain().extents());
47
48 DerivField dst_field = dst.span_view();
49 DerivField src_field = src.span_view();
50
51 dst_field.deepcopy(src_field);
52
53 return dst_field;
54}

63template <class ExecSpace, class FieldDst, class FieldSrc>
64auto deepcopy(ExecSpace const& execution_space, FieldDst&& dst, FieldSrc&& src)
65{

67 static_assert(is_borrowed_deriv_field_v<FieldSrc>);

73 DerivField src_field = get_field(src);
74
75 dst_field.deepcopy(execution_space, src_field);

179 template <std::size_t ArrayIndex, class Tag>
180 KOKKOS_FUNCTION constexpr Idx<Tag> get_chunk_subidx_range_1d_idx()
181 {
182 // The Tag describes a dimension for which a derivative is defined
183 if constexpr (ddc::in_tags_v<Tag, deriv_tags>) {
184 // If the Field at this index contains the derivatives of this dimension
185 if constexpr (ArrayIndex & (1 << ddc::type_seq_rank_v<Tag, deriv_tags>)) {
186 return Idx<Tag>(1);
187 }
188 // If the Field at this index doesn't contain derivatives of this dimension
189 else {
190 return Idx<Tag>(0);
191 }
192 } else {
193 // Empty IdxRange to be discarded
194 return Idx<Tag>();
195 }
196 }

323 template <class OElementType, class OLayoutStridedPolicy, class OMemorySpace>
324 void deepcopy(
325 DerivField<OElementType, index_range_type, OLayoutStridedPolicy, OMemorySpace> src)
326 {
327 for (int i(0); i < n_fields; ++i) {
328 auto kokkos_span = get_kokkos_view_from_internal_chunk(i);
329 auto src_kokkos_span = src.get_kokkos_view_from_internal_chunk(i);
330 Kokkos::deep_copy(kokkos_span, src_kokkos_span);
331 }
332 }
