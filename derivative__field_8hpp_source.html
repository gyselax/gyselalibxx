127  template <std::size_t ArrayIndex, class Tag>
128  ddc::DiscreteDomain<Tag> get_domain(
129  physical_domain_type val_domain,
130  physical_deriv_domain_type deriv_domains)
131  {
132  // If the Tag describes a dimension for which a derivative is defined
133  if constexpr (ddc::in_tags_v<Tag, physical_deriv_dims>) {
134  // If the Chunk at this index contains the derivatives of this dimension
135  if constexpr (ArrayIndex & (1 << ddc::type_seq_rank_v<Tag, physical_deriv_dims>)) {
136  return ddc::select<Tag>(deriv_domains);
137  }
138  // If the Chunk at this index doesn't contain derivatives of this dimension
139  else {
140  return ddc::select<Tag>(val_domain);
141  }
142  }
143  // If the Tag describes a derivative
144  else if constexpr (ddc::in_tags_v<Tag, deriv_tags>) {
145  // If the Chunk at this index contains the derivatives of this dimension
146  if constexpr (ArrayIndex & (1 << ddc::type_seq_rank_v<Tag, deriv_tags>)) {
147  return ddc::DiscreteDomain<
148  Tag>(ddc::DiscreteElement<Tag>(1), ddc::DiscreteVector<Tag>(NDerivs));
149  }
150  // If the Chunk at this index doesn't contain derivatives of this dimension
151  else {
152  return ddc::DiscreteDomain<
153  Tag>(ddc::DiscreteElement<Tag>(0), ddc::DiscreteVector<Tag>(1));
154  }
155  }
156  // If the Tag describes a dimension for which derivatives are not defined.
157  else {
158  return ddc::select<Tag>(val_domain);
159  }
160  }

173  template <class QueryDDim, std::size_t ArrayIndex>
174  std::size_t get_span_size()
175  {
176  if constexpr (ddc::in_tags_v<QueryDDim, deriv_tags>) {
177  if (ArrayIndex & (1 << ddc::type_seq_rank_v<QueryDDim, deriv_tags>)) {
178  return NDerivs;
179  } else {
180  return 1;
181  }
182  } else if constexpr (ddc::in_tags_v<QueryDDim, physical_deriv_dims>) {
183  if constexpr (
184  ArrayIndex & (1 << ddc::type_seq_rank_v<ddc::Deriv<QueryDDim>, deriv_tags>)) {
185  DiscreteSubDomain<QueryDDim> local_dom(base_type::m_cross_derivative_domain);
186  return local_dom.extents().value();
187  } else {
188  ddc::DiscreteDomain<QueryDDim> local_dom(base_type::m_physical_domain);
189  return local_dom.extents().value();
190  }
191  } else {
192  ddc::DiscreteDomain<QueryDDim> local_dom(base_type::m_physical_domain);
193  return local_dom.extents().value();
194  }

212  return internal_mdspan_type(ptr, mapping_s);
