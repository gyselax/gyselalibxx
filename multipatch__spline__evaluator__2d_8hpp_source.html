522 template <class EvalType1, class EvalType2, class StoringPatch>
523 void apply_evaluator(
524 ValuesOnPatch<StoringPatch> const& patch_values,
525 CoordConstFieldOnPatch<StoringPatch> const& patch_coords,
526 MultipatchSplineCoeff const patches_splines) const
527 {
528 assert(get_idx_range(patch_values) == get_idx_range(patch_coords));
529
530 using Index =
531 typename batched_evaluation_idx_range_type<StoringPatch>::discrete_element_type;
532 batched_evaluation_idx_range_type<StoringPatch> idx_range = get_idx_range(patch_values);
533
534 ddc::parallel_for_each(
535 exec_space(),
536 idx_range,
537 KOKKOS_CLASS_LAMBDA(Index const& idx) {
538 CoordOnPatch<StoringPatch> const coord = patch_coords(idx);
539 int const patch_idx = get_patch_idx(coord);
540 if (patch_idx < 0
541 && !((std::is_same_v<EvalType1, eval_type>)&&(
542 std::is_same_v<EvalType2, eval_type>))) {
543 Kokkos::abort("The evaluation coordinate has to be on a patch."
544 "No extrapolation rule for derivatives. \n");
545 }
546 patch_values(idx) = recursive_dispatch_patch_function<
547 EvalType1,
548 EvalType2>(coord, patches_splines, patch_idx);
549 });
550 }

557 template <class Patch>
558 void apply_integrate(double& integral, SplineCoeffOnPatch<Patch> const& spline_coef) const
559 {
560 static_assert(
561 std::is_same_v<exec_space, Kokkos::DefaultHostExecutionSpace>,
562 "Can only be called on CPU: .integrals() not defined yet on GPU.");
563 static_assert(
564 Kokkos::SpaceAccessibility<exec_space, memory_space>::accessible,
565 "Execution space and memory space have to be compatible.");
566
567 using bsplines_1 = bsplines_type1<Patch>;
568 using bsplines_2 = bsplines_type2<Patch>;
569 using IdxBS12 = typename spline_idx_range_type<Patch>::discrete_element_type;
570
571 DFieldMem<IdxRange<bsplines_1>, memory_space> values1_alloc(
572 get_idx_range<bsplines_1>(spline_coef));
573 DField<IdxRange<bsplines_1>, std::experimental::layout_right, memory_space> values1
574 = get_field(values1_alloc);
575 DFieldMem<IdxRange<bsplines_2>, memory_space> values2_alloc(
576 get_idx_range<bsplines_2>(spline_coef));
577 DField<IdxRange<bsplines_2>, std::experimental::layout_right, memory_space> values2
578 = get_field(values2_alloc);
579 ddc::integrals(exec_space(), values1);
580 ddc::integrals(exec_space(), values2);
581
582 integral = ddc::parallel_transform_reduce(
583 exec_space(),
584 get_idx_range(spline_coef),
585 0.0,
586 ddc::reducer::sum<double>(),
587 KOKKOS_LAMBDA(IdxBS12 const i12) {
588 return spline_coef(i12) * values1(ddc::select<bsplines_1>(i12))
589 * values2(ddc::select<bsplines_2>(i12));
590 });
591 }

596 // Recursive method to dispatch the coordinates on the right patch ---------------------------
597
599 template <class EvalType1, class EvalType2, class Dim1, class Dim2, int TestPatchIdx = 0>
600 KOKKOS_INLINE_FUNCTION double recursive_dispatch_patch_function(
601 Coord<Dim1, Dim2> coord,
602 MultipatchSplineCoeff const& patches_splines,
603 int const patch_idx) const
604 {
605 if constexpr (TestPatchIdx == ddc::type_seq_size_v<PatchOrdering>) {
606 if (patch_idx >= 0) {
607 Kokkos::abort("The recursion has reached the end without finding where "
608 "the coordinate is physically located.");
609 }
610 // Coord not on patch. Stop recursing.
611 if constexpr (
612 std::is_same_v<EvalType1, eval_type> && std::is_same_v<EvalType2, eval_type>) {
613 /* The operator currently works only for the case where the continuous
614 dimensions of all the patches are the same. So the equivalent coordinates
615 are the same on any patches.
616 Add a method to determine the patch from continuous dimensions, once
617 we would like to deal with other cases.
618 */
619 using AnyPatch = ddc::type_seq_element_t<0, PatchOrdering>;
620 replace_periodic_coord_inside<AnyPatch>(coord);
621 return m_extrapolation_rule(coord, patches_splines, patch_idx);
622 } else {
623 Kokkos::abort("The spline derivatives cannot be evaluated at coordinates "
624 "outside of the domain.");
625 }
626 } else {
627 if (patch_idx == TestPatchIdx) {
628 using TestPatch = ddc::type_seq_element_t<TestPatchIdx, PatchOrdering>;
629 CoordOnPatch<TestPatch> test_coord = get_equivalent_coord<
630 typename TestPatch::Dim1,
631 typename TestPatch::Dim2,
632 Dim1,
633 Dim2>(coord);
634 replace_periodic_coord_inside<TestPatch>(test_coord);
635
636 SplineCoeffOnPatch<TestPatch> const test_spline
637 = patches_splines.template get<TestPatch>();
638 return eval_no_bc<EvalType1, EvalType2, TestPatch>(test_coord, test_spline);
639 } else {
640 return recursive_dispatch_patch_function<
641 EvalType1,
642 EvalType2,
643 Dim1,
644 Dim2,
645 TestPatchIdx + 1>(coord, patches_splines, patch_idx);
646 }
647 }
648 }

665 template <class TargetDim1, class TargetDim2, class CurrentDim1, class CurrentDim2>
666 KOKKOS_INLINE_FUNCTION Coord<TargetDim1, TargetDim2> get_equivalent_coord(
667 Coord<CurrentDim1, CurrentDim2> const& current_coord) const
668 {
669 if constexpr (std::is_same_v<
670 Coord<TargetDim1, TargetDim2>,
671 Coord<CurrentDim1, CurrentDim2>>) {
672 return current_coord;
673 } else {
674 using CurrentMapping =
675 typename PatchLocator::get_mapping_on_logical_dim_t<CurrentDim1, CurrentDim2>;
676 using TargetMapping =
677 typename PatchLocator::get_mapping_on_logical_dim_t<TargetDim1, TargetDim2>;
678
679 static_assert((std::is_same_v<
680 Coord<typename CurrentMapping::curvilinear_tag_r,
681 typename CurrentMapping::curvilinear_tag_theta>,
682 Coord<CurrentDim1, CurrentDim2>>));
683 static_assert((std::is_same_v<
684 Coord<typename TargetMapping::curvilinear_tag_r,
685 typename TargetMapping::curvilinear_tag_theta>,
686 Coord<TargetDim1, TargetDim2>>));
687
688 CurrentMapping const current_mapping(
689 m_patch_locator
690 .template get_mapping_on_logical_dim<CurrentDim1, CurrentDim2>());
691 TargetMapping const target_mapping(
692 m_patch_locator.template get_mapping_on_logical_dim<TargetDim1, TargetDim2>());
693
694 return target_mapping(current_mapping(current_coord));
695 }

699 template <class Patch>
700 KOKKOS_INLINE_FUNCTION void replace_periodic_coord_inside(CoordOnPatch<Patch>& coord) const
701 {
702 using bsplines_1 = bsplines_type1<Patch>;
703 using bsplines_2 = bsplines_type2<Patch>;
704
705 using Dim1 = continuous_dimension_type1<Patch>;
706 using Dim2 = continuous_dimension_type2<Patch>;
707
708 Coord<Dim1> coord_1(coord);
709 Coord<Dim2> coord_2(coord);
710
711 ddcHelper::restrict_to_bspline_domain<bsplines_1>(coord_1);
712 ddcHelper::restrict_to_bspline_domain<bsplines_2>(coord_2);
713
714 coord = CoordOnPatch<Patch>(coord_1, coord_2);
715 }

722 template <class EvalType1, class EvalType2, class Patch, class Layout>
723 KOKKOS_INLINE_FUNCTION double eval_no_bc(
724 CoordOnPatch<Patch> const& coord_eval,
725 DConstField<spline_idx_range_type<Patch>, Layout, memory_space> const& spline_coef)
726 const
727 {
728 static_assert(
729 std::is_same_v<EvalType1, eval_type> || std::is_same_v<EvalType1, eval_deriv_type>);
730 static_assert(
731 std::is_same_v<EvalType2, eval_type> || std::is_same_v<EvalType2, eval_deriv_type>);
732
733 using bsplines_1 = bsplines_type1<Patch>;
734 using bsplines_2 = bsplines_type2<Patch>;
735
736 Idx<bsplines_1> jmin1;
737 Idx<bsplines_2> jmin2;
738
739 std::array<double, bsplines_1::degree() + 1> vals1_ptr;
740 DSpan1D const vals1(vals1_ptr.data(), bsplines_1::degree() + 1);
741 std::array<double, bsplines_2::degree() + 1> vals2_ptr;
742 DSpan1D const vals2(vals2_ptr.data(), bsplines_2::degree() + 1);
743
744 Coord<continuous_dimension_type1<Patch>> coord_eval_interest1
745 = ddc::select<continuous_dimension_type1<Patch>>(coord_eval);
746 Coord<continuous_dimension_type2<Patch>> coord_eval_interest2
747 = ddc::select<continuous_dimension_type2<Patch>>(coord_eval);
748
749 if constexpr (std::is_same_v<EvalType1, eval_type>) {
750 jmin1 = ddc::discrete_space<bsplines_1>().eval_basis(vals1, coord_eval_interest1);
751 } else if constexpr (std::is_same_v<EvalType1, eval_deriv_type>) {
752 jmin1 = ddc::discrete_space<bsplines_1>().eval_deriv(vals1, coord_eval_interest1);
753 }
754
755 if constexpr (std::is_same_v<EvalType2, eval_type>) {
756 jmin2 = ddc::discrete_space<bsplines_2>().eval_basis(vals2, coord_eval_interest2);
757 } else if constexpr (std::is_same_v<EvalType2, eval_deriv_type>) {
758 jmin2 = ddc::discrete_space<bsplines_2>().eval_deriv(vals2, coord_eval_interest2);
759 }
760
761 double y = 0.0;
762 for (std::size_t i = 0; i < bsplines_1::degree() + 1; ++i) {
763 for (std::size_t j = 0; j < bsplines_2::degree() + 1; ++j) {
764 y += spline_coef(jmin1 + i, jmin2 + j) * vals1[i] * vals2[j];
765 }
766 }
767 return y;
768 }
