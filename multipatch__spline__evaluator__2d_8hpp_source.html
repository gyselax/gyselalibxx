517 template <class EvalType1, class EvalType2, class StoringPatch>
518 void apply_evaluator(
519 ValuesOnPatch<StoringPatch> const& patch_values,
520 CoordConstFieldOnPatch<StoringPatch> const& patch_coords,
521 MultipatchSplineCoeff const patches_splines) const
522 {
523 assert(get_idx_range(patch_values) == get_idx_range(patch_coords));
524
525 using Index =
526 typename batched_evaluation_idx_range_type<StoringPatch>::discrete_element_type;
527 batched_evaluation_idx_range_type<StoringPatch> idx_range = get_idx_range(patch_values);
528
529 ddc::parallel_for_each(
530 exec_space(),
531 idx_range,
532 KOKKOS_CLASS_LAMBDA(Index const& idx) {
533 CoordOnPatch<StoringPatch> const coord = patch_coords(idx);
534 int const patch_idx = get_patch_idx(coord);
535 if (patch_idx < 0
536 && !((std::is_same_v<EvalType1, eval_type>)&&(
537 std::is_same_v<EvalType2, eval_type>))) {
538 Kokkos::abort("The evaluation coordinate has to be on a patch."
539 "No extrapolation rule for derivatives. \n");
540 }
541 patch_values(idx) = recursive_dispatch_patch_function<
542 EvalType1,
543 EvalType2>(coord, patches_splines, patch_idx);
544 });
545 }

552 template <class Patch>
553 void apply_integrate(double& integral, SplineCoeffOnPatch<Patch> const& spline_coef) const
554 {
555 static_assert(
556 std::is_same_v<exec_space, Kokkos::DefaultHostExecutionSpace>,
557 "Can only be called on CPU: .integrals() not defined yet on GPU.");
558 static_assert(
559 Kokkos::SpaceAccessibility<exec_space, memory_space>::accessible,
560 "Execution space and memory space have to be compatible.");
561
562 using bsplines_1 = bsplines_type1<Patch>;
563 using bsplines_2 = bsplines_type2<Patch>;
564 using IdxBS12 = typename spline_idx_range_type<Patch>::discrete_element_type;
565
566 DFieldMem<IdxRange<bsplines_1>, memory_space> values1_alloc(
567 get_idx_range<bsplines_1>(spline_coef));
568 DField<IdxRange<bsplines_1>, memory_space> values1 = get_field(values1_alloc);
569 DFieldMem<IdxRange<bsplines_2>, memory_space> values2_alloc(
570 get_idx_range<bsplines_2>(spline_coef));
571 DField<IdxRange<bsplines_2>, memory_space> values2 = get_field(values2_alloc);
572 ddc::integrals(exec_space(), values1);
573 ddc::integrals(exec_space(), values2);
574
575 integral = ddc::parallel_transform_reduce(
576 exec_space(),
577 get_idx_range(spline_coef),
578 0.0,
579 ddc::reducer::sum<double>(),
580 KOKKOS_LAMBDA(IdxBS12 const i12) {
581 return spline_coef(i12) * values1(ddc::select<bsplines_1>(i12))
582 * values2(ddc::select<bsplines_2>(i12));
583 });
584 }

589 // Recursive method to dispatch the coordinates on the right patch ---------------------------
590
592 template <class EvalType1, class EvalType2, class Dim1, class Dim2, int TestPatchIdx = 0>
593 KOKKOS_INLINE_FUNCTION double recursive_dispatch_patch_function(
594 Coord<Dim1, Dim2> coord,
595 MultipatchSplineCoeff const& patches_splines,
596 int const patch_idx) const
597 {
598 if constexpr (TestPatchIdx == ddc::type_seq_size_v<PatchOrdering>) {
599 if (patch_idx >= 0) {
600 Kokkos::abort("The recursion has reached the end without finding where "
601 "the coordinate is physically located.");
602 }
603 // Coord not on patch. Stop recursing.
604 if constexpr (
605 std::is_same_v<EvalType1, eval_type> && std::is_same_v<EvalType2, eval_type>) {
606 /* The operator currently works only for the case where the continuous
607 dimensions of all the patches are the same. So the equivalent coordinates
608 are the same on any patches.
609 Add a method to determine the patch from continuous dimensions, once
610 we would like to deal with other cases.
611 */
612 using AnyPatch = ddc::type_seq_element_t<0, PatchOrdering>;
613 replace_periodic_coord_inside<AnyPatch>(coord);
614 return m_extrapolation_rule(coord, patches_splines, patch_idx);
615 } else {
616 Kokkos::abort("The spline derivatives cannot be evaluated at coordinates "
617 "outside of the domain.");
618 }
619 } else {
620 if (patch_idx == TestPatchIdx) {
621 using TestPatch = ddc::type_seq_element_t<TestPatchIdx, PatchOrdering>;
622 CoordOnPatch<TestPatch> test_coord = get_equivalent_coord<
623 typename TestPatch::Dim1,
624 typename TestPatch::Dim2,
625 Dim1,
626 Dim2>(coord);
627 replace_periodic_coord_inside<TestPatch>(test_coord);
628
629 SplineCoeffOnPatch<TestPatch> const test_spline
630 = patches_splines.template get<TestPatch>();
631 return eval_no_bc<EvalType1, EvalType2, TestPatch>(test_coord, test_spline);
632 } else {
633 return recursive_dispatch_patch_function<
634 EvalType1,
635 EvalType2,
636 Dim1,
637 Dim2,
638 TestPatchIdx + 1>(coord, patches_splines, patch_idx);
639 }
640 }
641 }

649 KOKKOS_INLINE_FUNCTION int get_patch_idx(Coord<Dim1, Dim2> const coord) const
650 {

658 template <class TargetDim1, class TargetDim2, class CurrentDim1, class CurrentDim2>
659 KOKKOS_INLINE_FUNCTION Coord<TargetDim1, TargetDim2> get_equivalent_coord(
660 Coord<CurrentDim1, CurrentDim2> const& current_coord) const
661 {
662 if constexpr (std::is_same_v<
663 Coord<TargetDim1, TargetDim2>,
664 Coord<CurrentDim1, CurrentDim2>>) {
665 return current_coord;
666 } else {
667 using CurrentMapping = typename PatchLocator::
668 template get_mapping_on_logical_dim_t<CurrentDim1, CurrentDim2>;
669 using TargetMapping = typename PatchLocator::
670 template get_mapping_on_logical_dim_t<TargetDim1, TargetDim2>;
671
672 static_assert(is_curvilinear_2d_mapping_v<CurrentMapping>);
673 static_assert((std::is_same_v<
674 Coord<typename CurrentMapping::curvilinear_tag_r,
675 typename CurrentMapping::curvilinear_tag_theta>,
676 Coord<CurrentDim1, CurrentDim2>>));
677 static_assert(is_curvilinear_2d_mapping_v<TargetMapping>);
678 static_assert((std::is_same_v<
679 Coord<typename TargetMapping::curvilinear_tag_r,
680 typename TargetMapping::curvilinear_tag_theta>,
681 Coord<TargetDim1, TargetDim2>>));
682
683 CurrentMapping const current_mapping(
684 m_patch_locator
685 .template get_mapping_on_logical_dim<CurrentDim1, CurrentDim2>());
686 TargetMapping const target_mapping(
687 m_patch_locator.template get_mapping_on_logical_dim<TargetDim1, TargetDim2>());
688
689 return target_mapping(current_mapping(current_coord));
690 }

694 template <class Patch>
695 KOKKOS_INLINE_FUNCTION void replace_periodic_coord_inside(CoordOnPatch<Patch>& coord) const
696 {
697 using bsplines_1 = bsplines_type1<Patch>;
698 using bsplines_2 = bsplines_type2<Patch>;
699
700 using Dim1 = continuous_dimension_type1<Patch>;
701 using Dim2 = continuous_dimension_type2<Patch>;
702
703 Coord<Dim1> coord_1(coord);
704 Coord<Dim2> coord_2(coord);
705
706 ddcHelper::restrict_to_bspline_domain<bsplines_1>(coord_1);
707 ddcHelper::restrict_to_bspline_domain<bsplines_2>(coord_2);
708
709 coord = CoordOnPatch<Patch>(coord_1, coord_2);
710 }

717 template <class EvalType1, class EvalType2, class Patch, class Layout>
718 KOKKOS_INLINE_FUNCTION double eval_no_bc(
719 CoordOnPatch<Patch> const& coord_eval,
720 DConstField<spline_idx_range_type<Patch>, memory_space, Layout> const& spline_coef)
721 const
722 {
723 static_assert(
724 std::is_same_v<EvalType1, eval_type> || std::is_same_v<EvalType1, eval_deriv_type>);
725 static_assert(
726 std::is_same_v<EvalType2, eval_type> || std::is_same_v<EvalType2, eval_deriv_type>);
727
728 using bsplines_1 = bsplines_type1<Patch>;
729 using bsplines_2 = bsplines_type2<Patch>;
730
731 Idx<bsplines_1> jmin1;
732 Idx<bsplines_2> jmin2;
733
734 std::array<double, bsplines_1::degree() + 1> vals1_ptr;
735 DSpan1D const vals1(vals1_ptr.data(), bsplines_1::degree() + 1);
736 std::array<double, bsplines_2::degree() + 1> vals2_ptr;
737 DSpan1D const vals2(vals2_ptr.data(), bsplines_2::degree() + 1);
738
739 Coord<continuous_dimension_type1<Patch>> coord_eval_interest1
740 = ddc::select<continuous_dimension_type1<Patch>>(coord_eval);
741 Coord<continuous_dimension_type2<Patch>> coord_eval_interest2
742 = ddc::select<continuous_dimension_type2<Patch>>(coord_eval);
743
744 if constexpr (std::is_same_v<EvalType1, eval_type>) {
745 jmin1 = ddc::discrete_space<bsplines_1>().eval_basis(vals1, coord_eval_interest1);
746 } else if constexpr (std::is_same_v<EvalType1, eval_deriv_type>) {
747 jmin1 = ddc::discrete_space<bsplines_1>().eval_deriv(vals1, coord_eval_interest1);
748 }
749
750 if constexpr (std::is_same_v<EvalType2, eval_type>) {
751 jmin2 = ddc::discrete_space<bsplines_2>().eval_basis(vals2, coord_eval_interest2);
752 } else if constexpr (std::is_same_v<EvalType2, eval_deriv_type>) {
753 jmin2 = ddc::discrete_space<bsplines_2>().eval_deriv(vals2, coord_eval_interest2);
754 }
755
756 double y = 0.0;
757 for (std::size_t i = 0; i < bsplines_1::degree() + 1; ++i) {
758 for (std::size_t j = 0; j < bsplines_2::degree() + 1; ++j) {
759 y += spline_coef(jmin1 + i, jmin2 + j) * vals1[i] * vals2[j];
760 }
761 }
762 return y;
763 }
