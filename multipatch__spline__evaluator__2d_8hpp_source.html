517  void apply_evaluator(
518  ValuesOnPatch<StoringPatch> const& patch_values,
519  CoordConstFieldOnPatch<StoringPatch> const& patch_coords,

529  exec_space(),

531  KOKKOS_CLASS_LAMBDA(Index const& idx) {

551  template <class Patch>
552  void apply_integrate(double& integral, SplineCoeffOnPatch<Patch> const& spline_coef) const
553  {
554  static_assert(
555  std::is_same_v<exec_space, Kokkos::DefaultHostExecutionSpace>,
556  "Can only be called on CPU: .integrals() not defined yet on GPU.");
557  static_assert(
558  Kokkos::SpaceAccessibility<exec_space, memory_space>::accessible,
559  "Execution space and memory space have to be compatible.");
560 
561  using bsplines_1 = bsplines_type1<Patch>;
562  using bsplines_2 = bsplines_type2<Patch>;
563  using IdxBS12 = typename spline_idx_range_type<Patch>::discrete_element_type;
564 
565  DFieldMem<IdxRange<bsplines_1>, memory_space> values1_alloc(
566  get_idx_range<bsplines_1>(spline_coef));
567  DField<IdxRange<bsplines_1>, memory_space> values1 = get_field(values1_alloc);
568  DFieldMem<IdxRange<bsplines_2>, memory_space> values2_alloc(
569  get_idx_range<bsplines_2>(spline_coef));
570  DField<IdxRange<bsplines_2>, memory_space> values2 = get_field(values2_alloc);
571  ddc::integrals(exec_space(), values1);
572  ddc::integrals(exec_space(), values2);
573 
574  integral = ddc::parallel_transform_reduce(
575  exec_space(),
576  get_idx_range(spline_coef),
577  0.0,
578  ddc::reducer::sum<double>(),
579  KOKKOS_LAMBDA(IdxBS12 const i12) {
580  return spline_coef(i12) * values1(ddc::select<bsplines_1>(i12))
581  * values2(ddc::select<bsplines_2>(i12));
582  });
583  }

591  template <class EvalType1, class EvalType2, class Dim1, class Dim2, int TestPatchIdx = 0>
592  KOKKOS_INLINE_FUNCTION double recursive_dispatch_patch_function(
593  Coord<Dim1, Dim2> coord,
594  MultipatchSplineCoeff const& patches_splines,
595  int const patch_idx) const
596  {
597  if constexpr (TestPatchIdx == ddc::type_seq_size_v<PatchOrdering>) {
598  if (patch_idx >= 0) {
599  Kokkos::abort("The recursion has reached the end without finding where "
600  "the coordinate is physically located.");
601  }
602  // Coord not on patch. Stop recursing.
603  if constexpr (
604  std::is_same_v<EvalType1, eval_type> && std::is_same_v<EvalType2, eval_type>) {
605  /* The operator currently works only for the case where the continuous
606  dimensions of all the patches are the same. So the equivalent coordinates
607  are the same on any patches.
608  Add a method to determine the patch from continuous dimensions, once
609  we would like to deal with other cases.
610  */
611  using AnyPatch = ddc::type_seq_element_t<0, PatchOrdering>;
612  replace_periodic_coord_inside<AnyPatch>(coord);
613  return m_extrapolation_rule(coord, patches_splines, patch_idx);
614  } else {
615  Kokkos::abort("The spline derivatives cannot be evaluated at coordinates "
616  "outside of the domain.");
617  }
618  } else {
619  if (patch_idx == TestPatchIdx) {
620  using TestPatch = ddc::type_seq_element_t<TestPatchIdx, PatchOrdering>;
621  CoordOnPatch<TestPatch> test_coord = get_equivalent_coord<
622  typename TestPatch::Dim1,
623  typename TestPatch::Dim2,
624  Dim1,
625  Dim2>(coord);
626  replace_periodic_coord_inside<TestPatch>(test_coord);
627 
628  SplineCoeffOnPatch<TestPatch> const test_spline
629  = patches_splines.template get<TestPatch>();
630  return eval_no_bc<EvalType1, EvalType2, TestPatch>(test_coord, test_spline);
631  } else {
632  return recursive_dispatch_patch_function<
633  EvalType1,
634  EvalType2,
635  Dim1,
636  Dim2,
637  TestPatchIdx + 1>(coord, patches_splines, patch_idx);
638  }
639  }
640  }

657  template <class TargetDim1, class TargetDim2, class CurrentDim1, class CurrentDim2>
658  KOKKOS_INLINE_FUNCTION Coord<TargetDim1, TargetDim2> get_equivalent_coord(
659  Coord<CurrentDim1, CurrentDim2> const& current_coord) const
660  {
661  if constexpr (std::is_same_v<
662  Coord<TargetDim1, TargetDim2>,
663  Coord<CurrentDim1, CurrentDim2>>) {
664  return current_coord;
665  } else {
666  using CurrentMapping =
667  typename PatchLocator::get_mapping_on_logical_dim_t<CurrentDim1, CurrentDim2>;
668  using TargetMapping =
669  typename PatchLocator::get_mapping_on_logical_dim_t<TargetDim1, TargetDim2>;

671  static_assert((std::is_same_v<
672  Coord<typename CurrentMapping::curvilinear_tag_r,
673  typename CurrentMapping::curvilinear_tag_theta>,
674  Coord<CurrentDim1, CurrentDim2>>));
675  static_assert((std::is_same_v<
676  Coord<typename TargetMapping::curvilinear_tag_r,
677  typename TargetMapping::curvilinear_tag_theta>,
678  Coord<TargetDim1, TargetDim2>>));
679 
680  CurrentMapping const current_mapping(
681  m_patch_locator
682  .template get_mapping_on_logical_dim<CurrentDim1, CurrentDim2>());
683  TargetMapping const target_mapping(
684  m_patch_locator.template get_mapping_on_logical_dim<TargetDim1, TargetDim2>());
685 
686  return target_mapping(current_mapping(current_coord));

714  template <class EvalType1, class EvalType2, class Patch, class Layout>
715  KOKKOS_INLINE_FUNCTION double eval_no_bc(
716  CoordOnPatch<Patch> const& coord_eval,
717  DConstField<spline_idx_range_type<Patch>, memory_space, Layout> const& spline_coef)
718  const
719  {
720  static_assert(
721  std::is_same_v<EvalType1, eval_type> || std::is_same_v<EvalType1, eval_deriv_type>);
722  static_assert(
723  std::is_same_v<EvalType2, eval_type> || std::is_same_v<EvalType2, eval_deriv_type>);
724 
725  using bsplines_1 = bsplines_type1<Patch>;
726  using bsplines_2 = bsplines_type2<Patch>;
727 
728  Idx<bsplines_1> jmin1;
729  Idx<bsplines_2> jmin2;
730 
731  std::array<double, bsplines_1::degree() + 1> vals1_ptr;
732  DSpan1D const vals1(vals1_ptr.data(), bsplines_1::degree() + 1);
733  std::array<double, bsplines_2::degree() + 1> vals2_ptr;
734  DSpan1D const vals2(vals2_ptr.data(), bsplines_2::degree() + 1);
735 
736  Coord<continuous_dimension_type1<Patch>> coord_eval_interest1
737  = ddc::select<continuous_dimension_type1<Patch>>(coord_eval);
738  Coord<continuous_dimension_type2<Patch>> coord_eval_interest2
739  = ddc::select<continuous_dimension_type2<Patch>>(coord_eval);
740 
741  if constexpr (std::is_same_v<EvalType1, eval_type>) {
742  jmin1 = ddc::discrete_space<bsplines_1>().eval_basis(vals1, coord_eval_interest1);
743  } else if constexpr (std::is_same_v<EvalType1, eval_deriv_type>) {
744  jmin1 = ddc::discrete_space<bsplines_1>().eval_deriv(vals1, coord_eval_interest1);
745  }
746 
747  if constexpr (std::is_same_v<EvalType2, eval_type>) {
748  jmin2 = ddc::discrete_space<bsplines_2>().eval_basis(vals2, coord_eval_interest2);
749  } else if constexpr (std::is_same_v<EvalType2, eval_deriv_type>) {
750  jmin2 = ddc::discrete_space<bsplines_2>().eval_deriv(vals2, coord_eval_interest2);
751  }
752 
753  double y = 0.0;
754  for (std::size_t i = 0; i < bsplines_1::degree() + 1; ++i) {
755  for (std::size_t j = 0; j < bsplines_2::degree() + 1; ++j) {
756  y += spline_coef(jmin1 + i, jmin2 + j) * vals1[i] * vals2[j];
757  }
758  }
759  return y;
760  }
