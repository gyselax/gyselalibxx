61 
62  DFieldR nustar0_r = get_field(m_nustar0_r);
63  ddc::parallel_for_each(
64  Kokkos::DefaultExecutionSpace(),
65  get_idx_range(radial_profile),
66  KOKKOS_LAMBDA(Idx<GridR> idx) {
67  double rpeak_on_r = rpeak / radial_profile(idx);
68  double q_on_qrpeak = safety_factor(idx) / q_rpeak;
69  nustar0_r(idx)
70  = nustar0_rpeak * q_on_qrpeak * rpeak_on_r * Kokkos::sqrt(rpeak_on_r);
71  });
72  }
