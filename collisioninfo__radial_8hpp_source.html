57  void compute_nustar0()
58  {
59  double const rpeak = m_rpeak;
60  double const q_rpeak = m_q_rpeak;
61  DConstFieldR radial_profile = get_const_field(m_rg);
62  DConstFieldR safety_factor = get_const_field(m_safety_factor);
63  double const nustar0_rpeak = m_nustar0_rpeak;
64 
65  DFieldR nustar0 = get_field(m_nustar0);
66  ddc::parallel_for_each(
67  Kokkos::DefaultExecutionSpace(),
68  get_idx_range(radial_profile),
69  KOKKOS_LAMBDA(Idx<GridR> idx) {
70  double rpeak_on_r = rpeak / radial_profile(idx);
71  double q_on_qrpeak = safety_factor(idx) / q_rpeak;
72  nustar0(idx)
73  = nustar0_rpeak * q_on_qrpeak * rpeak_on_r * Kokkos::sqrt(rpeak_on_r);
74  });
75  }

94  Kokkos::DefaultExecutionSpace(),
95  get_idx_range(radial_profile),
96  KOKKOS_LAMBDA(Idx<GridR> idx) {
97  double eps = radial_profile(idx) / R0;
98  coeff_AD(idx) = Kokkos::sqrt(2.) * eps * Kokkos::sqrt(eps) * nustar0(idx)
99  / (safety_factor(idx) * R0);
100  });
