160 KOKKOS_FUNCTION DTensor<VectorIndexSet<X, Y>, VectorIndexSet<R_cov, Theta_cov>> jacobian_matrix(

162 {
163 DTensor<VectorIndexSet<X, Y>, VectorIndexSet<R_cov, Theta_cov>> jacobian_matrix;
164 ddcHelper::get<X, R_cov>(jacobian_matrix)

193 template <class IndexTag1, class IndexTag2>
194 KOKKOS_INLINE_FUNCTION double jacobian_component(Coord<R, Theta> coord) const
195 {
196 static_assert(ddc::in_tags_v<IndexTag1, VectorIndexSet<X, Y>>);
197 static_assert(ddc::in_tags_v<IndexTag2, VectorIndexSet<R_cov, Theta_cov>>);
198
199 if constexpr (std::is_same_v<IndexTag1, X> && std::is_same_v<IndexTag2, R_cov>) {
200 // Component (1,1), i.e dx/dr
201 return m_spline_evaluator
202 .deriv_dim_1(coord, get_const_field(m_x_spline_representation));
203 } else if constexpr (std::is_same_v<IndexTag1, X> && std::is_same_v<IndexTag2, Theta_cov>) {
204 // Component (1,2), i.e dx/dtheta
205 return m_spline_evaluator
206 .deriv_dim_2(coord, get_const_field(m_x_spline_representation));
207 } else if constexpr (std::is_same_v<IndexTag1, Y> && std::is_same_v<IndexTag2, R_cov>) {
208 // Component (2,1), i.e dy/dr
209 return m_spline_evaluator
210 .deriv_dim_1(coord, get_const_field(m_y_spline_representation));
211 } else {
212 // Component (2,2), i.e dy/dtheta
213 return m_spline_evaluator
214 .deriv_dim_2(coord, get_const_field(m_y_spline_representation));
215 }
216 }

226 KOKKOS_FUNCTION double jacobian(
227 Coord<curvilinear_tag_r, curvilinear_tag_theta> const& coord) const
228 {
229 Tensor J = jacobian_matrix(coord);
230 return ddcHelper::get<X, R_cov>(J) * ddcHelper::get<Y, Theta_cov>(J)
231 - ddcHelper::get<Y, R_cov>(J) * ddcHelper::get<X, Theta_cov>(J);
232 }

248 KOKKOS_INLINE_FUNCTION DTensor<VectorIndexSet<X, Y>, VectorIndexSet<R_cov, Theta_cov>>
249 first_order_jacobian_matrix_r_rtheta(
250 Coord<curvilinear_tag_r, curvilinear_tag_theta> const& coord) const
251 {
252 DTensor<VectorIndexSet<X, Y>, VectorIndexSet<R_cov, Theta_cov>> J;
253 ddcHelper::get<X, R_cov>(J)
254 = m_spline_evaluator.deriv_dim_1(coord, get_const_field(m_x_spline_representation));
255 ddcHelper::get<X, Theta_cov>(J)
256 = m_spline_evaluator
257 .deriv_1_and_2(coord, get_const_field(m_x_spline_representation));
258 ddcHelper::get<Y, R_cov>(J)
259 = m_spline_evaluator.deriv_dim_1(coord, get_const_field(m_y_spline_representation));
260 ddcHelper::get<Y, Theta_cov>(J)
261 = m_spline_evaluator
262 .deriv_1_and_2(coord, get_const_field(m_y_spline_representation));
263 return J;
264 }
