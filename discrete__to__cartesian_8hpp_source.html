310 KOKKOS_FUNCTION void to_pseudo_cartesian_jacobian_center_matrix(Matrix_2x2& matrix) const final
311 {
312 matrix[0][0] = 0;
313 matrix[0][1] = 0;
314 matrix[1][0] = 0;
315 matrix[1][1] = 0;
316
317 // Average the values at (r = 0, theta):
318 for (IdxTheta ip : m_idx_range_theta) {
319 const double th = ddc::coordinate(ip);
320 ddc::Coordinate<curvilinear_tag_r, curvilinear_tag_theta> const coord(0, th);
321 double const deriv_1_x
322 = m_spline_evaluator.deriv_dim_1(coord, m_x_spline_representation.span_cview());
323 double const deriv_1_2_x
324 = m_spline_evaluator
325 .deriv_1_and_2(coord, m_x_spline_representation.span_cview());
326 double const deriv_1_y
327 = m_spline_evaluator.deriv_dim_1(coord, m_y_spline_representation.span_cview());
328 double const deriv_1_2_y
329 = m_spline_evaluator
330 .deriv_1_and_2(coord, m_y_spline_representation.span_cview());
331
332 // Matrix from pseudo-Cart domain to physical domain by logical domain
333 double const j11 = deriv_1_x * Kokkos::cos(th) - deriv_1_2_x * Kokkos::sin(th);
334 double const j12 = deriv_1_x * Kokkos::sin(th) + deriv_1_2_x * Kokkos::cos(th);
335 double const j21 = deriv_1_y * Kokkos::cos(th) - deriv_1_2_y * Kokkos::sin(th);
336 double const j22 = deriv_1_y * Kokkos::sin(th) + deriv_1_2_y * Kokkos::cos(th);
337
338 double const jacobian = j11 * j22 - j12 * j21;
339 // Matrix from physical domain to pseudo_cart domain by logical domain
340 assert(fabs(jacobian) >= 1e-16);
341
342 matrix[0][0] += j22 / jacobian;
343 matrix[0][1] += -j12 / jacobian;
344 matrix[1][0] += -j21 / jacobian;
345 matrix[1][1] += j11 / jacobian;
346 }
347
348 int const theta_size = m_idx_range_theta.size();
349 matrix[0][0] /= theta_size;
350 matrix[0][1] /= theta_size;
351 matrix[1][0] /= theta_size;
352 matrix[1][1] /= theta_size;
353 }
