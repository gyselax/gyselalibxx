353 KOKKOS_FUNCTION void to_pseudo_cartesian_jacobian_center_matrix(Matrix_2x2& matrix) const final
354 {
355 matrix[0][0] = 0;
356 matrix[0][1] = 0;
357 matrix[1][0] = 0;
358 matrix[1][1] = 0;
359
360 // Average the values at (r = 0, theta):
361 for (IdxTheta ip : IdxRangeTheta(m_idx_range_singular_point)) {
362 const double th = ddc::coordinate(ip);
363 ddc::Coordinate<curvilinear_tag_r, curvilinear_tag_theta> const coord(0, th);
364 double const deriv_1_x
365 = m_spline_evaluator.deriv_dim_1(coord, m_x_spline_representation.span_cview());
366 double const deriv_1_2_x
367 = m_spline_evaluator
368 .deriv_1_and_2(coord, m_x_spline_representation.span_cview());
369 double const deriv_1_y
370 = m_spline_evaluator.deriv_dim_1(coord, m_y_spline_representation.span_cview());
371 double const deriv_1_2_y
372 = m_spline_evaluator
373 .deriv_1_and_2(coord, m_y_spline_representation.span_cview());
374
375 // Matrix from pseudo-Cart domain to physical domain by logical domain
376 double const j11 = deriv_1_x * Kokkos::cos(th) - deriv_1_2_x * Kokkos::sin(th);
377 double const j12 = deriv_1_x * Kokkos::sin(th) + deriv_1_2_x * Kokkos::cos(th);
378 double const j21 = deriv_1_y * Kokkos::cos(th) - deriv_1_2_y * Kokkos::sin(th);
379 double const j22 = deriv_1_y * Kokkos::sin(th) + deriv_1_2_y * Kokkos::cos(th);
380
381 double const jacobian = j11 * j22 - j12 * j21;
382 // Matrix from physical domain to pseudo_cart domain by logical domain
383 assert(fabs(jacobian) >= 1e-16);
384
385 matrix[0][0] += j22 / jacobian;
386 matrix[0][1] += -j12 / jacobian;
387 matrix[1][0] += -j21 / jacobian;
388 matrix[1][1] += j11 / jacobian;
389 }
390
391 int const theta_size = m_idx_range_singular_point.size();
392 matrix[0][0] /= theta_size;
393 matrix[0][1] /= theta_size;
394 matrix[1][0] /= theta_size;
395 matrix[1][1] /= theta_size;
396 }
