306  void to_pseudo_cartesian_jacobian_center_matrix(Matrix_2x2& matrix) const
307  {
308  matrix[0][0] = 0;
309  matrix[0][1] = 0;
310  matrix[1][0] = 0;
311  matrix[1][1] = 0;
312 
313  // Average the values at (r = 0, theta):
314  ddc::for_each(m_idx_range_theta, [&](auto const ip) {
315  const double th = ddc::coordinate(ip);
316  ddc::Coordinate<circular_tag_r, circular_tag_theta> const coord(0, th);
317  double const deriv_1_x
318  = m_spline_evaluator.deriv_dim_1(coord, m_x_spline_representation.span_cview());
319  double const deriv_1_2_x
320  = m_spline_evaluator
321  .deriv_1_and_2(coord, m_x_spline_representation.span_cview());
322  double const deriv_1_y
323  = m_spline_evaluator.deriv_dim_1(coord, m_y_spline_representation.span_cview());
324  double const deriv_1_2_y
325  = m_spline_evaluator
326  .deriv_1_and_2(coord, m_y_spline_representation.span_cview());
327 
328  // Matrix from pseudo-Cart domain to physical domain by logical domain
329  double const j11 = deriv_1_x * std::cos(th) - deriv_1_2_x * std::sin(th);
330  double const j12 = deriv_1_x * std::sin(th) + deriv_1_2_x * std::cos(th);
331  double const j21 = deriv_1_y * std::cos(th) - deriv_1_2_y * std::sin(th);
332  double const j22 = deriv_1_y * std::sin(th) + deriv_1_2_y * std::cos(th);
333 
334  double const jacobian = j11 * j22 - j12 * j21;
335  // Matrix from physical domain to pseudo_cart domain by logical domain
336  if (fabs(jacobian) <= 1e-16) {
337  std::cout << "WARNING! - Non invertible Jacobian matrix. ((r, theta) = (0, " << th
338  << "))" << std::endl;
339  }
340  assert(fabs(jacobian) >= 1e-16);
341 
342  matrix[0][0] += j22 / jacobian;
343  matrix[0][1] += -j12 / jacobian;
344  matrix[1][0] += -j21 / jacobian;
345  matrix[1][1] += j11 / jacobian;
346  });
347 
348  int const theta_size = m_idx_range_theta.size();
349  matrix[0][0] /= theta_size;
350  matrix[0][1] /= theta_size;
351  matrix[1][0] /= theta_size;
352  matrix[1][1] /= theta_size;
353  }
