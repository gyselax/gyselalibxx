309  void to_pseudo_cartesian_jacobian_center_matrix(Matrix_2x2& matrix) const final
310  {
311  matrix[0][0] = 0;
312  matrix[0][1] = 0;
313  matrix[1][0] = 0;
314  matrix[1][1] = 0;
315 
316  // Average the values at (r = 0, theta):
317  ddc::for_each(m_idx_range_theta, [&](auto const ip) {
318  const double th = ddc::coordinate(ip);
319  ddc::Coordinate<curvilinear_tag_r, curvilinear_tag_theta> const coord(0, th);
320  double const deriv_1_x
321  = m_spline_evaluator.deriv_dim_1(coord, m_x_spline_representation.span_cview());
322  double const deriv_1_2_x
323  = m_spline_evaluator
324  .deriv_1_and_2(coord, m_x_spline_representation.span_cview());
325  double const deriv_1_y
326  = m_spline_evaluator.deriv_dim_1(coord, m_y_spline_representation.span_cview());
327  double const deriv_1_2_y
328  = m_spline_evaluator
329  .deriv_1_and_2(coord, m_y_spline_representation.span_cview());
330 
331  // Matrix from pseudo-Cart domain to physical domain by logical domain
332  double const j11 = deriv_1_x * std::cos(th) - deriv_1_2_x * std::sin(th);
333  double const j12 = deriv_1_x * std::sin(th) + deriv_1_2_x * std::cos(th);
334  double const j21 = deriv_1_y * std::cos(th) - deriv_1_2_y * std::sin(th);
335  double const j22 = deriv_1_y * std::sin(th) + deriv_1_2_y * std::cos(th);
336 
337  double const jacobian = j11 * j22 - j12 * j21;
338  // Matrix from physical domain to pseudo_cart domain by logical domain
339  if (fabs(jacobian) <= 1e-16) {
340  std::cout << "WARNING! - Non invertible Jacobian matrix. ((r, theta) = (0, " << th
341  << "))" << std::endl;
342  }
343  assert(fabs(jacobian) >= 1e-16);
344 
345  matrix[0][0] += j22 / jacobian;
346  matrix[0][1] += -j12 / jacobian;
347  matrix[1][0] += -j21 / jacobian;
348  matrix[1][1] += j11 / jacobian;
349  });
350 
351  int const theta_size = m_idx_range_theta.size();
352  matrix[0][0] /= theta_size;
353  matrix[0][1] /= theta_size;
354  matrix[1][0] /= theta_size;
355  matrix[1][1] /= theta_size;
356  }
