161 KOKKOS_FUNCTION DTensor<VectorIndexSet<X, Y>, VectorIndexSet<R_cov, Theta_cov>> jacobian_matrix(

273 Coord<curvilinear_tag_r, curvilinear_tag_theta> const& coord) const
274 {
275 Tensor J = jacobian_matrix(coord);
276 return ddcHelper::get<X, R_cov>(J) * ddcHelper::get<Y, Theta_cov>(J)
277 - ddcHelper::get<Y, R_cov>(J) * ddcHelper::get<X, Theta_cov>(J);

294 KOKKOS_INLINE_FUNCTION DTensor<VectorIndexSet<X, Y>, VectorIndexSet<R_cov, Theta_cov>>
295 first_order_jacobian_matrix_r_rtheta(
296 Coord<curvilinear_tag_r, curvilinear_tag_theta> const& coord) const
297 {
298 DTensor<VectorIndexSet<X, Y>, VectorIndexSet<R_cov, Theta_cov>> J;
299 ddcHelper::get<X, R_cov>(J)
300 = m_spline_evaluator.deriv_dim_1(coord, get_const_field(m_x_spline_representation));
301 ddcHelper::get<X, Theta_cov>(J)
302 = m_spline_evaluator
303 .deriv_1_and_2(coord, get_const_field(m_x_spline_representation));
304 ddcHelper::get<Y, R_cov>(J)
305 = m_spline_evaluator.deriv_dim_1(coord, get_const_field(m_y_spline_representation));
306 ddcHelper::get<Y, Theta_cov>(J)
307 = m_spline_evaluator
308 .deriv_1_and_2(coord, get_const_field(m_y_spline_representation));
309 return J;
310 }
