311  KOKKOS_FUNCTION void to_pseudo_cartesian_jacobian_center_matrix(Matrix_2x2& matrix) const final
312  {
313  matrix[0][0] = 0;
314  matrix[0][1] = 0;
315  matrix[1][0] = 0;
316  matrix[1][1] = 0;
317 
318  // Average the values at (r = 0, theta):
319  for (IdxTheta ip : m_idx_range_theta) {
320  const double th = ddc::coordinate(ip);
321  ddc::Coordinate<curvilinear_tag_r, curvilinear_tag_theta> const coord(0, th);
322  double const deriv_1_x
323  = m_spline_evaluator.deriv_dim_1(coord, m_x_spline_representation.span_cview());
324  double const deriv_1_2_x
325  = m_spline_evaluator
326  .deriv_1_and_2(coord, m_x_spline_representation.span_cview());
327  double const deriv_1_y
328  = m_spline_evaluator.deriv_dim_1(coord, m_y_spline_representation.span_cview());
329  double const deriv_1_2_y
330  = m_spline_evaluator
331  .deriv_1_and_2(coord, m_y_spline_representation.span_cview());
332 
333  // Matrix from pseudo-Cart domain to physical domain by logical domain
334  double const j11 = deriv_1_x * Kokkos::cos(th) - deriv_1_2_x * Kokkos::sin(th);
335  double const j12 = deriv_1_x * Kokkos::sin(th) + deriv_1_2_x * Kokkos::cos(th);
336  double const j21 = deriv_1_y * Kokkos::cos(th) - deriv_1_2_y * Kokkos::sin(th);
337  double const j22 = deriv_1_y * Kokkos::sin(th) + deriv_1_2_y * Kokkos::cos(th);
338 
339  double const jacobian = j11 * j22 - j12 * j21;
340  // Matrix from physical domain to pseudo_cart domain by logical domain
341  assert(fabs(jacobian) >= 1e-16);
342 
343  matrix[0][0] += j22 / jacobian;
344  matrix[0][1] += -j12 / jacobian;
345  matrix[1][0] += -j21 / jacobian;
346  matrix[1][1] += j11 / jacobian;
347  }
348 
349  int const theta_size = m_idx_range_theta.size();
350  matrix[0][0] /= theta_size;
351  matrix[0][1] /= theta_size;
352  matrix[1][0] /= theta_size;
353  matrix[1][1] /= theta_size;
354  }
