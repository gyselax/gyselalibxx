130 KOKKOS_INLINE_FUNCTION Matrix_2x2 operator()() const
131 {
132 Matrix_2x2 J;
133 const double epsilon
134 = m_mapping.template get<CzarnyToCartesian<R, Theta, X, Y>>().epsilon();
135 const double e = m_mapping.template get<CzarnyToCartesian<R, Theta, X, Y>>().e();
136 const double xi = Kokkos::sqrt(1. / (1. - epsilon * epsilon * 0.25));
137 const double sqrt_eps_2 = Kokkos::sqrt(1. + epsilon * epsilon);
138 J[0][0] = -sqrt_eps_2;
139 J[0][1] = 0.;
140 J[1][0] = 0.;
141 J[1][1] = (2 - sqrt_eps_2) / e / xi;
142 return J;
143 }

175 using IdxRangeR = typename SplineEvaluator::evaluation_domain_type1;
176 using IdxRangeTheta = typename SplineEvaluator::evaluation_domain_type2;
177 using IdxRangeRTheta = typename SplineEvaluator::evaluation_domain_type;
178 using IdxR = typename IdxRangeR::discrete_element_type;
179 using IdxTheta = typename IdxRangeTheta::discrete_element_type;

229 KOKKOS_FUNCTION Matrix_2x2 operator()() const
230 {
231 DiscreteToCartesian<X, Y, SplineEvaluator, R, Theta, MemorySpace> const& discrete_mapping
232 = m_mapping.template get<
233 DiscreteToCartesian<X, Y, SplineEvaluator, R, Theta, MemorySpace>>();
234 Matrix_2x2 J;
235 J[0][0] = 0;
236 J[0][1] = 0;
237 J[1][0] = 0;
238 J[1][1] = 0;
239 IdxRangeRTheta idx_range_singular_point = discrete_mapping.idx_range_singular_point();
240 // Average the values at (r = 0, theta):
241 IdxR ir(idx_range_singular_point.front());
242 for (IdxTheta ip : IdxRangeTheta(idx_range_singular_point)) {
243 Coord<R, Theta> coord(ddc::coordinate(ir), ddc::coordinate(ip));
244 Matrix_2x2 J_first_order = discrete_mapping.first_order_jacobian_matrix_r_rtheta(coord);
245
246 double th = ddc::get<Theta>(coord);
247 Matrix_2x2 J_circ_r_rtheta;
248 J_circ_r_rtheta[0][0] = Kokkos::cos(th);
249 J_circ_r_rtheta[0][1] = Kokkos::sin(th);
250 J_circ_r_rtheta[1][0] = -Kokkos::sin(th);
251 J_circ_r_rtheta[1][1] = Kokkos::cos(th);
252
253 Matrix_2x2 J_theta;
254 J_theta = inverse(mat_mul(J_first_order, J_circ_r_rtheta));
255
256 J[0][0] += J_theta[0][0];
257 J[0][1] += J_theta[0][1];
258 J[1][0] += J_theta[1][0];
259 J[1][1] += J_theta[1][1];
260 }
261
262 int const theta_size = idx_range_singular_point.size();
263 J[0][0] /= theta_size;
264 J[0][1] /= theta_size;
265 J[1][0] /= theta_size;
266 J[1][1] /= theta_size;
267 return J;
268 }
269};
