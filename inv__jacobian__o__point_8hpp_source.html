144 KOKKOS_INLINE_FUNCTION DTensor<VectorIndexSet<Xpc, Ypc>, VectorIndexSet<X_cov, Y_cov>>
145 operator()() const
146 {
147 const double epsilon = m_mapping.template get<CzarnyToCart>().epsilon();
148 const double e = m_mapping.template get<CzarnyToCart>().e();
149 const double xi = Kokkos::sqrt(1. / (1. - epsilon * epsilon * 0.25));
150 const double sqrt_eps_2 = Kokkos::sqrt(1. + epsilon * epsilon);
151
152 DTensor<VectorIndexSet<Xpc, Ypc>, VectorIndexSet<X_cov, Y_cov>> J;
153 ddcHelper::get<Xpc, X_cov>(J) = -sqrt_eps_2;
154 ddcHelper::get<Xpc, Y_cov>(J) = 0.;
155 ddcHelper::get<Ypc, X_cov>(J) = 0.;
156 ddcHelper::get<Ypc, Y_cov>(J) = (2 - sqrt_eps_2) / e / xi;
157 return J;
158 }

254 KOKKOS_FUNCTION DTensor<VectorIndexSet<Xpc, Ypc>, VectorIndexSet<X_cov, Y_cov>> operator()()
255 const
256 {
257 DiscreteToCartesian<X, Y, SplineEvaluator, R, Theta, MemorySpace> const& discrete_mapping
258 = m_mapping.template get<
259 DiscreteToCartesian<X, Y, SplineEvaluator, R, Theta, MemorySpace>>();
260 DTensor<VectorIndexSet<Xpc, Ypc>, VectorIndexSet<X_cov, Y_cov>> J(0.0);
261 IdxRangeRTheta idx_range_singular_point = discrete_mapping.idx_range_singular_point();
262 // Average the values at (r = 0, theta):
263 IdxR ir(idx_range_singular_point.front());
264 for (IdxTheta itheta : IdxRangeTheta(idx_range_singular_point)) {
265 Coord<R, Theta> coord(ddc::coordinate(ir), ddc::coordinate(itheta));
266 DTensor<VectorIndexSet<X, Y>, VectorIndexSet<R_cov, Theta_cov>> J_first_order
267 = discrete_mapping.first_order_jacobian_matrix_r_rtheta(coord);
268
269 double th = ddc::get<Theta>(coord);
270 DTensor<VectorIndexSet<R, Theta>, VectorIndexSet<Xpc_cov, Ypc_cov>> J_circ_r_rtheta;
271 ddcHelper::get<R, Xpc_cov>(J_circ_r_rtheta) = Kokkos::cos(th);
272 ddcHelper::get<R, Ypc_cov>(J_circ_r_rtheta) = Kokkos::sin(th);
273 ddcHelper::get<Theta, Xpc_cov>(J_circ_r_rtheta) = -Kokkos::sin(th);
274 ddcHelper::get<Theta, Ypc_cov>(J_circ_r_rtheta) = Kokkos::cos(th);
275
276 Tensor J_theta = inverse(
277 tensor_mul(index<'i', 'j'>(J_first_order), index<'j', 'k'>(J_circ_r_rtheta)));
278
279 J += J_theta;
280 }
281
282 int const theta_size = idx_range_singular_point.size();
283 J /= theta_size;
284 return J;
285 }
