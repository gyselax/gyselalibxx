170template <char... ids, class TensorType>

172{

175 "The number of indices provided must be equal to the rank of the tensor.");

177 using TypeSeqCharIds = ddc::detail::TypeSeq<std::integral_constant<char, ids>...>;

193template <class... IndexedTensorType>
194KOKKOS_FUNCTION auto tensor_mul(IndexedTensorType... tensor_to_mul)
195{
196 using namespace tensor_tools;
197 static_assert(
198 (is_indexed_tensor_v<IndexedTensorType> && ...),
199 "A tensor multiplication must be carried out over IndexedTensor objects");
200 // Get the TypeSeq of VectorIndexIdMaps describing all the indices which appear in the calculation.
201 using AllIndexIdMaps = type_seq_cat_t<typename IndexedTensorType::index_pattern...>;
202 // Use the first tensor argument to extract the element type of the result.
203 using ElementType = std::tuple_element_t<
204 0,
205 std::tuple<typename IndexedTensorType::tensor_type::element_type...>>;
206 static_assert(
207 (std::is_same_v<
208 typename IndexedTensorType::tensor_type::element_type,
209 ElementType> && ...),
210 "All tensors must have the same element type");
211 using ResultIndexTypeSeq = non_repeated_indices_t<AllIndexIdMaps>;
212 if constexpr (ddc::type_seq_size_v<ResultIndexTypeSeq> == 0) {
213 ElementType result = 0.0;
214 details::internal_tensor_mul<AllIndexIdMaps>(
215 result,
216 std::make_index_sequence<
217 details::CalculateSize<unique_indices_t<AllIndexIdMaps>>::value>(),
218 tensor_to_mul...);
219 return result;
220 } else {
221 using ResultTensorType
222 = to_tensor_t<ElementType, get_type_seq_vector_index_set_t<ResultIndexTypeSeq>>;
223 ResultTensorType result(0.0);
224 IndexedTensor<ResultTensorType, ResultIndexTypeSeq> indexed_result(result);
225 details::internal_tensor_mul<AllIndexIdMaps>(
226 indexed_result,
227 std::make_index_sequence<
228 details::CalculateSize<unique_indices_t<AllIndexIdMaps>>::value>(),
229 tensor_to_mul...);
230 return result;
231 }
