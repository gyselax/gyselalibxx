34template <
35 class ElementType,
36 class Extents,
37 class Layout,
38 class Accessor,
39 std::size_t I0,
40 std::size_t... Is>
41std::ostream& stream_impl(
42 std::ostream& os,
43 Kokkos::mdspan<ElementType, Extents, Layout, Accessor> const& s,
44 std::index_sequence<I0, Is...>)
45{
46 if constexpr (sizeof...(Is) > 0) {
47 os << '[';
48 for (std::size_t i0 = 0; i0 < s.extent(I0); ++i0) {
49 stream_impl(
50 os,
51 Kokkos::submdspan(s, i0, ((void)Is, Kokkos::full_extent)...),
52 std::make_index_sequence<sizeof...(Is)>());
53 }
54 os << ']';
55 } else {
56 os << '[';
57 for (std::size_t i0 = 0; i0 < s.extent(I0) - 1; ++i0) {
58 os << s(i0) << ',';
59 }
60 os << s(s.extent(I0) - 1) << ']';
61 }
62 return os;
63}
