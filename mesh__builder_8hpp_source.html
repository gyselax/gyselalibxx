9template <class Dim, class Grid1D>
10std::vector<Coord<Dim>> build_uniform_break_points(
11 Coord<Dim> min,
12 Coord<Dim> max,
13 IdxStep<Grid1D> n_cells)
14{
15 static_assert(std::is_same_v<typename Grid1D::continuous_dimension_type, Dim>);
16 std::vector<Coord<Dim>> break_points(n_cells + 1);
17 double const delta((max - min) / double(n_cells));
18
19 break_points[0] = min;
20 for (int i(1); i < n_cells; ++i) {
21 break_points[i] = min + i * delta;
22 }
23 break_points[n_cells] = max;
24 return break_points;
25}

39
40 break_points[0] = min;
