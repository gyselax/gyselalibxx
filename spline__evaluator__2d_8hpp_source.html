502  template <class EvalType1, class EvalType2>
503  double eval_no_bc(
504  ddc::Coordinate<Dim1> const& coord_eval1,
505  ddc::Coordinate<Dim2> const& coord_eval2,
506  ddc::ChunkSpan<double const, ddc::DiscreteDomain<BSplinesType1, BSplinesType2>> const
507  spline_coef,
508  DSpan1D const vals1,
509  DSpan1D const vals2,
510  EvalType1 const eval_type_1,
511  EvalType2 const eval_type_2) const
512  {
513  static_assert(
514  std::is_same_v<EvalType1, eval_type> || std::is_same_v<EvalType1, eval_deriv_type>);
515  static_assert(
516  std::is_same_v<EvalType2, eval_type> || std::is_same_v<EvalType2, eval_deriv_type>);
517  ddc::DiscreteElement<BSplinesType1> jmin1;
518  ddc::DiscreteElement<BSplinesType2> jmin2;
519 
520  if constexpr (std::is_same_v<EvalType1, eval_type>) {
521  jmin1 = ddc::discrete_space<bsplines_type1>().eval_basis(vals1, coord_eval1);
522  } else if constexpr (std::is_same_v<EvalType1, eval_deriv_type>) {
523  jmin1 = ddc::discrete_space<bsplines_type1>().eval_deriv(vals1, coord_eval1);
524  }
525  if constexpr (std::is_same_v<EvalType2, eval_type>) {
526  jmin2 = ddc::discrete_space<bsplines_type2>().eval_basis(vals2, coord_eval2);
527  } else if constexpr (std::is_same_v<EvalType2, eval_deriv_type>) {
528  jmin2 = ddc::discrete_space<bsplines_type2>().eval_deriv(vals2, coord_eval2);
529  }
530 
531  double y = 0.0;
532  for (std::size_t i = 0; i < bsplines_type1::degree() + 1; ++i) {
533  for (std::size_t j = 0; j < bsplines_type2::degree() + 1; ++j) {
534  y += spline_coef(jmin1 + i, jmin2 + j) * vals1(i) * vals2(j);
535  }
536  }
537  return y;
538  }
