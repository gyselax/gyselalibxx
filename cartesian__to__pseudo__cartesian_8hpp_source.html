87  KOKKOS_FUNCTION void jacobian_matrix(CoordRTheta const& coord_rtheta, Matrix_2x2& J) const
88  {
89  double r = ddc::get<R>(coord_rtheta);
90 
91  Matrix_2x2 inv_jacobian_from_cartesian;
92  Matrix_2x2 jacobian_from_pseudo_cartesian;
93  if (r < m_epsilon) {
94  Matrix_2x2 J_0;
95  m_mapping_to_cartesian.to_pseudo_cartesian_jacobian_center_matrix(J_0);
96 
97  CoordRTheta coord_eps(m_epsilon, ddc::get<Theta>(coord_rtheta));
98 
99  m_mapping_to_cartesian.inv_jacobian_matrix(coord_eps, inv_jacobian_from_cartesian);
100  m_mapping_to_pseudo_cartesian
101  .jacobian_matrix(coord_eps, jacobian_from_pseudo_cartesian);
102  Matrix_2x2 J_eps = mat_mul(jacobian_from_pseudo_cartesian, inv_jacobian_from_cartesian);
103 
104  J[0][0] = (1 - r / m_epsilon) * J_0[0][0] + r / m_epsilon * J_eps[0][0];
105  J[0][1] = (1 - r / m_epsilon) * J_0[0][1] + r / m_epsilon * J_eps[0][1];
106  J[1][0] = (1 - r / m_epsilon) * J_0[1][0] + r / m_epsilon * J_eps[1][0];
107  J[1][1] = (1 - r / m_epsilon) * J_0[1][1] + r / m_epsilon * J_eps[1][1];
108  } else {
109  m_mapping_to_cartesian.inv_jacobian_matrix(coord_rtheta, inv_jacobian_from_cartesian);
110  m_mapping_to_pseudo_cartesian
111  .jacobian_matrix(coord_rtheta, jacobian_from_pseudo_cartesian);
112  J = mat_mul(jacobian_from_pseudo_cartesian, inv_jacobian_from_cartesian);
113  }
