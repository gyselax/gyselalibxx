98 KOKKOS_FUNCTION void jacobian_matrix(CoordRTheta const& coord_rtheta, Matrix_2x2& J) const
99 {
100 double r = ddc::get<R>(coord_rtheta);
101
102 Matrix_2x2 inv_jacobian_from_cartesian;
103 Matrix_2x2 jacobian_from_pseudo_cartesian;
104 InverseJacobianMatrix inv_jacobian_cartesian(m_mapping_to_cartesian);
105 if (r < m_epsilon) {
106 Matrix_2x2 J_0;
107 m_mapping_to_cartesian.to_pseudo_cartesian_jacobian_center_matrix(J_0);
108
109 CoordRTheta coord_eps(m_epsilon, ddc::get<Theta>(coord_rtheta));
110
111 inv_jacobian_from_cartesian = inv_jacobian_cartesian(coord_eps);
112 m_mapping_to_pseudo_cartesian
113 .jacobian_matrix(coord_eps, jacobian_from_pseudo_cartesian);
114 Matrix_2x2 J_eps = mat_mul(jacobian_from_pseudo_cartesian, inv_jacobian_from_cartesian);
115
116 J[0][0] = (1 - r / m_epsilon) * J_0[0][0] + r / m_epsilon * J_eps[0][0];
117 J[0][1] = (1 - r / m_epsilon) * J_0[0][1] + r / m_epsilon * J_eps[0][1];
118 J[1][0] = (1 - r / m_epsilon) * J_0[1][0] + r / m_epsilon * J_eps[1][0];
119 J[1][1] = (1 - r / m_epsilon) * J_0[1][1] + r / m_epsilon * J_eps[1][1];
120 } else {
121 inv_jacobian_from_cartesian = inv_jacobian_cartesian(coord_rtheta);
122 m_mapping_to_pseudo_cartesian
123 .jacobian_matrix(coord_rtheta, jacobian_from_pseudo_cartesian);
124 J = mat_mul(jacobian_from_pseudo_cartesian, inv_jacobian_from_cartesian);
125 }
126 }

206 KOKKOS_FUNCTION void inv_jacobian_matrix(CoordRTheta const& coord_rtheta, Matrix_2x2& J) const
207 {
208 double r = ddc::get<R>(coord_rtheta);
209
210 Matrix_2x2 jacobian_from_cartesian;
211 Matrix_2x2 inv_jacobian_from_pseudo_cartesian;
212 InverseJacobianMatrix inv_jacobian_pseudo_cartesian(m_mapping_to_pseudo_cartesian);
213 if (r < m_epsilon) {
214 Matrix_2x2 J_0;
215 m_mapping_to_cartesian.to_pseudo_cartesian_jacobian_center_matrix(J_0);
216 Matrix_2x2 J_0_inv = inverse(J_0);
217
218 CoordRTheta coord_eps(m_epsilon, ddc::get<Theta>(coord_rtheta));
219
220 m_mapping_to_cartesian.jacobian_matrix(coord_eps, jacobian_from_cartesian);
221 inv_jacobian_from_pseudo_cartesian = inv_jacobian_pseudo_cartesian(coord_eps);
222 Matrix_2x2 J_eps = mat_mul(inv_jacobian_from_pseudo_cartesian, jacobian_from_cartesian);
223
224 J[0][0] = (1 - r / m_epsilon) * J_0_inv[0][0] + r / m_epsilon * J_eps[0][0];
225 J[0][1] = (1 - r / m_epsilon) * J_0_inv[0][1] + r / m_epsilon * J_eps[0][1];
226 J[1][0] = (1 - r / m_epsilon) * J_0_inv[1][0] + r / m_epsilon * J_eps[1][0];
227 J[1][1] = (1 - r / m_epsilon) * J_0_inv[1][1] + r / m_epsilon * J_eps[1][1];
228 } else {
229 m_mapping_to_cartesian.jacobian_matrix(coord_rtheta, jacobian_from_cartesian);
230 inv_jacobian_from_pseudo_cartesian = inv_jacobian_pseudo_cartesian(coord_rtheta);
231 J = mat_mul(jacobian_from_cartesian, inv_jacobian_from_pseudo_cartesian);
232 }
233 }
