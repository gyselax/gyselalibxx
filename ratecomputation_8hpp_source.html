10template <typename CoeffView>
11DFieldSpX compute_rate_electron_density_temperature(
12 Kokkos::DefaultExecutionSpace exec_space,
13 DFieldSpX rate,
14 DConstFieldSpX density,
15 DConstFieldSpX temperature,
16 CoeffView slope_coefficients,
17 CoeffView intercept_coefficients,
18 double norm_coeff_rate)
19{
20 ddc::parallel_for_each(
21 exec_space,
22 get_idx_range(rate),
23 KOKKOS_LAMBDA(IdxSpX const ifspx) {
24 double temperature_log10
25 = Kokkos::log10(temperature(ielec(), ddc::select<GridX>(ifspx)));
26 double density_log10 = Kokkos::log10(density(ielec(), ddc::select<GridX>(ifspx)));
27
28 double rate_log10 = 0.0;
29 for (int i = 0; i < slope_coefficients.extent(0); ++i) {
30 double polynomial_cs
31 = slope_coefficients(i) * density_log10 + intercept_coefficients(i);
32 rate_log10 += polynomial_cs * Kokkos::pow(temperature_log10, i);
33 }
34 rate(ifspx) = Kokkos::pow(10, rate_log10) * norm_coeff_rate;
35 });
36 return rate;
37}
