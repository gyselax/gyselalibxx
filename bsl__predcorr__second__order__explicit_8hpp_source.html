143  DSpanRP operator()(DSpanRP allfdistribu, double const dt, int const steps) const
144  {
145  std::chrono::time_point<std::chrono::system_clock> start_time
146  = std::chrono::system_clock::now();
147  std::chrono::time_point<std::chrono::system_clock> end_time;
148 
149  // Grid. ------------------------------------------------------------------------------------------
150  IDomainRP const grid(allfdistribu.domain<IDimR, IDimP>());
151 
152  FieldRP<CoordRP> coords(grid);
153  ddc::for_each(grid, [&](IndexRP const irp) { coords(irp) = ddc::coordinate(irp); });
154 
155  BSDomainR radial_bsplines(ddc::discrete_space<BSplinesR>().full_domain().remove_first(
156  ddc::DiscreteVector<BSplinesR> {PolarBSplinesRP::continuity + 1}));
157  BSDomainP polar_domain(ddc::discrete_space<BSplinesP>().full_domain());
158 
159  // --- Electrostatic potential (phi). -------------------------------------------------------------
160  DFieldRP electrical_potential(grid);
161 
162  SplinePolar electrostatic_potential_coef(
163  PolarBSplinesRP::singular_domain(),
164  BSDomainRP(radial_bsplines, polar_domain));
165 
166  PolarSplineEvaluator<PolarBSplinesRP> polar_spline_evaluator(
167  g_polar_null_boundary_2d<PolarBSplinesRP>);
168 
169  // --- For the computation of advection field from the electrostatic potential (phi): -------------
170  VectorDFieldRP<RDimX, RDimY> electric_field(grid);
171  VectorDFieldRP<RDimX, RDimY> electric_field_predicted(grid);
172  VectorDFieldRP<RDimX, RDimY> advection_field(grid);
173  VectorDFieldRP<RDimX, RDimY> advection_field_predicted(grid);
174 
175  AdvectionFieldFinder advection_field_computer(m_mapping);
176 
177 
178 
179  // --- Parameter for linearisation of advection field: --------------------------------------------
180  start_time = std::chrono::system_clock::now();
181  for (int iter(0); iter < steps; ++iter) {
182  double const time = iter * dt;
183  // STEP 1: From rho^n, we compute phi^n: Poisson equation
184  Spline2D allfdistribu_coef(m_builder.spline_domain());
185  m_builder(allfdistribu_coef, allfdistribu);
186  PoissonRHSFunction const charge_density_coord_1(allfdistribu_coef, m_evaluator);
187  m_poisson_solver(charge_density_coord_1, electrostatic_potential_coef);
188 
189  polar_spline_evaluator(
190  electrical_potential.span_view(),
191  coords.span_cview(),
192  electrostatic_potential_coef);
193 
194  ddc::PdiEvent("iteration")
195  .with("iter", iter)
196  .and_with("time", time)
197  .and_with("density", allfdistribu)
198  .and_with("electrical_potential", electrical_potential);
199 
200  // STEP 2: From phi^n, we compute A^n:
201  advection_field_computer(electrostatic_potential_coef, advection_field);
202 
203 
204  // STEP 3: From rho^n and A^n, we compute rho^P: Vlasov equation
205  // --- Copy rho^n because it will be modified:
206  DFieldRP allfdistribu_predicted(grid);
207  ddc::parallel_deepcopy(allfdistribu_predicted.span_view(), allfdistribu);
208  m_advection_solver(allfdistribu_predicted.span_view(), advection_field.span_view(), dt);
209 
210  // --- advect also the feet because it is needed for the next step
211  FieldRP<CoordRP> feet_coords(grid);
212  ddc::for_each(grid, [&](IndexRP const irp) {
213  feet_coords(irp) = CoordRP(ddc::coordinate(irp));
214  });
215  m_find_feet(feet_coords.span_view(), advection_field.span_view(), dt);
216 
217 
218  // STEP 4: From rho^P, we compute phi^P: Poisson equation
219  m_builder(allfdistribu_coef, allfdistribu);
220  PoissonRHSFunction const charge_density_coord_4(allfdistribu_coef, m_evaluator);
221  m_poisson_solver(charge_density_coord_4, electrostatic_potential_coef);
222 
223  // STEP 5: From phi^P, we compute A^P:
224  advection_field_computer(electrostatic_potential_coef, advection_field_predicted);
225 
226 
227  // --- we evaluate the advection field A^n at the characteristic feet X^P
228  VectorDFieldRP<RDimX, RDimY> advection_field_evaluated(grid);
229  VectorSpline2D<RDimX, RDimY> advection_field_coefs(m_builder.spline_domain());
230 
231  m_builder(
232  ddcHelper::get<RDimX>(advection_field_coefs),
233  ddcHelper::get<RDimX>(advection_field));
234  m_builder(
235  ddcHelper::get<RDimY>(advection_field_coefs),
236  ddcHelper::get<RDimY>(advection_field));
237 
238  m_evaluator(
239  ddcHelper::get<RDimX>(advection_field_evaluated).span_view(),
240  feet_coords.span_cview(),
241  ddcHelper::get<RDimX>(advection_field_coefs));
242  m_evaluator(
243  ddcHelper::get<RDimY>(advection_field_evaluated).span_view(),
244  feet_coords.span_cview(),
245  ddcHelper::get<RDimY>(advection_field_coefs));
246 
247 
248  // STEP 6: From rho^n and (A^n(X^P) + A^P(X^n))/2, we compute rho^{n+1}: Vlasov equation
249  ddc::for_each(grid, [&](IndexRP const irp) {
250  ddcHelper::get<RDimX>(advection_field)(irp)
251  = (ddcHelper::get<RDimX>(advection_field_evaluated)(irp)
252  + ddcHelper::get<RDimX>(advection_field_predicted)(irp))
253  / 2.;
254  ddcHelper::get<RDimY>(advection_field)(irp)
255  = (ddcHelper::get<RDimY>(advection_field_evaluated)(irp)
256  + ddcHelper::get<RDimY>(advection_field_predicted)(irp))
257  / 2.;
258  });
259 
260 
261  m_advection_solver(allfdistribu, advection_field.span_view(), dt);
262  }
263 
264  // STEP 1: From rho^n, we compute phi^n: Poisson equation
265  Spline2D allfdistribu_coef(m_builder.spline_domain());
266  m_builder(allfdistribu_coef, allfdistribu);
267  PoissonRHSFunction const charge_density_coord(allfdistribu_coef, m_evaluator);
268  m_poisson_solver(charge_density_coord, coords, electrical_potential);
269 
270  ddc::PdiEvent("last_iteration")
271  .with("iter", steps)
272  .and_with("time", steps * dt)
273  .and_with("density", allfdistribu)
274  .and_with("electrical_potential", electrical_potential);
275 
276 
277  end_time = std::chrono::system_clock::now();
278  display_time_difference("Iterations time: ", start_time, end_time);
279 
280 
281  return allfdistribu;
282  }
