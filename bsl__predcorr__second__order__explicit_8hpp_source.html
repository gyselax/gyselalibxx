148 host_t<DFieldRTheta> operator()(
149 host_t<DFieldRTheta> allfdistribu,
150 double const dt,
151 int const steps) const final
152 {
153 std::chrono::time_point<std::chrono::system_clock> start_time;
154 std::chrono::time_point<std::chrono::system_clock> end_time;
155
156 // Grid. ------------------------------------------------------------------------------------------
157 IdxRangeRTheta const grid(get_idx_range<GridR, GridTheta>(allfdistribu));
158
159 host_t<FieldMemRTheta<CoordRTheta>> coords(grid);
160 ddc::for_each(grid, [&](IdxRTheta const irp) { coords(irp) = ddc::coordinate(irp); });
161
162 IdxRangeBSR radial_bsplines(ddc::discrete_space<BSplinesR>().full_domain().remove_first(
163 IdxStep<BSplinesR> {PolarBSplinesRTheta::continuity + 1}));
164 IdxRangeBSTheta polar_idx_range(ddc::discrete_space<BSplinesTheta>().full_domain());
165
166 // --- Electrostatic potential (phi). -------------------------------------------------------------
167 DFieldMemRTheta electrical_potential(grid);
168 host_t<DFieldMemRTheta> electrical_potential_host(grid);
169
170 host_t<PolarSplineMemRTheta> electrostatic_potential_coef(
171 PolarBSplinesRTheta::singular_idx_range<PolarBSplinesRTheta>(),
172 IdxRangeBSRTheta(radial_bsplines, polar_idx_range));
173
174 ddc::NullExtrapolationRule extrapolation_rule;
175 PolarSplineEvaluator<PolarBSplinesRTheta, ddc::NullExtrapolationRule>
176 polar_spline_evaluator(extrapolation_rule);
177
178 // --- For the computation of advection field from the electrostatic potential (phi): -------------
179 host_t<DVectorFieldMemRTheta<X, Y>> electric_field_alloc(grid);
180 host_t<DVectorFieldMemRTheta<X, Y>> electric_field_predicted_alloc(grid);
181 host_t<DVectorFieldMemRTheta<X, Y>> advection_field_alloc(grid);
182 host_t<DVectorFieldMemRTheta<X, Y>> advection_field_predicted_alloc(grid);
183
184 host_t<DVectorFieldRTheta<X, Y>> electric_field(electric_field_alloc);
185 host_t<DVectorFieldRTheta<X, Y>> electric_field_predicted(electric_field_predicted_alloc);
186 host_t<DVectorFieldRTheta<X, Y>> advection_field(advection_field_alloc);
187 host_t<DVectorFieldRTheta<X, Y>> advection_field_predicted(advection_field_predicted_alloc);
188
189 AdvectionFieldFinder advection_field_computer(m_logical_to_physical);
190
191
192
193 // --- Parameter for linearisation of advection field: --------------------------------------------
194 start_time = std::chrono::system_clock::now();
195 for (int iter(0); iter < steps; ++iter) {
196 double const time = iter * dt;
197 // STEP 1: From rho^n, we compute phi^n: Poisson equation
198 host_t<Spline2DMem> allfdistribu_coef(get_spline_idx_range(m_builder));
199 m_builder(get_field(allfdistribu_coef), get_const_field(allfdistribu));
200 PoissonLikeRHSFunction const
201 charge_density_coord_1(get_const_field(allfdistribu_coef), m_evaluator);
202 m_poisson_solver(charge_density_coord_1, electrostatic_potential_coef);
203
204 polar_spline_evaluator(
205 get_field(electrical_potential_host),
206 get_const_field(coords),
207 get_const_field(electrostatic_potential_coef));
208
209 ddc::PdiEvent("iteration")
210 .with("iter", iter)
211 .and_with("time", time)
212 .and_with("density", allfdistribu)
213 .and_with("electrical_potential", electrical_potential_host);
214
215 // STEP 2: From phi^n, we compute A^n:
216 advection_field_computer(electrostatic_potential_coef, advection_field);
217
218
219 // STEP 3: From rho^n and A^n, we compute rho^P: Vlasov equation
220 // --- Copy rho^n because it will be modified:
221 host_t<DFieldMemRTheta> allfdistribu_predicted(grid);
222 ddc::parallel_deepcopy(get_field(allfdistribu_predicted), allfdistribu);
223 m_advection_solver(get_field(allfdistribu_predicted), get_field(advection_field), dt);
224
225 // --- advect also the feet because it is needed for the next step
226 host_t<FieldMemRTheta<CoordRTheta>> feet_coords(grid);
227 ddc::for_each(grid, [&](IdxRTheta const irp) {
228 feet_coords(irp) = CoordRTheta(ddc::coordinate(irp));
229 });
230 m_find_feet(get_field(feet_coords), get_field(advection_field), dt);
231
232
233 // STEP 4: From rho^P, we compute phi^P: Poisson equation
234 m_builder(get_field(allfdistribu_coef), get_const_field(allfdistribu_predicted));
235 PoissonLikeRHSFunction const
236 charge_density_coord_4(get_const_field(allfdistribu_coef), m_evaluator);
237 m_poisson_solver(charge_density_coord_4, electrostatic_potential_coef);
238
239 // STEP 5: From phi^P, we compute A^P:
240 advection_field_computer(electrostatic_potential_coef, advection_field_predicted);
241
242
243 // --- we evaluate the advection field A^n at the characteristic feet X^P
244 host_t<DVectorFieldMemRTheta<X, Y>> advection_field_evaluated(grid);
245 host_t<VectorSplineCoeffsMem2D<X, Y>> advection_field_coefs(
246 get_spline_idx_range(m_builder));
247
248 m_builder(
249 ddcHelper::get<X>(advection_field_coefs),
250 ddcHelper::get<X>(get_const_field(advection_field)));
251 m_builder(
252 ddcHelper::get<Y>(advection_field_coefs),
253 ddcHelper::get<Y>(get_const_field(advection_field)));
254
255 m_evaluator(
256 get_field(ddcHelper::get<X>(advection_field_evaluated)),
257 get_const_field(feet_coords),
258 ddcHelper::get<X>(get_const_field(advection_field_coefs)));
259 m_evaluator(
260 get_field(ddcHelper::get<Y>(advection_field_evaluated)),
261 get_const_field(feet_coords),
262 ddcHelper::get<Y>(get_const_field(advection_field_coefs)));
263
264
265 // STEP 6: From rho^n and (A^n(X^P) + A^P(X^n))/2, we compute rho^{n+1}: Vlasov equation
266 ddc::for_each(grid, [&](IdxRTheta const irp) {
267 ddcHelper::get<X>(advection_field)(irp)
268 = (ddcHelper::get<X>(advection_field_evaluated)(irp)
269 + ddcHelper::get<X>(advection_field_predicted)(irp))
270 / 2.;
271 ddcHelper::get<Y>(advection_field)(irp)
272 = (ddcHelper::get<Y>(advection_field_evaluated)(irp)
273 + ddcHelper::get<Y>(advection_field_predicted)(irp))
274 / 2.;
275 });
276
277
278 m_advection_solver(allfdistribu, get_field(advection_field), dt);
279 }
280
281 // STEP 1: From rho^n, we compute phi^n: Poisson equation
282 host_t<Spline2DMem> allfdistribu_coef(get_spline_idx_range(m_builder));
283 m_builder(get_field(allfdistribu_coef), get_const_field(allfdistribu));
284 PoissonLikeRHSFunction const
285 charge_density_coord(get_const_field(allfdistribu_coef), m_evaluator);
286 m_poisson_solver(charge_density_coord, get_field(electrical_potential));
287 ddc::parallel_deepcopy(electrical_potential_host, electrical_potential);
288 ddc::PdiEvent("last_iteration")
289 .with("iter", steps)
290 .and_with("time", steps * dt)
291 .and_with("density", allfdistribu)
292 .and_with("electrical_potential", electrical_potential_host);
293
294
295 end_time = std::chrono::system_clock::now();
296 display_time_difference("Iterations time: ", start_time, end_time);
297
298
299 return allfdistribu;
300 }
