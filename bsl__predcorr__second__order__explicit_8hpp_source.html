138  DFieldRTheta operator()(DFieldRTheta allfdistribu, double const dt, int const steps) const
139  {
140  std::chrono::time_point<std::chrono::system_clock> start_time
141  = std::chrono::system_clock::now();
142  std::chrono::time_point<std::chrono::system_clock> end_time;
143 
144  // Grid. ------------------------------------------------------------------------------------------
145  IdxRangeRTheta const grid(get_idx_range<GridR, GridTheta>(allfdistribu));
146 
147  FieldMemRTheta<CoordRTheta> coords(grid);
148  ddc::for_each(grid, [&](IdxRTheta const irp) { coords(irp) = ddc::coordinate(irp); });
149 
150  BSIdxRangeR radial_bsplines(ddc::discrete_space<BSplinesR>().full_domain().remove_first(
151  IdxStep<BSplinesR> {PolarBSplinesRTheta::continuity + 1}));
152  BSIdxRangeTheta polar_idx_range(ddc::discrete_space<BSplinesTheta>().full_domain());
153 
154  // --- Electrostatic potential (phi). -------------------------------------------------------------
155  DFieldMemRTheta electrical_potential(grid);
156 
157  SplinePolar electrostatic_potential_coef(
158  PolarBSplinesRTheta::singular_idx_range<PolarBSplinesRTheta>(),
159  BSIdxRangeRTheta(radial_bsplines, polar_idx_range));
160 
161  ddc::NullExtrapolationRule extrapolation_rule;
162  PolarSplineEvaluator<PolarBSplinesRTheta, ddc::NullExtrapolationRule>
163  polar_spline_evaluator(extrapolation_rule);
164 
165  // --- For the computation of advection field from the electrostatic potential (phi): -------------
166  DVectorFieldMemRTheta<X, Y> electric_field(grid);
167  DVectorFieldMemRTheta<X, Y> electric_field_predicted(grid);
168  DVectorFieldMemRTheta<X, Y> advection_field(grid);
169  DVectorFieldMemRTheta<X, Y> advection_field_predicted(grid);
170 
171  AdvectionFieldFinder advection_field_computer(m_mapping);
172 
173 
174 
175  // --- Parameter for linearisation of advection field: --------------------------------------------
176  start_time = std::chrono::system_clock::now();
177  for (int iter(0); iter < steps; ++iter) {
178  double const time = iter * dt;
179  // STEP 1: From rho^n, we compute phi^n: Poisson equation
180  Spline2D allfdistribu_coef(get_spline_idx_range(m_builder));
181  m_builder(get_field(allfdistribu_coef), get_const_field(allfdistribu));
182  PoissonLikeRHSFunction const
183  charge_density_coord_1(get_const_field(allfdistribu_coef), m_evaluator);
184  m_poisson_solver(charge_density_coord_1, electrostatic_potential_coef);
185 
186  polar_spline_evaluator(
187  get_field(electrical_potential),
188  get_const_field(coords),
189  electrostatic_potential_coef);
190 
191  ddc::PdiEvent("iteration")
192  .with("iter", iter)
193  .and_with("time", time)
194  .and_with("density", allfdistribu)
195  .and_with("electrical_potential", electrical_potential);
196 
197  // STEP 2: From phi^n, we compute A^n:
198  advection_field_computer(electrostatic_potential_coef, advection_field);
199 
200 
201  // STEP 3: From rho^n and A^n, we compute rho^P: Vlasov equation
202  // --- Copy rho^n because it will be modified:
203  DFieldMemRTheta allfdistribu_predicted(grid);
204  ddc::parallel_deepcopy(get_field(allfdistribu_predicted), allfdistribu);
205  m_advection_solver(get_field(allfdistribu_predicted), get_field(advection_field), dt);
206 
207  // --- advect also the feet because it is needed for the next step
208  FieldMemRTheta<CoordRTheta> feet_coords(grid);
209  ddc::for_each(grid, [&](IdxRTheta const irp) {
210  feet_coords(irp) = CoordRTheta(ddc::coordinate(irp));
211  });
212  m_find_feet(get_field(feet_coords), get_field(advection_field), dt);
213 
214 
215  // STEP 4: From rho^P, we compute phi^P: Poisson equation
216  m_builder(get_field(allfdistribu_coef), get_const_field(allfdistribu));
217  PoissonLikeRHSFunction const
218  charge_density_coord_4(get_const_field(allfdistribu_coef), m_evaluator);
219  m_poisson_solver(charge_density_coord_4, electrostatic_potential_coef);
220 
221  // STEP 5: From phi^P, we compute A^P:
222  advection_field_computer(electrostatic_potential_coef, advection_field_predicted);
223 
224 
225  // --- we evaluate the advection field A^n at the characteristic feet X^Theta
226  DVectorFieldMemRTheta<X, Y> advection_field_evaluated(grid);
227  VectorSplineCoeffsMem2D<X, Y> advection_field_coefs(get_spline_idx_range(m_builder));
228 
229  m_builder(
230  ddcHelper::get<X>(advection_field_coefs),
231  ddcHelper::get<X>(get_const_field(advection_field)));
232  m_builder(
233  ddcHelper::get<Y>(advection_field_coefs),
234  ddcHelper::get<Y>(get_const_field(advection_field)));
235 
236  m_evaluator(
237  ddcHelper::get<X>(advection_field_evaluated).span_view(),
238  get_const_field(feet_coords),
239  ddcHelper::get<X>(get_const_field(advection_field_coefs)));
240  m_evaluator(
241  ddcHelper::get<Y>(advection_field_evaluated).span_view(),
242  get_const_field(feet_coords),
243  ddcHelper::get<Y>(get_const_field(advection_field_coefs)));
244 
245 
246  // STEP 6: From rho^n and (A^n(X^P) + A^P(X^n))/2, we compute rho^{n+1}: Vlasov equation
247  ddc::for_each(grid, [&](IdxRTheta const irp) {
248  ddcHelper::get<X>(advection_field)(irp)
249  = (ddcHelper::get<X>(advection_field_evaluated)(irp)
250  + ddcHelper::get<X>(advection_field_predicted)(irp))
251  / 2.;
252  ddcHelper::get<Y>(advection_field)(irp)
253  = (ddcHelper::get<Y>(advection_field_evaluated)(irp)
254  + ddcHelper::get<Y>(advection_field_predicted)(irp))
255  / 2.;
256  });
257 
258 
259  m_advection_solver(allfdistribu, get_field(advection_field), dt);
260  }
261 
262  // STEP 1: From rho^n, we compute phi^n: Poisson equation
263  Spline2D allfdistribu_coef(get_spline_idx_range(m_builder));
264  m_builder(get_field(allfdistribu_coef), get_const_field(allfdistribu));
265  PoissonLikeRHSFunction const
266  charge_density_coord(get_const_field(allfdistribu_coef), m_evaluator);
267  m_poisson_solver(charge_density_coord, coords, electrical_potential);
268 
269  ddc::PdiEvent("last_iteration")
270  .with("iter", steps)
271  .and_with("time", steps * dt)
272  .and_with("density", allfdistribu)
273  .and_with("electrical_potential", electrical_potential);
274 
275 
276  end_time = std::chrono::system_clock::now();
277  display_time_difference("Iterations time: ", start_time, end_time);
278 
279 
280  return allfdistribu;
281  }
