138 host_t<DFieldRTheta> operator()(
139 host_t<DFieldRTheta> allfdistribu,
140 double const dt,
141 int const steps) const
142 {
143 std::chrono::time_point<std::chrono::system_clock> start_time
144 = std::chrono::system_clock::now();
145 std::chrono::time_point<std::chrono::system_clock> end_time;
146
147 // Grid. ------------------------------------------------------------------------------------------
148 IdxRangeRTheta const grid(get_idx_range<GridR, GridTheta>(allfdistribu));
149
150 host_t<FieldMemRTheta<CoordRTheta>> coords(grid);
151 ddc::for_each(grid, [&](IdxRTheta const irp) { coords(irp) = ddc::coordinate(irp); });
152
153 IdxRangeBSR radial_bsplines(ddc::discrete_space<BSplinesR>().full_domain().remove_first(
154 IdxStep<BSplinesR> {PolarBSplinesRTheta::continuity + 1}));
155 IdxRangeBSTheta polar_idx_range(ddc::discrete_space<BSplinesTheta>().full_domain());
156
157 // --- Electrostatic potential (phi). -------------------------------------------------------------
158 host_t<DFieldMemRTheta> electrical_potential(grid);
159
160 SplinePolar electrostatic_potential_coef(
161 PolarBSplinesRTheta::singular_idx_range<PolarBSplinesRTheta>(),
162 IdxRangeBSRTheta(radial_bsplines, polar_idx_range));
163
164 ddc::NullExtrapolationRule extrapolation_rule;
165 PolarSplineEvaluator<PolarBSplinesRTheta, ddc::NullExtrapolationRule>
166 polar_spline_evaluator(extrapolation_rule);
167
168 // --- For the computation of advection field from the electrostatic potential (phi): -------------
169 host_t<DVectorFieldMemRTheta<X, Y>> electric_field(grid);
170 host_t<DVectorFieldMemRTheta<X, Y>> electric_field_predicted(grid);
171 host_t<DVectorFieldMemRTheta<X, Y>> advection_field(grid);
172 host_t<DVectorFieldMemRTheta<X, Y>> advection_field_predicted(grid);
173
174 AdvectionFieldFinder advection_field_computer(m_mapping);
175
176
177
178 // --- Parameter for linearisation of advection field: --------------------------------------------
179 start_time = std::chrono::system_clock::now();
180 for (int iter(0); iter < steps; ++iter) {
181 double const time = iter * dt;
182 // STEP 1: From rho^n, we compute phi^n: Poisson equation
183 host_t<Spline2DMem> allfdistribu_coef(get_spline_idx_range(m_builder));
184 m_builder(get_field(allfdistribu_coef), get_const_field(allfdistribu));
185 PoissonLikeRHSFunction const
186 charge_density_coord_1(get_const_field(allfdistribu_coef), m_evaluator);
187 m_poisson_solver(charge_density_coord_1, electrostatic_potential_coef);
188
189 polar_spline_evaluator(
190 get_field(electrical_potential),
191 get_const_field(coords),
192 electrostatic_potential_coef);
193
194 ddc::PdiEvent("iteration")
195 .with("iter", iter)
196 .and_with("time", time)
197 .and_with("density", allfdistribu)
198 .and_with("electrical_potential", electrical_potential);
199
200 // STEP 2: From phi^n, we compute A^n:
201 advection_field_computer(electrostatic_potential_coef, advection_field);
202
203
204 // STEP 3: From rho^n and A^n, we compute rho^P: Vlasov equation
205 // --- Copy rho^n because it will be modified:
206 host_t<DFieldMemRTheta> allfdistribu_predicted(grid);
207 ddc::parallel_deepcopy(get_field(allfdistribu_predicted), allfdistribu);
208 m_advection_solver(get_field(allfdistribu_predicted), get_field(advection_field), dt);
209
210 // --- advect also the feet because it is needed for the next step
211 host_t<FieldMemRTheta<CoordRTheta>> feet_coords(grid);
212 ddc::for_each(grid, [&](IdxRTheta const irp) {
213 feet_coords(irp) = CoordRTheta(ddc::coordinate(irp));
214 });
215 m_find_feet(get_field(feet_coords), get_field(advection_field), dt);
216
217
218 // STEP 4: From rho^P, we compute phi^P: Poisson equation
219 m_builder(get_field(allfdistribu_coef), get_const_field(allfdistribu));
220 PoissonLikeRHSFunction const
221 charge_density_coord_4(get_const_field(allfdistribu_coef), m_evaluator);
222 m_poisson_solver(charge_density_coord_4, electrostatic_potential_coef);
223
224 // STEP 5: From phi^P, we compute A^P:
225 advection_field_computer(electrostatic_potential_coef, advection_field_predicted);
226
227
228 // --- we evaluate the advection field A^n at the characteristic feet X^Theta
229 host_t<DVectorFieldMemRTheta<X, Y>> advection_field_evaluated(grid);
230 host_t<VectorSplineCoeffsMem2D<X, Y>> advection_field_coefs(
231 get_spline_idx_range(m_builder));
232
233 m_builder(
234 ddcHelper::get<X>(advection_field_coefs),
235 ddcHelper::get<X>(get_const_field(advection_field)));
236 m_builder(
237 ddcHelper::get<Y>(advection_field_coefs),
238 ddcHelper::get<Y>(get_const_field(advection_field)));
239
240 m_evaluator(
241 get_field(ddcHelper::get<X>(advection_field_evaluated)),
242 get_const_field(feet_coords),
243 ddcHelper::get<X>(get_const_field(advection_field_coefs)));
244 m_evaluator(
245 get_field(ddcHelper::get<Y>(advection_field_evaluated)),
246 get_const_field(feet_coords),
247 ddcHelper::get<Y>(get_const_field(advection_field_coefs)));
248
249
250 // STEP 6: From rho^n and (A^n(X^P) + A^P(X^n))/2, we compute rho^{n+1}: Vlasov equation
251 ddc::for_each(grid, [&](IdxRTheta const irp) {
252 ddcHelper::get<X>(advection_field)(irp)
253 = (ddcHelper::get<X>(advection_field_evaluated)(irp)
254 + ddcHelper::get<X>(advection_field_predicted)(irp))
255 / 2.;
256 ddcHelper::get<Y>(advection_field)(irp)
257 = (ddcHelper::get<Y>(advection_field_evaluated)(irp)
258 + ddcHelper::get<Y>(advection_field_predicted)(irp))
259 / 2.;
260 });
261
262
263 m_advection_solver(allfdistribu, get_field(advection_field), dt);
264 }
265
266 // STEP 1: From rho^n, we compute phi^n: Poisson equation
267 host_t<Spline2DMem> allfdistribu_coef(get_spline_idx_range(m_builder));
268 m_builder(get_field(allfdistribu_coef), get_const_field(allfdistribu));
269 PoissonLikeRHSFunction const
270 charge_density_coord(get_const_field(allfdistribu_coef), m_evaluator);
271 m_poisson_solver(charge_density_coord, coords, electrical_potential);
272
273 ddc::PdiEvent("last_iteration")
274 .with("iter", steps)
275 .and_with("time", steps * dt)
276 .and_with("density", allfdistribu)
277 .and_with("electrical_potential", electrical_potential);
278
279
280 end_time = std::chrono::system_clock::now();
281 display_time_difference("Iterations time: ", start_time, end_time);
282
283
284 return allfdistribu;
285 }
