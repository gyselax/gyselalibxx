62  template <class Domain>
63  MPITransposeAllToAll(Domain global_domain, MPI_Comm comm)
64  : IMPITranspose<Layout1, Layout2>(comm)
65  {
66  static_assert(
67  std::is_same_v<Domain, domain_type1> || std::is_same_v<Domain, domain_type2>,
68  "The initialisation global domain should be described by one of the layouts");
69  domain_type1 global_domain_layout_1(global_domain);
70  domain_type2 global_domain_layout_2(global_domain);
71  int rank;
72  MPI_Comm_size(comm, &m_comm_size);
73  MPI_Comm_rank(comm, &rank);
74  distributed_domain_type1 distrib_domain(global_domain_layout_1);
75  int n_cores_max = distrib_domain.size();
76  assert(m_comm_size <= n_cores_max);
77  int n_elems = n_cores_max / m_comm_size;
78  // Ensure that the load balancing is good
79  assert(n_cores_max - m_comm_size * n_elems == 0);
80  m_local_domain_1 = m_layout_1.distribute_domain(global_domain_layout_1, m_comm_size, rank);
81  m_local_domain_2 = m_layout_2.distribute_domain(global_domain_layout_2, m_comm_size, rank);
82  m_layout_1_mpi_domain = get_distribution(
83  distributed_domain_type1(m_local_domain_1),
84  distributed_domain_type1(global_domain_layout_1));
85  m_layout_2_mpi_domain = get_distribution(
86  distributed_domain_type2(m_local_domain_2),
87  distributed_domain_type2(global_domain_layout_2));
88  }

162  template <class OutLayout, class ElementType, class MemSpace, class ExecSpace, class InDomain>
163  void transpose_to(
164  ExecSpace const& execution_space,
165  ddc::ChunkSpan<
166  ElementType,
167  typename OutLayout::discrete_domain_type,
168  chunk_layout_type,
169  MemSpace> recv_span,
170  ddc::ChunkView<ElementType, InDomain, chunk_layout_type, MemSpace> send_span)
171  {
172  using InLayout = std::conditional_t<std::is_same_v<OutLayout, Layout1>, Layout2, Layout1>;
173  /*****************************************************************
174  * Validate input
175  *****************************************************************/
176  static_assert((std::is_same_v<InLayout, Layout1>) || (std::is_same_v<InLayout, Layout2>));
177  static_assert((std::is_same_v<OutLayout, Layout1>) || (std::is_same_v<OutLayout, Layout2>));
178  static_assert(Kokkos::SpaceAccessibility<ExecSpace, MemSpace>::accessible);
179 
180  static_assert(std::is_same_v<InDomain, typename InLayout::discrete_domain_type>);
181  using OutDomain = typename OutLayout::discrete_domain_type;
182 
183  /*****************************************************************
184  * Define groups of tags to build necessary transform
185  *****************************************************************/
186  // Currently distributed dims will be gathered
187  using dims_to_gather = typename InLayout::distributed_type_seq;
188  // Future distributed dims must be scattered
189  using dims_to_scatter = typename OutLayout::distributed_type_seq;
190  // Get MPI tags (e.g. MPI<DimX> indicates where DimX will be gathered from)
191  using gather_mpi_dims = ddcHelper::
192  apply_template_to_type_seq_t<MPIDim, typename InLayout::distributed_type_seq>;
193  using scatter_mpi_dims = ddcHelper::
194  apply_template_to_type_seq_t<MPIDim, typename OutLayout::distributed_type_seq>;
195  // Get complete set of tags from domains
196  using input_ordered_dims = ddc::to_type_seq_t<InDomain>;
197  using output_ordered_dims = ddc::to_type_seq_t<OutDomain>;
198  // Find tags that are neither scattered nor gathered
199  using batch_dims = ddc::type_seq_remove_t<
200  ddc::type_seq_remove_t<input_ordered_dims, dims_to_scatter>,
201  dims_to_gather>;
202  // Insert the MPI tags into the received data to introduce the dimension split
203  using input_mpi_domain_tags
204  = insert_mpi_tags_into_seq_t<scatter_mpi_dims, input_ordered_dims>;
205  using output_mpi_domain_tags
206  = insert_mpi_tags_into_seq_t<gather_mpi_dims, output_ordered_dims>;
207  // During the alltoall call the MPI tags must be first to correctly send/receive data
208  // but data does not reorder itself
209  using input_alltoall_dim_order
210  = ddc::type_seq_merge_t<scatter_mpi_dims, input_ordered_dims>;
211  using output_alltoall_dim_order
212  = ddc::type_seq_merge_t<gather_mpi_dims, input_ordered_dims>;
213 
214  /*****************************************************************
215  * Define useful domain types
216  *****************************************************************/
217  // Get the domains of objects which are distributed across processes
218  using gather_domain_type = typename InLayout::distributed_subdomain;
219  // Get the domains of objects which will be distributed across processes
220  using scatter_domain_type = typename OutLayout::distributed_subdomain;
221  // Get the domain of objects that are not distributed across processes
222  using batch_domain_type = ddc::detail::convert_type_seq_to_discrete_domain<batch_dims>;
223  using gather_mpi_domain_type
224  = ddc::detail::convert_type_seq_to_discrete_domain<gather_mpi_dims>;
225  using scatter_mpi_domain_type
226  = ddc::detail::convert_type_seq_to_discrete_domain<scatter_mpi_dims>;
227  // Get the domain containing MPI tags which can be used to describe the function input
228  using input_mpi_domain_type
229  = ddc::detail::convert_type_seq_to_discrete_domain<input_mpi_domain_tags>;
230  // Get the domain containing MPI tags which can be used to describe the function output
231  using output_mpi_domain_type
232  = ddc::detail::convert_type_seq_to_discrete_domain<output_mpi_domain_tags>;
233  // Get the domain that should be used when scattering data
234  using input_alltoall_domain_type
235  = ddc::detail::convert_type_seq_to_discrete_domain<input_alltoall_dim_order>;
236  // Get the domain that should be used when gathering data
237  using output_alltoall_domain_type
238  = ddc::detail::convert_type_seq_to_discrete_domain<output_alltoall_dim_order>;
239 
240  /*****************************************************************
241  * Build domains
242  *****************************************************************/
243  // Collect the artificial dimension describing the MPI rank where the scattered information
244  // will be sent to or where the gathered information will be collected from
245  gather_mpi_domain_type gather_mpi_domain;
246  scatter_mpi_domain_type scatter_mpi_domain;
247  if constexpr (std::is_same_v<InLayout, Layout1>) {
248  gather_mpi_domain = m_layout_1_mpi_domain;
249  scatter_mpi_domain = m_layout_2_mpi_domain;
250  } else {
251  gather_mpi_domain = m_layout_2_mpi_domain;
252  scatter_mpi_domain = m_layout_1_mpi_domain;
253  }
254 
255  // Collect the useful subdomains described in the fields
256  batch_domain_type batch_domain(send_span.domain());
257  scatter_domain_type scatter_domain(recv_span.domain());
258  gather_domain_type gather_domain(send_span.domain());
259 
260  // Build the domains describing the function inputs but including the MPI rank information
261  input_mpi_domain_type
262  input_mpi_domain(scatter_mpi_domain, scatter_domain, gather_domain, batch_domain);
263  output_mpi_domain_type
264  output_mpi_domain(gather_mpi_domain, scatter_domain, gather_domain, batch_domain);
265  assert(input_mpi_domain.size() == send_span.size());
266  assert(output_mpi_domain.size() == recv_span.size());
267 
268  // Create the domains used during the alltoall call (the MPI rank domain is first in this layout)
269  input_alltoall_domain_type input_alltoall_domain(input_mpi_domain);
270  output_alltoall_domain_type output_alltoall_domain(output_mpi_domain);
271 
272  /*****************************************************************
273  * Create views on the function inputs with the domains including the MPI rank information
274  *****************************************************************/
275  ddc::ChunkView<ElementType, input_mpi_domain_type, chunk_layout_type, MemSpace>
276  send_mpi_span(send_span.data_handle(), input_mpi_domain);
277  ddc::ChunkSpan<ElementType, output_mpi_domain_type, chunk_layout_type, MemSpace>
278  recv_mpi_span(recv_span.data_handle(), output_mpi_domain);
279 
280  /*****************************************************************
281  * Transpose data (both on the rank and between ranks)
282  *****************************************************************/
283  // Create views or copies of the function inputs such that they are laid out on the
284  // domain used during the alltoall call
285  auto alltoall_send_buffer = ddcHelper::create_transpose_mirror_view_and_copy<
286  input_alltoall_domain_type>(execution_space, send_mpi_span);
287  auto alltoall_recv_buffer = ddcHelper::create_transpose_mirror<
288  output_alltoall_domain_type>(execution_space, recv_mpi_span);
289 
290  // Call the MPI AlltoAll routine
291  call_all_to_all(
292  execution_space,
293  alltoall_recv_buffer.span_view(),
294  alltoall_send_buffer.span_cview());
295 
296  // If alltoall_recv_buffer owns its data (not a view) then copy the results back to
297  // recv_mpi_span which is a view on recv_span, the function output
298  if constexpr (!ddc::is_borrowed_chunk_v<decltype(alltoall_recv_buffer)>) {
299  transpose_layout(execution_space, recv_mpi_span, alltoall_recv_buffer.span_cview());
300  }
301  }

303 private:
305  template <
306  class ElementType,
307  class MPIRecvDomain,
308  class MPISendDomain,
309  class MemSpace,
310  class ExecSpace>
311  void call_all_to_all(
312  ExecSpace const& execution_space,
313  ddc::ChunkSpan<ElementType, MPIRecvDomain, chunk_layout_type, MemSpace> recv_span,
314  ddc::ChunkView<ElementType, MPISendDomain, chunk_layout_type, MemSpace> send_span)
315  {
316  // No Cuda-aware MPI yet
317  auto send_buffer = ddc::create_mirror_view_and_copy(send_span);
318  auto recv_buffer = ddc::create_mirror_view(recv_span);
319  MPI_Alltoall(
320  send_buffer.data_handle(),
321  send_buffer.size() / m_comm_size,
322  MPI_type_descriptor_t<ElementType>,
323  recv_buffer.data_handle(),
324  recv_buffer.size() / m_comm_size,
325  MPI_type_descriptor_t<ElementType>,
326  IMPITranspose<Layout1, Layout2>::m_comm);
327  if constexpr (!ddc::is_borrowed_chunk_v<decltype(recv_buffer)>) {
328  ddc::parallel_deepcopy(execution_space, recv_span, recv_buffer);
329  }
330  }
