19 class HasIdxRange
20 {
21 private:
22  // Class for SFINAE deduction
23  template <typename U>
24  class check
25  {
26  };
27 
28  // Function that will be chosen if ClassType has a method called idx_range with 0 arguments
29  template <typename ClassType>
30  static char f(check<decltype(std::declval<ClassType>().idx_range())>*);
31 
32  // Function that will be chosen by default

36 public:

118 template <class... QueryGrids, class FieldType>
119 auto get_idx_range(FieldType const& field) noexcept
120 {
121  static_assert(
122  ddc::is_chunk_v<FieldType> || has_idx_range_v<FieldType>,
123  "Not a DDC field (ddc::ChunkSpan) type");
124  if constexpr (ddc::is_chunk_v<FieldType>) {
125  if constexpr (sizeof...(QueryGrids) == 0) {
126  return field.domain();
127  } else {
128  return field.template domain<QueryGrids...>();
129  }
130  } else {
131  if constexpr (sizeof...(QueryGrids) == 0) {
132  return field.idx_range();
133  } else {
134  return field.template idx_range<QueryGrids...>();
135  }
136  }
137 }
