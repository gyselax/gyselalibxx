15/*
16 * A class using template magic to determine if a type has an idx_range function.
17 */
18template <typename Type>
19class HasIdxRange
20{
21private:
22 // Class for SFINAE deduction
23 template <typename U>
24 class check
25 {
26 };
27
28 // Function that will be chosen if ClassType has a method called idx_range with 0 arguments
29 template <typename ClassType>
30 static char f(check<decltype(std::declval<ClassType>().idx_range())>*);
31
32 // Function that will be chosen by default
33 template <typename ClassType>
34 static long f(...);
35
36public:
37 static constexpr bool value = (sizeof(f<Type>(0)) == sizeof(char));
38};

40/*
41 * A class using template magic to determine if a type is a kind of Field defined in Gyselalibxx.
42 * Such a field should define a function get_const_field.
43 */
44template <typename Type>
45class IsGslxField
46{
47private:
48 // Class for SFINAE deduction
49 template <typename U>
50 class check
51 {
52 };
53
54 // Function that will be chosen if ClassType has a method called get_const_field with 0 arguments
55 template <typename ClassType>
56 static char f(check<decltype(std::declval<ClassType>().get_const_field())>*);
57
58 // Function that will be chosen by default
59 template <typename ClassType>
60 static long f(...);
61
62public:
63 static constexpr bool value = (sizeof(f<Type>(0)) == sizeof(char));
64};

120template <class... QueryGrids, class FieldType>
121auto get_idx_range(FieldType const& field) noexcept
122{
123 static_assert(
124 ddc::is_chunk_v<FieldType> || has_idx_range_v<FieldType>,
125 "Not a DDC field (ddc::ChunkSpan) type");
126 if constexpr (ddc::is_chunk_v<FieldType>) {
127 if constexpr (sizeof...(QueryGrids) == 0) {
128 return field.domain();
129 } else {
130 return field.template domain<QueryGrids...>();
131 }
132 } else {
133 if constexpr (sizeof...(QueryGrids) == 0) {
134 return field.idx_range();
135 } else {
136 return field.template idx_range<QueryGrids...>();
137 }
138 }
139}
