13/*
14 * A class using template magic to determine if a type has an idx_range function.
15 */
16template <typename Type>
17class HasIdxRange
18{
19private:
20 // Class for SFINAE deduction
21 template <typename U>
22 class check
23 {
24 };
25
26 // Function that will be chosen if ClassType has a method called idx_range with 0 arguments
27 template <typename ClassType>
28 static char f(check<decltype(std::declval<ClassType>().idx_range())>*);
29
30 // Function that will be chosen by default
31 template <typename ClassType>
32 static long f(...);
33
34public:
35 static constexpr bool value = (sizeof(f<Type>(0)) == sizeof(char));
36};

104template <class... QueryGrids, class FieldType>
105auto get_idx_range(FieldType const& field) noexcept
106{
107 static_assert(
108 ddc::is_chunk_v<FieldType> || has_idx_range_v<FieldType>,
109 "Not a DDC field (ddc::ChunkSpan) type");
110 if constexpr (ddc::is_chunk_v<FieldType>) {
111 if constexpr (sizeof...(QueryGrids) == 0) {
112 return field.domain();
113 } else {
114 return field.template domain<QueryGrids...>();
115 }
116 } else {
117 if constexpr (sizeof...(QueryGrids) == 0) {
118 return field.idx_range();
119 } else {
120 return field.template idx_range<QueryGrids...>();
121 }
122 }
123}
