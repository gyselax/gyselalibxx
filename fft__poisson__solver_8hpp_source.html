193  template <class Layout>
194  void solve_poisson_equation(
195  fourier_span_type intermediate_chunk,
196  ddc::ChunkSpan<double, laplacian_domain_type, Layout, memory_space> rho) const
197  {
198  // Compute FFT(rho)
199  ddc::fft(ExecSpace(), intermediate_chunk, rho, ddc::kwArgs_fft {m_norm});
200 
201  fourier_domain_type const k_mesh = intermediate_chunk.domain();
202 
203  // Solve Poisson's equation -\Delta phi = -(\sum_j \partial_j^2) \phi = rho
204  // in Fourier space as -(\sum_j i*k_i * i*k_i) FFT(Phi) = FFT(rho))
205  ddc::parallel_for_each(
206  ExecSpace(),
207  k_mesh,
208  KOKKOS_LAMBDA(fourier_element_type const ik) {
209  if (ik != k_mesh.front()) {
210  intermediate_chunk(ik) = intermediate_chunk(ik) / get_laplace_operator(ik);
211  } else {
212  intermediate_chunk(ik) = 0.;
213  }
214  });
215  }

227  void negative_differentiate_equation(fourier_span_type derivative, fourier_span_type values)
228  const
229  {
230  Kokkos::complex<double> imaginary_unit(0.0, 1.0);
231  ddc::parallel_for_each(
232  ExecSpace(),
233  values.domain(),
234  KOKKOS_LAMBDA(fourier_element_type const ik) {
235  ddc::DiscreteElement<Dim> const ikx = ddc::select<Dim>(ik);
236  derivative(ik) = -imaginary_unit * ddc::coordinate(ikx) * values(ik);
237  });
