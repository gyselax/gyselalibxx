187  template <class Layout>
188  void solve_poisson_equation(
189  fourier_span_type intermediate_chunk,
190  ddc::ChunkSpan<double, laplacian_domain_type, Layout, memory_space> rho) const
191  {
192  // Compute FFT(rho)
193  ddc::fft(ExecSpace(), intermediate_chunk, rho, ddc::kwArgs_fft {m_norm});
194 
195  fourier_domain_type const k_mesh = intermediate_chunk.domain();
196 
197  // Solve Poisson's equation -\Delta phi = -(\sum_j \partial_j^2) \phi = rho
198  // in Fourier space as -(\sum_j i*k_i * i*k_i) FFT(Phi) = FFT(rho))
199  ddc::parallel_for_each(
200  ExecSpace(),
201  k_mesh,
202  KOKKOS_LAMBDA(fourier_element_type const ik) {
203  if (ik != k_mesh.front()) {
204  intermediate_chunk(ik) = intermediate_chunk(ik) / get_laplace_operator(ik);
205  } else {
206  intermediate_chunk(ik) = 0.;
207  }
208  });
209  }

221  void negative_differentiate_equation(fourier_span_type derivative, fourier_span_type values)
222  const
223  {
224  Kokkos::complex<double> imaginary_unit(0.0, 1.0);
225  ddc::parallel_for_each(
226  ExecSpace(),
227  values.domain(),
228  KOKKOS_LAMBDA(fourier_element_type const ik) {
229  ddc::DiscreteElement<Dim> const ikx = ddc::select<Dim>(ik);
230  derivative(ik) = -imaginary_unit * ddc::coordinate(ikx) * values(ik);
231  });
232  }
