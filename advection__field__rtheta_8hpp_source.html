208 template <class SplineType, class Evaluator>
209 void compute_advection_field_XY(
210 Evaluator evaluator,
211 SplineType& electrostatic_potential_coef,
212 host_t<DVectorFieldRTheta<X, Y>> advection_field_xy) const
213 {
214 static_assert(
215 (std::is_same_v<
216 Evaluator,
217 SplineRThetaEvaluatorNullBound_host> && std::is_same_v<SplineType, host_t<Spline2D>>)
218 || (std::is_same_v<
219 Evaluator,
220 PolarSplineEvaluator<
221 PolarBSplinesRTheta,
222 ddc::NullExtrapolationRule>> && std::is_same_v<SplineType, host_t<PolarSplineMemRTheta>>));
223
224 IdxRangeRTheta const grid = get_idx_range(advection_field_xy);
225 host_t<DVectorFieldMemRTheta<X, Y>> electric_field(grid);
226
227 host_t<FieldMemRTheta<CoordRTheta>> coords(grid);
228 ddc::for_each(grid, [&](IdxRTheta const irtheta) {
229 coords(irtheta) = ddc::coordinate(irtheta);
230 });
231
232 // > computation of the phi derivatives
233 host_t<DFieldMemRTheta> deriv_r_phi(grid);
234 host_t<DFieldMemRTheta> deriv_theta_phi(grid);
235
236 evaluator.deriv_dim_1(
237 get_field(deriv_r_phi),
238 get_const_field(coords),
239 get_const_field(electrostatic_potential_coef));
240 evaluator.deriv_dim_2(
241 get_field(deriv_theta_phi),
242 get_const_field(coords),
243 get_const_field(electrostatic_potential_coef));
244
245 InverseJacobianMatrix<Mapping, CoordRTheta> inv_jacobian_matrix(m_mapping);
246
247 // > computation of the electric field
248 ddc::for_each(grid, [&](IdxRTheta const irtheta) {
249 double const r = ddc::coordinate(ddc::select<GridR>(irtheta));
250 double const th = ddc::coordinate(ddc::select<GridTheta>(irtheta));
251
252 if (r > m_epsilon) {
253 CoordRTheta const coord_rtheta(r, th);
254
255 Matrix_2x2 inv_J = inv_jacobian_matrix(coord_rtheta);
256
257 // Gradient of phi in the physical domain (Cartesian domain)
258 double const deriv_x_phi = deriv_r_phi(irtheta) * inv_J[0][0]
259 + deriv_theta_phi(irtheta) * inv_J[1][0];
260 double const deriv_y_phi = deriv_r_phi(irtheta) * inv_J[0][1]
261 + deriv_theta_phi(irtheta) * inv_J[1][1];
262
263 // E = -grad phi
264 ddcHelper::get<X>(electric_field)(irtheta) = -deriv_x_phi;
265 ddcHelper::get<Y>(electric_field)(irtheta) = -deriv_y_phi;
266
267 } else {
268 // Linearisation of the electric field
269 double const th1 = M_PI / 4.;
270 double const th2 = -M_PI / 4. + 2 * M_PI;
271
272 // --- Value at r = 0:
273 CoordRTheta const coord_1_0(0, th1);
274 CoordRTheta const coord_2_0(0, th2);
275
276 double const dr_x_1 = m_mapping.jacobian_11(coord_1_0); // dr_x (0, th1)
277 double const dr_y_1 = m_mapping.jacobian_21(coord_1_0); // dr_y (0, th1)
278
279 double const dr_x_2 = m_mapping.jacobian_11(coord_2_0); // dr_x (0, th2)
280 double const dr_y_2 = m_mapping.jacobian_21(coord_2_0); // dr_y (0, th2)
281
282 double deriv_r_phi_1 = evaluator.deriv_dim_1(
283 coord_1_0,
284 get_const_field(electrostatic_potential_coef));
285 double deriv_r_phi_2 = evaluator.deriv_dim_1(
286 coord_2_0,
287 get_const_field(electrostatic_potential_coef));
288
289 double const determinant = dr_x_1 * dr_y_2 - dr_x_2 * dr_y_1;
290
291 double const deriv_x_phi_0
292 = (dr_y_2 * deriv_r_phi_1 - dr_y_1 * deriv_r_phi_2) / determinant;
293 double const deriv_y_phi_0
294 = (-dr_x_2 * deriv_r_phi_1 + dr_x_1 * deriv_r_phi_2) / determinant;
295
296 // E = -grad phi
297 double const electric_field_x_0 = -deriv_x_phi_0;
298 double const electric_field_y_0 = -deriv_y_phi_0;
299
300
301
302 // --- Value at r = m_epsilon:
303 CoordRTheta const coord_rtheta_epsilon(m_epsilon, th);
304
305 Matrix_2x2 inv_J_eps = inv_jacobian_matrix(coord_rtheta_epsilon);
306
307 double const deriv_r_phi_epsilon = evaluator.deriv_dim_1(
308 coord_rtheta_epsilon,
309 get_const_field(electrostatic_potential_coef));
310 double const deriv_theta_phi_epsilon = evaluator.deriv_dim_2(
311 coord_rtheta_epsilon,
312 get_const_field(electrostatic_potential_coef));
313
314 // Gradient of phi in the physical domain (Cartesian domain)
315 double const deriv_x_phi_epsilon = deriv_r_phi_epsilon * inv_J_eps[0][0]
316 + deriv_theta_phi_epsilon * inv_J_eps[1][0];
317 double const deriv_y_phi_epsilon = deriv_r_phi_epsilon * inv_J_eps[0][1]
318 + deriv_theta_phi_epsilon * inv_J_eps[1][1];
319
320 // E = -grad phi
321 double const electric_field_x_epsilon = -deriv_x_phi_epsilon;
322 double const electric_field_y_epsilon = -deriv_y_phi_epsilon;
323
324
325 // --- Linearisation:
326 ddcHelper::get<X>(electric_field)(irtheta)
327 = electric_field_x_0 * (1 - r / m_epsilon)
328 + electric_field_x_epsilon * r / m_epsilon;
329 ddcHelper::get<Y>(electric_field)(irtheta)
330 = electric_field_y_0 * (1 - r / m_epsilon)
331 + electric_field_y_epsilon * r / m_epsilon;
332 }
333
334 // > computation of the advection field
335 ddcHelper::get<X>(advection_field_xy)(irtheta)
336 = -ddcHelper::get<Y>(electric_field)(irtheta);
337 ddcHelper::get<Y>(advection_field_xy)(irtheta)
338 = ddcHelper::get<X>(electric_field)(irtheta);
339 });
340 }

443 template <class SplineType, class Evaluator>
444 void compute_advection_field_RTheta(
445 Evaluator evaluator,
446 SplineType& electrostatic_potential_coef,
447 host_t<DVectorFieldRTheta<R_cov, Theta_cov>> advection_field_rtheta,
448 CoordXY& advection_field_xy_centre) const
449 {
450 static_assert(
451 (std::is_same_v<
452 Evaluator,
453 SplineRThetaEvaluatorNullBound_host> && std::is_same_v<SplineType, host_t<Spline2D>>)
454 || (std::is_same_v<
455 Evaluator,
456 PolarSplineEvaluator<
457 PolarBSplinesRTheta,
458 ddc::NullExtrapolationRule>> && std::is_same_v<SplineType, host_t<PolarSplineMemRTheta>>));
459
460 IdxRangeRTheta const grid_without_Opoint = get_idx_range(advection_field_rtheta);
461
462 host_t<FieldMemRTheta<CoordRTheta>> coords(grid_without_Opoint);
463 ddc::for_each(grid_without_Opoint, [&](IdxRTheta const irtheta) {
464 coords(irtheta) = ddc::coordinate(irtheta);
465 });
466
467 // > computation of the phi derivatives
468 host_t<DFieldMemRTheta> deriv_r_phi(grid_without_Opoint);
469 host_t<DFieldMemRTheta> deriv_theta_phi(grid_without_Opoint);
470
471 evaluator.deriv_dim_1(
472 get_field(deriv_r_phi),
473 get_const_field(coords),
474 get_const_field(electrostatic_potential_coef));
475 evaluator.deriv_dim_2(
476 get_field(deriv_theta_phi),
477 get_const_field(coords),
478 get_const_field(electrostatic_potential_coef));
479
480 MetricTensorEvaluator<Mapping, CoordRTheta> metric_tensor(m_mapping);
481
482 // > computation of the advection field
483 ddc::for_each(grid_without_Opoint, [&](IdxRTheta const irtheta) {
484 CoordRTheta const coord_rtheta(ddc::coordinate(irtheta));
485
486 DTensor<VectorIndexSet<R_cov, Theta_cov>, VectorIndexSet<R_cov, Theta_cov>> inv_G
487 = metric_tensor.inverse(coord_rtheta);
488
489 // E = -grad phi
490 double const electric_field_r
491 = -deriv_r_phi(irtheta) * ddcHelper::get<R_cov, R_cov>(inv_G)
492 - deriv_theta_phi(irtheta) * ddcHelper::get<R_cov, Theta_cov>(inv_G);
493 double const electric_field_theta
494 = -deriv_r_phi(irtheta) * ddcHelper::get<Theta_cov, R_cov>(inv_G)
495 - deriv_theta_phi(irtheta) * ddcHelper::get<Theta_cov, Theta_cov>(inv_G);
496
497 // A = E \wedge e_z
498 ddcHelper::get<R_cov>(advection_field_rtheta)(irtheta) = -electric_field_theta;
499 ddcHelper::get<Theta_cov>(advection_field_rtheta)(irtheta) = electric_field_r;
500 });
501
502 // SPECIAL TREATMENT FOR THE O-POINT =====================================================
503 // Linearisation of the electric field
504 double const th1 = M_PI / 4.;
505 double const th2 = -M_PI / 4. + 2 * M_PI;
506
507 // --- Value at r = 0:
508 CoordRTheta const coord_1_0(0, th1);
509 CoordRTheta const coord_2_0(0, th2);
510
511 double const dr_x_1 = m_mapping.jacobian_11(coord_1_0); // dr_x (0, th1)
512 double const dr_y_1 = m_mapping.jacobian_21(coord_1_0); // dr_y (0, th1)
513
514 double const dr_x_2 = m_mapping.jacobian_11(coord_2_0); // dr_x (0, th2)
515 double const dr_y_2 = m_mapping.jacobian_21(coord_2_0); // dr_y (0, th2)
516
517 double const deriv_r_phi_1
518 = evaluator.deriv_dim_1(coord_1_0, get_const_field(electrostatic_potential_coef));
519 double const deriv_r_phi_2
520 = evaluator.deriv_dim_1(coord_2_0, get_const_field(electrostatic_potential_coef));
521
522 double const determinant = dr_x_1 * dr_y_2 - dr_x_2 * dr_y_1;
523
524 double const deriv_x_phi_0
525 = (dr_y_2 * deriv_r_phi_1 - dr_y_1 * deriv_r_phi_2) / determinant;
526 double const deriv_y_phi_0
527 = (-dr_x_2 * deriv_r_phi_1 + dr_x_1 * deriv_r_phi_2) / determinant;
528
529 advection_field_xy_centre = CoordXY(deriv_y_phi_0, -deriv_x_phi_0);
530 }
