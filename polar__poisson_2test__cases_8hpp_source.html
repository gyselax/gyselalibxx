117template <class CurvilinearToCartesian>
118class CartesianSolution : public PoissonSolution<CurvilinearToCartesian>
119{
120private:
121 InverseJacobianMatrix<CurvilinearToCartesian, Coord<R, Theta>> m_inverse_jacobian;
122
123public:
131 explicit CartesianSolution(CurvilinearToCartesian const& coordinate_converter)
132 : PoissonSolution<CurvilinearToCartesian>(coordinate_converter)
133 , m_inverse_jacobian(coordinate_converter)
134 {
135 }
136
137 double operator()(Coord<R, Theta> const& coord) const final
138 {
139 const double s = ddc::get<R>(coord);
140 const Coord<X, Y> cart_coord
141 = PoissonSolution<CurvilinearToCartesian>::m_coordinate_converter(coord);
142 const double x = ddc::get<X>(cart_coord);
143 const double y = ddc::get<Y>(cart_coord);
144 return 1e-4 * ipow(1 + s, 6) * ipow(1 - s, 6) * std::cos(2 * M_PI * x)
145 * std::sin(2 * M_PI * y) / ipow(0.5, 12);
146 }
147
156 double derivative_x(Coord<R, Theta> const& coord) const
157 {
158 const Coord<X, Y> cart_coord
159 = PoissonSolution<CurvilinearToCartesian>::m_coordinate_converter(coord);
160 const double x = ddc::get<X>(cart_coord);
161 const double y = ddc::get<Y>(cart_coord);
162
163 const double r = ddc::get<R>(coord);
164 const double dx_r = m_inverse_jacobian.inv_jacobian_11(coord);
165
166 const double C = 1e-4 * ipow(2., 12);
167 return C * std::sin(2 * M_PI * y)
168 * (12. * r * dx_r * ipow(r * r - 1, 5) * std::cos(2 * M_PI * x)
169 - 2 * M_PI * ipow(r * r - 1, 6) * std::sin(2 * M_PI * x));
170 }
171
180 double derivative_y(Coord<R, Theta> const& coord) const
181 {
182 const Coord<X, Y> cart_coord
183 = PoissonSolution<CurvilinearToCartesian>::m_coordinate_converter(coord);
184 const double x = ddc::get<X>(cart_coord);
185 const double y = ddc::get<Y>(cart_coord);
186
187 const double r = ddc::get<R>(coord);
188 const double dy_r = m_inverse_jacobian.inv_jacobian_12(coord);
189
190 const double C = 1e-4 * ipow(2., 12);
191 return C * std::cos(2 * M_PI * x)
192 * (12. * r * dy_r * ipow(r * r - 1, 5) * std::sin(2 * M_PI * y)
193 + 2 * M_PI * ipow(r * r - 1, 6) * std::cos(2 * M_PI * y));
194 }
195};
