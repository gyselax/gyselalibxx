244 {
245 const double r = ddc::get<R>(coord);
246 const double theta = ddc::get<Theta>(coord);
247 assert(fabs(r) >= 1e-15);
248 matrix[0][0] = Kokkos::cos(theta);
249 matrix[0][1] = Kokkos::sin(theta);
250 matrix[1][0] = -1 / r * Kokkos::sin(theta);
251 matrix[1][1] = 1 / r * Kokkos::cos(theta);
252 }
