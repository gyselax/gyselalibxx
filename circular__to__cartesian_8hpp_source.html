154 {
155 const double r = ddc::get<R>(coord);
156 const double theta = ddc::get<Theta>(coord);
157 DTensor<VectorIndexSet<X, Y>, VectorIndexSet<R_cov, Theta_cov>> jacobian_matrix;
158 ddcHelper::get<X, R_cov>(jacobian_matrix) = Kokkos::cos(theta);

254 KOKKOS_FUNCTION DTensor<VectorIndexSet<R, Theta>, VectorIndexSet<X_cov, Y_cov>>
255 inv_jacobian_matrix(Coord<R, Theta> const& coord) const
256 {
257 const double r = ddc::get<R>(coord);
258 const double theta = ddc::get<Theta>(coord);
259 assert(fabs(r) >= 1e-15);
260
261 DTensor<VectorIndexSet<R, Theta>, VectorIndexSet<X_cov, Y_cov>> matrix;
262 ddcHelper::get<R, X_cov>(matrix) = Kokkos::cos(theta);
263 ddcHelper::get<R, Y_cov>(matrix) = Kokkos::sin(theta);
264 ddcHelper::get<Theta, X_cov>(matrix) = -1 / r * Kokkos::sin(theta);
265 ddcHelper::get<Theta, Y_cov>(matrix) = 1 / r * Kokkos::cos(theta);
266 return matrix;
267 }
