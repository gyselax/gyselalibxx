151 KOKKOS_FUNCTION DTensor<VectorIndexSet<X, Y>, VectorIndexSet<R_cov, Theta_cov>> jacobian_matrix(
152 Coord<R, Theta> const& coord) const
153 {
154 const double r = ddc::get<R>(coord);
155 const double theta = ddc::get<Theta>(coord);
156 DTensor<VectorIndexSet<X, Y>, VectorIndexSet<R_cov, Theta_cov>> jacobian_matrix;
157 ddcHelper::get<X, R_cov>(jacobian_matrix) = Kokkos::cos(theta);
158 ddcHelper::get<X, Theta_cov>(jacobian_matrix) = -r * Kokkos::sin(theta);
159 ddcHelper::get<Y, R_cov>(jacobian_matrix) = Kokkos::sin(theta);
160 ddcHelper::get<Y, Theta_cov>(jacobian_matrix) = r * Kokkos::cos(theta);
161 return jacobian_matrix;
162 }

172 template <class IndexTag1, class IndexTag2>
173 KOKKOS_FUNCTION double jacobian_component(Coord<R, Theta> const& coord) const
174 {
175 static_assert(ddc::in_tags_v<IndexTag1, VectorIndexSet<X, Y>>);
176 static_assert(ddc::in_tags_v<IndexTag2, VectorIndexSet<R_cov, Theta_cov>>);
177
178 const double theta = ddc::get<Theta>(coord);
179
180 if constexpr (std::is_same_v<IndexTag1, X> && std::is_same_v<IndexTag2, R_cov>) {
181 //Compute the (1,1) coefficient of the Jacobian matrix, i.e J^x_r.
182 return Kokkos::cos(theta);
183 } else if constexpr (std::is_same_v<IndexTag1, X> && std::is_same_v<IndexTag2, Theta_cov>) {
184 //Compute the (1,2) coefficient of the Jacobian matrix, i.e J^x_theta.
185 const double r = ddc::get<R>(coord);
186 return -r * Kokkos::sin(theta);
187 } else if constexpr (std::is_same_v<IndexTag1, Y> && std::is_same_v<IndexTag2, R_cov>) {
188 //Compute the (2,1) coefficient of the Jacobian matrix, i.e J^y_r.
189 return Kokkos::sin(theta);
190 } else {
191 //Compute the (2,2) coefficient of the Jacobian matrix, i.e J^y_theta.
192 const double r = ddc::get<R>(coord);
193 return r * Kokkos::cos(theta);
194 }
195 }

216 KOKKOS_FUNCTION DTensor<VectorIndexSet<R, Theta>, VectorIndexSet<X_cov, Y_cov>>
217 inv_jacobian_matrix(Coord<R, Theta> const& coord) const
218 {
219 const double r = ddc::get<R>(coord);
220 const double theta = ddc::get<Theta>(coord);
221 assert(fabs(r) >= 1e-15);
222
223 DTensor<VectorIndexSet<R, Theta>, VectorIndexSet<X_cov, Y_cov>> matrix;
224 ddcHelper::get<R, X_cov>(matrix) = Kokkos::cos(theta);
225 ddcHelper::get<R, Y_cov>(matrix) = Kokkos::sin(theta);
226 ddcHelper::get<Theta, X_cov>(matrix) = -1 / r * Kokkos::sin(theta);
227 ddcHelper::get<Theta, Y_cov>(matrix) = 1 / r * Kokkos::cos(theta);
228 return matrix;
229 }
