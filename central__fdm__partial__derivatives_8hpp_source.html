25class CentralFDMPartialDerivative : public IPartialDerivative<IdxRangeFull, DerivativeDimension>

27private:

34 using typename base_type::DFieldType;

54 explicit CentralFDMPartialDerivative(DConstFieldType const field_ref)
55 : m_field(get_idx_range(field_ref))
56 {
57 ddc::parallel_deepcopy(get_field(m_field), field_ref);

67 void operator()(DFieldType differentiated_field) const final
68 {
69 using IdxFull = typename IdxRangeFull::discrete_element_type;
70 using IdxDeriv = typename IdxRangeDeriv::discrete_element_type;
71 using IdxBatch = typename IdxRangeBatch::discrete_element_type;
72 using IdxStepDeriv = typename IdxRangeDeriv::discrete_vector_type;
73
74 IdxRangeFull idxrange_full = get_idx_range(m_field);
75 IdxRangeDeriv idxrange_deriv(idxrange_full);
76 IdxRangeBatch idxrange_batch(idxrange_full);
77
78 DConstFieldType const field_proxy = get_const_field(m_field);
79
80 IdxStepDeriv const step(1);
81
82 // front batched derivative
83 {
84 IdxDeriv const ix(idxrange_deriv.front());
85 double const h1 = ddc::coordinate(ix + step) - ddc::coordinate(ix);
86 double const h2 = ddc::coordinate(ix + 2 * step) - ddc::coordinate(ix + step);
87 double const c3 = -h1 / (h2 * (h1 + h2));
88 double const c2 = 1. / h1 + 1. / h2;
89 double const c1 = -c3 - c2;
90 ddc::parallel_for_each(
91 Kokkos::DefaultExecutionSpace(),
92 idxrange_batch,
93 KOKKOS_LAMBDA(IdxBatch ib) {
94 IdxFull ibx(ib, ix);
95 differentiated_field(ibx) = c1 * field_proxy(ibx)
96 + c2 * field_proxy(ib, ix + step)
97 + c3 * field_proxy(ib, ix + 2 * step);
98 });
99 }
100
101 // back batched derivative
102 {
103 IdxDeriv const ix = idxrange_deriv.back();
104 double const h1 = ddc::coordinate(ix) - ddc::coordinate(ix - step);
105 double const h2 = ddc::coordinate(ix - step) - ddc::coordinate(ix - 2 * step);
106 double const c3 = h1 / (h2 * (h1 + h2));
107 double const c2 = -(h1 + h2) / (h1 * h2);
108 double const c1 = -c3 - c2;
109 ddc::parallel_for_each(
110 Kokkos::DefaultExecutionSpace(),
111 idxrange_batch,
112 KOKKOS_LAMBDA(IdxBatch ib) {
113 IdxFull ibx(ib, ix);
114 differentiated_field(ibx) = c1 * field_proxy(ibx)
115 + c2 * field_proxy(ib, ix - step)
116 + c3 * field_proxy(ib, ix - 2 * step);
117 });
118 }
119
120 // central domain batched derivative
121 {
122 IdxRangeDeriv idxrange_deriv_central = idxrange_deriv.remove(step, step);
123 IdxRangeFull idxrange_central(idxrange_deriv_central, idxrange_batch);
124 ddc::parallel_for_each(
125 Kokkos::DefaultExecutionSpace(),
126 idxrange_central,
127 KOKKOS_LAMBDA(IdxFull ibx) {
128 IdxBatch ib(ibx);
129 IdxDeriv ix(ibx);
130 double const h1 = ddc::coordinate(ix) - ddc::coordinate(ix - step);
131 double const h2 = ddc::coordinate(ix + step) - ddc::coordinate(ix);
132 double const c3 = h1 / (h2 * (h1 + h2));
133 double const c2 = 1. / h1 - 1. / h2;
134 double const c1 = -c3 - c2;
135 differentiated_field(ibx) = c1 * field_proxy(ib, ix - step)
136 + c2 * field_proxy(ibx)
137 + c3 * field_proxy(ib, ix + step);
138 });
139 }
140 }
141};
