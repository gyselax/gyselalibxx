99  void find_equilibrium(
100  DFieldRTheta sigma,
101  DFieldRTheta phi_eq,
102  DFieldRTheta rho_eq,
103  std::function<double(double const)> const& function,
104  double const phi_max, // ToDo: ADD CASE WHERE RHO_MAX IS GIVEN.
105  double const tau,
106  int count_max = 25) const
107  {
108  DFieldMemRTheta phi_star(m_grid);
109  DFieldMemRTheta ci(m_grid);
110 
111  IdxRangeBSRTheta idx_range_bsplinesRTheta = get_spline_idx_range(m_builder);
112  Spline2D rho_coef(idx_range_bsplinesRTheta);
113 
114  FieldMemRTheta<CoordRTheta> coords(m_grid);
115  ddc::for_each(m_grid, [&](IdxRTheta const irp) { coords(irp) = ddc::coordinate(irp); });
116 
117  double difference_sigma(0.);
118  int count = 0;
119 
120  do {
121  count += 1;
122  // STEP 1: compute rho^i
123  ddc::for_each(m_grid, [&](IdxRTheta const irp) {
124  rho_eq(irp) = sigma(irp) * function(phi_eq(irp));
125  });
126 
127 
128  // STEP 2: compute phi_star^i with PDE solver
129  m_builder(get_field(rho_coef), get_const_field(rho_eq));
130  PoissonLikeRHSFunction poisson_rhs(rho_coef, m_evaluator);
131  m_poisson_solver(poisson_rhs, get_const_field(coords), get_field(phi_star));
132 
133  // STEP 3: compute c^i
134  // If phi_max is given:
135  double norm_Linf_phi_star(0.);
136  ddc::for_each(m_grid, [&](IdxRTheta const irp) {
137  double const abs_phi_star = fabs(phi_star(irp));
138  norm_Linf_phi_star
139  = norm_Linf_phi_star > abs_phi_star ? norm_Linf_phi_star : abs_phi_star;
140  });
141 
142  ddc::for_each(m_grid, [&](IdxRTheta const irp) {
143  ci(irp) = phi_max / norm_Linf_phi_star;
144  });
145 
146 
147  // STEP 4: update sigma and phi
148  difference_sigma = 0.;
149  ddc::for_each(m_grid, [&](IdxRTheta const irp) {
150  double const abs_diff_sigma = fabs(sigma(irp) - ci(irp) * sigma(irp));
151  difference_sigma
152  = difference_sigma > abs_diff_sigma ? difference_sigma : abs_diff_sigma;
153 
154  sigma(irp) = ci(irp) * sigma(irp);
155  phi_eq(irp) = ci(irp) * phi_star(irp);
156  });
157 
158  } while ((difference_sigma > tau) and (count < count_max));
159 
160 
161  // STEP 1: compute rho^i
162  ddc::for_each(m_grid, [&](IdxRTheta const irp) {
163  rho_eq(irp) = sigma(irp) * function(phi_eq(irp));
164  });
165 
166  // Unify at the center point:
167  IdxRangeR r_idx_range = get_idx_range<GridR>(rho_eq);
168  IdxRangeTheta theta_idx_range = get_idx_range<GridTheta>(rho_eq);
169  if (std::fabs(ddc::coordinate(r_idx_range.front())) < 1e-15) {
170  ddc::for_each(theta_idx_range, [&](const IdxTheta ip) {
171  rho_eq(r_idx_range.front(), ip)
172  = rho_eq(r_idx_range.front(), theta_idx_range.front());
173  });
174  }
175  };
