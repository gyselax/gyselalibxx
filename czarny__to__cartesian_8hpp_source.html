152 KOKKOS_FUNCTION ddc::Coordinate<X, Y> operator()(
153 ddc::Coordinate<R, Theta> const& coord) const final
154 {
155 const double r = ddc::get<R>(coord);
156 const double theta = ddc::get<Theta>(coord);
157 const double tmp1
158 = Kokkos::sqrt(m_epsilon * (m_epsilon + 2.0 * r * Kokkos::cos(theta)) + 1.0);
159
160 const double x = (1.0 - tmp1) / m_epsilon;
161 const double y = m_e * r * Kokkos::sin(theta)
162 / (Kokkos::sqrt(1.0 - 0.25 * m_epsilon * m_epsilon) * (2.0 - tmp1));
163
164 return ddc::Coordinate<X, Y>(x, y);
165 }

167 KOKKOS_FUNCTION ddc::Coordinate<R, Theta> operator()(
168 ddc::Coordinate<X, Y> const& coord) const final
169 {
170 const double x = ddc::get<X>(coord);
171 const double y = ddc::get<Y>(coord);
172 const double ex = 1. + m_epsilon * x;
173 const double ex2 = (m_epsilon * x * x - 2. * x - m_epsilon);
174 const double xi2 = 1. / (1. - m_epsilon * m_epsilon * 0.25);
175 const double xi = Kokkos::sqrt(xi2);
176 const double r = Kokkos::sqrt(y * y * ex * ex / (m_e * m_e * xi2) + ex2 * ex2 * 0.25);
177 double theta
178 = Kokkos::atan2(2. * y * ex, (m_e * xi * (m_epsilon * x * x - 2. * x - m_epsilon)));
179 if (theta < 0) {
180 theta = 2 * M_PI + theta;
181 }
182 return ddc::Coordinate<R, Theta>(r, theta);
183 }

186 {

188 const double theta = ddc::get<Theta>(coord);

215 KOKKOS_FUNCTION void jacobian_matrix(ddc::Coordinate<R, Theta> const& coord, Matrix_2x2& matrix)
216 const final
217 {
218 const double r = ddc::get<R>(coord);
219 const double theta = ddc::get<Theta>(coord);
220
221 const double sin_theta = Kokkos::sin(theta);
222 const double cos_theta = Kokkos::cos(theta);
223 const double xi2 = 1. / (1. - m_epsilon * m_epsilon * 0.25);
224 const double xi = Kokkos::sqrt(xi2);
225 const double sqrt_eps = Kokkos::sqrt(m_epsilon * (m_epsilon + 2.0 * r * cos_theta) + 1.0);
226 const double sqrt_eps_2 = 2.0 - sqrt_eps;
227
228 matrix[0][0] = -cos_theta / sqrt_eps;
229 matrix[0][1] = r * sin_theta / sqrt_eps;
230 matrix[1][0] = m_e * m_epsilon * r * sin_theta * cos_theta * xi
231 / (sqrt_eps_2 * sqrt_eps_2 * sqrt_eps)
232 + m_e * sin_theta * xi / sqrt_eps_2;
233 matrix[1][1] = r
234 * (-m_e * m_epsilon * r * sin_theta * sin_theta * xi
235 / (sqrt_eps_2 * sqrt_eps_2 * sqrt_eps)
236 + m_e * cos_theta * xi / sqrt_eps_2);
237 }

305 KOKKOS_FUNCTION void inv_jacobian_matrix(
306 ddc::Coordinate<R, Theta> const& coord,
307 Matrix_2x2& matrix) const final
308 {
309 const double r = ddc::get<R>(coord);
310 const double theta = ddc::get<Theta>(coord);
311
312 assert(r >= 1e-15);
313
314 const double sin_theta = Kokkos::sin(theta);
315 const double cos_theta = Kokkos::cos(theta);
316 const double xi = Kokkos::sqrt(1. / (1. - m_epsilon * m_epsilon * 0.25));
317 const double divisor = 2 - Kokkos::sqrt(1 + m_epsilon * (m_epsilon + 2.0 * r * cos_theta));
318
319 const double fact_1 = 1 / Kokkos::sqrt(1 + m_epsilon * (m_epsilon + 2.0 * r * cos_theta));
320 const double fact_2 = m_e * m_epsilon * xi * r * sin_theta * fact_1 / divisor / divisor;
321 const double fact_3 = m_e * xi / divisor;
322
323 matrix[0][0] = -1 / fact_1 * (-sin_theta * fact_2 + cos_theta * fact_3) / fact_3;
324 matrix[0][1] = sin_theta / fact_3;
325 matrix[1][0] = 1 / r / fact_1 * (cos_theta * fact_2 + sin_theta * fact_3) / fact_3;
326 matrix[1][1] = 1 / r * cos_theta / fact_3;
327 }
