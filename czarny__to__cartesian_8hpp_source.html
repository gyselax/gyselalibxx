157 KOKKOS_FUNCTION Coord<X, Y> operator()(Coord<R, Theta> const& coord) const
158 {
159 const double r = ddc::get<R>(coord);
160 const double theta = ddc::get<Theta>(coord);
161 const double tmp1
162 = Kokkos::sqrt(m_epsilon * (m_epsilon + 2.0 * r * Kokkos::cos(theta)) + 1.0);
163
164 const double x = (1.0 - tmp1) / m_epsilon;
165 const double y = m_e * r * Kokkos::sin(theta)
166 / (Kokkos::sqrt(1.0 - 0.25 * m_epsilon * m_epsilon) * (2.0 - tmp1));
167
168 return Coord<X, Y>(x, y);
169 }

179 KOKKOS_FUNCTION double jacobian(Coord<R, Theta> const& coord) const
180 {
181 const double r = ddc::get<R>(coord);
182 const double theta = ddc::get<Theta>(coord);
183 const double xi = Kokkos::sqrt(1. / (1. - m_epsilon * m_epsilon * 0.25));
184 return -r / Kokkos::sqrt(1 + m_epsilon * (m_epsilon + 2.0 * r * Kokkos::cos(theta))) * m_e
185 * xi
186 / (2 - Kokkos::sqrt(1 + m_epsilon * (m_epsilon + 2.0 * r * Kokkos::cos(theta))));
187 }

208 KOKKOS_FUNCTION void jacobian_matrix(Coord<R, Theta> const& coord, Matrix_2x2& matrix) const
209 {
210 const double r = ddc::get<R>(coord);
211 const double theta = ddc::get<Theta>(coord);
212
213 const double sin_theta = Kokkos::sin(theta);
214 const double cos_theta = Kokkos::cos(theta);
215 const double xi2 = 1. / (1. - m_epsilon * m_epsilon * 0.25);
216 const double xi = Kokkos::sqrt(xi2);
217 const double sqrt_eps = Kokkos::sqrt(m_epsilon * (m_epsilon + 2.0 * r * cos_theta) + 1.0);
218 const double sqrt_eps_2 = 2.0 - sqrt_eps;
219
220 matrix[0][0] = -cos_theta / sqrt_eps;
221 matrix[0][1] = r * sin_theta / sqrt_eps;
222 matrix[1][0] = m_e * m_epsilon * r * sin_theta * cos_theta * xi
223 / (sqrt_eps_2 * sqrt_eps_2 * sqrt_eps)
224 + m_e * sin_theta * xi / sqrt_eps_2;
225 matrix[1][1] = r
226 * (-m_e * m_epsilon * r * sin_theta * sin_theta * xi
227 / (sqrt_eps_2 * sqrt_eps_2 * sqrt_eps)
228 + m_e * cos_theta * xi / sqrt_eps_2);
229 }

341 KOKKOS_FUNCTION void inv_jacobian_matrix(Coord<R, Theta> const& coord, Matrix_2x2& matrix) const
342 {
343 const double r = ddc::get<R>(coord);
344 const double theta = ddc::get<Theta>(coord);
345
346 assert(r >= 1e-15);
347
348 const double sin_theta = Kokkos::sin(theta);
349 const double cos_theta = Kokkos::cos(theta);
350 const double xi = Kokkos::sqrt(1. / (1. - m_epsilon * m_epsilon * 0.25));
351 const double divisor = 2 - Kokkos::sqrt(1 + m_epsilon * (m_epsilon + 2.0 * r * cos_theta));
352
353 const double fact_1 = 1 / Kokkos::sqrt(1 + m_epsilon * (m_epsilon + 2.0 * r * cos_theta));
354 const double fact_2 = m_e * m_epsilon * xi * r * sin_theta * fact_1 / divisor / divisor;
355 const double fact_3 = m_e * xi / divisor;
356
357 matrix[0][0] = -1 / fact_1 * (-sin_theta * fact_2 + cos_theta * fact_3) / fact_3;
358 matrix[0][1] = sin_theta / fact_3;
359 matrix[1][0] = 1 / r / fact_1 * (cos_theta * fact_2 + sin_theta * fact_3) / fact_3;
360 matrix[1][1] = 1 / r * cos_theta / fact_3;
361 }
