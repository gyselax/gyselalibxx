167 KOKKOS_FUNCTION Coord<X, Y> operator()(Coord<R, Theta> const& coord) const
168 {
169 const double r = ddc::get<R>(coord);
170 const double theta = ddc::get<Theta>(coord);
171 const double tmp1
172 = Kokkos::sqrt(m_epsilon * (m_epsilon + 2.0 * r * Kokkos::cos(theta)) + 1.0);
173
174 const double x = (1.0 - tmp1) / m_epsilon;
175 const double y = m_e * r * Kokkos::sin(theta)
176 / (Kokkos::sqrt(1.0 - 0.25 * m_epsilon * m_epsilon) * (2.0 - tmp1));
177
178 return Coord<X, Y>(x, y);
179 }

189 KOKKOS_FUNCTION double jacobian(Coord<R, Theta> const& coord) const
190 {
191 const double r = ddc::get<R>(coord);
192 const double theta = ddc::get<Theta>(coord);
193 const double xi = Kokkos::sqrt(1. / (1. - m_epsilon * m_epsilon * 0.25));
194 return -r / Kokkos::sqrt(1 + m_epsilon * (m_epsilon + 2.0 * r * Kokkos::cos(theta))) * m_e
195 * xi
196 / (2 - Kokkos::sqrt(1 + m_epsilon * (m_epsilon + 2.0 * r * Kokkos::cos(theta))));
197 }

217 KOKKOS_FUNCTION DTensor<VectorIndexSet<X, Y>, VectorIndexSet<R_cov, Theta_cov>> jacobian_matrix(
218 Coord<R, Theta> const& coord) const
219 {
220 const double r = ddc::get<R>(coord);
221 const double theta = ddc::get<Theta>(coord);
222
223 const double sin_theta = Kokkos::sin(theta);
224 const double cos_theta = Kokkos::cos(theta);
225 const double xi2 = 1. / (1. - m_epsilon * m_epsilon * 0.25);
226 const double xi = Kokkos::sqrt(xi2);
227 const double sqrt_eps = Kokkos::sqrt(m_epsilon * (m_epsilon + 2.0 * r * cos_theta) + 1.0);
228 const double sqrt_eps_2 = 2.0 - sqrt_eps;
229
230 DTensor<VectorIndexSet<X, Y>, VectorIndexSet<R_cov, Theta_cov>> jacobian_matrix;
231 ddcHelper::get<X, R_cov>(jacobian_matrix) = -cos_theta / sqrt_eps;
232 ddcHelper::get<X, Theta_cov>(jacobian_matrix) = r * sin_theta / sqrt_eps;
233 ddcHelper::get<Y, R_cov>(jacobian_matrix) = m_e * m_epsilon * r * sin_theta * cos_theta * xi
234 / (sqrt_eps_2 * sqrt_eps_2 * sqrt_eps)
235 + m_e * sin_theta * xi / sqrt_eps_2;
236 ddcHelper::get<Y, Theta_cov>(jacobian_matrix)
237 = r
238 * (-m_e * m_epsilon * r * sin_theta * sin_theta * xi
239 / (sqrt_eps_2 * sqrt_eps_2 * sqrt_eps)
240 + m_e * cos_theta * xi / sqrt_eps_2);
241 return jacobian_matrix;
242 }

293 KOKKOS_FUNCTION double jacobian_21(Coord<R, Theta> const& coord) const
294 {
295 const double r = ddc::get<R>(coord);
296 const double theta = ddc::get<Theta>(coord);

353 KOKKOS_FUNCTION DTensor<VectorIndexSet<R, Theta>, VectorIndexSet<X_cov, Y_cov>>
354 inv_jacobian_matrix(Coord<R, Theta> const& coord) const
355 {
356 const double r = ddc::get<R>(coord);
357 const double theta = ddc::get<Theta>(coord);
358
359 assert(r >= 1e-15);
360
361 const double sin_theta = Kokkos::sin(theta);
362 const double cos_theta = Kokkos::cos(theta);
363 const double xi = Kokkos::sqrt(1. / (1. - m_epsilon * m_epsilon * 0.25));
364 const double divisor = 2 - Kokkos::sqrt(1 + m_epsilon * (m_epsilon + 2.0 * r * cos_theta));
365
366 const double fact_1 = 1 / Kokkos::sqrt(1 + m_epsilon * (m_epsilon + 2.0 * r * cos_theta));
367 const double fact_2 = m_e * m_epsilon * xi * r * sin_theta * fact_1 / divisor / divisor;
368 const double fact_3 = m_e * xi / divisor;
369
370 DTensor<VectorIndexSet<R, Theta>, VectorIndexSet<X_cov, Y_cov>> matrix;
371 ddcHelper::get<R, X_cov>(matrix)
372 = -1 / fact_1 * (-sin_theta * fact_2 + cos_theta * fact_3) / fact_3;
373 ddcHelper::get<R, Y_cov>(matrix) = sin_theta / fact_3;
374 ddcHelper::get<Theta, X_cov>(matrix)
375 = 1 / r / fact_1 * (cos_theta * fact_2 + sin_theta * fact_3) / fact_3;
376 ddcHelper::get<Theta, Y_cov>(matrix) = 1 / r * cos_theta / fact_3;
377 return matrix;
378 }
