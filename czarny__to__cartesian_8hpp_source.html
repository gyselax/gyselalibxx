163  ddc::Coordinate<DimR, DimP> operator()(ddc::Coordinate<DimX, DimY> const& coord) const
164  {
165  const double x = ddc::get<DimX>(coord);
166  const double y = ddc::get<DimY>(coord);
167  const double ex = 1. + m_epsilon * x;
168  const double ex2 = (m_epsilon * x * x - 2. * x - m_epsilon);
169  const double xi2 = 1. / (1. - m_epsilon * m_epsilon * 0.25);
170  const double xi = std::sqrt(xi2);
171  const double r = std::sqrt(y * y * ex * ex / (m_e * m_e * xi2) + ex2 * ex2 * 0.25);
172  double theta
173  = std::atan2(2. * y * ex, (m_e * xi * (m_epsilon * x * x - 2. * x - m_epsilon)));
174  if (theta < 0) {
175  theta = 2 * M_PI + theta;
176  }
177  return ddc::Coordinate<DimR, DimP>(r, theta);

200  const double sqrt_eps_2 = 2.0 - sqrt_eps;
