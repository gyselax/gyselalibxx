168  KOKKOS_FUNCTION ddc::Coordinate<R, Theta> operator()(
169  ddc::Coordinate<X, Y> const& coord) const final
170  {
171  const double x = ddc::get<X>(coord);
172  const double y = ddc::get<Y>(coord);
173  const double ex = 1. + m_epsilon * x;
174  const double ex2 = (m_epsilon * x * x - 2. * x - m_epsilon);
175  const double xi2 = 1. / (1. - m_epsilon * m_epsilon * 0.25);
176  const double xi = Kokkos::sqrt(xi2);
177  const double r = Kokkos::sqrt(y * y * ex * ex / (m_e * m_e * xi2) + ex2 * ex2 * 0.25);
178  double theta
179  = Kokkos::atan2(2. * y * ex, (m_e * xi * (m_epsilon * x * x - 2. * x - m_epsilon)));
180  if (theta < 0) {
181  theta = 2 * M_PI + theta;
182  }
183  return ddc::Coordinate<R, Theta>(r, theta);
