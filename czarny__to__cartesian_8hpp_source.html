159  {

161  const double theta = ddc::get<Theta>(coord);
162  const double tmp1
163  = Kokkos::sqrt(m_epsilon * (m_epsilon + 2.0 * r * Kokkos::cos(theta)) + 1.0);

209  KOKKOS_FUNCTION void jacobian_matrix(ddc::Coordinate<R, Theta> const& coord, Matrix_2x2& matrix)
210  const
211  {
212  const double r = ddc::get<R>(coord);
213  const double theta = ddc::get<Theta>(coord);
214 
215  const double sin_theta = Kokkos::sin(theta);
216  const double cos_theta = Kokkos::cos(theta);
217  const double xi2 = 1. / (1. - m_epsilon * m_epsilon * 0.25);
218  const double xi = Kokkos::sqrt(xi2);
219  const double sqrt_eps = Kokkos::sqrt(m_epsilon * (m_epsilon + 2.0 * r * cos_theta) + 1.0);
220  const double sqrt_eps_2 = 2.0 - sqrt_eps;
221 
222  matrix[0][0] = -cos_theta / sqrt_eps;
223  matrix[0][1] = r * sin_theta / sqrt_eps;
224  matrix[1][0] = m_e * m_epsilon * r * sin_theta * cos_theta * xi
225  / (sqrt_eps_2 * sqrt_eps_2 * sqrt_eps)
226  + m_e * sin_theta * xi / sqrt_eps_2;
227  matrix[1][1] = r
228  * (-m_e * m_epsilon * r * sin_theta * sin_theta * xi
229  / (sqrt_eps_2 * sqrt_eps_2 * sqrt_eps)
230  + m_e * cos_theta * xi / sqrt_eps_2);
231  }

343  KOKKOS_FUNCTION void inv_jacobian_matrix(
344  ddc::Coordinate<R, Theta> const& coord,
345  Matrix_2x2& matrix) const
346  {
347  const double r = ddc::get<R>(coord);
348  const double theta = ddc::get<Theta>(coord);
349 
350  assert(r >= 1e-15);
351 
352  const double sin_theta = Kokkos::sin(theta);
353  const double cos_theta = Kokkos::cos(theta);
354  const double xi = Kokkos::sqrt(1. / (1. - m_epsilon * m_epsilon * 0.25));
355  const double divisor = 2 - Kokkos::sqrt(1 + m_epsilon * (m_epsilon + 2.0 * r * cos_theta));
356 
357  const double fact_1 = 1 / Kokkos::sqrt(1 + m_epsilon * (m_epsilon + 2.0 * r * cos_theta));
358  const double fact_2 = m_e * m_epsilon * xi * r * sin_theta * fact_1 / divisor / divisor;
359  const double fact_3 = m_e * xi / divisor;
360 
361  matrix[0][0] = -1 / fact_1 * (-sin_theta * fact_2 + cos_theta * fact_3) / fact_3;
362  matrix[0][1] = sin_theta / fact_3;
363  matrix[1][0] = 1 / r / fact_1 * (cos_theta * fact_2 + sin_theta * fact_3) / fact_3;
364  matrix[1][1] = 1 / r * cos_theta / fact_3;
365  }
