154  DSpanRP electrostatic_potential,
155  VectorFieldSpan<double, IDomainRP, NDTag<RDimX, RDimY>> electric_field,
156  DViewRP allfdistribu) const
157  {
158  auto const grid = ddc::get_domain<IDimR, IDimP>(allfdistribu);
159 
160  FieldRP<CoordRP> coords(grid);
161  ddc::for_each(grid, [&](IndexRP const irp) { coords(irp) = ddc::coordinate(irp); });
162 
163  auto const dom_bsplinesRP = m_rhs_builder.spline_domain();
164 
165  Spline2D allfdistribu_coef(dom_bsplinesRP);
166 
167  BSDomainR radial_bsplines(ddc::discrete_space<BSplinesR>().full_domain().remove_first(
168  ddc::DiscreteVector<BSplinesR> {n_overlap_cells}));
169  BSDomainP polar_domain(ddc::discrete_space<BSplinesP>().full_domain());
170  SplinePolar electrostatic_potential_coef(
171  PolarBSplinesRP::singular_domain(),
172  BSDomainRP(radial_bsplines, polar_domain));
173 
174 
175  m_rhs_builder(allfdistribu_coef, allfdistribu);
176  PoissonRHSFunction const charge_density_coord(allfdistribu_coef, m_rhs_evaluator);
177 
178  m_poisson_solver(charge_density_coord, electrostatic_potential_coef);
179 
180 
181  m_polar_spline_evaluator(
182  electrostatic_potential.span_view(),
183  coords.span_cview(),
184  electrostatic_potential_coef);
185 
186 
187  // Computation of the advection_field:
188  DFieldRP deriv_r_phi(grid);
189  DFieldRP deriv_p_phi(grid);
190 
191 
192  m_polar_spline_evaluator.deriv_dim_1(
193  deriv_r_phi.span_view(),
194  coords.span_cview(),
195  electrostatic_potential_coef);
196  m_polar_spline_evaluator.deriv_dim_2(
197  deriv_p_phi.span_view(),
198  coords.span_cview(),
199  electrostatic_potential_coef);
200 
201  ddc::for_each(grid, [&](IndexRP const irp) {
202  double const r = ddc::coordinate(ddc::select<IDimR>(irp));
203  double const th = ddc::coordinate(ddc::select<IDimP>(irp));
204 
205  if (r > m_epsilon) {
206  CoordRP const coord_rp(r, th);
207 
208  Matrix_2x2 J; // Jacobian matrix
209  m_mapping.jacobian_matrix(coord_rp, J);
210  Matrix_2x2 inv_G; // Inverse metric tensor
211  m_mapping.inverse_metric_tensor(coord_rp, inv_G);
212 
213  // E = -grad phi
214  double const electric_field_r
215  = -deriv_r_phi(irp) * inv_G[0][0] - deriv_p_phi(irp) * inv_G[1][0];
216  double const electric_field_th
217  = -deriv_r_phi(irp) * inv_G[0][1] - deriv_p_phi(irp) * inv_G[1][1];
218 
219  // compute the electric field in the physical domain (Cartesian domain)
220  ddcHelper::get<RDimX>(electric_field)(irp)
221  = electric_field_r * J[0][0] + electric_field_th * J[0][1];
222  ddcHelper::get<RDimY>(electric_field)(irp)
223  = electric_field_r * J[1][0] + electric_field_th * J[1][1];
224 
225  } else {
226  // Linearisation of the electric field
227  double const th1 = M_PI / 4.;
228  double const th2 = -M_PI / 4. + 2 * M_PI;
229 
230  // --- Value at r = 0:
231  CoordRP const coord_1_0(0, th1);
232  CoordRP const coord_2_0(0, th2);
233 
234  double const dr_x_1 = m_mapping.jacobian_11(coord_1_0); // dr_x (0, th1)
235  double const dr_y_1 = m_mapping.jacobian_21(coord_1_0); // dr_y (0, th1)
236 
237  double const dr_x_2 = m_mapping.jacobian_11(coord_2_0); // dr_x (0, th2)
238  double const dr_y_2 = m_mapping.jacobian_21(coord_2_0); // dr_y (0, th2)
239 
240  double deriv_r_phi_1
241  = m_polar_spline_evaluator
242  .deriv_dim_1(coord_1_0, electrostatic_potential_coef);
243  double deriv_r_phi_2
244  = m_polar_spline_evaluator
245  .deriv_dim_1(coord_2_0, electrostatic_potential_coef);
246 
247  double const deriv_y_phi_0 = 1 / (dr_y_2 - dr_y_1 * dr_x_2 / dr_x_1)
248  * (deriv_r_phi_2 - deriv_r_phi_1 * dr_x_2 / dr_x_1);
249 
250  double const deriv_x_phi_0 = 1 / dr_x_1 * (deriv_r_phi_1 - deriv_y_phi_0 * dr_y_1);
251 
252  // E = -grad phi
253  double const electric_field_x_0 = -deriv_x_phi_0;
254  double const electric_field_y_0 = -deriv_y_phi_0;
255 
256 
257 
258  // --- Value at r = m_epsilon:
259  CoordRP const coord_rp_epsilon(m_epsilon, th);
260  Matrix_2x2 J_eps; // Jacobian matrix
261  m_mapping.jacobian_matrix(coord_rp_epsilon, J_eps);
262  Matrix_2x2 inv_G_eps; // Inverse metric tensor
263  m_mapping.inverse_metric_tensor(coord_rp_epsilon, inv_G_eps);
264 
265  double deriv_r_phi_epsilon
266  = m_polar_spline_evaluator
267  .deriv_dim_1(coord_rp_epsilon, electrostatic_potential_coef);
268  double deriv_p_phi_epsilon
269  = m_polar_spline_evaluator
270  .deriv_dim_2(coord_rp_epsilon, electrostatic_potential_coef);
271 
272 
273  // E = -grad phi
274  double const electric_field_r = -deriv_r_phi_epsilon * inv_G_eps[0][0]
275  - deriv_p_phi_epsilon * inv_G_eps[1][0];
276  double const electric_field_th = -deriv_r_phi_epsilon * inv_G_eps[0][1]
277  - deriv_p_phi_epsilon * inv_G_eps[1][1];
278 
279  // compute the electric field in the physical domain (cartesian domain)
280  double const electric_field_x_epsilon
281  = electric_field_r * J_eps[0][0] + electric_field_th * J_eps[0][1];
282  double const electric_field_y_epsilon
283  = electric_field_r * J_eps[1][0] + electric_field_th * J_eps[1][1];
284 
285 
286  // --- Linearisation:
287  ddcHelper::get<RDimX>(electric_field)(irp)
288  = electric_field_x_0 * (1 - r / m_epsilon)
289  + electric_field_x_epsilon * r / m_epsilon;
290  ddcHelper::get<RDimY>(electric_field)(irp)
291  = electric_field_y_0 * (1 - r / m_epsilon)
292  + electric_field_y_epsilon * r / m_epsilon;
293  }
294  });
295  }
