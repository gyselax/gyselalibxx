129 void do_configuration_data_initialisation(
130 double input_B_norm,
131 DConstField<IdxRangeSpVparType> input_Bstar_s)
132 {
133 {
134 /* input_B_norm is a scalar when there is no Theta nor R dimensions.
135 */
136 ddc::parallel_fill(
137 Kokkos::DefaultExecutionSpace(),
138 get_field(m_operator_quantities.m_B_norm_allocation),
139 input_B_norm);
140 }
141
142 {
143 /* Koliop's Bstar_s is needed in a format layout different from the
144 * rest of the code. It is needed in layout right
145 * [sp, theta, r, vpar] where as the input is in [sp, vpar].
146 * We simply iterate over [sp, theta, r, vpar] and copy the input
147 * [sp, vpar] for every theta, r.
148 */
149 CollisionConfigurationDataType::transpose_replicate_deep_copy(
150 get_field(m_operator_quantities.m_Bstar_s_allocation),
151 input_Bstar_s);
152 }
153
154 {
155 /* Computation of the radial profile of ADi that is required for
156 * koliop interface
157 * @f[ AD(rpeak) = \sqrt(2)*eps_rpeak^(3/2)/(q_rpeak R0)*nustar0_rpeak @f]
158 * eps_rpeak = eps(rpeak) with eps(r) = r/R0 is the aspect ratio.
159 *
160 * In this specific case, where AD is a scalar all values are forced
161 * to 1, so coeff_AD_atrpeak = sqrt(2)
162 */
163 double const coeff_AD = Kokkos::sqrt(2.0);
164
165 /* [TODO] According to Peter D., the places where coeff_AD is needed
166 * can resynthesise the quantities from a scalar and some other
167 * fields that koliop is already aware of. This change has not
168 * landed in Gysela Fortran yet. For koliop to make do with a scalar
169 * and not a radial profile, it would require koliop to modify its
170 * interface and the Cpar koliop operator computation to be modified
171 * slightly. For now, the uniform coeff_AD_rpeak is applied to
172 * produce a koliop compatible radial profile.
173 */
174 ddc::parallel_fill(
175 Kokkos::DefaultExecutionSpace(),
176 get_field(m_operator_quantities.m_coeff_AD_allocation),
177 coeff_AD);
178 }
179
180 Kokkos::fence();
181 }
