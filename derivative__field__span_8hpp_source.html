175  template <std::size_t ArrayIndex, class Tag>
176  KOKKOS_FUNCTION constexpr ddc::DiscreteElement<Tag> get_chunk_subdomain_1d_idx()
177  {
178  // The Tag describes a dimension for which a derivative is defined
179  if constexpr (ddc::in_tags_v<Tag, deriv_tags>) {
180  // If the Chunk at this index contains the derivatives of this dimension
181  if constexpr (ArrayIndex & (1 << ddc::type_seq_rank_v<Tag, deriv_tags>)) {
182  return ddc::DiscreteElement<Tag>(1);
183  }
184  // If the Chunk at this index doesn't contain derivatives of this dimension
185  else {
186  return ddc::DiscreteElement<Tag>(0);

191  }
