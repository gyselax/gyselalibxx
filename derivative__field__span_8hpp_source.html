169  template <std::size_t ArrayIndex, class Tag>
170  KOKKOS_FUNCTION constexpr ddc::DiscreteElement<Tag> get_chunk_subdomain_1d_idx()
171  {
172  // The Tag describes a dimension for which a derivative is defined
173  if constexpr (ddc::in_tags_v<Tag, deriv_tags>) {
174  // If the Chunk at this index contains the derivatives of this dimension
175  if constexpr (ArrayIndex & (1 << ddc::type_seq_rank_v<Tag, deriv_tags>)) {
176  return ddc::DiscreteElement<Tag>(1);
177  }
178  // If the Chunk at this index doesn't contain derivatives of this dimension
179  else {
180  return ddc::DiscreteElement<Tag>(0);
181  }
182  } else {
183  // Empty DiscreteDomain to be discarded
184  return ddc::DiscreteElement<Tag>();
185  }
186  }
