265  double norm(int batch_idx) const
266  {
267  int const tmp_mat_size = get_size();
268  int const tmp_batch_size = get_batch_size();
269  int const non_zeros = m_batch_matrix_ell->get_num_stored_elements_per_row();
270  double* vals_proxy = m_batch_matrix_ell->get_values();
271  Kokkos::LayoutStride values_layout(
272  tmp_batch_size,
273  non_zeros * tmp_mat_size,
274  tmp_mat_size,
275  1,
276  non_zeros,
277  tmp_mat_size);
278  Kokkos::View<double***, Kokkos::LayoutStride, typename ExecSpace::memory_space>
279  vals_view(vals_proxy, values_layout);
280 
281  double result = 0;
282  Kokkos::parallel_reduce(
283  "L-infinitty norm",
284  Kokkos::RangePolicy<ExecSpace>(0, tmp_mat_size),
285  KOKKOS_LAMBDA(int i, double& res) {
286  double row_sum = 0.;
287  for (int k = 0; k < non_zeros; k++) {
288  row_sum += Kokkos::abs(vals_view(batch_idx, i, k));
289  }
290  if (row_sum > res) {
291  res = row_sum;
292  }
293  },
294  Kokkos::Max<double>(result));
295  return result;
296  }

299  void save_logger(
300  DKokkosView2D const x_view,
301  DKokkosView2D const b_view,
302  std::shared_ptr<const gko::batch::log::BatchConvergence<double>> logger) const
303  {
304  std::shared_ptr const gko_exec = m_solver->get_executor();
305  // allocate the residual
306  auto res = gko::batch::MultiVector<double>::
307  create(gko_exec, gko::batch_dim<2>(get_batch_size(), gko::dim<2>(get_size(), 1)));
308  res->copy_from(to_gko_multivector(gko_exec, b_view));
309 
310  auto norm_dim = gko::batch_dim<2>(get_batch_size(), gko::dim<2>(1, 1));
311  // allocate rhs norm on host.
312  auto b_norm_host
313  = gko::batch::MultiVector<double>::create(gko_exec->get_master(), norm_dim);
314  b_norm_host->fill(0.0);
315  // allocate the residual norm on host.
316  auto res_norm_host
317  = gko::batch::MultiVector<double>::create(gko_exec->get_master(), norm_dim);
318  res_norm_host->fill(0.0);
319  // compute rhs norm.
320  to_gko_multivector(gko_exec, b_view)->compute_norm2(b_norm_host);
321  // we need constants on the device
322  auto one = gko::batch::MultiVector<double>::create(gko_exec, norm_dim);
323  one->fill(1.0);
324  auto neg_one = gko::batch::MultiVector<double>::create(gko_exec, norm_dim);
325  neg_one->fill(-1.0);
326  //to estimate the "true" residual, the apply function below computes Ax-res, and stores the result in res.
327  m_batch_matrix_ell->apply(one, to_gko_multivector(gko_exec, x_view), neg_one, res);
328  //compute residual norm.
329  res->compute_norm2(res_norm_host);
330 
331  auto log_resid_host
332  = gko::make_temporary_clone(gko_exec->get_master(), &logger->get_residual_norm());
333  auto log_iters_host
334  = gko::make_temporary_clone(gko_exec->get_master(), &logger->get_num_iterations());
335 
336  std::fstream log_file("logger.txt", std::fstream::out);
337  // "unbatch" converts a batch object into a vector
338  // of objects of the corresponding single type.
339  auto unb_res_norm = unbatch(res_norm_host.get());
340  auto unb_bnorm = unbatch(b_norm_host.get());
341  for (int i = 0; i < get_batch_size(); ++i) {
342  // Logger output
343  log_file << " System no. " << i
344  << ": Ax-b residual norm = " << unb_res_norm[i]->at(0, 0)
345  << ", implicit residual norm = " << log_resid_host->get_const_data()[i]
346  << ", iterations = " << log_iters_host->get_const_data()[i] << std::endl;
347  log_file << " unbatched bnorm at(i,0)" << unb_bnorm[i]->at(0, 0) << std::endl;
348  log_file << " unbatched residual norm at(i,0)" << unb_res_norm[i]->at(0, 0)
349  << std::endl;
350  if (!(unb_res_norm[i]->at(0, 0) <= m_tol)) {
351  log_file << "System " << i << " converged only to " << unb_res_norm[i]->at(0, 0)
352  << " relative residual." << std::endl;
353  }
354  }
355  log_file.close();
356  }

364  void check_conv(std::shared_ptr<const gko::batch::log::BatchConvergence<double>> logger) const
365  {

377  },
378  Kokkos::LAnd<bool>(has_converged));
379  if (!has_converged) {
