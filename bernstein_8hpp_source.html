155template <class DDim, class MemorySpace>
156void TriangularBernsteinPolynomialBasis<X, Y, Corner1Tag, Corner2Tag, Corner3Tag, D>::Impl<
157 DDim,
158 MemorySpace>::eval_basis(host_t<DField<IdxRange<DDim>>> values, Coord<X, Y> const& x) const
159{
160 const Coord<Corner1Tag, Corner2Tag, Corner3Tag> bary_coords = m_coord_changer(x);
161 const double l1 = ddc::get<Corner1Tag>(bary_coords);
162 const double l2 = ddc::get<Corner2Tag>(bary_coords);
163 const double l3 = ddc::get<Corner3Tag>(bary_coords);
164 assert(values.size() == nbasis());
165
166 Idx<DDim> idx(0);
167 for (std::size_t i(0); i < D + 1; ++i) {
168 for (std::size_t j(0); j < D + 1 - i; ++j, ++idx) {
169 const int k = D - i - j;
170 const double multinomial_coefficient
171 = factorial(D) / (factorial(i) * factorial(j) * factorial(k));
172 values(idx) = multinomial_coefficient * ipow(l1, i) * ipow(l2, j) * ipow(l3, k);
173 }
174 }
