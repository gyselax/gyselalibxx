46 template <typename Type>
47 class Is2DMapping
48 {
49  template <typename ClassType>
50  using coord_arg_type = typename ClassType::CoordArg;
51  template <typename ClassType>
52  using coord_result_type = typename ClassType::CoordResult;
53 
54  static bool constexpr is_2d_mapping()
55  {
56  constexpr bool success
57  = CheckClassAttributeExistence<Type, coord_arg_type>::has_attribute
58  && CheckClassAttributeExistence<Type, coord_result_type>::has_attribute;
59  if constexpr (success) {
60  using CoordArg = typename Type::CoordArg;
61  using CoordResult = typename Type::CoordResult;
62  return std::is_invocable_r_v<CoordResult, Type, CoordArg>;
63  }
64  return true;
65  }
66 
67 public:
69  static constexpr bool value = is_2d_mapping();
70 };
