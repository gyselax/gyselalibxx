60 explicit KOKKOS_DEFAULTED_FUNCTION ElectrostaticPotentialSimulation_decentred_rotation(

74 KOKKOS_FUNCTION double operator()(CoordXY const coord, double const t) const
75 {
76 double const x = ddc::get<X>(coord);
77 double const y = ddc::get<Y>(coord);
78 return m_omega * (0.5 * x * x - m_xc * x + 0.5 * y * y - m_yc * y);
79 }
80
91 KOKKOS_FUNCTION CoordXY exact_feet(CoordXY coord, double const t) const
92 {
93 double const x = ddc::get<X>(coord);
94 double const y = ddc::get<Y>(coord);
95 double const foot_x = m_xc + (x - m_xc) * Kokkos::cos(m_omega * -t)
96 - (y - m_yc) * Kokkos::sin(m_omega * -t);
97 double const foot_y = m_yc + (x - m_xc) * Kokkos::sin(m_omega * -t)
98 + (y - m_yc) * Kokkos::cos(m_omega * -t);
99 return CoordXY(foot_x, foot_y);
100 }
101};

238 KOKKOS_FUNCTION double operator()(CoordXY const coord, double const t) const
239 {
240 double const x = ddc::get<X>(coord);

242 return 0.5 * m_vtheta * (x * x + y * y);
243 }
244
255 KOKKOS_FUNCTION CoordXY exact_feet(CoordXY coord_xy, double const t) const
256 {
257 CoordRTheta const coord_rtheta(m_mapping(coord_xy));
258 CoordRTheta const velocity(m_vr, m_vtheta);

260 return logical_to_physical_mapping(coord_rtheta - t * velocity);
