77class ElectrostaticalPotentialSimulation_decentred_rotation
78 : public ElectrostaticalPotentialSimulation
79{
80private:
81 double const m_omega;
82 double const m_xc;
83 double const m_yc;
84 double const m_x_bar;
85 double const m_y_bar;
86
87public:
92 ElectrostaticalPotentialSimulation_decentred_rotation()
93 : m_omega(2 * M_PI)
94 , m_xc(0.25)
95 , m_yc(0.)
96 , m_x_bar(0.)
97 , m_y_bar(0.) {};
98 ~ElectrostaticalPotentialSimulation_decentred_rotation() {};
99
100 double operator()(CoordXY const coord, double const t) const final
101 {
102 double const x = ddc::get<X>(coord);
103 double const y = ddc::get<Y>(coord);
104 return m_omega * (-0.5 * x * x + m_xc * x - 0.5 * y * y + m_yc * y);
105 };
106
107 CoordXY exact_feet(CoordXY coord, double const t) const final
108 {
109 double const x = ddc::get<X>(coord);
110 double const y = ddc::get<Y>(coord);
111 double const foot_x
112 = m_xc + (x - m_xc) * std::cos(m_omega * -t) - (y - m_yc) * std::sin(m_omega * -t);
113 double const foot_y
114 = m_yc + (x - m_xc) * std::sin(m_omega * -t) + (y - m_yc) * std::cos(m_omega * -t);
115 return CoordXY(foot_x, foot_y);
116 };
117};

136class ElectrostaticalPotentialSimulation_translation : public ElectrostaticalPotentialSimulation

150 ElectrostaticalPotentialSimulation_translation(CoordVx vx, CoordVy vy)

152 ~ElectrostaticalPotentialSimulation_translation() {};
153
154 double operator()(CoordXY const coord, double const t) const final
155 {
156 double const vx = ddc::get<X>(m_velocity);
157 double const vy = ddc::get<Y>(m_velocity);
158 return -vy * ddc::get<X>(coord) + vx * ddc::get<Y>(coord);
159 };
160
161 CoordXY exact_feet(CoordXY coord, double const t) const final
162 {
163 return coord - t * m_velocity;
164 };

188class ElectrostaticalPotentialSimulation_rotation : public ElectrostaticalPotentialSimulation
189{
190private:
191 double const m_vr;
192 double const m_vtheta;
193 CircularToCartesian<X, Y, R, Theta> const m_mapping;
194
195public:
202 ElectrostaticalPotentialSimulation_rotation(CoordVtheta vtheta)
203 : m_vr(0)
204 , m_vtheta(vtheta)
205 , m_mapping() {};
206 ~ElectrostaticalPotentialSimulation_rotation() {};
207
208 double operator()(CoordXY const coord, double const t) const final
209 {
210 CoordRTheta const coord_rp(m_mapping(coord));
211 double const r = ddc::get<R>(coord_rp);
212 return -0.5 * r * r * m_vtheta;
213 };
214
215 CoordXY exact_feet(CoordXY coord_xy, double const t) const final
216 {
217 CoordRTheta const coord_rp(m_mapping(coord_xy));
218 CoordRTheta const velocity(m_vr, m_vtheta);
219 return m_mapping(coord_rp - t * velocity);
220 };
221};
