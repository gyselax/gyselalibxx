75 KOKKOS_FUNCTION double operator()(CoordXY const coord, double const t) const
76 {
77 double const x = ddc::get<X>(coord);
78 double const y = ddc::get<Y>(coord);
79 return m_omega * (-0.5 * x * x + m_xc * x - 0.5 * y * y + m_yc * y);
80 }

92 KOKKOS_FUNCTION CoordXY exact_feet(CoordXY coord, double const t) const
93 {
94 double const x = ddc::get<X>(coord);
95 double const y = ddc::get<Y>(coord);
96 double const foot_x = m_xc + (x - m_xc) * Kokkos::cos(m_omega * -t)
97 - (y - m_yc) * Kokkos::sin(m_omega * -t);
98 double const foot_y = m_yc + (x - m_xc) * Kokkos::sin(m_omega * -t)
99 + (y - m_yc) * Kokkos::cos(m_omega * -t);
100 return CoordXY(foot_x, foot_y);
101 }
102};
