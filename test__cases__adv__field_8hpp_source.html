36class ElectrostaticPotentialSimulation_decentred_rotation
37{
38private:
39 double const m_omega;
40 double const m_xc;
41 double const m_yc;
42 double const m_x_bar;
43 double const m_y_bar;
44
45public:
47 KOKKOS_FUNCTION ElectrostaticPotentialSimulation_decentred_rotation()
48 : m_omega(2 * M_PI)
49 , m_xc(0.25)
50 , m_yc(0.)
51 , m_x_bar(0.)
52 , m_y_bar(0.)
53 {
54 }
55
57 explicit KOKKOS_DEFAULTED_FUNCTION ElectrostaticPotentialSimulation_decentred_rotation(
58 ElectrostaticPotentialSimulation_decentred_rotation const&)
59 = default;
60
71 KOKKOS_FUNCTION double operator()(CoordXY const coord, double const t) const
72 {
73 double const x = ddc::get<X>(coord);
74 double const y = ddc::get<Y>(coord);
75 return m_omega * (-0.5 * x * x + m_xc * x - 0.5 * y * y + m_yc * y);
76 }
77
88 KOKKOS_FUNCTION CoordXY exact_feet(CoordXY coord, double const t) const
89 {
90 double const x = ddc::get<X>(coord);
91 double const y = ddc::get<Y>(coord);
92 double const foot_x = m_xc + (x - m_xc) * Kokkos::cos(m_omega * -t)
93 - (y - m_yc) * Kokkos::sin(m_omega * -t);
94 double const foot_y = m_yc + (x - m_xc) * Kokkos::sin(m_omega * -t)
95 + (y - m_yc) * Kokkos::cos(m_omega * -t);
96 return CoordXY(foot_x, foot_y);
97 }
98};

193class ElectrostaticPotentialSimulation_rotation
194{
195private:
196 double const m_vr;
197 double const m_vtheta;
198 CartesianToCircular<X, Y, R, Theta> const m_mapping;
199
200public:
207 explicit ElectrostaticPotentialSimulation_rotation(CoordVtheta vtheta)
208 : m_vr(0)
209 , m_vtheta(vtheta)
210 , m_mapping()
211 {
212 }
213
215 explicit KOKKOS_DEFAULTED_FUNCTION ElectrostaticPotentialSimulation_rotation(
216 ElectrostaticPotentialSimulation_rotation const&)
217 = default;
218
229 KOKKOS_FUNCTION double operator()(CoordXY const coord, double const t) const
230 {
231 CoordRTheta const coord_rtheta(m_mapping(coord));
232 double const r = ddc::get<R>(coord_rtheta);
233 return -0.5 * r * r * m_vtheta;
234 }
235
246 KOKKOS_FUNCTION CoordXY exact_feet(CoordXY coord_xy, double const t) const
247 {
248 CoordRTheta const coord_rtheta(m_mapping(coord_xy));
249 CoordRTheta const velocity(m_vr, m_vtheta);
250 CircularToCartesian<R, Theta, X, Y> logical_to_physical_mapping;
251 return logical_to_physical_mapping(coord_rtheta - t * velocity);
252 }
253};
