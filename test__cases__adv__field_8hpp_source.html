77class ElectrostaticalPotentialSimulation_decentred_rotation
78 : public ElectrostaticalPotentialSimulation
79{
80private:
81 double const m_omega;
82 double const m_xc;
83 double const m_yc;
84 double const m_x_bar;
85 double const m_y_bar;
86
87public:
92 ElectrostaticalPotentialSimulation_decentred_rotation()
93 : m_omega(2 * M_PI)
94 , m_xc(0.25)
95 , m_yc(0.)
96 , m_x_bar(0.)
97 , m_y_bar(0.)
98 {
99 }
100
101 double operator()(CoordXY const coord, double const t) const final
102 {
103 double const x = ddc::get<X>(coord);
104 double const y = ddc::get<Y>(coord);
105 return m_omega * (-0.5 * x * x + m_xc * x - 0.5 * y * y + m_yc * y);
106 }
107
108 CoordXY exact_feet(CoordXY coord, double const t) const final
109 {
110 double const x = ddc::get<X>(coord);
111 double const y = ddc::get<Y>(coord);
112 double const foot_x
113 = m_xc + (x - m_xc) * std::cos(m_omega * -t) - (y - m_yc) * std::sin(m_omega * -t);
114 double const foot_y
115 = m_yc + (x - m_xc) * std::sin(m_omega * -t) + (y - m_yc) * std::cos(m_omega * -t);
116 return CoordXY(foot_x, foot_y);
117 }
118};

137class ElectrostaticalPotentialSimulation_translation : public ElectrostaticalPotentialSimulation
138{
139private:
140 CoordXY const m_velocity;
141
142public:
151 ElectrostaticalPotentialSimulation_translation(CoordVx vx, CoordVy vy)
152 : m_velocity(ddc::get<Vx>(vx), ddc::get<Vy>(vy))
153 {
154 }
155
156 double operator()(CoordXY const coord, double const t) const final
157 {
158 double const vx = ddc::get<X>(m_velocity);
159 double const vy = ddc::get<Y>(m_velocity);
160 return -vy * ddc::get<X>(coord) + vx * ddc::get<Y>(coord);
161 }
162
163 CoordXY exact_feet(CoordXY coord, double const t) const final
164 {
165 return coord - t * m_velocity;
166 }
167};

190class ElectrostaticalPotentialSimulation_rotation : public ElectrostaticalPotentialSimulation
191{
192private:
193 double const m_vr;
194 double const m_vtheta;
195 CartesianToCircular<X, Y, R, Theta> const m_mapping;
196
197public:
204 explicit ElectrostaticalPotentialSimulation_rotation(CoordVtheta vtheta)
205 : m_vr(0)
206 , m_vtheta(vtheta)
207 , m_mapping()
208 {
209 }
210
211 double operator()(CoordXY const coord, double const t) const final
212 {
213 CoordRTheta const coord_rp(m_mapping(coord));
214 double const r = ddc::get<R>(coord_rp);
215 return -0.5 * r * r * m_vtheta;
216 }
217
218 CoordXY exact_feet(CoordXY coord_xy, double const t) const final
219 {
220 CoordRTheta const coord_rp(m_mapping(coord_xy));
221 CoordRTheta const velocity(m_vr, m_vtheta);
222 CircularToCartesian<R, Theta, X, Y> logical_to_physical_mapping;
223 return logical_to_physical_mapping(coord_rp - t * velocity);
224 }
225};
